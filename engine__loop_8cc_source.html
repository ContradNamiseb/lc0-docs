47void RunEngine(SearchFactory* factory) {
48 CERR << "Search algorithm: " << factory->GetName();
49 StdoutUciResponder uci_responder;
50
51 // Populate options from various sources.
52 OptionsParser options_parser;
53 options_parser.Add<StringOption>(kLogFileId);
54 ConfigFile::PopulateOptions(&options_parser);
55 Engine::PopulateOptions(&options_parser);
56 if (factory) factory->PopulateParams(&options_parser); // Search params.
57 uci_responder.PopulateParams(&options_parser); // UCI params.
58 SharedBackendParams::Populate(&options_parser);
59
60 // Parse flags, show help, initialize logging, read config etc.
61 if (!ConfigFile::Init() || !options_parser.ProcessAllFlags()) return;
62 const auto options = options_parser.GetOptionsDict();
63 Logging::Get().SetFilename(options.Get<std::string>(kLogFileId));
64
65 // Create engine.
66 Engine engine(*factory, options);
67 UciLoop loop(&uci_responder, &options_parser, &engine);
68
69 // Run the stdin loop.
70 std::cout.setf(std::ios::unitbuf);
71 std::string line;
72 while (std::getline(std::cin, line)) {
73 LOGFILE << ">> " << line;
74 try {
75 if (!loop.ProcessLine(line)) break;
76 // Set the log filename for the case it was set in UCI option.
77 Logging::Get().SetFilename(options.Get<std::string>(kLogFileId));
78 } catch (Exception& ex) {
79 uci_responder.SendRawResponse(std::string("error ") + ex.what());
80 }
81 }
82}
