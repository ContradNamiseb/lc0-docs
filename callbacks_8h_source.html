42// Is sent when search decides on the best move.
43struct BestMoveInfo {
44 BestMoveInfo(Move bestmove, Move ponder = Move{})
45 : bestmove(bestmove), ponder(ponder) {}
46 Move bestmove;
47 Move ponder;
48 // Those are extensions and not really UCI protocol.
49 // 1 if it's "player1", 2 if it's "player2"
50 int player = -1;
51 // Index of the game in the tournament (0-based).
52 int game_id = -1;
53 // The color of the player, if known.
54 std::optional<bool> is_black;
55};

150// The responder which forwards the output to another responder, with
151// observer-like subscription model.
152class UciResponderForwarder : public UciResponder {
153 public:
154 void OutputBestMove(BestMoveInfo* info) override {
155 if (wrapped_) wrapped_->OutputBestMove(info);
156 }
157 void OutputThinkingInfo(std::vector<ThinkingInfo>* infos) override {
158 if (wrapped_) wrapped_->OutputThinkingInfo(infos);
159 }
160 void Register(UciResponder* wrapped) {
161 if (wrapped_) {
162 throw Exception("UciResponderForwarder already has a wrapped responder");
163 }
164 wrapped_ = wrapped;
165 }
166 void Unregister(UciResponder* wrapped) {
167 if (wrapped_ != wrapped) {
168 throw Exception(
169 "UciResponderForwarder doesn't have this wrapped responder");
170 }
171 wrapped_ = nullptr;
172 }
173
174 private:
175 UciResponder* wrapped_ = nullptr;
176};

178// The responder which calls callbacks. Used for easier transition from old
179// code.
180class CallbackUciResponder : public UciResponder {
181 public:
182 using ThinkingCallback =
183 std::function<void(const std::vector<ThinkingInfo>&)>;
184 using BestMoveCallback = std::function<void(const BestMoveInfo&)>;
185
186 CallbackUciResponder(BestMoveCallback bestmove, ThinkingCallback info)
187 : bestmove_callback_(bestmove), info_callback_(info) {}
188
189 void OutputBestMove(BestMoveInfo* info) { bestmove_callback_(*info); }
190 void OutputThinkingInfo(std::vector<ThinkingInfo>* infos) {
191 info_callback_(*infos);
192 }
193
194 private:
195 const BestMoveCallback bestmove_callback_;
196 const ThinkingCallback info_callback_;
197};

215// Base class for uci response transformations.
216class TransformingUciResponder : public UciResponder {
217 public:
218 TransformingUciResponder(std::unique_ptr<UciResponder> parent)
219 : parent_(std::move(parent)) {}
220
221 virtual void TransformBestMove(BestMoveInfo*) {}
222 virtual void TransformThinkingInfo(std::vector<ThinkingInfo>*) {}
223
224 private:
225 virtual void OutputBestMove(BestMoveInfo* info) {
226 TransformBestMove(info);
227 parent_->OutputBestMove(info);
228 }
229 virtual void OutputThinkingInfo(std::vector<ThinkingInfo>* infos) {
230 TransformThinkingInfo(infos);
231 parent_->OutputThinkingInfo(infos);
232 }
233 std::unique_ptr<UciResponder> parent_;
234};
