106TEST_F(EngineTest, BackendReloadByUpdateBackendConfig) {
107 WaitingUciResponder uci_responder;
108 Engine engine(search_factory_, *options_);
109 engine.RegisterUciResponder(&uci_responder);
110 EXPECT_EQ(backend_, nullptr); // Backend not created before the search.
111 EXPECT_CALL(*search_, StartSearch(_)).WillRepeatedly([&](const GoParams&) {
112 std::this_thread::sleep_for(std::chrono::milliseconds(30));
113 static BestMoveInfo bestmove_info(Move::White(kSquareE1, kSquareA1));
114 search_->GetUciResponder()->OutputBestMove(&bestmove_info);
115 });
116 engine.Go(GoParams{.nodes = 10});
117 uci_responder.Wait();
118 EXPECT_NE(backend_, nullptr); // Backend created after the search.
119 Backend* prev_backend = backend_;
120 EXPECT_CALL(*backend_, UpdateConfiguration(_))
121 .WillOnce(Return(Backend::UPDATE_OK));
122 engine.NewGame();
123 engine.Go(GoParams{.nodes = 10});
124 uci_responder.Wait();
125 EXPECT_EQ(backend_, prev_backend); // Backend not recreated.
126 EXPECT_CALL(*backend_, UpdateConfiguration(_))
127 .WillOnce(Return(Backend::NEED_RESTART));
128 engine.Go(GoParams{.nodes = 10}); // Go alone should not restart the backend.
129 uci_responder.Wait();
130 EXPECT_EQ(backend_, prev_backend);
131 engine.NewGame();
132 EXPECT_NE(backend_, prev_backend); // Backend recreated.
133}
