72inline uint16_t FP32toFP16(float f32) {
73 uint32_t x = bit_cast<uint32_t>(f32);
74 uint32_t sign = (x & 0x80000000) >> 16;
75 x &= 0x7fffffff;
76 if (x < 0x477ff000) {
77 if (x >= 0x387ff000) {
78 // Normal fp16 result. Adjust exponent and round to nearest even.
79 // Branchless idea from <https://gist.github.com/rygorous/2156668>.
80 x += (x >> 13) & 1;
81 x -= 0x37fff001;
82 x >>= 13;
83 } else {
84 // Subnormal or zero. The result is the last bits of fabs(f32) + 0.5f.
85 x = bit_cast<uint32_t>(bit_cast<float>(x) + 0.5f);
86 }
87 } else {
88 if (x > 0x7f800000) {
89 // NaN
90 x = ((x >> 13) - 0x38000) | 0x200;
91 } else {
92 // Inf
93 x = 0x7c00;
94 }
95 }
96 return x | sign;
97}

99inline float FP16toFP32(uint16_t f16) {
100 int32_t s = static_cast<int16_t>(f16);
101 uint32_t x;
102 float f;
103 if ((s & 0x7c00) == 0) {
104 // Subnormal or zero. Scale to float.
105 x = s & 0x7fff;
106 f = 5.9604645e-8f * x;
107 x = bit_cast<uint32_t>(f);
108 if (s & 0x8000) x |= 0x80000000;
109 } else if ((s & 0x7c00) == 0x7c00) {
110 // Inf or NaN. Adjust exponent and shift.
111 if (s & 0x1ff) s |= 0x200; // Change sNaN to qNaN as intel does.
112 x = ((s & 0x47fff) + 0x38000) << 13;
113 } else {
114 // Normal. Adjust exponent and shift.
115 x = ((s & 0x47fff) + 0x1c000U) << 13;
116 }
117 return bit_cast<float>(x);
118}
