36Square SingleSquare(BitBoard input) {
37 for (auto sq : input) {
38 return sq;
39 }
40 assert(false);
41 return Square();
42}

44BitBoard MaskDiffWithMirror(const InputPlane& cur, const InputPlane& prev) {
45 auto to_mirror = BitBoard(prev.mask);
46 to_mirror.Mirror();
47 return BitBoard(cur.mask ^ to_mirror.as_int());
48}

50Square OldPosition(const InputPlane& prev, BitBoard mask_diff) {
51 auto to_mirror = BitBoard(prev.mask);
52 to_mirror.Mirror();
53 return SingleSquare(to_mirror & mask_diff);
54}

58void PopulateBoard(pblczero::NetworkFormat::InputFormat input_format,
59 InputPlanes planes, ChessBoard* board, int* rule50,
60 int* gameply) {
61 auto pawnsOurs = BitBoard(planes[0].mask);
62 auto knightsOurs = BitBoard(planes[1].mask);
63 auto bishopOurs = BitBoard(planes[2].mask);
64 auto rookOurs = BitBoard(planes[3].mask);
65 auto queenOurs = BitBoard(planes[4].mask);
66 auto kingOurs = BitBoard(planes[5].mask);
67 auto pawnsTheirs = BitBoard(planes[6].mask);
68 auto knightsTheirs = BitBoard(planes[7].mask);
69 auto bishopTheirs = BitBoard(planes[8].mask);
70 auto rookTheirs = BitBoard(planes[9].mask);
71 auto queenTheirs = BitBoard(planes[10].mask);
72 auto kingTheirs = BitBoard(planes[11].mask);
73 ChessBoard::Castlings castlings;
74 switch (input_format) {
75 case pblczero::NetworkFormat::INPUT_CLASSICAL_112_PLANE: {
76 if (planes[kAuxPlaneBase + 0].mask != 0) {
77 castlings.set_we_can_000();
78 }
79 if (planes[kAuxPlaneBase + 1].mask != 0) {
80 castlings.set_we_can_00();
81 }
82 if (planes[kAuxPlaneBase + 2].mask != 0) {
83 castlings.set_they_can_000();
84 }
85 if (planes[kAuxPlaneBase + 3].mask != 0) {
86 castlings.set_they_can_00();
87 }
88 break;
89 }
90 case pblczero::NetworkFormat::INPUT_112_WITH_CASTLING_PLANE:
91 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION:
92 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_HECTOPLIES:
93 case pblczero::NetworkFormat::
94 INPUT_112_WITH_CANONICALIZATION_HECTOPLIES_ARMAGEDDON:
95 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_V2:
96 case pblczero::NetworkFormat::
97 INPUT_112_WITH_CANONICALIZATION_V2_ARMAGEDDON: {
98 File our_queenside = kFileA;
99 File their_queenside = kFileA;
100 File our_kingside = kFileH;
101 File their_kingside = kFileH;
102 if (planes[kAuxPlaneBase + 0].mask != 0) {
103 auto mask = planes[kAuxPlaneBase + 0].mask;
104 if ((mask & 0xFFLL) != 0) {
105 our_queenside = File::FromIdx(GetLowestBit(mask & 0xFFLL));
106 castlings.set_we_can_000();
107 }
108 if (mask >> 56 != 0) {
109 their_queenside = File::FromIdx(GetLowestBit(mask >> 56));
110 castlings.set_they_can_000();
111 }
112 }
113 if (planes[kAuxPlaneBase + 1].mask != 0) {
114 auto mask = planes[kAuxPlaneBase + 1].mask;
115 if ((mask & 0xFFLL) != 0) {
116 our_kingside.idx = GetLowestBit(mask & 0xFFLL);
117 castlings.set_we_can_00();
118 }
119 if (mask >> 56 != 0) {
120 their_kingside.idx = GetLowestBit(mask >> 56);
121 castlings.set_they_can_00();
122 }
123 }
124 castlings.our_kingside_rook = our_kingside;
125 castlings.our_queenside_rook = our_queenside;
126 castlings.their_kingside_rook = their_kingside;
127 castlings.their_queenside_rook = their_queenside;
128 break;
129 }
130
131 default:
132 throw Exception("Unsupported input plane encoding " +
133 std::to_string(input_format));
134 }
135 std::string fen;
136 // Canonical input has no sense of side to move, so we should simply assume
137 // the starting position is always white.
138 bool black_to_move =
139 !IsCanonicalFormat(input_format) && planes[kAuxPlaneBase + 4].mask != 0;
140 if (black_to_move) {
141 // Flip to white perspective rather than side to move perspective.
142 std::swap(pawnsOurs, pawnsTheirs);
143 std::swap(knightsOurs, knightsTheirs);
144 std::swap(bishopOurs, bishopTheirs);
145 std::swap(rookOurs, rookTheirs);
146 std::swap(queenOurs, queenTheirs);
147 std::swap(kingOurs, kingTheirs);
148 pawnsOurs.Mirror();
149 pawnsTheirs.Mirror();
150 knightsOurs.Mirror();
151 knightsTheirs.Mirror();
152 bishopOurs.Mirror();
153 bishopTheirs.Mirror();
154 rookOurs.Mirror();
155 rookTheirs.Mirror();
156 queenOurs.Mirror();
157 queenTheirs.Mirror();
158 kingOurs.Mirror();
159 kingTheirs.Mirror();
160 castlings.Mirror();
161 }
162 for (int row = 7; row >= 0; --row) {
163 int emptycounter = 0;
164 for (int col = 0; col < 8; ++col) {
165 char piece = '\0';
166 Square square(File::FromIdx(col), Rank::FromIdx(row));
167 if (pawnsOurs.get(square)) {
168 piece = 'P';
169 } else if (pawnsTheirs.get(square)) {
170 piece = 'p';
171 } else if (knightsOurs.get(square)) {
172 piece = 'N';
173 } else if (knightsTheirs.get(square)) {
174 piece = 'n';
175 } else if (bishopOurs.get(square)) {
176 piece = 'B';
177 } else if (bishopTheirs.get(square)) {
178 piece = 'b';
179 } else if (rookOurs.get(square)) {
180 piece = 'R';
181 } else if (rookTheirs.get(square)) {
182 piece = 'r';
183 } else if (queenOurs.get(square)) {
184 piece = 'Q';
185 } else if (queenTheirs.get(square)) {
186 piece = 'q';
187 } else if (kingOurs.get(square)) {
188 piece = 'K';
189 } else if (kingTheirs.get(square)) {
190 piece = 'k';
191 }
192 if (emptycounter > 0 && piece) {
193 fen += std::to_string(emptycounter);
194 emptycounter = 0;
195 }
196 if (piece) {
197 fen += piece;
198 } else {
199 emptycounter++;
200 }
201 }
202 if (emptycounter > 0) fen += std::to_string(emptycounter);
203 if (row > 0) fen += "/";
204 }
205 fen += " ";
206 fen += black_to_move ? "b" : "w";
207 fen += " ";
208 fen += castlings.as_string();
209 fen += " ";
210 if (IsCanonicalFormat(input_format)) {
211 // Canonical format helpfully has the en passant details ready for us.
212 if (planes[kAuxPlaneBase + 4].mask == 0) {
213 fen += "-";
214 } else {
215 File file =
216 File::FromIdx(GetLowestBit(planes[kAuxPlaneBase + 4].mask >> 56));
217 fen += Square(file, kRank6).ToString();
218 }
219 } else {
220 auto pawndiff = BitBoard(planes[6].mask ^ planes[kPlanesPerBoard + 6].mask);
221 // If no pawns then 2 pawns, history isn't filled properly and we shouldn't
222 // try and infer enpassant.
223 if (pawndiff.count() == 2 && planes[kPlanesPerBoard + 6].mask != 0) {
224 auto from =
225 SingleSquare(planes[kPlanesPerBoard + 6].mask & pawndiff.as_int());
226 auto to = SingleSquare(planes[6].mask & pawndiff.as_int());
227 if (from.file() != to.file() || std::abs(from.rank() - to.rank()) != 2) {
228 fen += "-";
229 } else {
230 // TODO: Ensure enpassant is legal rather than setting it blindly?
231 // Doesn't matter for rescoring use case as only legal moves will be
232 // performed afterwards.
233 fen += Square(to.file(),
234 (planes[kAuxPlaneBase + 4].mask != 0) ? kRank3 : kRank6)
235 .ToString();
236 }
237 } else {
238 fen += "-";
239 }
240 }
241 fen += " ";
242 int rule50plane = (int)planes[kAuxPlaneBase + 5].value;
243 if (IsHectopliesFormat(input_format)) {
244 rule50plane = (int)(100.0f * planes[kAuxPlaneBase + 5].value);
245 }
246 fen += std::to_string(rule50plane);
247 // Reuse the 50 move rule as gameply since we don't know better.
248 fen += " ";
249 fen += std::to_string(rule50plane);
250 board->SetFromFen(fen, rule50, gameply);
251}

253Move DecodeMoveFromInput(const InputPlanes& planes, const InputPlanes& prior) {
254 auto pawndiff = MaskDiffWithMirror(planes[6], prior[0]);
255 auto knightdiff = MaskDiffWithMirror(planes[7], prior[1]);
256 auto bishopdiff = MaskDiffWithMirror(planes[8], prior[2]);
257 auto rookdiff = MaskDiffWithMirror(planes[9], prior[3]);
258 auto queendiff = MaskDiffWithMirror(planes[10], prior[4]);
259 // Handle Promotion.
260 if (pawndiff.count() == 1) {
261 auto from = SingleSquare(pawndiff);
262 if (knightdiff.count() == 1) {
263 auto to = SingleSquare(knightdiff);
264 return Move::WhitePromotion(from, to, kKnight);
265 }
266 if (bishopdiff.count() == 1) {
267 auto to = SingleSquare(bishopdiff);
268 return Move::WhitePromotion(from, to, kBishop);
269 }
270 if (rookdiff.count() == 1) {
271 auto to = SingleSquare(rookdiff);
272 return Move::WhitePromotion(from, to, kRook);
273 }
274 if (queendiff.count() == 1) {
275 auto to = SingleSquare(queendiff);
276 return Move::WhitePromotion(from, to, kQueen);
277 }
278 assert(false);
279 return Move();
280 }
281 // check king first as castling moves both king and rook.
282 auto kingdiff = MaskDiffWithMirror(planes[11], prior[5]);
283 if (kingdiff.count() == 2) {
284 if (rookdiff.count() == 2) {
285 auto from = OldPosition(prior[5], kingdiff);
286 auto to = OldPosition(prior[3], rookdiff);
287 Move m = Move::WhiteCastling(from.file(), to.file());
288 if (from.rank() == kRank8) m.Flip();
289 return m;
290 }
291 auto from = OldPosition(prior[5], kingdiff);
292 auto to = SingleSquare(planes[11].mask & kingdiff.as_int());
293 if (std::abs(from.file() - to.file()) > 1) {
294 // Chess 960 castling can leave the rook in place, but the king has moved
295 // from one side of the rook to the other - thus has gone at least 2
296 // squares, which is impossible for a normal king move. Can't work out the
297 // rook location from rookdiff since its empty, but it is known given the
298 // direction of the king movement and the knowledge that the rook hasn't
299 // moved.
300 if (from.file() > to.file()) {
301 to = Square(to.file() + 1, from.rank());
302 } else {
303 to = Square(to.file() - 1, from.rank());
304 }
305 Move m = Move::WhiteCastling(from.file(), to.file());
306 if (from.rank() == kRank8) m.Flip();
307 return m;
308 }
309 return Move::White(from, to);
310 }
311 if (queendiff.count() == 2) {
312 auto from = OldPosition(prior[4], queendiff);
313 auto to = SingleSquare(planes[10].mask & queendiff.as_int());
314 return Move::White(from, to);
315 }
316 if (rookdiff.count() == 2) {
317 auto from = OldPosition(prior[3], rookdiff);
318 auto to = SingleSquare(planes[9].mask & rookdiff.as_int());
319 // Only one king, so we can simply grab its current location directly.
320 auto kingpos = SingleSquare(planes[11].mask);
321 if (from.rank() == kingpos.rank() && to.rank() == kingpos.rank() &&
322 ((from.file() < kingpos.file() && to.file() > kingpos.file()) ||
323 (from.file() > kingpos.file() && to.file() < kingpos.file()))) {
324 // If the king hasn't moved, this could still be a chess 960 castling move
325 // if the rook has passed through the king.
326 // Destination of the castling move is where the rook started.
327 to = from;
328 // And since the king didn't move it forms the start position.
329 from = kingpos;
330 Move m = Move::WhiteCastling(from.file(), to.file());
331 if (from.rank() == kRank8) m.Flip();
332 return m;
333 }
334 return Move::White(from, to);
335 }
336 if (bishopdiff.count() == 2) {
337 auto from = OldPosition(prior[2], bishopdiff);
338 auto to = SingleSquare(planes[8].mask & bishopdiff.as_int());
339 return Move::White(from, to);
340 }
341 if (knightdiff.count() == 2) {
342 auto from = OldPosition(prior[1], knightdiff);
343 auto to = SingleSquare(planes[7].mask & knightdiff.as_int());
344 return Move::White(from, to);
345 }
346 if (pawndiff.count() == 2) {
347 auto from = OldPosition(prior[0], pawndiff);
348 auto to = SingleSquare(planes[6].mask & pawndiff.as_int());
349 // Check for enpassant.
350 auto targets = BitBoard(prior[6].mask | prior[7].mask | prior[8].mask |
351 prior[9].mask | prior[10].mask);
352 targets.Mirror();
353 if (from.file() != to.file() && (targets & pawndiff) == 0) {
354 return Move::WhiteEnPassant(from, to);
355 }
356 return Move::White(from, to);
357 }
358 assert(false);
359 throw Exception("Invalid move encoding");
360}
