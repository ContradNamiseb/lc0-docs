38float LayerAdapter::Iterator::ExtractValue(const std::byte* ptr,
39 const LayerAdapter* adapter) {
40 switch (adapter->encoding_) {
41 case pblczero::Weights::Layer::LINEAR16: {
42 float theta =
43 *reinterpret_cast<const uint16_t*>(ptr) / static_cast<float>(0xffff);
44 return adapter->min_ * (1 - theta) + adapter->max_ * theta;
45 }
46 case pblczero::Weights::Layer::FLOAT16:
47 return FP16toFP32(*reinterpret_cast<const uint16_t*>(ptr));
48 case pblczero::Weights::Layer::BFLOAT16:
49 return BF16toFP32(*reinterpret_cast<const uint16_t*>(ptr));
50 case pblczero::Weights::Layer::FLOAT32: {
51 return *reinterpret_cast<const float*>(ptr);
52 }
53 [[unlikely]] default: // To silence a couple of warnings.
54#if defined(ABSL_UNREACHABLE)
55 ABSL_UNREACHABLE();
56#elif defined(__GNUC__)
57 __builtin_unreachable();
58#else
59 __assume(false);
60#endif
61 }
62}

64LayerAdapter::LayerAdapter(const pblczero::Weights::Layer& layer)
65 : encoding_(layer.has_encoding() ? layer.encoding()
66 : pblczero::Weights::Layer::LINEAR16),
67 element_size_(encoding_ == pblczero::Weights::Layer::FLOAT32
68 ? sizeof(float)
69 : sizeof(uint16_t)),
70 data_(reinterpret_cast<const std::byte*>(layer.params().data())),
71 size_(layer.params().size() / element_size_),
72 min_(layer.min_val()),
73 max_(layer.max_val()) {
74 switch (encoding_) {
75 case pblczero::Weights::Layer::LINEAR16:
76 case pblczero::Weights::Layer::FLOAT16:
77 case pblczero::Weights::Layer::BFLOAT16:
78 case pblczero::Weights::Layer::FLOAT32:
79 break;
80 default:
81 throw Exception("Unknown layer encoding " +
82 pblczero::Weights::Layer::Encoding_Name(encoding_));
83 }
84}
