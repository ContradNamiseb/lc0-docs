37inline unsigned long GetLowestBit(std::uint64_t value) {
38#if defined(_MSC_VER) && defined(_WIN64)
39 unsigned long result;
40 _BitScanForward64(&result, value);
41 return result;
42#elif defined(_MSC_VER)
43 unsigned long result;
44 if (value & 0xFFFFFFFF) {
45 _BitScanForward(&result, value);
46 } else {
47 _BitScanForward(&result, value >> 32);
48 result += 32;
49 }
50 return result;
51#else
52 return __builtin_ctzll(value);
53#endif
54}

66inline uint64_t ReverseBitsInBytes(uint64_t v) {
67 v = ((v >> 1) & 0x5555555555555555ull) | ((v & 0x5555555555555555ull) << 1);
68 v = ((v >> 2) & 0x3333333333333333ull) | ((v & 0x3333333333333333ull) << 2);
69 v = ((v >> 4) & 0x0F0F0F0F0F0F0F0Full) | ((v & 0x0F0F0F0F0F0F0F0Full) << 4);
70 return v;
71}

73inline uint64_t ReverseBytesInBytes(uint64_t v) {
74 v = (v & 0x00000000FFFFFFFF) << 32 | (v & 0xFFFFFFFF00000000) >> 32;
75 v = (v & 0x0000FFFF0000FFFF) << 16 | (v & 0xFFFF0000FFFF0000) >> 16;
76 v = (v & 0x00FF00FF00FF00FF) << 8 | (v & 0xFF00FF00FF00FF00) >> 8;
77 return v;
78}

80// Transpose across the diagonal connecting bit 7 to bit 56.
81inline uint64_t TransposeBitsInBytes(uint64_t v) {
82 v = (v & 0xAA00AA00AA00AA00ULL) >> 9 | (v & 0x0055005500550055ULL) << 9 |
83 (v & 0x55AA55AA55AA55AAULL);
84 v = (v & 0xCCCC0000CCCC0000ULL) >> 18 | (v & 0x0000333300003333ULL) << 18 |
85 (v & 0x3333CCCC3333CCCCULL);
86 v = (v & 0xF0F0F0F000000000ULL) >> 36 | (v & 0x000000000F0F0F0FULL) << 36 |
87 (v & 0x0F0F0F0FF0F0F0F0ULL);
88 return v;
89}

91// Iterates over all set bits of the value, lower to upper. The value of
92// dereferenced iterator is bit number (lower to upper, 0 bazed)
93template <typename T, typename Convert = std::identity>
94class BitIterator {
95 public:
96 using iterator_category = std::forward_iterator_tag;
97 using difference_type = T;
98 using value_type = T;
99 using pointer = T*;
100 using reference = T&;
101
102 BitIterator(std::uint64_t value) : value_(value) {};
103 bool operator!=(const BitIterator& other) { return value_ != other.value_; }
104
105 void operator++() { value_ &= (value_ - 1); }
106 T operator*() const { return Convert()(GetLowestBit(value_)); }
107
108 private:
109 std::uint64_t value_;
110};
