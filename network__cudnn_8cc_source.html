52static size_t getMaxAttentionHeadSize(
53 const MultiHeadWeights::PolicyHead& weights, int N) {
54 const size_t embedding_op_size = weights.ip_pol_b.size();
55 const size_t policy_d_model = weights.ip2_pol_b.size();
56 assert(policy_d_model == weights.ip3_pol_b.size());
57
58 size_t encoder_d_model = 0;
59 size_t encoder_dff = 0;
60
61 if (weights.pol_encoder.size() > 0) {
62 encoder_d_model = weights.pol_encoder[0].mha.q_b.size();
63 encoder_dff = weights.pol_encoder[0].ffn.dense1_b.size();
64
65 assert(encoder_d_model == weights.pol_encoder[0].mha.k_b.size());
66 assert(encoder_d_model == weights.pol_encoder[0].mha.v_b.size());
67 assert(embedding_op_size == weights.pol_encoder[0].ffn.dense2_b.size());
68 }
69
70 const size_t encoder_heads = weights.pol_encoder_head_count;
71
72 size_t size =
73 N * 64 *
74 std::max(std::max(embedding_op_size, encoder_dff), policy_d_model);
75
76 // size of matmul_qk matrix = encoder_heads_ * Batch * 64 * 64
77 const size_t matmul_qk_size = encoder_heads * N * 64 * 64;
78 const size_t output_size = N * (64 * 64 + 8 * 24);
79 size = std::max(size, std::max(matmul_qk_size, output_size));
80
81 size_t qkv_size = N * 64 * encoder_d_model;
82 // We store qkv in single allocation, and other intermediate tensors are
83 // sometimes stored by splitting an allocation into two halves.
84 size = std::max(2 * size, 3 * qkv_size);
85 return size;
86}

88template <typename DataType>
89class CudnnNetworkComputation : public NetworkComputation {
90 public:
91 CudnnNetworkComputation(CudnnNetwork<DataType>* network, bool wdl,
92 bool moves_left);
93 ~CudnnNetworkComputation();
94
95 void AddInput(InputPlanes&& input) override {
96 const auto iter_mask =
97 &inputs_outputs_->input_masks_mem_[batch_size_ * kInputPlanes];
98 const auto iter_val =
99 &inputs_outputs_->input_val_mem_[batch_size_ * kInputPlanes];
100
101 for (int i = 0; i < kInputPlanes; i++) {
102 const auto& plane = input[i];
103 iter_mask[i] = plane.mask;
104 ToType(iter_val[i], plane.value);
105 }
106
107 batch_size_++;
108 }
109
110 void ComputeBlocking() override;
111
112 void CaptureGraph(std::unique_lock<std::mutex>&& lock = {});
113
114 int GetBatchSize() const override { return batch_size_; }
115
116 float GetQVal(int sample) const override {
117 if (wdl_) {
118 const float* wdl =
119 sizeof(inputs_outputs_->op_value_mem_[0]) == sizeof(float)
120 ? (float*)inputs_outputs_->op_value_mem_
121 : inputs_outputs_->wdl_cpu_softmax_.get();
122 return wdl[2 * sample];
123 }
124 return FromType(inputs_outputs_->op_value_mem_[sample]);
125 }
126
127 float GetDVal(int sample) const override {
128 if (wdl_) {
129 const float* wdl =
130 sizeof(inputs_outputs_->op_value_mem_[0]) == sizeof(float)
131 ? (float*)inputs_outputs_->op_value_mem_
132 : inputs_outputs_->wdl_cpu_softmax_.get();
133 return wdl[2 * sample + 1];
134 }
135 return 0.0f;
136 }
137
138 float GetPVal(int sample, int move_id) const override {
139 return FromType(
140 inputs_outputs_->op_policy_mem_[sample * kNumOutputPolicy + move_id]);
141 }
142
143 float GetMVal(int sample) const override {
144 if (moves_left_) {
145 return FromType(inputs_outputs_->op_moves_left_mem_[sample]);
146 }
147 return 0.0f;
148 }
149
150 private:
151 // Memory holding inputs, outputs.
152 std::unique_ptr<InputsOutputs<DataType>> inputs_outputs_;
153 int batch_size_;
154 bool wdl_;
155 bool moves_left_;
156
157 CudnnNetwork<DataType>* network_;
158};

160template <typename DataType>
161class CudnnNetwork : public Network {
162 public:
163 CudnnNetwork(const WeightsFile& file, const OptionsDict& options)
164 : capabilities_{file.format().network_format().input(),
165 file.format().network_format().output(),
166 file.format().network_format().moves_left()} {
167 MultiHeadWeights weights(file.weights());
168 gpu_id_ = options.GetOrDefault<int>("gpu", 0);
169 enable_graph_capture_ = options.GetOrDefault<bool>("graph_capture", true);
170
171 conv_policy_ = file.format().network_format().policy() ==
172 pblczero::NetworkFormat::POLICY_CONVOLUTION;
173
174 attn_policy_ = file.format().network_format().policy() ==
175 pblczero::NetworkFormat::POLICY_ATTENTION;
176
177 max_batch_size_ = options.GetOrDefault<int>("max_batch", 1024);
178 // min_batch_size_ is chosen as 4 as it is common that for sizes less than
179 // 4 that there is no performance gain, but there is variance in the
180 // outputs, which means that there is extra non-determinism in some
181 // scenarios, including using the multiplexing backend.
182 min_batch_size_ =
183 options.GetOrDefault<int>("min_batch", std::min(4, max_batch_size_));
184 if (max_batch_size_ < min_batch_size_)
185 throw Exception("Max batch must not be less than min_batch setting.");
186
187 showInfo();
188
189 int total_gpus;
190 ReportCUDAErrors(cudaGetDeviceCount(&total_gpus));
191
192 if (gpu_id_ >= total_gpus)
193 throw Exception("Invalid GPU Id: " + std::to_string(gpu_id_));
194
195 cudaDeviceProp deviceProp = {};
196 cudaGetDeviceProperties(&deviceProp, gpu_id_);
197 showDeviceInfo(deviceProp, gpu_id_);
198
199 // Select GPU to run on (for *the current* thread).
200 ReportCUDAErrors(cudaSetDevice(gpu_id_));
201
202 ReportCUDNNErrors(cudnnCreate(&cudnn_));
203 ReportCUBLASErrors(cublasCreate(&cublas_));
204
205 // Default layout is nchw.
206 nhwc_ = false;
207 bool hasTensorCores = false;
208 constexpr bool fp16 = std::is_same<half, DataType>::value;
209
210 if (fp16) {
211 // Check if the GPU support FP16.
212
213 if ((deviceProp.major == 6 && deviceProp.minor != 1) ||
214 (deviceProp.major == 5 && deviceProp.minor == 3)) {
215 // FP16 without tensor cores supported on GP100 (SM 6.0) and Jetson
216 // (SM 5.3 and 6.2). SM 6.1 GPUs also have FP16, but slower than FP32.
217 // nhwc_ remains false.
218 } else if (deviceProp.major >= 7) {
219 // NHWC layout is faster with Tensor Cores when using cudnn's implicit
220 // gemm algorithm.
221 // Supported on Volta and Turing (and hopefully future GPUs too).
222
223 // Some GPUs (GTX 16xx) are SM 7.5 but don't have tensor cores
224 // enabling TENSOR_OP_MATH or nhwc_ layout for them works but is
225 // very very slow (likely because the system emulates it).
226 if (!strstr(deviceProp.name, "GTX 16")) {
227 hasTensorCores = true;
228 nhwc_ = true;
229 }
230 } else {
231 throw Exception("Your GPU doesn't support FP16");
232 }
233
234 // Override if forced from backend option
235 if (options.Exists<bool>("nhwc")) nhwc_ = options.Get<bool>("nhwc");
236 }
237 ReportCUDAErrors(
238 cudaStreamCreateWithFlags(&compute_stream_, cudaStreamNonBlocking));
239 ReportCUDAErrors(
240 cudaStreamCreateWithFlags(&upload_stream_, cudaStreamNonBlocking));
241 ReportCUDAErrors(
242 cudaStreamCreateWithFlags(&download_stream_, cudaStreamNonBlocking));
243 ReportCUDAErrors(cudaEventCreateWithFlags(&compute_ordering_event_,
244 cudaEventDisableTiming));
245
246 ReportCUBLASErrors(cublasSetStream(cublas_, compute_stream_));
247 ReportCUDNNErrors(cudnnSetStream(cudnn_, compute_stream_));
248
249 if (hasTensorCores)
250 ReportCUBLASErrors(cublasSetMathMode(
251 cublas_,
252 CUBLAS_TENSOR_OP_MATH)); // Deprecated on CUDA 11.0 and later
253 else if (fp16)
254 ReportCUBLASErrors(cublasSetMathMode(
255 cublas_,
256 CUBLAS_PEDANTIC_MATH)); // Explicitly set PEDANTIC_MATH mode to
257 // avoid cublas bug of making use of tensor
258 // core math on TU11x GPUs that don't
259 // support it.
260
261 const int kNumInputPlanes = kInputPlanes;
262 const int kNumFilters = (int)weights.input.biases.size();
263 numBlocks_ = (int)weights.residual.size();
264
265 // Use our custom winograd for residual tower convolutions for most cases:
266 //
267 // 1. Should be always faster than cudnn's winograd that we use for fp32,
268 // and for fp16 on GPUs without tensor cores
269 //
270 // 2. Should also be faster than cudnn's implicit GEMM on GPUs with tensor
271 // cores too, but only for networks with 256 or higher no. of filters.
272 //
273 // 3. Currently a bug in cublas makes it slower on RTX GPUs with fp16 so
274 // it's disabled. TODO: Enable it once the bug has been fixed and it's
275 // tested to be faster. Putting check for cuda 11 for now.
276
277 if (fp16) {
278 int cuda_version;
279 cudaRuntimeGetVersion(&cuda_version);
280 if (!hasTensorCores)
281 use_custom_winograd_ = false;
282 else if (kNumFilters >= 256 &&
283 !(deviceProp.major == 7 && deviceProp.minor == 5 &&
284 cuda_version < 11000))
285 use_custom_winograd_ = true;
286 else
287 use_custom_winograd_ = false;
288 } else {
289 use_custom_winograd_ = true;
290 }
291
292 // Warn if the memory required for storing transformed weights is
293 // going to exceed 40% of total video memory, force custom_winograd off
294 // if it's going to exceed 50% of memory.
295 size_t residual_single_layer_weight_size =
296 3 * 3 * kNumFilters * kNumFilters * sizeof(DataType);
297 size_t residual_weight_size =
298 residual_single_layer_weight_size * numBlocks_ * 2;
299 size_t transformed_residual_weight_size = residual_weight_size * 4;
300
301 if (residual_weight_size > 0.6 * deviceProp.totalGlobalMem) {
302 CERR << "Low video memory detected. You may run into OOM errors. Please "
303 "consider using a smaller network.";
304 }
305
306 const bool custom_winograd_override =
307 options.Exists<bool>("custom_winograd");
308
309 if (!custom_winograd_override && use_custom_winograd_ &&
310 transformed_residual_weight_size > 0.5 * deviceProp.totalGlobalMem) {
311 CERR << "WARNING: Low GPU video memory. Turning off custom_winograd "
312 "path. You may still run into OOM errors. "
313 "Please consider using a smaller network.";
314 use_custom_winograd_ = false;
315 }
316
317 // Override if set in backend-opts.
318 if (custom_winograd_override)
319 use_custom_winograd_ = options.Get<bool>("custom_winograd");
320
321 if (use_custom_winograd_ &&
322 transformed_residual_weight_size > 0.4 * deviceProp.totalGlobalMem) {
323 CERR << "WARNING: Low GPU video memory. You may still run into OOM "
324 "errors. Try with backend-opts=custom_winograd=false, or "
325 "using a smaller network.";
326 }
327
328 // Winograd needs nchw tensor layout.
329 if (use_custom_winograd_) nhwc_ = false;
330
331 use_res_block_winograd_fuse_opt_ = false;
332 if (use_custom_winograd_) {
333 // Disable res block fusing for fp32 for now.
334 // TODO: make it work for filters not a multiple of 32.
335 if (kNumFilters % 32 == 0 && fp16) {
336 use_res_block_winograd_fuse_opt_ = true;
337 }
338 // Override if set in backend-opts.
339 if (options.Exists<bool>("res_block_fusing")) {
340 use_res_block_winograd_fuse_opt_ =
341 options.Get<bool>("res_block_fusing");
342 }
343 }
344
345 const bool use_gemm_ex = deviceProp.major >= 5;
346
347 // 0. Check for SE.
348 has_se_ = false;
349 if (weights.residual[0].has_se) {
350 has_se_ = true;
351 }
352
353 const bool mish_net = file.format().network_format().default_activation() ==
354 pblczero::NetworkFormat::DEFAULT_ACTIVATION_MISH;
355
356 // 1. Allocate scratch space (used internally by cudnn to run convolutions,
357 // and also for format/layout conversion for weights).
358 cudnnFilterDescriptor_t wDesc;
359 cudnnConvolutionDescriptor_t convDesc;
360 cudnnTensorDescriptor_t xDesc;
361 cudnnCreateFilterDescriptor(&wDesc);
362 cudnnCreateConvolutionDescriptor(&convDesc);
363 cudnnCreateTensorDescriptor(&xDesc);
364 cudnnConvolutionFwdAlgo_t conv_algo;
365
366 const int maxChannels = std::max(kInputPlanes, kNumFilters);
367
368 const cudnnDataType_t datatype = fp16 ? CUDNN_DATA_HALF : CUDNN_DATA_FLOAT;
369 const cudnnTensorFormat_t layout =
370 nhwc_ ? CUDNN_TENSOR_NHWC : CUDNN_TENSOR_NCHW;
371
372 ReportCUDNNErrors(cudnnSetFilter4dDescriptor(
373 wDesc, datatype, layout, maxChannels, maxChannels, 3, 3));
374
375 ReportCUDNNErrors(cudnnSetTensor4dDescriptor(
376 xDesc, layout, datatype, max_batch_size_, maxChannels, 8, 8));
377
378 ReportCUDNNErrors(cudnnSetConvolution2dDescriptor(
379 convDesc, 1, 1, 1, 1, 1, 1, CUDNN_CROSS_CORRELATION, datatype));
380
381 // It will fall back to non-tensor math if not supported.
382 ReportCUDNNErrors(
383 cudnnSetConvolutionMathType(convDesc, CUDNN_TENSOR_OP_MATH));
384
385 if (nhwc_) {
386 conv_algo = CUDNN_CONVOLUTION_FWD_ALGO_IMPLICIT_PRECOMP_GEMM;
387 } else {
388 conv_algo = CUDNN_CONVOLUTION_FWD_ALGO_WINOGRAD_NONFUSED;
389 }
390
391 // Query expected scratch space from cudnn.
392 ReportCUDNNErrors(cudnnGetConvolutionForwardWorkspaceSize(
393 cudnn_, xDesc, wDesc, convDesc, xDesc, conv_algo, &scratch_size_));
394
395 // Have some minumum as we also use this for transforming weights.
396 size_t max_weight_size = 128 * 1024 * 1024;
397
398 // parts from scratch allocation are suballocated to hold various weights
399 // and biases when transforming winograd weights (one layer at a time), 128
400 // MB is way more than that what we need but make sure it's at least 3x of
401 // single layer's weight size to be safe.
402 if (max_weight_size < 3 * residual_single_layer_weight_size)
403 max_weight_size = 3 * residual_single_layer_weight_size;
404
405 if (scratch_size_ < max_weight_size) scratch_size_ = max_weight_size;
406
407 size_t transformed_tensor_size = 0;
408 if (use_custom_winograd_) {
409 // Need additional space for transformed input/outputs which are 36/16
410 // times size (4x4 block transformed into 6x6).
411 transformed_tensor_size = (size_t)(max_batch_size_ * kNumFilters * 64 *
412 (36.0 / 16.0) * sizeof(DataType));
413 scratch_size_ = std::max(scratch_size_, 2 * transformed_tensor_size);
414 }
415
416 // Attention policy head may need more memory
417 const size_t attentionSize =
418 getMaxAttentionHeadSize(weights.policy_heads.at("vanilla"),
419 max_batch_size_) *
420 sizeof(DataType);
421 scratch_size_ = std::max(scratch_size_, attentionSize);
422
423 ReportCUDAErrors(cudaMalloc(&scratch_mem_, scratch_size_));
424#ifdef DEBUG_RAW_NPS
425 CERR << "allocated " << scratch_size_ << " bytes for scratch memory";
426#endif
427
428 // 2. Build the network, and copy the weights to GPU memory.
429
430 // Input.
431 if (use_custom_winograd_) {
432 auto inputConv = std::make_unique<FusedWinogradConvSELayer<DataType>>(
433 nullptr, kNumFilters, 8, 8, kNumInputPlanes,
434 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true, false, false, 0,
435 use_gemm_ex, use_res_block_winograd_fuse_opt_);
436 inputConv->LoadWeights(&weights.input.weights[0],
437 &weights.input.biases[0], scratch_mem_);
438 network_.emplace_back(std::move(inputConv));
439 } else {
440 auto inputConv = std::make_unique<ConvLayer<DataType>>(
441 nhwc_, kNumFilters, 8, 8, 3, kNumInputPlanes,
442 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true);
443 inputConv->LoadWeights(&weights.input.weights[0],
444 &weights.input.biases[0], scratch_mem_);
445 network_.emplace_back(std::move(inputConv));
446 }
447
448 // Residual block.
449 for (int block = 0; block < numBlocks_; block++) {
450 if (use_custom_winograd_) {
451 bool has_se = weights.residual[block].has_se;
452 int se_k = (int)weights.residual[block].se.b1.size();
453
454 if (use_res_block_winograd_fuse_opt_) {
455 auto layer = std::make_unique<ResidualBlock<DataType>>(
456 getLastLayer(), kNumFilters, has_se, se_k, use_gemm_ex,
457 block == 0, block == (numBlocks_ - 1),
458 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU,
459 deviceProp.sharedMemPerBlockOptin);
460 layer->LoadWeights0(&weights.residual[block].conv1.weights[0],
461 &weights.residual[block].conv1.biases[0],
462 scratch_mem_);
463 layer->LoadWeights1(&weights.residual[block].conv2.weights[0],
464 &weights.residual[block].conv2.biases[0],
465 scratch_mem_);
466 if (has_se)
467 layer->LoadSEWeights(&weights.residual[block].se.w1[0],
468 &weights.residual[block].se.b1[0],
469 &weights.residual[block].se.w2[0],
470 &weights.residual[block].se.b2[0],
471 scratch_mem_);
472 network_.emplace_back(std::move(layer));
473 } else {
474 auto conv1 = std::make_unique<FusedWinogradConvSELayer<DataType>>(
475 getLastLayer(), kNumFilters, 8, 8, kNumFilters,
476 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true, false, false,
477 0, use_gemm_ex);
478 conv1->LoadWeights(&weights.residual[block].conv1.weights[0],
479 &weights.residual[block].conv1.biases[0],
480 scratch_mem_);
481 network_.emplace_back(std::move(conv1));
482
483 auto conv2 = std::make_unique<FusedWinogradConvSELayer<DataType>>(
484 getLastLayer(), kNumFilters, 8, 8, kNumFilters,
485 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true, true, has_se,
486 se_k, use_gemm_ex);
487 conv2->LoadWeights(&weights.residual[block].conv2.weights[0],
488 &weights.residual[block].conv2.biases[0],
489 scratch_mem_);
490 if (has_se)
491 conv2->LoadSEWeights(&weights.residual[block].se.w1[0],
492 &weights.residual[block].se.b1[0],
493 &weights.residual[block].se.w2[0],
494 &weights.residual[block].se.b2[0],
495 scratch_mem_);
496 network_.emplace_back(std::move(conv2));
497 }
498
499 } else {
500 auto conv1 = std::make_unique<ConvLayer<DataType>>(
501 getLastLayer(), kNumFilters, 8, 8, 3, kNumFilters,
502 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true);
503 conv1->LoadWeights(&weights.residual[block].conv1.weights[0],
504 &weights.residual[block].conv1.biases[0],
505 scratch_mem_);
506 network_.emplace_back(std::move(conv1));
507
508 // Relu and bias of second convolution is handled by SELayer.
509 bool useReluAndBias = weights.residual[block].has_se ? false : true;
510
511 auto conv2 = std::make_unique<ConvLayer<DataType>>(
512 getLastLayer(), kNumFilters, 8, 8, 3, kNumFilters,
513 useReluAndBias ? (mish_net ? ACTIVATION_MISH : ACTIVATION_RELU)
514 : ACTIVATION_NONE,
515 useReluAndBias);
516 conv2->LoadWeights(
517 &weights.residual[block].conv2.weights[0],
518 useReluAndBias ? &weights.residual[block].conv2.biases[0] : nullptr,
519 scratch_mem_);
520 network_.emplace_back(std::move(conv2));
521
522 if (weights.residual[block].has_se) {
523 int numFCOut = (int)weights.residual[block].se.b1.size();
524 auto se = std::make_unique<SELayer<DataType>>(
525 getLastLayer(), numFCOut, false,
526 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU);
527 se->LoadWeights(&weights.residual[block].se.w1[0],
528 &weights.residual[block].se.b1[0],
529 &weights.residual[block].se.w2[0],
530 &weights.residual[block].se.b2[0],
531 &weights.residual[block].conv2.biases[0],
532 scratch_mem_);
533 network_.emplace_back(std::move(se));
534 }
535 }
536 }
537
538 resi_last_ = getLastLayer();
539
540 // Policy head.
541 {
542 MultiHeadWeights::PolicyHead& head = weights.policy_heads.at("vanilla");
543 if (attn_policy_) {
544 auto AttentionPolicy = std::make_unique<AttentionPolicyHead<DataType>>(
545 getLastLayer(), head, scratch_mem_, false, ACTIVATION_SELU,
546 max_batch_size_, use_gemm_ex);
547 network_.emplace_back(std::move(AttentionPolicy));
548
549 auto policymap = std::make_unique<PolicyMapLayer<DataType>>(
550 getLastLayer(), kNumOutputPolicy, 1, 1, 64 * 64 + 8 * 24, true);
551 policymap->LoadWeights(kAttnPolicyMap, scratch_mem_);
552 network_.emplace_back(std::move(policymap));
553 } else if (conv_policy_) {
554 auto conv1 = std::make_unique<ConvLayer<DataType>>(
555 resi_last_, kNumFilters, 8, 8, 3, kNumFilters,
556 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true);
557 conv1->LoadWeights(&head.policy1.weights[0], &head.policy1.biases[0],
558 scratch_mem_);
559 network_.emplace_back(std::move(conv1));
560
561 auto pol_channels = head.policy.biases.size();
562
563 // No relu
564 auto conv2 = std::make_unique<ConvLayer<DataType>>(
565 getLastLayer(), pol_channels, 8, 8, 3, kNumFilters, ACTIVATION_NONE,
566 true);
567 conv2->LoadWeights(&head.policy.weights[0], &head.policy.biases[0],
568 scratch_mem_);
569 network_.emplace_back(std::move(conv2));
570
571 auto policymap = std::make_unique<PolicyMapLayer<DataType>>(
572 getLastLayer(), kNumOutputPolicy, 1, 1, 73 * 8 * 8, false);
573 policymap->LoadWeights(kConvPolicyMap, scratch_mem_);
574
575 network_.emplace_back(std::move(policymap));
576 } else {
577 auto convPol = std::make_unique<ConvLayer<DataType>>(
578 resi_last_, head.policy.biases.size(), 8, 8, 1, kNumFilters,
579 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true);
580 convPol->LoadWeights(&head.policy.weights[0], &head.policy.biases[0],
581 scratch_mem_);
582 network_.emplace_back(std::move(convPol));
583
584 auto FCPol = std::make_unique<FCLayer<DataType>>(
585 getLastLayer(), head.ip_pol_b.size(), 1, 1, true, ACTIVATION_NONE);
586 FCPol->LoadWeights(&head.ip_pol_w[0], &head.ip_pol_b[0], scratch_mem_);
587 network_.emplace_back(std::move(FCPol));
588 }
589 policy_out_ = getLastLayer();
590 }
591
592 // Value head.
593 {
594 MultiHeadWeights::ValueHead& head = weights.value_heads.at("winner");
595 auto convVal = std::make_unique<ConvLayer<DataType>>(
596 resi_last_, head.value.biases.size(), 8, 8, 1, kNumFilters,
597 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true);
598 convVal->LoadWeights(&head.value.weights[0], &head.value.biases[0],
599 scratch_mem_);
600 network_.emplace_back(std::move(convVal));
601
602 auto FCVal1 = std::make_unique<FCLayer<DataType>>(
603 getLastLayer(), head.ip1_val_b.size(), 1, 1, true,
604 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU);
605 FCVal1->LoadWeights(&head.ip1_val_w[0], &head.ip1_val_b[0], scratch_mem_);
606 network_.emplace_back(std::move(FCVal1));
607
608 wdl_ = file.format().network_format().value() ==
609 pblczero::NetworkFormat::VALUE_WDL;
610 auto fc2_tanh = !wdl_;
611
612 auto FCVal2 = std::make_unique<FCLayer<DataType>>(
613 getLastLayer(), head.ip2_val_b.size(), 1, 1, true,
614 fc2_tanh ? ACTIVATION_TANH : ACTIVATION_NONE);
615 FCVal2->LoadWeights(&head.ip2_val_w[0], &head.ip2_val_b[0], scratch_mem_);
616 network_.emplace_back(std::move(FCVal2));
617 }
618 value_out_ = getLastLayer();
619
620 // Moves left head
621 moves_left_ = (file.format().network_format().moves_left() ==
622 pblczero::NetworkFormat::MOVES_LEFT_V1) &&
623 options.GetOrDefault<bool>("mlh", true);
624 if (moves_left_) {
625 auto convMov = std::make_unique<ConvLayer<DataType>>(
626 resi_last_, weights.moves_left.biases.size(), 8, 8, 1, kNumFilters,
627 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU, true);
628 convMov->LoadWeights(&weights.moves_left.weights[0],
629 &weights.moves_left.biases[0], scratch_mem_);
630 network_.emplace_back(std::move(convMov));
631
632 auto FCMov1 = std::make_unique<FCLayer<DataType>>(
633 getLastLayer(), weights.ip1_mov_b.size(), 1, 1, true,
634 mish_net ? ACTIVATION_MISH : ACTIVATION_RELU);
635 FCMov1->LoadWeights(&weights.ip1_mov_w[0], &weights.ip1_mov_b[0],
636 scratch_mem_);
637 network_.emplace_back(std::move(FCMov1));
638
639 auto FCMov2 = std::make_unique<FCLayer<DataType>>(getLastLayer(), 1, 1, 1,
640 true, ACTIVATION_RELU);
641 FCMov2->LoadWeights(&weights.ip2_mov_w[0], &weights.ip2_mov_b[0],
642 scratch_mem_);
643 network_.emplace_back(std::move(FCMov2));
644 }
645 moves_left_out_ = getLastLayer();
646
647 // 3. Allocate GPU memory for running the network:
648 // - three buffers of max size are enough (one to hold input, second to
649 // hold output and third to hold skip connection's input).
650
651 // size of input to the network
652 size_t maxSize = max_batch_size_ * kNumInputPlanes * 64 * sizeof(DataType);
653
654 // take max size of all layers
655 for (auto& layer : network_) {
656 maxSize = std::max(maxSize, layer->GetOutputSize(max_batch_size_));
657 }
658
659 // when this optimization is enabled, we write transformed outputs to
660 // intermediate tensor memory
661 if (use_res_block_winograd_fuse_opt_ && transformed_tensor_size > maxSize)
662 maxSize = transformed_tensor_size;
663
664 if (attn_policy_ && scratch_size_ > maxSize) {
665 maxSize = scratch_size_;
666 }
667
668 for (auto& mem : tensor_mem_) {
669 ReportCUDAErrors(cudaMalloc(&mem, maxSize));
670 ReportCUDAErrors(cudaMemset(mem, 0, maxSize));
671 }
672
673 cudnnDestroyFilterDescriptor(wDesc);
674 cudnnDestroyConvolutionDescriptor(convDesc);
675 cudnnDestroyTensorDescriptor(xDesc);
676
677#ifdef DEBUG_RAW_NPS
678 CERR << "allocated " << 3 * maxSize
679 << " bytes of GPU memory to run the network";
680#endif
681
682 // pre-allocate cuda graphs for search threads
683 auto allocateCudaGraphs = [&] {
684 CudnnNetworkComputation<DataType> comp(this, wdl_, moves_left_);
685 comp.AddInput(InputPlanes{(size_t)kNumInputPlanes});
686 // Make sure cublas is initialized in this thread.
687 comp.ComputeBlocking();
688 for (int i = 0; i < GetMiniBatchSize(); i++) {
689 comp.AddInput(InputPlanes{(size_t)kNumInputPlanes});
690 auto lock = LockEval();
691 comp.CaptureGraph(std::move(lock));
692 }
693 };
694 std::thread t2(allocateCudaGraphs);
695 allocateCudaGraphs();
696 t2.join();
697 }

737 void forwardEval(InputsOutputs<DataType>* io, int batchSize,
738 bool capture = false) {
739 // It is safe to evaluate larger than the batchSize
740 // as all buffers are designed to handle max_batch_size
741 // and the extra invalid results are never read.
742 if (batchSize < min_batch_size_) batchSize = min_batch_size_;
743
744#ifdef DEBUG_RAW_NPS
745 auto t_start = std::chrono::high_resolution_clock::now();
746#endif
747
748 // TODO: consider supporting multi-stream path for cudnn backend too.
749 cudaStream_t compute_stream = compute_stream_;
750 cudaStream_t download_stream = download_stream_;
751
752 // Expand packed planes to full planes.
753 const uint64_t* ipDataMasks = io->input_masks_mem_gpu_;
754 const auto* ipDataValues = io->input_val_mem_gpu_;
755
756 if (nhwc_)
757 expandPlanes_NHWC(tensor_mem_[0], ipDataMasks, ipDataValues,
758 batchSize * kInputPlanes, compute_stream);
759 else
760 expandPlanes_NCHW(tensor_mem_[0], ipDataMasks, ipDataValues,
761 batchSize * kInputPlanes, compute_stream);
762
763 // debug code example
764 // dumpTensor(tensor_mem_[0], 1024, "After expand Planes", fp16);
765
766 auto* opPol = io->op_policy_mem_gpu_;
767 auto* opVal = io->op_value_mem_gpu_;
768 auto* opMov = io->op_moves_left_mem_gpu_;
769
770 int l = 0;
771 // Input.
772 network_[l++]->Eval(
773 batchSize,
774 use_res_block_winograd_fuse_opt_ ? tensor_mem_[1] : tensor_mem_[2],
775 tensor_mem_[0], nullptr, scratch_mem_, scratch_size_, cudnn_, cublas_,
776 compute_stream); // input conv
777
778 // Residual block.
779 for (int block = 0; block < numBlocks_; block++) {
780 if (use_res_block_winograd_fuse_opt_) {
781 network_[l++]->Eval(batchSize, tensor_mem_[2], tensor_mem_[1], nullptr,
782 scratch_mem_, scratch_size_, cudnn_, cublas_,
783 compute_stream); // block
784 } else {
785 network_[l++]->Eval(batchSize, tensor_mem_[0], tensor_mem_[2], nullptr,
786 scratch_mem_, scratch_size_, cudnn_, cublas_,
787 compute_stream); // conv1
788
789 if (use_custom_winograd_) {
790 network_[l++]->Eval(batchSize, tensor_mem_[2], tensor_mem_[0],
791 tensor_mem_[2], scratch_mem_, scratch_size_,
792 cudnn_, cublas_, compute_stream); // conv2
793 } else {
794 // For SE Resnet, skip connection is added after SE (and bias is added
795 // as part of SE).
796 if (has_se_) {
797 network_[l++]->Eval(batchSize, tensor_mem_[1], tensor_mem_[0],
798 nullptr, scratch_mem_, scratch_size_, cudnn_,
799 cublas_, compute_stream); // conv2
800 } else {
801 network_[l++]->Eval(batchSize, tensor_mem_[2], tensor_mem_[0],
802 tensor_mem_[2], scratch_mem_, scratch_size_,
803 cudnn_, cublas_, compute_stream); // conv2
804 }
805
806 if (has_se_) {
807 network_[l++]->Eval(batchSize, tensor_mem_[2], tensor_mem_[1],
808 tensor_mem_[2], scratch_mem_, scratch_size_,
809 cudnn_, cublas_, compute_stream); // SE layer
810 }
811 }
812 }
813 }
814
815 // Policy head.
816 if (attn_policy_) {
817 network_[l++]->Eval(
818 batchSize, tensor_mem_[0], tensor_mem_[2], tensor_mem_[1],
819 scratch_mem_, scratch_size_, nullptr, cublas_, compute_stream,
820 &head_offset_pointers_); // Entire Attention policy head except for
821 // the policy map
822 network_[l++]->Eval(batchSize, (DataType*)opPol, tensor_mem_[0], nullptr,
823 scratch_mem_, scratch_size_, nullptr, cublas_,
824 compute_stream); // policy map layer
825 // POLICY output
826
827 } else if (conv_policy_) {
828 network_[l++]->Eval(batchSize, tensor_mem_[0], tensor_mem_[2], nullptr,
829 scratch_mem_, scratch_size_, cudnn_, cublas_,
830 compute_stream); // policy conv1
831
832 network_[l++]->Eval(batchSize, tensor_mem_[1], tensor_mem_[0], nullptr,
833 scratch_mem_, scratch_size_, cudnn_, cublas_,
834 compute_stream); // policy conv2
835
836 network_[l++]->Eval(
837 batchSize, (DataType*)opPol, tensor_mem_[1], nullptr, scratch_mem_,
838 scratch_size_, cudnn_, cublas_,
839 compute_stream); // policy map layer // POLICY output
840 } else {
841 network_[l++]->Eval(batchSize, tensor_mem_[0], tensor_mem_[2], nullptr,
842 scratch_mem_, scratch_size_, cudnn_, cublas_,
843 compute_stream); // pol conv
844
845 network_[l++]->Eval(batchSize, (DataType*)opPol, tensor_mem_[0], nullptr,
846 scratch_mem_, scratch_size_, cudnn_, cublas_,
847 compute_stream); // pol FC // POLICY
848 }
849
850 ReportCUDAErrors(cudaEventRecord(io->policy_done_event_, compute_stream));
851 ReportCUDAErrors(
852 cudaStreamWaitEvent(download_stream, io->policy_done_event_, 0));
853
854 // Copy policy output from device memory to host memory.
855 ReportCUDAErrors(cudaMemcpyAsync(
856 io->op_policy_mem_, io->op_policy_mem_gpu_,
857 sizeof(io->op_policy_mem_[0]) * kNumOutputPolicy * batchSize,
858 cudaMemcpyDeviceToHost, download_stream));
859
860 // value head
861 network_[l++]->Eval(batchSize, tensor_mem_[0], tensor_mem_[2], nullptr,
862 scratch_mem_, scratch_size_, cudnn_, cublas_,
863 compute_stream); // value conv
864
865 network_[l++]->Eval(batchSize, tensor_mem_[1], tensor_mem_[0], nullptr,
866 scratch_mem_, scratch_size_, cudnn_, cublas_,
867 compute_stream); // value FC1
868
869 network_[l++]->Eval(batchSize, (DataType*)opVal, tensor_mem_[1], nullptr,
870 scratch_mem_, scratch_size_, cudnn_, cublas_,
871 compute_stream); // value FC2 // VALUE
872
873 ReportCUDAErrors(cudaEventRecord(io->value_done_event_, compute_stream));
874 ReportCUDAErrors(
875 cudaStreamWaitEvent(download_stream, io->value_done_event_, 0));
876 ReportCUDAErrors(cudaMemcpyAsync(
877 io->op_value_mem_, io->op_value_mem_gpu_,
878 sizeof(io->op_value_mem_[0]) * (wdl_ ? 3 : 1) * batchSize,
879 cudaMemcpyDeviceToHost, download_stream));
880
881 if (moves_left_) {
882 // Moves left head
883 network_[l++]->Eval(batchSize, tensor_mem_[0], tensor_mem_[2], nullptr,
884 scratch_mem_, scratch_size_, cudnn_, cublas_,
885 compute_stream); // moves conv
886
887 network_[l++]->Eval(batchSize, tensor_mem_[1], tensor_mem_[0], nullptr,
888 scratch_mem_, scratch_size_, cudnn_, cublas_,
889 compute_stream); // moves FC1
890
891 // Moves left FC2
892 network_[l++]->Eval(batchSize, (DataType*)opMov, tensor_mem_[1], nullptr,
893 scratch_mem_, scratch_size_, cudnn_, cublas_,
894 compute_stream);
895
896 ReportCUDAErrors(
897 cudaEventRecord(io->moves_left_done_event_, compute_stream));
898 ReportCUDAErrors(
899 cudaStreamWaitEvent(download_stream, io->moves_left_done_event_, 0));
900 ReportCUDAErrors(
901 cudaMemcpyAsync(io->op_moves_left_mem_, io->op_moves_left_mem_gpu_,
902 sizeof(io->op_moves_left_mem_[0]) * batchSize,
903 cudaMemcpyDeviceToHost, download_stream));
904 }
905
906 if (!capture) {
907 ReportCUDAErrors(
908 cudaEventRecord(io->download_done_event_, download_stream));
909 }
910 }

912 void finishEval(InputsOutputs<DataType>* io, int batchSize) {
913 ReportCUDAErrors(cudaEventSynchronize(io->download_done_event_));
914 if (wdl_) {
915 // Value softmax done cpu side.
916 for (int i = 0; i < batchSize; i++) {
917 float* wdl = sizeof(io->op_value_mem_[0]) == sizeof(float)
918 ? (float*)io->op_value_mem_
919 : io->wdl_cpu_softmax_.get();
920 float w = FromType(io->op_value_mem_[3 * i + 0]);
921 float d = FromType(io->op_value_mem_[3 * i + 1]);
922 float l = FromType(io->op_value_mem_[3 * i + 2]);
923 float m = std::max({w, d, l});
924 w = std::exp(w - m);
925 d = std::exp(d - m);
926 l = std::exp(l - m);
927 float sum = w + d + l;
928 w /= sum;
929 l /= sum;
930 d /= sum;
931 wdl[2 * i + 0] = w - l;
932 wdl[2 * i + 1] = d;
933 }
934 }
935
936#ifdef DEBUG_RAW_NPS
937 const int reportingCalls = 100;
938 static int numCalls = 0;
939 static int sumBatchSize = 0;
940 static double totalTime = 0;
941
942 sumBatchSize += batchSize;
943 numCalls++;
944
945 auto t_end = std::chrono::high_resolution_clock::now();
946
947 double dt = std::chrono::duration<double>(t_end - t_start).count();
948 totalTime += dt;
949 if (numCalls == reportingCalls) {
950 double avgBatchSize = ((double)sumBatchSize) / numCalls;
951 double nps = sumBatchSize / totalTime;
952 CERR << "Avg batch size: " << avgBatchSize
953 << ", NN eval time: " << totalTime << " seconds per " << sumBatchSize
954 << " evals. NPS: " << nps;
955 sumBatchSize = 0;
956 totalTime = 0;
957 numCalls = 0;
958 }
959#endif
960 }

1166 if (!network_->GetGraphCaptureEnabled()) return;
