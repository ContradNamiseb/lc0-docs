130class Output {
131 public:
132 // Not exposed.
133 Output(const NetworkComputation& computation, int idx) {
134 for (int i = 0; i < 1858; ++i) p_[i] = computation.GetPVal(idx, i);
135 q_ = computation.GetQVal(idx);
136 d_ = computation.GetDVal(idx);
137 m_ = computation.GetMVal(idx);
138 }
139 float q() const { return q_; }
140 float d() const { return d_; }
141 float m() const { return m_; }
142 std::vector<float> p_raw(const std::vector<int>& indicies) {
143 std::vector<float> result(indicies.size());
144 for (size_t i = 0; i < indicies.size(); ++i) {
145 int idx = indicies[i];
146 if (idx < 0 || idx > 1857) {
147 throw Exception("Policy index must be between 0 and 1857.");
148 }
149 result[i] = p_[idx];
150 }
151 return result;
152 }
153
154 std::vector<float> p_softmax(const std::vector<int>& indicies) {
155 auto p_vals = p_raw(indicies);
156 float max_p = -std::numeric_limits<float>::infinity();
157 for (auto x : p_vals) max_p = std::max(max_p, x);
158
159 float total = 0.0;
160 for (auto& x : p_vals) {
161 x = FastExp(x - max_p);
162 total += x;
163 }
164 // Normalize P values to add up to 1.0.
165 if (total > 0.0f) {
166 const float scale = 1.0f / total;
167 for (auto& x : p_vals) x *= scale;
168 }
169 return p_vals;
170 }
171
172 private:
173 float p_[1858];
174 float q_;
175 float d_;
176 float m_;
177};

192class Backend {
193 public:
194 // Exported methods.
195 static inline std::vector<std::string> available_backends() {
196 return NetworkFactory::Get()->GetBackendsList();
197 }
198
199 Backend(const Weights* weights, const std::optional<std::string>& backend,
200 const std::optional<std::string>& options) {
201 std::optional<WeightsFile> w;
202 if (weights) w = weights->weights();
203 const auto& backends = GetAvailableBackends();
204 const std::string be =
205 backend.value_or(backends.empty() ? "<none>" : backends[0]);
206 OptionsDict network_options;
207 if (options) network_options.AddSubdictFromString(*options);
208 network_ = NetworkFactory::Get()->Create(be, w, network_options);
209 }
210
211 BackendCapabilities capabilities() const {
212 return BackendCapabilities(network_->GetCapabilities());
213 }
214
215 std::vector<std::unique_ptr<Output>> evaluate(
216 const std::vector<Input*>& inputs) const {
217 if (inputs.empty()) return {};
218 auto computation = network_->NewComputation();
219 for (const auto* input : inputs) {
220 InputPlanes input_copy = input->GetPlanes();
221 computation->AddInput(std::move(input_copy));
222 }
223 computation->ComputeBlocking();
224 std::vector<std::unique_ptr<Output>> result;
225 for (int i = 0; i < computation->GetBatchSize(); ++i) {
226 result.push_back(std::make_unique<Output>(*computation, i));
227 }
228 return result;
229 }
230
231 private:
232 std::unique_ptr<::lczero::Network> network_;
233};

235class GameState {
236 public:
237 GameState(const std::optional<std::string> startpos,
238 const std::vector<std::string>& moves,
239 const bool is_c960): is_c960_(is_c960) {
240 ChessBoard starting_board;
241 int no_capture_ply;
242 int full_moves;
243 starting_board.SetFromFen(startpos.value_or(ChessBoard::kStartposFen),
244 &no_capture_ply, &full_moves);
245
246 history_.Reset(starting_board, no_capture_ply,
247 full_moves * 2 - (starting_board.flipped() ? 1 : 2));
248
249 for (const auto& m : moves) {
250 auto board = history_.Last().GetBoard();
251 Move move = board.ParseMove(m);
252 history_.Append(move);
253 }
254 }
255
256 GameState(const std::optional<std::string> startpos,
257 const std::vector<std::string>& moves)
258 : GameState(startpos, moves, false) {}
259
260 std::unique_ptr<Input> as_input(const Backend& backend) const {
261 int tmp;
262 return std::make_unique<Input>(
263 EncodePositionForNN(static_cast<pblczero::NetworkFormat::InputFormat>(
264 backend.capabilities().input_format()),
265 history_, 8, FillEmptyHistory::FEN_ONLY, &tmp));
266 }
267
268 std::vector<std::string> moves() const {
269 auto ms = history_.Last().GetBoard().GenerateLegalMoves();
270 bool is_black = history_.IsBlackToMove();
271 std::vector<std::string> result;
272 for (auto m : ms) {
273 if (is_black) m.Flip();
274 result.push_back(m.ToString(is_c960_));
275 }
276 return result;
277 }
278
279 std::vector<int> policy_indices() const {
280 auto ms = history_.Last().GetBoard().GenerateLegalMoves();
281 std::vector<int> result;
282 for (auto m : ms) {
283 result.push_back(MoveToNNIndex(m, /* transform= */ 0));
284 }
285 return result;
286 }
287
288 std::string as_string() const {
289 auto board = history_.Last().GetBoard();
290 if (history_.IsBlackToMove()) board.Mirror();
291 return board.DebugString();
292 }
293
294 private:
295 PositionHistory history_;
296 bool is_c960_;
297};
