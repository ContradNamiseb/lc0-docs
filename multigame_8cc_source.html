32class PolicyEvaluator : public Evaluator {
33 public:
34 void Reset(const PlayerOptions& player) override {
35 comp_ = player.backend->CreateComputation();
36 }
37 void Gather(classic::NodeTree* tree) override {
38 const auto& history = tree->GetPositionHistory();
39 moves_.clear();
40 for (auto edge : tree->GetCurrentHead()->Edges()) {
41 moves_.push_back(edge.GetMove());
42 }
43 p_.resize(moves_.size());
44 comp_->AddInput(
45 EvalPosition{
46 .pos = history.GetPositions(),
47 .legal_moves = moves_,
48 },
49 EvalResultPtr{.p = p_});
50 }
51 void Run() override { comp_->ComputeBlocking(); }
52 void MakeBestMove(classic::NodeTree* tree) override {
53 size_t best_idx = std::max_element(p_.begin(), p_.end()) - p_.begin();
54 tree->MakeMove(moves_[best_idx]);
55 }
56
57 std::unique_ptr<BackendComputation> comp_;
58 std::vector<Move> moves_;
59 std::vector<float> p_;
60};

62class ValueEvaluator : public Evaluator {
63 public:
64 void Reset(const PlayerOptions& player) override {
65 comp_ = player.backend->CreateComputation();
66 }
67 void Gather(classic::NodeTree* tree) override {
68 PositionHistory history = tree->GetPositionHistory();
69 q_.clear();
70 q_.reserve(tree->GetCurrentHead()->GetNumEdges());
71 moves_.clear();
72 for (auto edge : tree->GetCurrentHead()->Edges()) {
73 moves_.push_back(edge.GetMove());
74 history.Append(edge.GetMove());
75 auto result = history.ComputeGameResult();
76 if (result == GameResult::UNDECIDED) {
77 comp_->AddInput(
78 EvalPosition{
79 .pos = history.GetPositions(),
80 .legal_moves = {},
81 },
82 EvalResultPtr{.q = &q_.emplace_back()});
83 } else if (result == GameResult::DRAW) {
84 q_.push_back(0);
85 } else {
86 // A legal move to a non-drawn terminal without tablebases must be a
87 // win.
88 q_.push_back(1);
89 }
90 history.Pop();
91 }
92 }
93 void Run() override { comp_->ComputeBlocking(); }
94 void MakeBestMove(classic::NodeTree* tree) override {
95 size_t best_idx = std::max_element(q_.begin(), q_.end()) - q_.begin();
96 tree->MakeMove(moves_[best_idx]);
97 }
98
99 std::unique_ptr<BackendComputation> comp_;
100 std::vector<Move> moves_;
101 std::vector<float> q_;
102};

104MultiSelfPlayGames::MultiSelfPlayGames(PlayerOptions player1,
105 PlayerOptions player2,
106 const std::vector<Opening>& openings,
107 SyzygyTablebase* syzygy_tb,
108 bool use_value)
109 : options_{player1, player2}, syzygy_tb_(syzygy_tb) {
110 eval_ = use_value
111 ? std::unique_ptr<Evaluator>(std::make_unique<ValueEvaluator>())
112 : std::unique_ptr<Evaluator>(std::make_unique<PolicyEvaluator>());
113 trees_.reserve(openings.size());
114 for (auto opening : openings) {
115 trees_.push_back(std::make_shared<classic::NodeTree>());
116 trees_.back()->ResetToPosition(opening.start_fen, {});
117 results_.push_back(GameResult::UNDECIDED);
118
119 for (Move m : opening.moves) {
120 trees_.back()->MakeMove(m);
121 }
122 }
123}

130void MultiSelfPlayGames::Play() {
131 while (true) {
132 {
133 std::lock_guard<std::mutex> lock(mutex_);
134 if (abort_) break;
135 }
136 bool all_done = true;
137 bool blacks_move = false;
138 for (size_t i = 0; i < trees_.size(); i++) {
139 const auto& tree = trees_[i];
140 if (results_[i] == GameResult::UNDECIDED) {
141 if (tree->GetPositionHistory().ComputeGameResult() !=
142 GameResult::UNDECIDED) {
143 results_[i] = tree->GetPositionHistory().ComputeGameResult();
144 continue;
145 }
146 if (syzygy_tb_ != nullptr) {
147 auto board = tree->GetPositionHistory().Last().GetBoard();
148 if (board.castlings().no_legal_castle() &&
149 (board.ours() | board.theirs()).count() <=
150 syzygy_tb_->max_cardinality()) {
151 auto tb_side_black = (tree->GetPlyCount() % 2) == 1;
152 ProbeState state;
153 const WDLScore wdl = syzygy_tb_->probe_wdl(
154 tree->GetPositionHistory().Last(), &state);
155 // Only fail state means the WDL is wrong, probe_wdl may produce
156 // correct result with a stat other than OK.
157 if (state != FAIL) {
158 if (wdl == WDL_WIN) {
159 results_[i] = tb_side_black ? GameResult::BLACK_WON
160 : GameResult::WHITE_WON;
161 } else if (wdl == WDL_LOSS) {
162 results_[i] = tb_side_black ? GameResult::WHITE_WON
163 : GameResult::BLACK_WON;
164 } else { // Cursed wins and blessed losses count as draws.
165 results_[i] = GameResult::DRAW;
166 }
167 continue;
168 }
169 }
170 }
171 if (all_done) {
172 all_done = false;
173 blacks_move = (tree->GetPlyCount() % 2) == 1;
174 // Don't break as we need to update result state for everything.
175 }
176 }
177 }
178 if (all_done) break;
179 const int idx = blacks_move ? 1 : 0;
180 eval_->Reset(options_[idx]);
181 for (size_t i = 0; i < trees_.size(); i++) {
182 const auto& tree = trees_[i];
183 if (results_[i] != GameResult::UNDECIDED) {
184 continue;
185 }
186 if (((tree->GetPlyCount() % 2) == 1) != blacks_move) continue;
187 const auto& board = tree->GetPositionHistory().Last().GetBoard();
188 auto legal_moves = board.GenerateLegalMoves();
189 tree->GetCurrentHead()->CreateEdges(legal_moves);
190 eval_->Gather(tree.get());
191 }
192 eval_->Run();
193 for (size_t i = 0; i < trees_.size(); i++) {
194 const auto& tree = trees_[i];
195 if (results_[i] != GameResult::UNDECIDED) {
196 continue;
197 }
198 if (((tree->GetPlyCount() % 2) == 1) != blacks_move) continue;
199 eval_->MakeBestMove(tree.get());
200 }
201 }
202}
