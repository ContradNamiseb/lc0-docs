130Params::MovesLeftEstimator CreateMovesLeftEstimator(const OptionsDict& params) {
131 // The only estimator we have now is MLE-legacy (Moves left estimator).
132 const OptionsDict& mle_dict = params.HasSubdict("mle-legacy")
133 ? params.GetSubdict("mle-legacy")
134 : params;
135 return [midpoint = mle_dict.GetOrDefault<float>("midpoint", 45.2f),
136 steepness = mle_dict.GetOrDefault<float>("steepness", 5.93f)](
137 const Position& position) {
138 const auto ply = position.GetGamePly();
139 return ComputeEstimatedMovesToGo(ply, midpoint, steepness);
140 };

175// Returns the updated value of @from, towards @to by the number of halves
176// equal to number of @steps in @value. E.g. if value=1*step, returns
177// (from+to)/2, if value=2*step, return (1*from + 3*to)/4, if
178// value=3*step, return (1*from + 7*to)/7, if value=0, returns from.
179float ExponentialDecay(float from, float to, float step, float value) {
180 return to - (to - from) * std::pow(0.5f, value / step);
181}
182
183float LinearInterpolate(float a, float b, float theta) {
184 if (theta <= 0) return a;
185 if (theta >= 1) return b;
186 return a + (b - a) * theta;
187}
188
189// @cur_value -- current "average", returned from the previous call of this
190// function.
191// @target_value -- current "correct but noisy" value, that we slowly go to.
192// @time_since_reliable -- time since we had reliable nps in target_value. For
193// the case of nps, end of the previous move is considered reliable point).
194// @time_since_last_update_ms -- how long ago was this function last called.
195// @window_size_ms -- in what time should the output converge to @target_value.
196float LinearDecay(float cur_value, float target_value,
197 float time_since_reliable_ms, float time_since_last_update_ms,
198 float window_size_ms) {
199 // If target is higher than cur value, just return target without any
200 // smoothing. That makes sense for nps, but for other averages we should look
201 // a bit closer.
202 if (cur_value <= target_value) return target_value;
203 // If no time passed since last update, return previous value.
204 if (time_since_last_update_ms <= 0) return cur_value;
205 // If window is large enough, we consider
206 if (time_since_reliable_ms >= window_size_ms) return target_value;
207 if (time_since_last_update_ms > time_since_reliable_ms) {
208 // Should never happen, but just for the case.
209 time_since_last_update_ms = time_since_reliable_ms;
210 }
211 // time_since_reliable_ms during the previous call of this function.
212 const float prev_time_since_reliable_ms =
213 time_since_reliable_ms - time_since_last_update_ms;
214 // The same, but as fraction of window size.
215 const float prev_reliable_frac = prev_time_since_reliable_ms / window_size_ms;
216 // What was the value when it was reliable.
217 const float reliable_value = (prev_reliable_frac * target_value - cur_value) /
218 (prev_reliable_frac - 1);
219 // Now we know window width, nps when it was reliable, and time since that,
220 // and time to converge, interpolate.
221 return LinearInterpolate(reliable_value, target_value,
222 time_since_reliable_ms / window_size_ms);
223}

277 float UpdateNps(int64_t time_since_movestart_ms,
278 int64_t nodes_since_movestart) {
279 Mutex::Lock lock(mutex_);
280 if (time_since_movestart_ms <= 0) return nps_;
281 if (nps_is_reliable_ && time_since_movestart_ms > last_time_) {
282 const float nps =
283 1000.0f * nodes_since_movestart / time_since_movestart_ms;
284 nps_ = LinearDecay(nps_, nps, time_since_movestart_ms,
285 time_since_movestart_ms - last_time_,
286 params_.nps_update_seconds() * 1000.0f);
287 } else {
288 nps_ = 1000.0f * nodes_since_movestart / time_since_movestart_ms;
289 }
290 last_time_ = time_since_movestart_ms;
291 return nps_;
292 }

294 void UpdateEndOfMoveStats(int64_t total_move_time, bool used_piggybank,
295 int64_t time_budget, int64_t total_nodes) {
296 Mutex::Lock lock(mutex_);
297 // Whatever is in nps_ after the first move, is truth now.
298 nps_is_reliable_ = true;
299 // How different was this move from an average move
300 const float this_move_time_fraction =
301 avg_ms_per_move_ <= 0.0f ? 0.0f : total_move_time / avg_ms_per_move_;
302 // Update time_use estimation.
303 const float this_move_time_use =
304 move_allocated_time_ms_ <= 0.0f
305 ? 1.0f
306 : total_move_time / move_allocated_time_ms_;
307 // Recompute expected move time for logging.
308 const float expected_move_time = move_allocated_time_ms_ * timeuse_;
309
310 int64_t piggybank_time_used = 0;
311 if (used_piggybank) {
312 piggybank_time_used = std::max(int64_t(), total_move_time - time_budget);
313 piggybank_time_ -= piggybank_time_used;
314 }
315 // If piggybank was used, time use is 100%.
316 timeuse_ =
317 ExponentialDecay(timeuse_, used_piggybank ? 1.0f : this_move_time_use,
318 params_.smartpruning_timeuse_halfupdate_moves(),
319 this_move_time_fraction);
320 if (timeuse_ < params_.min_smartpruning_timeuse()) {
321 timeuse_ = params_.min_smartpruning_timeuse();
322 }
323 // Remember final number of nodes for tree reuse estimation.
324 last_move_final_nodes_ = total_nodes;
325
326 LOGFILE << std::fixed << "TMGR: Updating endmove stats. actual_move_time="
327 << total_move_time
328 << "ms, allocated_move_time=" << move_allocated_time_ms_
329 << "ms (ratio=" << this_move_time_use
330 << "), expected_move_time=" << expected_move_time
331 << "ms. New time_use=" << timeuse_
332 << ", update_rate=" << this_move_time_fraction
333 << " (avg_move_time=" << avg_ms_per_move_ << "ms)."
334 << " piggybank_used=" << piggybank_time_used << "ms";
335 }
336
337 private:
338 std::unique_ptr<SearchStopper> GetStopper(const GoParams& params,
339 const Position& position,
340 size_t /*total_memory*/,
341 size_t /*avg_node_size*/,
342 uint32_t current_nodes) override {
343 const bool is_black = position.IsBlackToMove();
344 const std::optional<int64_t>& time =
345 (is_black ? params.btime : params.wtime);
346 // If no time limit is given, don't stop on this condition.
347 if (params.infinite || params.ponder || !time) return nullptr;
348
349 Mutex::Lock lock(mutex_);
350
351 if (is_first_move_) {
352 piggybank_time_ =
353 static_cast<int64_t>(*time * params_.initial_piggybank_fraction());
354 is_first_move_ = false;
355 }
356
357 if (last_move_final_nodes_ && last_time_ && avg_ms_per_move_ >= 0.0f) {
358 UpdateTreeReuseFactor(current_nodes);
359 }
360
361 last_time_ = 0;
362
363 // Get remaining moves estimation.
364 float remaining_moves = params_.moves_left_estimator()(position);
365
366 // If the number of moves remaining until the time control are less than
367 // the estimated number of moves left in the game, then use the number of
368 // moves until the time control instead.
369 if (params.movestogo &&
370 *params.movestogo > 0 && // Ignore non-standard uci command.
371 *params.movestogo < remaining_moves) {
372 remaining_moves = *params.movestogo;
373 }
374
375 const std::optional<int64_t>& inc = is_black ? params.binc : params.winc;
376 const int increment = inc ? std::max(int64_t(0), *inc) : 0;
377
378 // Maximum time allowed in piggybank.
379 const float max_piggybank_time =
380 params_.max_piggybank_moves() *
381 std::max(0.0f, *time + increment * (remaining_moves - 1) -
382 params_.move_overhead_ms()) /
383 remaining_moves;
384
385 if (piggybank_time_ > max_piggybank_time) {
386 piggybank_time_ = max_piggybank_time;
387 }
388
389 // Total time, including increments, until time control.
390 const auto total_remaining_ms = std::max(
391 0.0f, *time - piggybank_time_ + increment * (remaining_moves - 1) -
392 params_.move_overhead_ms());
393
394 // Total remaining nodes that we'll have chance to compute in a game.
395 const float remaining_game_nodes = total_remaining_ms * nps_ / 1000.0f;
396 // Total (fresh) nodes, in average, to processed per move.
397 const float avg_nodes_per_move = remaining_game_nodes / remaining_moves;
398 // Average time that will be spent per move.
399 avg_ms_per_move_ = total_remaining_ms / remaining_moves;
400 // As some part of a tree is usually reused, we can aim to a larger target.
401 const float nodes_per_move_including_reuse =
402 avg_nodes_per_move / (1.0f - tree_reuse_);
403 // Subtract what we already have, and get what we need to compute.
404 const float move_estimate_nodes =
405 std::max(0.0f, nodes_per_move_including_reuse - current_nodes);
406 // This is what time we think will be really spent thinking.
407 const float expected_movetime_ms_brutto =
408 move_estimate_nodes / nps_ * 1000.0f;
409 // Share of this move that will go to piggybank.
410 const float time_to_piggybank_ms = std::min(
411 max_piggybank_time - piggybank_time_,
412 expected_movetime_ms_brutto * params_.per_move_piggybank_fraction());
413 // Some share of this time will go to a piggybank, so a move gets less.
414 const float expected_movetime_ms =
415 expected_movetime_ms_brutto - time_to_piggybank_ms;
416 // When we need to use piggybank, we can use it.
417 int64_t allowed_piggybank_time_ms =
418 piggybank_time_ * params_.max_piggybank_use();
419 // This is what is the actual budget as we hope that the search will be
420 // shorter due to smart pruning.
421 move_allocated_time_ms_ = expected_movetime_ms / timeuse_;
422
423 if (move_allocated_time_ms_ >
424 *time * params_.max_single_move_time_fraction()) {
425 move_allocated_time_ms_ = *time * params_.max_single_move_time_fraction();
426 }
427 if (move_allocated_time_ms_ > *time - params_.move_overhead_ms()) {
428 move_allocated_time_ms_ = std::max(static_cast<int64_t>(0LL),
429 *time - params_.move_overhead_ms());
430 }
431 if (allowed_piggybank_time_ms >
432 *time - params_.move_overhead_ms() - move_allocated_time_ms_) {
433 allowed_piggybank_time_ms =
434 *time - params_.move_overhead_ms() - move_allocated_time_ms_;
435 }
436 piggybank_time_ += time_to_piggybank_ms;
437
438 LOGFILE << std::fixed << std::setprecision(1)
439 << "TMGR: MOVE TIME: allocated()=" << move_allocated_time_ms_
440 << "ms, expected(brutto)=" << expected_movetime_ms_brutto
441 << "ms, expected(netto)=" << expected_movetime_ms
442 << std::setprecision(5) << "ms, timeuse=" << timeuse_;
443
444 LOGFILE << std::fixed << std::setprecision(1)
445 << "TMGR: MOVE NODES: avg_total_with_reuse(brutto)="
446 << nodes_per_move_including_reuse
447 << ", avg_total_without_reuse(brutto)=" << avg_nodes_per_move
448 << ", reused=" << current_nodes
449 << ", expected_new(brutto)=" << move_estimate_nodes
450 << std::setprecision(5) << ", tree_reuse=" << tree_reuse_;
451
452 LOGFILE << std::fixed << std::setprecision(1)
453 << "TMGR: PIGGYBANK: total=" << piggybank_time_
454 << "ms, increment=" << time_to_piggybank_ms
455 << "ms, longthink=" << allowed_piggybank_time_ms
456 << "ms, max=" << max_piggybank_time << "ms";
457
458 LOGFILE << std::fixed << std::setprecision(1)
459 << "TMGR: REMAINING GAME: nodes=" << remaining_game_nodes
460 << ", moves=" << remaining_moves << ", time=" << total_remaining_ms
461 << "ms, nps=" << nps_;
462
463 return std::make_unique<SmoothStopper>(
464 move_allocated_time_ms_, allowed_piggybank_time_ms,
465 params_.trend_nps_update_period_ms(), params_.bestmove_optimism(),
466 params_.overtaker_optimism(), params_.force_piggybank_ms(), this);
467 }

469 void UpdateTreeReuseFactor(int64_t new_move_nodes) REQUIRES(mutex_) {
470 // How different was this move from an average move
471 const float this_move_time_fraction =
472 avg_ms_per_move_ <= 0.0f ? 0.0f : last_time_ / avg_ms_per_move_;
473
474 const float this_move_tree_reuse =
475 static_cast<float>(new_move_nodes) / last_move_final_nodes_;
476 tree_reuse_ = ExponentialDecay(tree_reuse_, this_move_tree_reuse,
477 params_.tree_reuse_halfupdate_moves(),
478 this_move_time_fraction);
479 if (tree_reuse_ > params_.max_tree_reuse()) {
480 tree_reuse_ = params_.max_tree_reuse();
481 }
482 LOGFILE << std::fixed << "TMGR: Updating tree reuse. last_move_nodes="
483 << last_move_final_nodes_ << ", this_move_nodes=" << new_move_nodes
484 << " (tree_reuse=" << this_move_tree_reuse
485 << "). avg_tree_reuse=" << tree_reuse_
486 << ", update_rate=" << this_move_time_fraction
487 << " (avg_move_time=" << avg_ms_per_move_
488 << "ms, actual_move_time=" << last_time_ << "ms)";
489 }

493 Mutex mutex_;
494 // Fraction of a tree which usually survives a full move (and is reused).
495 float tree_reuse_ GUARDED_BY(mutex_) = params_.initial_tree_reuse();
496 // Current NPS estimation.
497 float nps_ GUARDED_BY(mutex_) = 20000.0f;
498 // NPS is unreliable until the end of the first move.
499 bool nps_is_reliable_ GUARDED_BY(mutex_) = false;
500 // Fraction of a allocated time usually used.
501 float timeuse_ GUARDED_BY(mutex_) = params_.initial_smartpruning_timeuse();
502
503 // Average amount of time per move. Used to compute ratio for timeuse and
504 // tree reuse updates.
505 float avg_ms_per_move_ GUARDED_BY(mutex_) = 0.0f;
506 // Total amount of time allocated for the current move. Used to update
507 // timeuse_ when the move ends.
508 float move_allocated_time_ms_ GUARDED_BY(mutex_) = 0.0f;
509 // Total amount of nodes in the end of the previous search. Used to compute
510 // tree reuse factor when a new search starts.
511 int64_t last_move_final_nodes_ GUARDED_BY(mutex_) = 0;
512 // Time of the last report, since the beginning of the move.
513 int64_t last_time_ GUARDED_BY(mutex_) = 0;
514 // Time in piggybank (to use when search requests it).
515 int64_t piggybank_time_ GUARDED_BY(mutex_) = 0;
516 // Tracks whether it's called for the first time, meaning there's a need to
517 // do initialization.
518 bool is_first_move_ GUARDED_BY(mutex_) = true;

521void VisitsTrendWatcher::Update(uint64_t timestamp,
522 const std::vector<uint32_t>& visits) {
523 Mutex::Lock lock(mutex_);
524 if (timestamp <= last_timestamp_) return;
525 if (prev_visits_.empty()) {
526 prev_visits_ = visits;
527 cur_visits_ = visits;
528 prev_timestamp_ = timestamp;
529 cur_timestamp_ = timestamp;
530 }
531 last_timestamp_ = timestamp;
532 last_visits_ = visits;
533 if (cur_timestamp_ + nps_update_period_ >= timestamp) {
534 prev_timestamp_ = cur_timestamp_;
535 prev_visits_ = std::move(cur_visits_);
536 cur_visits_ = last_visits_;
537 cur_timestamp_ = last_timestamp_;
538 }

541bool VisitsTrendWatcher::IsBestmoveBeingOvertaken(
542 uint64_t by_which_time) const {
543 Mutex::Lock lock(mutex_);
544 // If we don't have any stats yet, we cannot stop the search.
545 if (prev_timestamp_ >= last_timestamp_) return false;
546 if (by_which_time <= last_timestamp_) return false;
547 std::vector<float> npms;
548 npms.reserve(last_visits_.size());
549 for (size_t i = 0; i < last_visits_.size(); ++i) {
550 npms.push_back(static_cast<float>(last_visits_[i] - prev_visits_[i]) /
551 (last_timestamp_ - prev_timestamp_));
552 }
553 const size_t bestmove_idx =
554 std::max_element(last_visits_.begin(), last_visits_.end()) -
555 last_visits_.begin();
556 const auto planned_bestmove_visits =
557 last_visits_[bestmove_idx] + bestmove_optimism_ * npms[bestmove_idx] *
558 (by_which_time - last_timestamp_);
559 for (size_t i = 0; i < last_visits_.size(); ++i) {
560 if (i == bestmove_idx) continue;
561 const auto planned_visits =
562 last_visits_[i] +
563 overtaker_optimism_ * npms[i] * (by_which_time - last_timestamp_);
564 if (planned_visits > planned_bestmove_visits) return true;
565 }
566 return false;
567}

584bool SmoothStopper::ShouldStop(const IterationStats& stats,
585 StoppersHints* hints) {
586 const auto nps = manager_->UpdateNps(stats.time_since_first_batch,
587 stats.nodes_since_movestart);
588 if (stats.time_usage_hint_ == IterationStats::TimeUsageHint::kImmediateMove) {
589 LOGFILE << "Search requested immediate stop, alrite.";
590 return true;
591 }
592
593 visits_trend_watcher_.Update(stats.time_since_movestart, stats.edge_n);
594 const auto deadline_with_piggybank = deadline_ms_ + allowed_piggybank_use_ms_;
595 const bool force_use_piggybank =
596 stats.time_since_first_batch <= forced_piggybank_use_ms_;
597 const bool use_piggybank =
598 (stats.time_usage_hint_ == IterationStats::TimeUsageHint::kNeedMoreTime ||
599 force_use_piggybank ||
600 visits_trend_watcher_.IsBestmoveBeingOvertaken(deadline_with_piggybank));
601 const int64_t time_limit =
602 use_piggybank ? deadline_with_piggybank : deadline_ms_;
603 hints->UpdateEstimatedNps(nps);
604 hints->UpdateEstimatedRemainingTimeMs(time_limit -
605 stats.time_since_movestart);
606 if (use_piggybank && stats.time_since_movestart >= deadline_ms_) {
607 // It's not entirely correct as due to extended remaining time smart pruning
608 // will trigger later and we spend more time than if use_piggyback was
609 // false, even before reaching the deadline.
610 if (!used_piggybank_.test_and_set()) {
611 LOGFILE << "Entering piggybank, reason: "
612 << (stats.time_usage_hint_ ==
613 IterationStats::TimeUsageHint::kNeedMoreTime
614 ? "requested by search."
615 : force_use_piggybank
616 ? "forced used in the beginning of the move."
617 : "bestmove can be overtaken.");
618 }
619 }
620 if (stats.time_since_movestart >= time_limit) {
621 LOGFILE << "Stopping search: Ran out of time. elapsed=" << std::fixed
622 << stats.time_since_movestart << " limit=" << time_limit
623 << " piggy=" << use_piggybank;
624 return true;
625 }
626 return false;
627}
