59std::string DecompressGzip(const std::string& filename) {
60 const int kStartingSize = 8 * 1024 * 1024; // 8M
61 std::string buffer;
62 buffer.resize(kStartingSize);
63 int bytes_read = 0;
64
65 // Read whole file into a buffer.
66 FILE* fp = fopen(filename.c_str(), "rb");
67 if (!fp) {
68 throw Exception("Cannot read weights from " + filename);
69 }
70 if (filename == CommandLine::BinaryName()) {
71 // The network file should be appended at the end of the lc0 executable,
72 // followed by the network file size and a "Lc0!" (0x2130634c) magic.
73 int32_t size, magic;
74 if (fseek(fp, -8, SEEK_END) || fread(&size, 4, 1, fp) != 1 ||
75 fread(&magic, 4, 1, fp) != 1 || magic != 0x2130634c) {
76 fclose(fp);
77 throw Exception("No embedded file detected.");
78 }
79 fseek(fp, -size - 8, SEEK_END);
80 }
81 fflush(fp);
82 gzFile file = gzdopen(dup(fileno(fp)), "rb");
83 fclose(fp);
84 if (!file) {
85 throw Exception("Cannot process file " + filename);
86 }
87 while (true) {
88 const int sz =
89 gzread(file, &buffer[bytes_read], buffer.size() - bytes_read);
90 if (sz < 0) {
91 int errnum;
92 throw Exception(gzerror(file, &errnum));
93 }
94 if (sz == static_cast<int>(buffer.size()) - bytes_read) {
95 bytes_read = buffer.size();
96 buffer.resize(buffer.size() * 2);
97 } else {
98 bytes_read += sz;
99 buffer.resize(bytes_read);
100 break;
101 }
102 }
103 gzclose(file);
104
105 return buffer;
106}

108void FixOlderWeightsFile(WeightsFile* file) {
109 using nf = pblczero::NetworkFormat;
110 auto network_format = file->format().network_format().network();
111 const auto has_network_format = file->format().has_network_format();
112
113 auto* net = file->mutable_format()->mutable_network_format();
114 if (!has_network_format) {
115 // Older protobufs don't have format definition.
116 net->set_input(nf::INPUT_CLASSICAL_112_PLANE);
117 net->set_output(nf::OUTPUT_CLASSICAL);
118 net->set_network(nf::NETWORK_CLASSICAL_WITH_HEADFORMAT);
119 net->set_value(nf::VALUE_CLASSICAL);
120 net->set_policy(nf::POLICY_CLASSICAL);
121 } else if (network_format == nf::NETWORK_CLASSICAL) {
122 // Populate policyFormat and valueFormat fields in old protobufs
123 // without these fields.
124 net->set_network(nf::NETWORK_CLASSICAL_WITH_HEADFORMAT);
125 net->set_value(nf::VALUE_CLASSICAL);
126 net->set_policy(nf::POLICY_CLASSICAL);
127 } else if (network_format == nf::NETWORK_SE) {
128 net->set_network(nf::NETWORK_SE_WITH_HEADFORMAT);
129 net->set_value(nf::VALUE_CLASSICAL);
130 net->set_policy(nf::POLICY_CLASSICAL);
131 } else if (network_format == nf::NETWORK_SE_WITH_HEADFORMAT &&
132 file->weights().encoder().size() > 0) {
133 // Attention body network made with old protobuf.
134 auto* net = file->mutable_format()->mutable_network_format();
135 net->set_network(nf::NETWORK_ATTENTIONBODY_WITH_HEADFORMAT);
136 if (file->weights().has_smolgen_w()) {
137 // Need to override activation defaults for smolgen.
138 net->set_ffn_activation(nf::ACTIVATION_RELU_2);
139 net->set_smolgen_activation(nf::ACTIVATION_SWISH);
140 }
141 } else if (network_format == nf::NETWORK_AB_LEGACY_WITH_MULTIHEADFORMAT) {
142 net->set_network(nf::NETWORK_ATTENTIONBODY_WITH_MULTIHEADFORMAT);
143 }
144
145 // Get updated network format.
146 if (file->format().network_format().network() ==
147 nf::NETWORK_ATTENTIONBODY_WITH_HEADFORMAT) {
148 auto weights = file->weights();
149 if (weights.has_policy_heads() && weights.has_value_heads()) {
150 CERR << "Weights file has multihead format, updating format flag";
151 net->set_network(nf::NETWORK_ATTENTIONBODY_WITH_MULTIHEADFORMAT);
152 net->set_input_embedding(nf::INPUT_EMBEDDING_PE_DENSE);
153 }
154 if (!file->format().network_format().has_input_embedding()) {
155 net->set_input_embedding(nf::INPUT_EMBEDDING_PE_MAP);
156 }
157 }
158}

160WeightsFile ParseWeightsProto(const std::string& buffer) {
161 WeightsFile net;
162 net.ParseFromString(buffer);
163
164 if (net.magic() != kWeightMagic) {
165 throw Exception("Invalid weight file: bad header.");
166 }
167
168 const auto min_version =
169 GetVersionStr(net.min_version().major(), net.min_version().minor(),
170 net.min_version().patch(), "", "");
171 const auto lc0_ver = GetVersionInt();
172 const auto net_ver =
173 GetVersionInt(net.min_version().major(), net.min_version().minor(),
174 net.min_version().patch());
175
176 FixOlderWeightsFile(&net);
177
178 // Weights files with this signature are also compatible.
179 if (net_ver != 0x5c99973 && net_ver > lc0_ver) {
180 throw Exception("Invalid weight file: lc0 version >= " + min_version +
181 " required.");
182 }
183
184 if (net.has_weights() &&
185 net.format().weights_encoding() != pblczero::Format::LINEAR16 &&
186 net_ver < GetVersionInt(0, 33, 0)) {
187 throw Exception("Invalid weight file: unsupported encoding.");
188 }
189
190 return net;
191}

195WeightsFile LoadWeightsFromFile(const std::string& filename) {
196 FloatVectors vecs;
197 auto buffer = DecompressGzip(filename);
198
199 if (buffer.size() < 2) {
200 throw Exception("Invalid weight file: too small.");
201 }
202 if (buffer[0] == '1' && buffer[1] == '\n') {
203 throw Exception("Invalid weight file: no longer supported.");
204 }
205 if (buffer[0] == '2' && buffer[1] == '\n') {
206 throw Exception(
207 "Text format weights files are no longer supported. Use a command line "
208 "tool to convert it to the new format.");
209 }
210
211 return ParseWeightsProto(buffer);
212}

214std::optional<WeightsFile> LoadWeights(std::string_view location) {
215 std::string net_path = std::string(location);
216 if (net_path == SharedBackendParams::kAutoDiscover) {
217 net_path = DiscoverWeightsFile();
218 } else if (net_path == SharedBackendParams::kEmbed) {
219 net_path = CommandLine::BinaryName();
220 }
221 if (net_path.empty()) return std::nullopt;
222 if (location == SharedBackendParams::kEmbed) {
223 CERR << "Using embedded weights from binary: " << net_path;
224 } else {
225 CERR << "Loading weights file from: " << net_path;
226 }
227 return LoadWeightsFromFile(net_path);
228}

230std::string DiscoverWeightsFile() {
231 const int kMinFileSize = 500000; // 500 KB
232
233 std::vector<std::string> data_dirs = {CommandLine::BinaryDirectory()};
234 const std::string user_data_path = GetUserDataDirectory();
235 if (!user_data_path.empty()) {
236 data_dirs.emplace_back(user_data_path + "lc0");
237 }
238 for (const auto& dir : GetSystemDataDirectoryList()) {
239 data_dirs.emplace_back(dir + (dir.back() == '/' ? "" : "/") + "lc0");
240 }
241
242 for (const auto& dir : data_dirs) {
243 // Open all files in <dir> amd <dir>/networks,
244 // ones which are >= kMinFileSize are candidates.
245 std::vector<std::pair<time_t, std::string> > time_and_filename;
246 for (const auto& path : {"", "/networks"}) {
247 for (const auto& file : GetFileList(dir + path)) {
248 const std::string filename = dir + path + "/" + file;
249 if (GetFileSize(filename) < kMinFileSize) continue;
250 time_and_filename.emplace_back(GetFileTime(filename), filename);
251 }
252 }
253
254 std::sort(time_and_filename.rbegin(), time_and_filename.rend());
255
256 // Open all candidates, from newest to oldest, possibly gzipped, and try to
257 // read version for it. If version is 2 or if the file is our protobuf,
258 // return it.
259 for (const auto& candidate : time_and_filename) {
260 const gzFile file = gzopen(candidate.second.c_str(), "rb");
261
262 if (!file) continue;
263 unsigned char buf[256];
264 int sz = gzread(file, buf, 256);
265 gzclose(file);
266 if (sz < 0) continue;
267
268 // First byte of the protobuf stream is 0x0d for fixed32, so we ignore it
269 // as our own magic should suffice.
270 const auto magic = buf[1] | (static_cast<uint32_t>(buf[2]) << 8) |
271 (static_cast<uint32_t>(buf[3]) << 16) |
272 (static_cast<uint32_t>(buf[4]) << 24);
273 if (magic == kWeightMagic) return candidate.second;
274 }
275 }
276 LOGFILE << "Network weights file not found.";
277 return {};
278}
