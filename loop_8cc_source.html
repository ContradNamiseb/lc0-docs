49void SelfPlayLoop::Run() {
50 SelfPlayTournament::PopulateOptions(&options_);
51
52 options_.Add<StringOption>(kLogFileId);
53
54 if (!options_.ProcessAllFlags()) return;
55
56 Logging::Get().SetFilename(
57 options_.GetOptionsDict().Get<std::string>(kLogFileId));
58
59 // Send id before starting tournament to allow wrapping client to know
60 // who we are.
61 uci_responder_->SendId();
62 SelfPlayTournament tournament(
63 options_.GetOptionsDict(), uci_responder_,
64 std::bind(&SelfPlayLoop::SendGameInfo, this, std::placeholders::_1),
65 std::bind(&SelfPlayLoop::SendTournament, this, std::placeholders::_1));
66 tournament.RunBlocking();
67}

69void SelfPlayLoop::SendGameInfo(const GameInfo& info) {
70 std::vector<std::string> responses;
71 // Send separate resign report before gameready as client gameready parsing
72 // will easily get confused by adding new parameters as both training file
73 // and move list potentially contain spaces.
74 if (info.min_false_positive_threshold) {
75 std::string resign_res = "resign_report";
76 resign_res +=
77 " fp_threshold " + std::to_string(*info.min_false_positive_threshold);
78 responses.push_back(resign_res);
79 }
80 std::string res = "gameready";
81 if (!info.training_filename.empty())
82 res += " trainingfile " + info.training_filename;
83 if (info.game_id != -1) res += " gameid " + std::to_string(info.game_id);
84 res += " play_start_ply " + std::to_string(info.play_start_ply);
85 if (info.is_black)
86 res += " player1 " + std::string(*info.is_black ? "black" : "white");
87 if (info.game_result != GameResult::UNDECIDED) {
88 res += std::string(" result ") +
89 ((info.game_result == GameResult::DRAW) ? "draw"
90 : (info.game_result == GameResult::WHITE_WON) ? "whitewon"
91 : "blackwon");
92 }
93 if (!info.moves.empty()) {
94 res += " moves";
95 for (const auto& move : info.moves) res += " " + move.ToString(true);
96 }
97 if (!info.initial_fen.empty() &&
98 info.initial_fen != ChessBoard::kStartposFen) {
99 res += " from_fen " + info.initial_fen;
100 }
101 responses.push_back(res);
102 uci_responder_->SendRawResponses(responses);
103}

105void SelfPlayLoop::SendTournament(const TournamentInfo& info) {
106 const int winp1 = info.results[0][0] + info.results[0][1];
107 const int losep1 = info.results[2][0] + info.results[2][1];
108 const int draws = info.results[1][0] + info.results[1][1];
109
110 // Initialize variables.
111 float percentage = -1;
112 std::optional<float> elo;
113 std::optional<float> los;
114
115 // Only caculate percentage if any games at all (avoid divide by 0).
116 if ((winp1 + losep1 + draws) > 0) {
117 percentage =
118 (static_cast<float>(draws) / 2 + winp1) / (winp1 + losep1 + draws);
119 }
120 // Calculate elo and los if percentage strictly between 0 and 1 (avoids divide
121 // by 0 or overflow).
122 if ((percentage < 1) && (percentage > 0))
123 elo = -400 * log(1 / percentage - 1) / log(10);
124 if ((winp1 + losep1) > 0) {
125 los = .5f +
126 .5f * std::erf((winp1 - losep1) / std::sqrt(2.0 * (winp1 + losep1)));
127 }
128 std::ostringstream oss;
129 oss << "tournamentstatus";
130 if (info.finished) oss << " final";
131 oss << " P1: +" << winp1 << " -" << losep1 << " =" << draws;
132
133 if (percentage > 0) {
134 oss << " Win: " << std::fixed << std::setw(5) << std::setprecision(2)
135 << (percentage * 100.0f) << "%";
136 }
137 if (elo) {
138 oss << " Elo: " << std::fixed << std::setw(5) << std::setprecision(2)
139 << (*elo);
140 }
141 if (los) {
142 oss << " LOS: " << std::fixed << std::setw(5) << std::setprecision(2)
143 << (*los * 100.0f) << "%";
144 }
145
146 oss << " P1-W: +" << info.results[0][0] << " -" << info.results[2][0] << " ="
147 << info.results[1][0];
148 oss << " P1-B: +" << info.results[0][1] << " -" << info.results[2][1] << " ="
149 << info.results[1][1];
150 oss << " npm " + std::to_string(static_cast<double>(info.nodes_total_) /
151 info.move_count_);
152 oss << " nodes " + std::to_string(info.nodes_total_);
153 oss << " moves " + std::to_string(info.move_count_);
154 uci_responder_->SendRawResponse(oss.str());
155}
