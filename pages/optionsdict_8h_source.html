42template <typename T>
43class TypeDict {
44 protected:
45 struct V {
46 const T& Get() const {
47 was_read_since_last_set_.store(true, std::memory_order::release);
48 return value_;
49 }
50 T& Get() {
51 was_read_since_last_set_.store(true, std::memory_order::release);
52 return value_;
53 }
54 void Set(const T& v) {
55 value_ = v;
56 was_read_since_last_set_.store(false, std::memory_order::release);
57 }
58 bool WasReadSinceLastSet() const {
59 return was_read_since_last_set_.load(std::memory_order::acquire);
60 }
61
62 V() = default;
63 V(const V& o) :
64 was_read_since_last_set_{o.was_read_since_last_set_.load(std::memory_order::acquire)},
65 value_{o.value_} {
66 }
67 V& operator=(const V& o) {
68 value_ = o.value_;
69 was_read_since_last_set_.store(o.was_read_since_last_set_.load(std::memory_order::acquire),
70 std::memory_order::release);
71 return *this;
72 }
73 V(const T& v) : value_{v} {}
74 private:
75 mutable std::atomic<bool> was_read_since_last_set_ = false;
76 T value_;
77 };
78 void EnsureNoUnusedOptions(const std::string& type_name,
79 const std::string& prefix) const {
80 for (auto const& option : dict_) {
81 if (!option.second.WasReadSinceLastSet()) {
82 throw Exception("Unknown " + type_name + " option: " + prefix +
83 option.first);
84 }
85 }
86 }
87
88 const std::unordered_map<std::string, V>& dict() const { return dict_; }
89 std::unordered_map<std::string, V>* mutable_dict() { return &dict_; }
90
91 private:
92 std::unordered_map<std::string, V> dict_;
93};

95class OptionId {
96 public:
97 enum VisibilityMode {
98 kSimpleMode = 1 << 0, // Simple mode.
99 kNormalMode = 1 << 1, // Normal mode.
100 kProMode = 1 << 2, // Pro mode.
101 };
102
103 enum VisibilityMask {
104 kSimpleOnly = kSimpleMode,
105 kDefaultVisibility = kNormalMode | kProMode,
106 kProOnly = kProMode,
107 kAlwaysVisible = kSimpleMode | kNormalMode | kProMode,
108 };
109
110 struct OptionsParams {
111 const char* long_flag = nullptr;
112 const char* uci_option = nullptr;
113 const char* help_text = nullptr;
114 char short_flag = '\0';
115 VisibilityMask visibility = kDefaultVisibility;
116 };
117
118 OptionId(const OptionsParams& params)
119 : long_flag_(params.long_flag),
120 uci_option_(params.uci_option),
121 help_text_(params.help_text),
122 short_flag_(params.short_flag),

131

141 private:

144 const char* const help_text_;

146 uint64_t visibility_mask_ = kDefaultVisibility;
147};

270template <typename T>
271T OptionsDict::Get(const std::string& key) const {
272 for (const auto* alias : aliases_) {
273 const auto value = alias->OwnGet<T>(key);
274 if (value) return *value;
275 }
276 if (parent_) return parent_->Get<T>(key);
277 throw Exception("Key [" + key + "] was not set in options.");
278}
279template <typename T>
280T OptionsDict::Get(const OptionId& option_id) const {
281 return Get<T>(GetOptionId(option_id));
282}
283template <typename T>
284std::optional<T> OptionsDict::OwnGet(const std::string& key) const {
285 const auto& dict = TypeDict<T>::dict();
286 auto iter = dict.find(key);
287 if (iter != dict.end()) {
288 return iter->second.Get();
289 }
290 return {};
291}
292template <typename T>
293std::optional<T> OptionsDict::OwnGet(const OptionId& option_id) const {
294 return OwnGet<T>(GetOptionId(option_id));
