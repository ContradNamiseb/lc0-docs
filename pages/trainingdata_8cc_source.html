33std::tuple<float, float> DriftCorrect(float q, float d) {
34 // Training data doesn't have a high number of nodes, so there shouldn't be
35 // too much drift. Highest known value not caused by backend bug was 1.5e-7.
36 const float allowed_eps = 0.000001f;
37 if (q > 1.0f) {
38 if (q > 1.0f + allowed_eps) {
39 CERR << "Unexpectedly large drift in q " << q;
40 }
41 q = 1.0f;
42 }
43 if (q < -1.0f) {
44 if (q < -1.0f - allowed_eps) {
45 CERR << "Unexpectedly large drift in q " << q;
46 }
47 q = -1.0f;
48 }
49 if (d > 1.0f) {
50 if (d > 1.0f + allowed_eps) {
51 CERR << "Unexpectedly large drift in d " << d;
52 }
53 d = 1.0f;
54 }
55 if (d < 0.0f) {
56 if (d < 0.0f - allowed_eps) {
57 CERR << "Unexpectedly large drift in d " << d;
58 }
59 d = 0.0f;
60 }
61 float w = (1.0f - d + q) / 2.0f;
62 float l = w - q;
63 // Assume q drift is rarer than d drift and apply all correction to d.
64 if (w < 0.0f || l < 0.0f) {
65 float drift = 2.0f * std::min(w, l);
66 if (drift < -allowed_eps) {
67 CERR << "Unexpectedly large drift correction for d based on q. " << drift;
68 }
69 d += drift;
70 // Since q is in range -1 to 1 - this correction should never push d outside
71 // of range, but precision could be lost in calculations so just in case.
72 if (d < 0.0f) {
73 d = 0.0f;
74 }
75 }
76 return {q, d};
77}

80void V6TrainingDataArray::Write(TrainingDataWriter* writer, GameResult result,
81 bool adjudicated) const {
82 if (training_data_.empty()) return;
83 // Base estimate off of best_m. If needed external processing can use a
84 // different approach.
85 float m_estimate = training_data_.back().best_m + training_data_.size() - 1;
86 for (auto chunk : training_data_) {
87 bool black_to_move = chunk.side_to_move_or_enpassant;
88 if (IsCanonicalFormat(static_cast<pblczero::NetworkFormat::InputFormat>(
89 chunk.input_format))) {
90 black_to_move = (chunk.invariance_info & (1u << 7)) != 0;
91 }
92 if (result == GameResult::WHITE_WON) {
93 chunk.result_q = black_to_move ? -1 : 1;
94 chunk.result_d = 0;
95 } else if (result == GameResult::BLACK_WON) {
96 chunk.result_q = black_to_move ? 1 : -1;
97 chunk.result_d = 0;
98 } else {
99 chunk.result_q = 0;
100 chunk.result_d = 1;
101 }
102 if (adjudicated) {
103 chunk.invariance_info |= 1u << 5; // Game adjudicated.
104 }
105 if (adjudicated && result == GameResult::UNDECIDED) {
106 chunk.invariance_info |= 1u << 4; // Max game length exceeded.
107 }
108 chunk.plies_left = m_estimate;
109 m_estimate -= 1.0f;
110 writer->WriteChunk(chunk);
111 }
112}

114void V6TrainingDataArray::Add(const classic::Node* node,
115 const PositionHistory& history,
116 classic::Eval best_eval,
117 classic::Eval played_eval, bool best_is_proven,
118 Move best_move, Move played_move,
119 std::span<Move> legal_moves,
120 const std::optional<EvalResult>& nneval,
121 float policy_softmax_temp) {
122 V6TrainingData result;
123 const auto& position = history.Last();
124
125 // Set version.
126 result.version = 6;
127 result.input_format = input_format_;
128
129 // Populate planes.
130 int transform;
131 InputPlanes planes = EncodePositionForNN(
132 input_format_, history, 8, fill_empty_history_[position.IsBlackToMove()],
133 &transform);
134 int plane_idx = 0;
135 for (auto& plane : result.planes) {
136 plane = ReverseBitsInBytes(planes[plane_idx++].mask);
137 }
138
139 // Populate probabilities.
140 auto total_n = node->GetChildrenVisits();
141 // Prevent garbage/invalid training data from being uploaded to server.
142 // It's possible to have N=0 when there is only one legal move in position
143 // (due to smart pruning).
144 if (total_n == 0 && node->GetNumEdges() != 1) {
145 throw Exception("Search generated invalid data!");
146 }
147 // Set illegal moves to have -1 probability.
148 std::fill(std::begin(result.probabilities), std::end(result.probabilities),
149 -1);
150 // Set moves probabilities according to their relative amount of visits.
151 // Compute Kullback-Leibler divergence in nats (between policy and visits).
152 float kld_sum = 0;
153 float total = 0.0;
154 for (const auto& child : node->Edges()) {
155 const Move move = child.GetMove();
156 float fracv = total_n > 0 ? child.GetN() / static_cast<float>(total_n) : 1;
157 if (nneval) {
158 size_t move_idx =
159 std::find(legal_moves.begin(), legal_moves.end(), move) -
160 legal_moves.begin();
161 // Undo any softmax temperature in the cached data.
162 float P = std::pow(nneval->p[move_idx], policy_softmax_temp);
163 if (fracv > 0) {
164 kld_sum += fracv * std::log(fracv / P);
165 }
166 total += P;
167 }
168 result.probabilities[MoveToNNIndex(move, transform)] = fracv;
169 }
170 if (nneval) {
171 // Add small epsilon for backward compatibility with earlier value of 0.
172 auto epsilon = std::numeric_limits<float>::min();
173 kld_sum = std::max(kld_sum + std::log(total), 0.0f) + epsilon;
174 }
175 result.policy_kld = kld_sum;
176
177 const auto& castlings = position.GetBoard().castlings();
178 // Populate castlings.
179 // For non-frc trained nets, just send 1 like we used to.
180 uint8_t our_queen_side = 1;
181 uint8_t our_king_side = 1;
182 uint8_t their_queen_side = 1;
183 uint8_t their_king_side = 1;
184 // If frc trained, send the bit mask representing rook position.
185 if (Is960CastlingFormat(input_format_)) {
186 our_queen_side <<= castlings.our_queenside_rook.idx;
187 our_king_side <<= castlings.our_kingside_rook.idx;
188 their_queen_side <<= castlings.their_queenside_rook.idx;
189 their_king_side <<= castlings.their_kingside_rook.idx;
190 }
191
192 result.castling_us_ooo = castlings.we_can_000() ? our_queen_side : 0;
193 result.castling_us_oo = castlings.we_can_00() ? our_king_side : 0;
194 result.castling_them_ooo = castlings.they_can_000() ? their_queen_side : 0;
195 result.castling_them_oo = castlings.they_can_00() ? their_king_side : 0;
196
197 // Other params.
198 if (IsCanonicalFormat(input_format_)) {
199 result.side_to_move_or_enpassant =
200 position.GetBoard().en_passant().as_int() >> 56;
201 if ((transform & FlipTransform) != 0) {
202 result.side_to_move_or_enpassant =
203 ReverseBitsInBytes(result.side_to_move_or_enpassant);
204 }
205 // Send transform in deprecated move count so rescorer can reverse it to
206 // calculate the actual move list from the input data.
207 result.invariance_info =
208 transform | (position.IsBlackToMove() ? (1u << 7) : 0u);
209 } else {
210 result.side_to_move_or_enpassant = position.IsBlackToMove() ? 1 : 0;
211 result.invariance_info = 0;
212 }
213 if (best_is_proven) {
214 result.invariance_info |= 1u << 3; // Best node is proven best;
215 }
216 result.dummy = 0;
217 result.rule50_count = position.GetRule50Ply();
218
219 // Game result is undecided.
220 result.result_q = 0;
221 result.result_d = 1;
222
223 classic::Eval orig_eval;
224 if (nneval) {
225 orig_eval.wl = nneval->q;
226 orig_eval.d = nneval->d;
227 orig_eval.ml = nneval->m;
228 } else {
229 orig_eval.wl = std::numeric_limits<float>::quiet_NaN();
230 orig_eval.d = std::numeric_limits<float>::quiet_NaN();
231 orig_eval.ml = std::numeric_limits<float>::quiet_NaN();
232 }
233
234 // Aggregate evaluation WL.
235 result.root_q = -node->GetWL();
236 result.best_q = best_eval.wl;
237 result.played_q = played_eval.wl;
238 result.orig_q = orig_eval.wl;
239
240 // Draw probability of WDL head.
241 result.root_d = node->GetD();
242 result.best_d = best_eval.d;
243 result.played_d = played_eval.d;
244 result.orig_d = orig_eval.d;
245
246 std::tie(result.best_q, result.best_d) =
247 DriftCorrect(result.best_q, result.best_d);
248 std::tie(result.root_q, result.root_d) =
249 DriftCorrect(result.root_q, result.root_d);
250 std::tie(result.played_q, result.played_d) =
251 DriftCorrect(result.played_q, result.played_d);
252
253 result.root_m = node->GetM();
254 result.best_m = best_eval.ml;
255 result.played_m = played_eval.ml;
256 result.orig_m = orig_eval.ml;
257
258 result.visits = node->GetN();
259 if (position.IsBlackToMove()) {
260 best_move.Flip();
261 played_move.Flip();
262 }
263 result.best_idx = MoveToNNIndex(best_move, transform);
264 result.played_idx = MoveToNNIndex(played_move, transform);
265 result.reserved = 0;
266
267 // Unknown here - will be filled in once the full data has been collected.
268 result.plies_left = 0;
269 training_data_.push_back(result);
270}
