58// Provides methods to load and probe syzygy tablebases.
59// Thread safe methods are thread safe subject to the non-thread sfaety
60// conditions of the init method.
61class SyzygyTablebase {
62 public:
63 SyzygyTablebase();
64 virtual ~SyzygyTablebase();
65 // Current maximum number of pieces on board that can be probed for. Will
66 // be 0 unless initialized with tablebase paths.
67 // Thread safe.
68 int max_cardinality() { return max_cardinality_; }
69 // Allows for the tablebases being used to be changed. This method is not
70 // thread safe, there must be no concurrent usage while this method is
71 // running. All other thread safe method calls must be strictly ordered with
72 // respect to this method.
73 bool init(const std::string& paths);
74 // Probes WDL tables for the given position to determine a WDLScore.
75 // Thread safe.
76 // Result is only strictly valid for positions with 0 ply 50 move counter.
77 // Probe state will return FAIL if the position is not in the tablebase.
78 WDLScore probe_wdl(const Position& pos, ProbeState* result);
79 // Probes DTZ tables for the given position to determine the number of ply
80 // before a zeroing move under optimal play.
81 // Thread safe.
82 // Probe state will return FAIL if the position is not in the tablebase.
83 int probe_dtz(const Position& pos, ProbeState* result);
84 // Probes DTZ tables to determine which moves are on the optimal play path.
85 // Assumes the position is one reached such that the side to move has been
86 // performing optimal play moves since the last 50 move counter reset.
87 // has_repeated should be whether there are any repeats since last 50 move
88 // counter reset.
89 // Thread safe.
90 // Returns false if the position is not in the tablebase.
91 // Safe moves are added to the safe_moves output paramater.
92 bool root_probe(const Position& pos, bool has_repeated, bool win_only,
93 std::vector<Move>* safe_moves);
94 // Probes WDL tables to determine which moves might be on the optimal play
95 // path. If 50 move ply counter is non-zero some (or maybe even all) of the
96 // returned safe moves in a 'winning' position, may actually be draws.
97 // Returns false if the position is not in the tablebase.
98 // Safe moves are added to the safe_moves output paramater.
99 bool root_probe_wdl(const Position& pos, std::vector<Move>* safe_moves);
100
101 private:
102 template <bool CheckZeroingMoves = false>
103 WDLScore search(const Position& pos, ProbeState* result);
104
105 std::string paths_;
106 // Caches the max_cardinality from the impl, as max_cardinality may be a hot
107 // path.
108 int max_cardinality_;
109 std::unique_ptr<SyzygyTablebaseImpl> impl_;
110};
