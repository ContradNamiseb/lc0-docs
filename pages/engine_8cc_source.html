75GameState MakeGameState(const std::string& fen,
76 const std::vector<std::string>& moves) {
77 GameState state;
78 state.startpos = Position::FromFen(fen);
79 ChessBoard cur_board = state.startpos.GetBoard();
80 state.moves.reserve(moves.size());
81 for (const auto& move : moves) {
82 Move m = cur_board.ParseMove(move);
83 state.moves.push_back(m);
84 cur_board.ApplyMove(m);
85 cur_board.Mirror();
86 }
87 return state;
88}

95 void OutputBestMove(BestMoveInfo* info) override {
96 if (!wrapped_) return;
97 wrapped_->OutputBestMove(info);
98 }
99 void OutputThinkingInfo(std::vector<ThinkingInfo>* infos) override {
100 if (!wrapped_) return;
101 if (engine_->last_go_params_ && engine_->last_go_params_->ponder) {
102 assert(engine_->last_position_ &&
103 !engine_->last_position_->moves.empty());
104 const Move ponder_move_ = engine_->last_position_->moves.back();
105 // Output all stats from main variation (not necessary the ponder move)
106 // but PV only from ponder move.
107 ThinkingInfo ponder_info;
108 for (const auto& info : *infos) {
109 if (info.multipv <= 1) {
110 ponder_info = info;
111 if (ponder_info.mate) ponder_info.mate = -*ponder_info.mate;
112 if (ponder_info.score) ponder_info.score = -*ponder_info.score;
113 if (ponder_info.depth > 1) ponder_info.depth--;
114 if (ponder_info.seldepth > 1) ponder_info.seldepth--;
115 if (ponder_info.wdl)
116 std::swap(ponder_info.wdl->w, ponder_info.wdl->l);
117 ponder_info.pv.clear();
118 }
119 if (!info.pv.empty() && info.pv[0] == ponder_move_) {
120 ponder_info.pv.assign(info.pv.begin() + 1, info.pv.end());
121 }
122 }
123 infos->clear();
124 infos->push_back(ponder_info);
125 }
126 wrapped_->OutputThinkingInfo(infos);
127 }
128
129 void Register(UciResponder* wrapped) {
130 if (wrapped_) {
131 throw Exception("UciPonderForwarder already has a wrapped responder");
132 }
133 wrapped_ = wrapped;
134 }
135 void Unregister(UciResponder* wrapped) {
136 if (wrapped_ != wrapped) {
137 throw Exception("UciPonderForwarder doesn't have this wrapped responder");
138 }
139 wrapped_ = nullptr;
140 }

142 private:
143 UciResponder* wrapped_ = nullptr;
144 Engine* const engine_;

164void Engine::UpdateBackendConfig() {
165 LOGFILE << "Update backend configuration.";
166 const std::string backend_name =
167 options_.Get<std::string>(SharedBackendParams::kBackendId);
168 if (!backend_ || backend_name != backend_name_ ||
169 backend_->UpdateConfiguration(options_) == Backend::NEED_RESTART) {
170 backend_name_ = backend_name;
171 backend_ = CreateMemCache(BackendManager::Get()->CreateFromParams(options_),
172 options_);
173 search_->SetBackend(backend_.get());
174 } else {
175 backend_->SetCacheSize(
176 options_.Get<int>(SharedBackendParams::kNNCacheSizeId));
177 }

180void Engine::EnsureSyzygyTablebasesLoaded() {
181 const std::string tb_paths = options_.Get<std::string>(kSyzygyTablebaseId);
182 if (tb_paths == previous_tb_paths_) return;
183 previous_tb_paths_ = tb_paths;
184
185 if (tb_paths.empty()) {
186 LOGFILE << "Reset Syzygy tablebases.";
187 syzygy_tb_.reset();
188 } else {
189 syzygy_tb_ = std::make_unique<SyzygyTablebase>();
190 CERR << "Loading Syzygy tablebases from " << tb_paths;
191 if (!syzygy_tb_->init(tb_paths)) {
192 CERR << "Failed to load Syzygy tablebases!";
193 syzygy_tb_.reset();
194 }
195 }
196
197 search_->SetSyzygyTablebase(syzygy_tb_.get());
198}

200// Initializes the search with either the specified position for the normal
201// search or the position one ply trimmed for the ponder search.
202void Engine::InitializeSearchPosition(bool for_ponder) {
203 LOGFILE << "Setting a new search position.";
204 assert(last_position_);
205 if (!for_ponder) {
206 search_->SetPosition(*last_position_);
207 return;
208 }
209 if (last_position_->moves.empty()) {
210 throw Exception("Ponder search requires at least one move.");
211 }
212 GameState position = *last_position_;
213 position.moves.pop_back();
214 search_->SetPosition(position);
215 return;
216}

218void Engine::SetPosition(const std::string& fen,
219 const std::vector<std::string>& moves) {
220 EnsureSearchStopped();
221 ponder_enabled_ = options_.Get<bool>(kPonderId);
222 strict_uci_timing_ = options_.Get<bool>(kStrictUciTiming);
223 isready_seen_ = false;
224 search_->StartClock();
225 UpdateBackendConfig();
226 EnsureSyzygyTablebasesLoaded();
227 last_position_ = MakeGameState(fen, moves);
228 if (!ponder_enabled_) InitializeSearchPosition(/*for_ponder=*/false);
229}

237void Engine::Go(const GoParams& params) {
238 if (!ponder_enabled_ && params.ponder) {
239 throw Exception(
240 "Ponder is not enabled, but the ponder search is requested.");
241 }
242 if ((strict_uci_timing_ && isready_seen_) ||
243 !(params.wtime || params.btime)) {
244 search_->StartClock();
245 }
246 if (!last_position_) NewGame();
247 if (ponder_enabled_) InitializeSearchPosition(params.ponder);
248 last_go_params_ = params;
249 search_->StartSearch(params);
250}

258void Engine::PonderHit() {
259 if (!last_go_params_ || !last_go_params_->ponder) {
260 throw Exception("ponderhit while not pondering");
261 }
262 EnsureSearchStopped();
263 search_->StartClock();
264 last_go_params_->ponder = false;
265 InitializeSearchPosition(/*ponder=*/false);
266 search_->StartSearch(*last_go_params_);
267}
