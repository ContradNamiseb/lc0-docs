40bool ProcessParameters(OptionsParser* options) {
41 options->Add<StringOption>(kWeightsFilenameId);
42 if (!options->ProcessAllFlags()) return false;
43 const OptionsDict& dict = options->GetOptionsDict();
44 dict.EnsureExists<std::string>(kWeightsFilenameId);
45
46 return true;
47}

146void ShowNetworkWeightsBodyInfo(const pblczero::Net& weights) {
147 const auto& w = weights.weights();
148 if (w.encoder_size() > 0) {
149 COUT << Justify("Encoders") << w.encoder_size();
150 COUT << Justify("Encoder heads") << w.headcount();
151 COUT << Justify("Embedding size") << w.ip_emb_b().params().size() / 2;
152 COUT << Justify("Dmodel") << w.encoder(0).mha().q_b().params().size() / 2;
153 COUT << Justify("Encoder DFF")
154 << w.encoder(0).ffn().dense1_b().params().size() / 2;
155 } else {
156 COUT << Justify("Blocks") << w.residual_size();
157 int se_count = 0;
158 for (size_t i = 0; i < w.residual_size(); i++) {
159 if (w.residual(i).has_se()) se_count++;
160 }
161 if (se_count > 0) {
162 COUT << Justify("SE blocks") << se_count;
163 }
164 COUT << Justify("Filters")
165 << w.input().weights().params().size() / 2 / 112 / 9;
166 }
167}

169void ShowNetworkWeightsPolicyInfo(const pblczero::Net& weights) {
170 using pblczero::NetworkFormat;
171 const auto& w = weights.weights();
172 const auto& format = weights.format().network_format();
173 auto pol_activation = NetworkFormat::ACTIVATION_DEFAULT;
174 if (format.policy() == NetworkFormat::POLICY_ATTENTION) {
175 // Non-attentionbody nets use hardcoded SELU as policy activation and FFN
176 // activations.
177 auto ffn_activation = format.ffn_activation();
178 if (w.encoder_size() == 0) {
179 pol_activation = NetworkFormat::ACTIVATION_SELU;
180 ffn_activation = NetworkFormat::ACTIVATION_SELU;
181 }
182
183 COUT << Justify("Policy") << "Attention";
184 COUT << Justify("Policy activation")
185 << NetworkFormat::ActivationFunction_Name(pol_activation);
186
187 if (w.pol_encoder_size() > 0) {
188 COUT << Justify("Policy encoders") << w.pol_encoder_size();
189 COUT << Justify("Policy encoder heads") << w.pol_headcount();
190 COUT << Justify("Policy encoder Dmodel")
191 << w.pol_encoder(0).mha().q_b().params().size() / 2;
192 COUT << Justify("Policy encoder DFF")
193 << w.pol_encoder(0).ffn().dense1_b().params().size() / 2;
194 COUT << Justify("Policy FFN activation")
195 << NetworkFormat::ActivationFunction_Name(ffn_activation);
196 }
197 COUT << Justify("Policy Dmodel") << w.ip2_pol_b().params().size() / 2;
198 } else {
199 COUT << Justify("Policy") << (w.has_policy1() ? "Convolution" : "Dense");
200 COUT << Justify("Policy activation")
201 << NetworkFormat::ActivationFunction_Name(pol_activation);
202 if (!w.has_policy1()) {
203 int policy_channels = w.policy().biases().params().size() / 2;
204 if (policy_channels == 0) {
205 policy_channels = w.policy().bn_means().params().size() / 2;
206 }
207 COUT << Justify("Policy channels") << policy_channels;
208 }
209 }

226void ShowNetworkOnnxInfo(const pblczero::Net& weights,
227 bool show_onnx_internals) {
228 if (!weights.has_onnx_model()) return;
229 const auto& onnx_model = weights.onnx_model();
230 COUT << "\nONNX interface";
231 COUT << "~~~~~~~~~~~~~~";
232 if (onnx_model.has_data_type()) {
233 COUT << Justify("Data type")
234 << pblczero::OnnxModel::DataType_Name(onnx_model.data_type());
235 }
236 if (onnx_model.has_input_planes()) {
237 COUT << Justify("Input planes") << onnx_model.input_planes();
238 }
239 if (onnx_model.has_output_value()) {
240 COUT << Justify("Output value") << onnx_model.output_value();
241 }
242 if (onnx_model.has_output_wdl()) {
243 COUT << Justify("Output WDL") << onnx_model.output_wdl();
244 }
245 if (onnx_model.has_output_policy()) {
246 COUT << Justify("Output Policy") << onnx_model.output_policy();
247 }
248 if (onnx_model.has_output_mlh()) {
249 COUT << Justify("Output MLH") << onnx_model.output_mlh();
250 }
251
252 if (!show_onnx_internals) return;
253 if (!onnx_model.has_model()) return;
254
255 pblczero::ModelProto onnx;
256 onnx.ParseFromString(onnx_model.model());
257 COUT << "\nONNX model";
258 COUT << "~~~~~~~~~~";
259
260 if (onnx.has_ir_version()) {
261 COUT << Justify("IR version") << onnx.ir_version();
262 }
263 if (onnx.has_producer_name()) {
264 COUT << Justify("Producer Name") << onnx.producer_name();
265 }
266 if (onnx.has_producer_version()) {
267 COUT << Justify("Producer Version") << onnx.producer_version();
268 }
269 if (onnx.has_domain()) {
270 COUT << Justify("Domain") << onnx.domain();
271 }
272 if (onnx.has_model_version()) {
273 COUT << Justify("Model Version") << onnx.model_version();
274 }
275 if (onnx.has_doc_string()) {
276 COUT << Justify("Doc String") << onnx.doc_string();
277 }
278 for (const auto& input : onnx.graph().input()) {
279 std::string name(input.name());
280 if (input.has_doc_string()) {
281 name += " (" + std::string(input.doc_string()) + ")";
282 }
283 COUT << Justify("Input") << name;
284 }
285 for (const auto& output : onnx.graph().output()) {
286 std::string name(output.name());
287 if (output.has_doc_string()) {
288 name += " (" + std::string(output.doc_string()) + ")";
289 }
290 COUT << Justify("Output") << name;
291 }
292 for (const auto& opset : onnx.opset_import()) {
293 std::string name;
294 if (opset.has_domain()) name += std::string(opset.domain()) + " ";
295 COUT << Justify("Opset") << name << opset.version();
296 }
297}
