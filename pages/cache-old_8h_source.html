39// Generic LRU cache. Thread-safe.
40template <class K, class V>
41class LruCache {
42 public:
43 LruCache(int capacity = 128) : capacity_(capacity) {}
44
45 // Inserts the element under key @key with value @val.
46 // If the element is pinned, old value is still kept (until fully unpinned),
47 // but new lookups will return updated value.
48 // If @pinned, pins inserted element, Unpin has to be called to unpin.
49 // In any case, puts element to front of the queue (makes it last to evict).
50 V* Insert(K key, std::unique_ptr<V> val, bool pinned = false) {
51 std::lock_guard<std::mutex> lock(mutex_);
52 V* new_val = val.get();
53 auto iter = lookup_.find(key);
54 if (iter != lookup_.end()) {
55 auto list_iter = iter->second;
56 MakePending(list_iter);
57 list_iter->pins = pinned ? 1 : 0;
58 list_iter->value = std::move(val);
59 BringToFront(list_iter);
60 } else {
61 MaybeCleanup(capacity_ - 1);
62 lru_.emplace_front(key, std::move(val), pinned ? 1 : 0);
63 lookup_[key] = lru_.begin();
64 }
65 return new_val;
66 }
67
68 // Looks up and pins the element by key. Returns nullptr if not found.
69 // If found, brings the element to the head of the queue (makes it last to
70 // evict).
71 V* Lookup(K key) {
72 std::lock_guard<std::mutex> lock(mutex_);
73 auto iter = lookup_.find(key);
74 if (iter == lookup_.end()) return nullptr;
75 ++iter->second->pins;
76 return iter->second->value.get();
77 }
78
79 // Checks whether a key exists. Doesn't lock. Of course the next moment the
80 // key may be evicted.
81 bool ContainsKey(K key) {
82 std::lock_guard<std::mutex> lock(mutex_);
83 auto iter = lookup_.find(key);
84 return iter != lookup_.end();
85 }
86
87 // Unpins the element given key and value.
88 void Unpin(K key, V* value) {
89 std::lock_guard<std::mutex> lock(mutex_);
90 auto pending_iter = pending_pins_.find({key, value});
91 if (pending_iter != pending_pins_.end()) {
92 if (--pending_iter->second.pins == 0) pending_pins_.erase(pending_iter);
93 return;
94 }
95 auto iter = lookup_.find(key);
96 --iter->second->pins;
97 }
98
99 // Sets the capacity of the cache. If new capacity is less than current size
100 // of the cache, oldest entries are evicted.
101 void SetCapacity(int capacity) {
102 std::lock_guard<std::mutex> lock(mutex_);
103 capacity_ = capacity;
104 MaybeCleanup(capacity);
105 }
106
107 int GetSize() const { return lru_.size(); }
108 int GetCapacity() const { return capacity_; }
109
110 private:
111 // Mutex should be locked when calling this function.
112 void MaybeCleanup(int size) {
113 if (size >= lookup_.size()) return;
114 int to_delete = lookup_.size() - size;
115 auto iter = std::prev(lru_.end());
116 for (int i = 0; i < to_delete; ++i) {
117 lookup_.erase(iter->key);
118 MakePending(iter);
119 iter = lru_.erase(iter);
120 --iter;
121 }
122 }
123
124 struct Item {
125 Item(K key, std::unique_ptr<V> value, int pins)
126 : key(key), value(std::move(value)), pins(pins) {}
127 K key;
128 std::unique_ptr<V> value;
129 int pins;
130 };
131
132 void MakePending(typename std::list<Item>::iterator iter) {
133 if (iter->pins > 0) {
134 K key = iter->key;
135 V* val = iter->value.get();
136 int pins = iter->pins;
137 pending_pins_.emplace(std::make_pair(key, val),
138 Item{key, std::move(iter->value), pins});
139 }
140 }
141
142 void BringToFront(typename std::list<Item>::iterator iter) {
143 if (iter != lru_.begin())
144 lru_.splice(lru_.begin(), lru_, iter, std::next(iter));
145 }
146 // Fresh in front, stale on back.
147 int capacity_;
148 std::list<Item> lru_;
149 using ListIter = typename std::list<Item>::iterator;
150 std::unordered_map<K, ListIter> lookup_;
151
152 struct PairHash {
153 std::size_t operator()(const std::pair<K, V*>& p) const {
154 return HashCat({std::hash<V*>{}(p.second), std::hash<K>{}(p.first)});
155 }
156 };
157 std::unordered_map<std::pair<K, V*>, Item, PairHash> pending_pins_;
158 std::mutex mutex_;
159};

161template <class K, class V>
162class LruCacheLock {
163 public:
164 // Looks up the value in @cache by @key and pins it if found.
165 LruCacheLock(LruCache<K, V>* cache, K key)
166 : cache_(cache), key_(key), value_(cache->Lookup(key_)) {}
167
168 // Unpins the cache entry (if holds).
169 ~LruCacheLock() {
170 if (value_) cache_->Unpin(key_, value_);
171 }
172
173 // Returns whether lock holds any value.
174 operator bool() const { return value_; }
175
176 // Gets the value.
177 V* operator->() const { return value_; }
178 V* operator*() const { return value_; }
179
180 LruCacheLock() {}
181 LruCacheLock(LruCacheLock&& other)
182 : cache_(other.cache_), key_(other.key_), value_(other.value_) {
183 other.value_ = nullptr;
184 }
185 void operator=(LruCacheLock&& other) {
186 cache_ = other.cache_;
187 key_ = other.key_;
188 value_ = other.value_;
189 other.value_ = nullptr;
190 }
191
192 private:
193 LruCache<K, V>* cache_ = nullptr;
194 K key_;
195 V* value_ = nullptr;
196};
