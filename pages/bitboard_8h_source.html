40// Represents a board as an array of 64 bits.
41// Bit enumeration goes from bottom to top, from left to right:
42// Square a1 is bit 0, square h1 is bit 7, square a2 is bit 8.
43class BitBoard {
44 public:
45 constexpr BitBoard(std::uint64_t board) : board_(board) {}
46 BitBoard() = default;
47 constexpr static BitBoard FromSquare(Square square) {
48 return BitBoard(1ULL << square.as_idx());
49 }
50
51 std::uint64_t as_int() const { return board_; }
52 void clear() { board_ = 0; }
53
54 // Counts the number of set bits in the BitBoard.
55 int count() const {
56#if defined(NO_POPCNT)
57 std::uint64_t x = board_;
58 x -= (x >> 1) & 0x5555555555555555;
59 x = (x & 0x3333333333333333) + ((x >> 2) & 0x3333333333333333);
60 x = (x + (x >> 4)) & 0x0F0F0F0F0F0F0F0F;
61 return (x * 0x0101010101010101) >> 56;
62#elif defined(_MSC_VER) && defined(_WIN64)
63 return _mm_popcnt_u64(board_);
64#elif defined(_MSC_VER)
65 return __popcnt(board_) + __popcnt(board_ >> 32);
66#else
67 return __builtin_popcountll(board_);
68#endif
69 }
70
71 // Like count() but using algorithm faster on a very sparse BitBoard.
72 // May be slower for more than 4 set bits, but still correct.
73 // Useful when counting bits in a Q, R, N or B BitBoard.
74 int count_few() const {
75#if defined(NO_POPCNT)
76 std::uint64_t x = board_;
77 int count;
78 for (count = 0; x != 0; ++count) {
79 // Clear the rightmost set bit.
80 x &= x - 1;
81 }
82 return count;
83#else
84 return count();
85#endif
86 }
87
88 // Sets the value for given square to 1 if cond is true.
89 // Otherwise does nothing (doesn't reset!).
90 void set_if(Square square, bool cond) {
91 board_ |= (static_cast<uint64_t>(cond) << square.as_idx());
92 }
93 // Sets value of given square to 1.
94 void set(Square square) { board_ |= (1ULL << square.as_idx()); }
95 // Sets value of given square to 0.
96 void reset(Square square) { board_ &= ~(1ULL << square.as_idx()); }
97 // Gets value of a square.
98 bool get(Square square) const { return board_ & (1ULL << square.as_idx()); }
99
100 // Returns whether all bits of a board are set to 0.
101 bool empty() const { return board_ == 0; }
102
103 // Checks whether two bitboards have common bits set.
104 bool intersects(const BitBoard& other) const { return board_ & other.board_; }
105
106 // Flips black and white side of a board.
107 void Mirror() { board_ = ReverseBytesInBytes(board_); }
108
109 bool operator==(const BitBoard& other) const = default;
110 bool operator!=(const BitBoard& other) const = default;
111
112 struct Uin64ToSquare {
113 constexpr Square operator()(uint64_t x) { return Square::FromIdx(x); }
114 };
115 using Iterator = BitIterator<Square, Uin64ToSquare>;
116 Iterator begin() const { return board_; }
117 Iterator end() const { return 0; }
118
119 std::string DebugString() const {
120 std::string res;
121 for (int i = 7; i >= 0; --i) {
122 for (int j = 0; j < 8; ++j)
123 res += get({File::FromIdx(i), Rank::FromIdx(j)}) ? '#' : '.';
124 res += '\n';
125 }
126 return res;
127 }
128
129 // Applies a mask to the bitboard (intersects).
130 BitBoard& operator&=(const BitBoard& a) {
131 board_ &= a.board_;
132 return *this;
133 }
134
135 friend void swap(BitBoard& a, BitBoard& b) {
136 using std::swap;
137 swap(a.board_, b.board_);
138 }
139
140 // Returns union (bitwise OR) of two boards.
141 friend BitBoard operator|(const BitBoard& a, const BitBoard& b) {
142 return {a.board_ | b.board_};
143 }
144
145 // Returns intersection (bitwise AND) of two boards.
146 friend BitBoard operator&(const BitBoard& a, const BitBoard& b) {
147 return {a.board_ & b.board_};
148 }
149
150 // Returns bitboard with one bit reset.
151 friend BitBoard operator-(const BitBoard& a, const Square& b) {
152 return {a.board_ & ~(1ULL << b.as_idx())};
153 }
154
155 // Returns difference (bitwise AND-NOT) of two boards.
156 friend BitBoard operator-(const BitBoard& a, const BitBoard& b) {
157 return {a.board_ & ~b.board_};
158 }
159
160 private:
161 std::uint64_t board_ = 0;
162};
