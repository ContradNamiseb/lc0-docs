46class SearchBase {
47 public:
48 SearchBase(UciResponder* responder) : uci_responder_(responder) {}
49 virtual ~SearchBase() = default;
50
51 // Sets objects needed by the search.
52 // They are guarnteed to be set before any other function is called, and after
53 // that, only can be changed while the search is stopped.
54 virtual void SetBackend(Backend* backend) { backend_ = backend; }
55 virtual void SetSyzygyTablebase(SyzygyTablebase* tb) { syzygy_tb_ = tb; }
56
57 // Resets search tree, and whatever else is needed to start a new game.
58 virtual void NewGame() {}
59 // Sets the position to search from in the future searches.
60 virtual void SetPosition(const GameState&) = 0;
61 // Start the search. Must not block, should return immediately.
62 virtual void StartSearch(const GoParams&) = 0;
63 // Starts the timer for the search. Must not block, should return immediately.
64 // It can be called either after or befor StartSearch(), particularly:
65 // - In the "strict timing" mode, it's called before SetPosition().
66 // - In normal mode, it's called before StartSearch().
67 // - In Ponder mode, it may potentially be called at `ponderhit` (although
68 // actually we'll stop the search, change the position and start again).
69 virtual void StartClock() = 0;
70 // Wait for the search to finish. This is blocking.
71 virtual void WaitSearch() = 0;
72 // Stops the search as soon as possible and responds with bestmove. Doesn't
73 // block.
74 virtual void StopSearch() = 0;
75 // Same as Stop(), but doesn't respond with bestmove. Doesn't block.
76 virtual void AbortSearch() = 0;
77 // Return the data needed to build a training data frame from the last search.
78 virtual SearchArtifacts GetArtifacts() const {
79 throw Exception(
80 "Training data generation is not supported for this search algorithm.");
81 }
82
83 protected:
84 UciResponder* uci_responder_ = nullptr;
85 Backend* backend_ = nullptr;
86 SyzygyTablebase* syzygy_tb_ = nullptr;
87};
