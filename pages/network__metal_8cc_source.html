63std::string activationString(pblczero::NetworkFormat::ActivationFunction act) {
64 switch (act) {
65 case pblczero::NetworkFormat::ACTIVATION_RELU:
66 return "relu";
67 case pblczero::NetworkFormat::ACTIVATION_MISH:
68 return "mish";
69 case pblczero::NetworkFormat::ACTIVATION_NONE:
70 return "none";
71 case pblczero::NetworkFormat::ACTIVATION_TANH:
72 return "tanh";
73 case pblczero::NetworkFormat::ACTIVATION_SIGMOID:
74 return "sigmoid";
75 case pblczero::NetworkFormat::ACTIVATION_SELU:
76 return "selu";
77 case pblczero::NetworkFormat::ACTIVATION_SWISH:
78 return "swish";
79 case pblczero::NetworkFormat::ACTIVATION_RELU_2:
80 return "relu_2";
81 case pblczero::NetworkFormat::ACTIVATION_SOFTMAX:
82 return "softmax";
83 default:
84 return "";
85 }

88MetalNetwork::MetalNetwork(const WeightsFile& file, const OptionsDict& options)
89 : capabilities_{file.format().network_format().input(),
90 file.format().network_format().output(),
91 file.format().network_format().moves_left()} {
92 MultiHeadWeights weights(file.weights());
93
94 try {
95 const int gpu_id = options.GetOrDefault<int>("gpu", 0);
96 builder_ = std::make_unique<MetalNetworkBuilder>();
97 std::string device = builder_->init(gpu_id);
98 CERR << "Initialized metal backend on device " << device;
99 } catch (...) {
100 throw Exception("There was an error initializing the GPU device.");
101 }
102
103 max_batch_size_ = options.GetOrDefault<int>("max_batch", 1024);
104 batch_size_ = options.GetOrDefault<int>("batch", 64);
105
106 conv_policy_ = file.format().network_format().policy() ==
107 pblczero::NetworkFormat::POLICY_CONVOLUTION;
108
109 attn_policy_ = file.format().network_format().policy() ==
110 pblczero::NetworkFormat::POLICY_ATTENTION;
111
112 wdl_ = file.format().network_format().value() ==
113 pblczero::NetworkFormat::VALUE_WDL;
114
115 moves_left_ = (file.format().network_format().moves_left() ==
116 pblczero::NetworkFormat::MOVES_LEFT_V1) &&
117 options.GetOrDefault<bool>("mlh", true);
118
119 bool attn_body =
120 (file.format().network_format().network()) ==
121 pblczero::NetworkFormat::NETWORK_ATTENTIONBODY_WITH_HEADFORMAT ||
122 (file.format().network_format().network()) ==
123 pblczero::NetworkFormat::NETWORK_ATTENTIONBODY_WITH_MULTIHEADFORMAT;
124
125 // Build MPS Graph.
126 Activations activations;
127 activations.default_activation =
128 file.format().network_format().default_activation() ==
129 pblczero::NetworkFormat::DEFAULT_ACTIVATION_MISH
130 ? "mish"
131 : "relu";
132 const auto smolgen_activation =
133 file.format().network_format().smolgen_activation();
134 activations.smolgen_activation =
135 smolgen_activation == pblczero::NetworkFormat::ACTIVATION_DEFAULT
136 ? activations.default_activation
137 : activationString(
138 static_cast<pblczero::NetworkFormat::ActivationFunction>(
139 smolgen_activation));
140 const auto ffn_activation = file.format().network_format().ffn_activation();
141 activations.ffn_activation =
142 ffn_activation == pblczero::NetworkFormat::ACTIVATION_DEFAULT
143 ? activations.default_activation
144 : activationString(
145 static_cast<pblczero::NetworkFormat::ActivationFunction>(
146 ffn_activation));
147
148 std::string policy_head =
149 options.GetOrDefault<std::string>("policy_head", "vanilla");
150 // Check that selected policy head exists.
151 if (weights.policy_heads.count(policy_head) == 0) {
152 throw Exception("The policy head you specified '" + policy_head +
153 "' does not exist in this net.");
154 }
155 std::string value_head =
156 options.GetOrDefault<std::string>("value_head", "winner");
157 // Check that selected value head exists.
158 if (weights.value_heads.count(value_head) == 0) {
159 throw Exception("The value head you specified '" + value_head +
160 "' does not exist in this net.");
161 }
162
163 auto embedding = static_cast<InputEmbedding>(
164 file.format().network_format().input_embedding());
165 builder_->build(kInputPlanes, weights, embedding, attn_body, attn_policy_,
166 conv_policy_, wdl_, moves_left_, activations, policy_head,
167 value_head);
168}

170void MetalNetwork::forwardEval(InputsOutputs* io, int batchSize) {
171 // Metal is not thread-safe, so lock is needed.
172 lock_.lock();
173
174 if (moves_left_) {
175 builder_->forwardEval(&io->input_val_mem_[0], &io->input_masks_mem_[0],
176 batchSize,
177 {&io->op_policy_mem_[0], &io->op_value_mem_[0],
178 &io->op_moves_left_mem_[0]});
179 } else {
180 builder_->forwardEval(&io->input_val_mem_[0], &io->input_masks_mem_[0],
181 batchSize,
182 {&io->op_policy_mem_[0], &io->op_value_mem_[0]});
183 }
184
185 // The next thread can start using the GPU now.
186 lock_.unlock();
187}

189std::unique_ptr<Network> MakeMetalNetwork(const std::optional<WeightsFile>& w,
190 const OptionsDict& options) {
191 if (!w) {
192 throw Exception("The Metal backend requires a network file.");
193 }
194 const WeightsFile& weights = *w;
195 auto nf = weights.format().network_format();
196 using NF = pblczero::NetworkFormat;
197 switch (nf.network()) {
198 case NF::NETWORK_CLASSICAL_WITH_HEADFORMAT:
199 case NF::NETWORK_SE_WITH_HEADFORMAT:
200 case NF::NETWORK_ATTENTIONBODY_WITH_HEADFORMAT:
201 case NF::NETWORK_ATTENTIONBODY_WITH_MULTIHEADFORMAT:
202 break;
203 default:
204 throw Exception("Network format " +
205 NF::NetworkStructure_Name(nf.network()) +
206 " is not supported by the Metal backend.");
207 }
208 switch (nf.policy()) {
209 case NF::POLICY_CLASSICAL:
210 case NF::POLICY_CONVOLUTION:
211 case NF::POLICY_ATTENTION:
212 break;
213 default:
214 throw Exception("Policy format " + NF::PolicyFormat_Name(nf.policy()) +
215 " is not supported by the Metal backend.");
216 }
217 switch (nf.value()) {
218 case NF::VALUE_CLASSICAL:
219 case NF::VALUE_WDL:
220 break;
221 default:
222 throw Exception("Value format " + NF::ValueFormat_Name(nf.value()) +
223 " is not supported by the Metal backend.");
224 }
225 switch (nf.moves_left()) {
226 case NF::MOVES_LEFT_NONE:
227 case NF::MOVES_LEFT_V1:
228 break;
229 default:
230 throw Exception("Moves left head format " +
231 NF::MovesLeftFormat_Name(nf.moves_left()) +
232 " is not supported by the Metal backend.");
233 }
234 switch (nf.default_activation()) {
235 case NF::DEFAULT_ACTIVATION_RELU:
236 case NF::DEFAULT_ACTIVATION_MISH:
237 break;
238 default:
239 throw Exception("Default activation " +
240 NF::DefaultActivation_Name(nf.default_activation()) +
241 " is not supported by the Metal backend.");
242 }
243 switch (nf.input_embedding()) {
244 case NF::INPUT_EMBEDDING_NONE:
245 case NF::INPUT_EMBEDDING_PE_MAP:
246 case NF::INPUT_EMBEDDING_PE_DENSE:
247 break;
248 default:
249 throw Exception("Input embedding " +
250 NF::InputEmbeddingFormat_Name(nf.input_embedding()) +
251 " is not supported by the Metal backend.");
252 }
253 return std::make_unique<MetalNetwork>(weights, options);
254}
