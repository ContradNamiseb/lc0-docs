210int count_pieces(const ChessBoard& pos, int type, bool theirs) {
211 const BitBoard all = theirs ? pos.theirs() : pos.ours();
212 switch (type) {
213 case KING:
214 return 1;
215 case QUEEN:
216 return (all & pos.queens()).count_few();
217 case ROOK:
218 return (all & pos.rooks()).count_few();
219 case BISHOP:
220 return (all & pos.bishops()).count_few();
221 case KNIGHT:
222 return (all & pos.knights()).count_few();
223 case PAWN:
224 return (all & pos.pawns()).count_few();
225 default:
226 assert(false);
227 }
228 return 0;
229}

252bool is_capture(const ChessBoard& pos, const Move& move) {
253 // Simple capture.
254 if (pos.theirs().get(move.to())) return true;
255 // Enpassant capture. Pawn moves other than straight it must be a capture.
256 if (pos.pawns().get(move.from()) && move.from().file() != move.to().file()) {
257 return true;
258 }
259 return false;
260}

267void prt_str(const ChessBoard& pos, char* str, bool flip) {
268 const bool first_theirs = flip ^ pos.flipped();

271 for (int i = count_pieces(pos, pt, first_theirs); i > 0; i--) {
272 *str++ = kPieceToChar[pt];

279 }
280 }
281 *str++ = 0;
282}

416void init_indices() {
417 // Binomial[k][n] = Bin(n, k)
418 for (int i = 0; i < 7; i++)
419 for (int j = 0; j < 64; j++) {
420 size_t f = 1;
421 size_t l = 1;
422 for (int k = 0; k < i; k++) {
423 f *= (j - k);
424 l *= (k + 1);
425 }
426 Binomial[i][j] = f / l;
427 }
428
429 for (int i = 0; i < 6; i++) {
430 size_t s = 0;
431 for (int j = 0; j < 24; j++) {
432 PawnIdx[0][i][j] = s;
433 s += Binomial[i][kPawnTwist[0][(1 + (j % 6)) * 8 + (j / 6)]];
434 if ((j + 1) % 6 == 0) {
435 PawnFactorFile[i][j / 6] = s;
436 s = 0;
437 }
438 }
439 }

470// Produce a 64-bit material key corresponding to the material combination
471// defined by pcs[16], where pcs[1], ..., pcs[6] are the number of white
472// pawns, ..., kings and pcs[9], ..., pcs[14] are the number of black
473// pawns, ..., kings.
474Key calc_key_from_pcs(int* pcs, bool flip) {
475 Key key = 0;
476
477 const int color = !flip ? 0 : 8;
478 for (int i = W_PAWN; i <= B_KING; i++) key += MaterialHash[i][pcs[i ^ color]];
479
480 return key;
481}

483// Produce a 64-bit material key corresponding to the material combination
484// piece[0], ..., piece[num - 1], where each value corresponds to a piece
485// (1-6 for white pawn-king, 9-14 for black pawn-king).
486Key calc_key_from_pieces(uint8_t* piece, int num) {
487 Key key = 0;
488
489 for (int i = 0; i < num; i++) {
490 if (piece[i]) key += MaterialHash[piece[i]][1];
491 }
492
493 return key;

515size_t encode(int* p, EncInfo* ei, BaseEntry* be, const int enc) {
516 const int n = be->num;
517 size_t idx;
518 int k;
519
520 if (p[0] & 0x04) {
521 for (int i = 0; i < n; i++) p[i] ^= 0x07;
522 }
523
524 if (enc == PIECE_ENC) {
525 if (p[0] & 0x20) {
526 for (int i = 0; i < n; i++) p[i] ^= 0x38;
527 }
528
529 for (int i = 0; i < n; i++) {
530 if (kOffDiag[p[i]]) {
531 if (kOffDiag[p[i]] > 0 && i < (be->kk_enc ? 2 : 3)) {
532 for (int j = 0; j < n; j++) p[j] = kFlipDiag[p[j]];
533 }
534 break;
535 }
536 }
537
538 if (be->kk_enc) {
539 idx = kKKIdx[kTriangle[p[0]]][p[1]];
540 k = 2;
541 } else {
542 const int s1 = (p[1] > p[0]);
543 const int s2 = (p[2] > p[0]) + (p[2] > p[1]);
544
545 if (kOffDiag[p[0]]) {
546 idx = kTriangle[p[0]] * 63 * 62 + (p[1] - s1) * 62 + (p[2] - s2);
547 } else if (kOffDiag[p[1]]) {
548 idx =
549 6 * 63 * 62 + kDiag[p[0]] * 28 * 62 + kLower[p[1]] * 62 + p[2] - s2;
550 } else if (kOffDiag[p[2]]) {
551 idx = 6 * 63 * 62 + 4 * 28 * 62 + kDiag[p[0]] * 7 * 28 +
552 (kDiag[p[1]] - s1) * 28 + kLower[p[2]];
553 } else {
554 idx = 6 * 63 * 62 + 4 * 28 * 62 + 4 * 7 * 28 + kDiag[p[0]] * 7 * 6 +
555 (kDiag[p[1]] - s1) * 6 + (kDiag[p[2]] - s2);
556 }
557 k = 3;
558 }
559 idx *= ei->factor[0];
560 } else {
561 for (int i = 1; i < be->pawns[0]; i++) {
562 for (int j = i + 1; j < be->pawns[0]; j++) {
563 if (kPawnTwist[enc - 1][p[i]] < kPawnTwist[enc - 1][p[j]]) {
564 Swap(p[i], p[j]);
565 }
566 }
567 }
568 k = be->pawns[0];
569 idx = PawnIdx[enc - 1][k - 1][kFlap[enc - 1][p[0]]];
570 for (int i = 1; i < k; i++) {
571 idx += Binomial[k - i][kPawnTwist[enc - 1][p[i]]];
572 }
573 idx *= ei->factor[0];
574
575 // Pawns of other color
576 if (be->pawns[1]) {
577 const int t = k + be->pawns[1];
578 for (int i = k; i < t; i++) {
579 for (int j = i + 1; j < t; j++) {
580 if (p[i] > p[j]) Swap(p[i], p[j]);
581 }
582 }
583 size_t s = 0;
584 for (int i = k; i < t; i++) {
585 const int sq = p[i];
586 int skips = 0;
587 for (int j = 0; j < k; j++) skips += (sq > p[j]);
588 s += Binomial[i - k + 1][sq - skips - 8];
589 }
590 idx += s * ei->factor[k];
591 k = t;
592 }
593 }
594
595 for (; k < n;) {
596 const int t = k + ei->norm[k];
597 for (int i = k; i < t; i++) {
598 for (int j = i + 1; j < t; j++) {
599 if (p[i] > p[j]) Swap(p[i], p[j]);
600 }
601 }
602 size_t s = 0;
603 for (int i = k; i < t; i++) {
604 const int sq = p[i];
605 int skips = 0;
606 for (int j = 0; j < k; j++) skips += (sq > p[j]);
607 s += Binomial[i - k + 1][sq - skips];
608 }
609 idx += s * ei->factor[k];
610 k = t;
611 }
612
613 return idx;
614}

640size_t init_enc_info(EncInfo* ei, BaseEntry* be, uint8_t* tb, int shift, int t,
641 const int enc) {
642 const bool more_pawns = enc != PIECE_ENC && be->pawns[1] > 0;
643
644 for (int i = 0; i < be->num; i++) {
645 ei->pieces[i] = (tb[i + 1 + more_pawns] >> shift) & 0x0f;
646 ei->norm[i] = 0;
647 }
648
649 const int order = (tb[0] >> shift) & 0x0f;
650 const int order2 = more_pawns ? (tb[1] >> shift) & 0x0f : 0x0f;
651
652 int k = ei->norm[0] = enc != PIECE_ENC ? be->pawns[0] : be->kk_enc ? 2 : 3;
653
654 if (more_pawns) {
655 ei->norm[k] = be->pawns[1];
656 k += ei->norm[k];
657 }
658
659 for (int i = k; i < be->num; i += ei->norm[i]) {
660 for (int j = i; j < be->num && ei->pieces[j] == ei->pieces[i]; j++) {
661 ei->norm[i]++;
662 }
663 }
664
665 int n = 64 - k;
666 size_t f = 1;
667
668 for (int i = 0; k < be->num || i == order || i == order2; i++) {
669 if (i == order) {
670 ei->factor[0] = f;
671 f *= enc == FILE_ENC ? PawnFactorFile[ei->norm[0] - 1][t]
672 : enc == RANK_ENC ? PawnFactorRank[ei->norm[0] - 1][t]
673 : be->kk_enc ? 462
674 : 31332;
675 } else if (i == order2) {
676 ei->factor[ei->norm[0]] = f;
677 f *= subfactor(ei->norm[ei->norm[0]], 48 - ei->norm[0]);
678 } else {
679 ei->factor[k] = f;
680 f *= subfactor(ei->norm[k], n);
681 n -= ei->norm[k];
682 k += ei->norm[k];
683 }
684 }
685
686 return f;
687}

689void calc_symLen(PairsData* d, uint32_t s, char* tmp) {
690 uint8_t* w = d->symPat + 3 * s;
691 const uint32_t s2 = (w[2] << 4) | (w[1] >> 4);
692 if (s2 == 0x0fff)
693 d->symLen[s] = 0;
694 else {

696 if (!tmp[s1]) calc_symLen(d, s1, tmp);
697 if (!tmp[s2]) calc_symLen(d, s2, tmp);
698 d->symLen[s] = d->symLen[s1] + d->symLen[s2] + 1;
699 }
700 tmp[s] = 1;
701}

712template <typename T, int Half = sizeof(T) / 2, int End = sizeof(T) - 1>
713T swap_endian(T val) {
714 static_assert(std::is_unsigned<T>::value,
715 "Argument of swap_endian not unsigned");
716 T x = val;
717 uint8_t tmp, *c = (uint8_t*)&x;
718 for (int i = 0; i < Half; ++i) {
719 tmp = c[i], c[i] = c[End - i], c[End - i] = tmp;
720 }
721 return x;
722}

748PairsData* setup_pairs(uint8_t** ptr, size_t tb_size, size_t* size,
749 uint8_t* flags, int type) {
750 PairsData* d;
751 uint8_t* data = *ptr;
752
753 *flags = data[0];
754 if (data[0] & 0x80) {
755 d = static_cast<PairsData*>(malloc(sizeof(*d)));
756 d->idxBits = 0;
757 d->constValue[0] = type == WDL ? data[1] : 0;
758 d->constValue[1] = 0;
759 *ptr = data + 2;
760 size[0] = size[1] = size[2] = 0;
761 return d;
762 }
763
764 const uint8_t block_size = data[1];
765 const uint8_t idx_bits = data[2];
766 const uint32_t real_num_blocks = read_le_u32(&data[4]);
767 const uint32_t num_blocks = real_num_blocks + data[3];
768 const int max_len = data[8];
769 const int min_len = data[9];
770 const int h = max_len - min_len + 1;
771 const uint32_t num_syms = read_le_u16(&data[10 + 2 * h]);
772 d = static_cast<PairsData*>(
773 malloc(sizeof(*d) + h * sizeof(uint64_t) + num_syms));
774 d->blockSize = block_size;
775 d->idxBits = idx_bits;
776 d->offset = reinterpret_cast<uint16_t*>(&data[10]);
777 d->symLen = reinterpret_cast<uint8_t*>(d) + sizeof(*d) + h * sizeof(uint64_t);
778 d->symPat = &data[12 + 2 * h];
779 d->minLen = min_len;
780 *ptr = &data[12 + 2 * h + 3 * num_syms + (num_syms & 1)];
781
782 const size_t num_indices = (tb_size + (1ULL << idx_bits) - 1) >> idx_bits;
783 size[0] = 6ULL * num_indices;
784 size[1] = 2ULL * num_blocks;
785 size[2] = static_cast<size_t>(real_num_blocks) << block_size;
786
787 std::vector<char> tmp;
788 tmp.resize(num_syms);
789 memset(tmp.data(), 0, num_syms);
790 for (uint32_t s = 0; s < num_syms; s++) {
791 if (!tmp[s]) calc_symLen(d, s, tmp.data());
792 }
793
794 d->base[h - 1] = 0;
795 for (int i = h - 2; i >= 0; i--) {
796 d->base[i] = (d->base[i + 1] +
797 read_le_u16(reinterpret_cast<uint8_t*>(d->offset + i)) -
798 read_le_u16(reinterpret_cast<uint8_t*>(d->offset + i + 1))) /
799 2;
800 }
801 for (int i = 0; i < h; i++) d->base[i] <<= 64 - (min_len + i);
802
803 d->offset -= d->minLen;
804
805 return d;
806}

808uint8_t* decompress_pairs(PairsData* d, size_t idx) {
809 if (!d->idxBits) return d->constValue;
810
811 const uint32_t main_idx = idx >> d->idxBits;
812 int lit_idx = (idx & ((static_cast<size_t>(1) << d->idxBits) - 1)) -
813 (static_cast<size_t>(1) << (d->idxBits - 1));
814 uint32_t block;
815 memcpy(&block, d->indexTable + 6 * main_idx, sizeof(block));
816 block = from_le_u32(block);
817
818 const uint16_t idx_offset =
819 *reinterpret_cast<uint16_t*>(d->indexTable + 6 * main_idx + 4);
820 lit_idx += from_le_u16(idx_offset);
821
822 if (lit_idx < 0) {
823 while (lit_idx < 0) lit_idx += d->sizeTable[--block] + 1;
824 } else {
825 while (lit_idx > d->sizeTable[block]) lit_idx -= d->sizeTable[block++] + 1;
826 }
827
828 uint32_t* ptr = reinterpret_cast<uint32_t*>(
829 d->data + (static_cast<size_t>(block) << d->blockSize));
830
831 const int m = d->minLen;
832 uint16_t* offset = d->offset;
833 uint64_t* base = d->base - m;
834 uint8_t* sym_len = d->symLen;
835 uint32_t sym;
836 uint32_t bit_cnt = 0; // number of "empty bits" in code
837
838 uint64_t code = from_be_u64(*reinterpret_cast<uint64_t*>(ptr));
839
840 ptr += 2;
841 for (;;) {
842 int l = m;
843 while (code < base[l]) l++;
844 sym = from_le_u16(offset[l]);
845 sym += (code - base[l]) >> (64 - l);
846 if (lit_idx < static_cast<int>(sym_len[sym]) + 1) break;
847 lit_idx -= static_cast<int>(sym_len[sym]) + 1;
848 code <<= l;
849 bit_cnt += l;
850 if (bit_cnt >= 32) {
851 bit_cnt -= 32;
852 const uint32_t tmp = from_be_u32(*ptr++);
853 code |= static_cast<uint64_t>(tmp) << bit_cnt;
854 }
855 }
856
857 uint8_t* symPat = d->symPat;
858 while (sym_len[sym] != 0) {
859 uint8_t* w = symPat + (3 * sym);
860 const int s1 = ((w[1] & 0xf) << 8) | w[0];
861 if (lit_idx < static_cast<int>(sym_len[s1]) + 1) {
862 sym = s1;
863 } else {
864 lit_idx -= static_cast<int>(sym_len[s1]) + 1;
865 sym = (w[2] << 4) | (w[1] >> 4);
866 }
867 }
868
869 return &symPat[3 * sym];
870}

872// p[i] is to contain the square 0-63 (A1-H8) for a piece of type
873// pc[i] ^ flip, where 1 = white pawn, ..., 14 = black king and pc ^ flip
874// flips between white and black if flip == true.
875// Pieces of the same type are guaranteed to be consecutive.
876int fill_squares(const ChessBoard& pos, uint8_t* pc, bool flip, int mirror,
877 int* p, int i) {
878 // if pos.flipped the board is already mirrored, so mirror it again.
879 if (pos.flipped()) mirror ^= 0x38;
880 BitBoard bb = pieces(pos, pc[i] & 7,
881 static_cast<bool>((pc[i] >> 3)) ^ flip ^ pos.flipped());
882 for (auto sq : bb) {
883 p[i++] = sq.as_idx() ^ mirror;
884 }
885 return i;
886}

1217 bool init_table(BaseEntry* be, const char* str, int type) {
1218 uint8_t* data =
1219 static_cast<uint8_t*>(map_tb(str, kSuffix[type], &be->mapping[type]));
1220 if (!data) return false;
1221
1222 if (read_le_u32(data) != kMagic[type]) {
1223 fprintf(stderr, "Corrupted table.\n");
1224 unmap_file(data, be->mapping[type]);
1225 return false;
1226 }
1227
1228 be->data[type] = data;
1229
1230 const bool split = type != DTZ && (data[4] & 0x01);
1231 if (type == DTM) be->dtmLossOnly = data[4] & 0x04;
1232
1233 data += 5;
1234
1235 size_t tb_size[6][2];
1236 const int num = num_tables(be, type);
1237 EncInfo* ei = first_ei(be, type);
1238 const int enc = !be->hasPawns ? PIECE_ENC
1239 : type != DTM ? FILE_ENC
1240 : RANK_ENC;
1241
1242 for (int t = 0; t < num; t++) {
1243 tb_size[t][0] = init_enc_info(&ei[t], be, data, 0, t, enc);
1244 if (split) {
1245 tb_size[t][1] = init_enc_info(&ei[num + t], be, data, 4, t, enc);
1246 }
1247 data += be->num + 1 + (be->hasPawns && be->pawns[1]);
1248 }
1249 data += (uintptr_t)data & 1;
1250
1251 size_t size[6][2][3];
1252 for (int t = 0; t < num; t++) {
1253 uint8_t flags;
1254 ei[t].precomp =
1255 setup_pairs(&data, tb_size[t][0], size[t][0], &flags, type);
1256 if (type == DTZ) {
1257 if (!be->hasPawns) {
1258 PIECE(be)->dtzFlags = flags;
1259 } else {
1260 PAWN(be)->dtzFlags[t] = flags;
1261 }
1262 }
1263 if (split) {
1264 ei[num + t].precomp =
1265 setup_pairs(&data, tb_size[t][1], size[t][1], &flags, type);
1266 } else if (type != DTZ) {
1267 ei[num + t].precomp = NULL;
1268 }
1269 }
1270
1271 if (type == DTM && !be->dtmLossOnly) {
1272 uint16_t* map = reinterpret_cast<uint16_t*>(data);
1273 *(be->hasPawns ? &PAWN(be)->dtmMap : &PIECE(be)->dtmMap) = map;
1274 uint16_t (*mapIdx)[2][2] =
1275 be->hasPawns ? &PAWN(be)->dtmMapIdx[0] : &PIECE(be)->dtmMapIdx;
1276 for (int t = 0; t < num; t++) {
1277 for (int i = 0; i < 2; i++) {
1278 mapIdx[t][0][i] = reinterpret_cast<uint16_t*>(data) + 1 - map;
1279 data += 2 + 2 * read_le_u16(data);
1280 }
1281 if (split) {
1282 for (int i = 0; i < 2; i++) {
1283 mapIdx[t][1][i] = reinterpret_cast<uint16_t*>(data) + 1 - map;
1284 data += 2 + 2 * read_le_u16(data);
1285 }
1286 }
1287 }
1288 }
1289
1290 if (type == DTZ) {
1291 void* map = data;
1292 *(be->hasPawns ? &PAWN(be)->dtzMap : &PIECE(be)->dtzMap) = map;
1293 uint16_t (*mapIdx)[4] =
1294 be->hasPawns ? &PAWN(be)->dtzMapIdx[0] : &PIECE(be)->dtzMapIdx;
1295 uint8_t* flags =
1296 be->hasPawns ? &PAWN(be)->dtzFlags[0] : &PIECE(be)->dtzFlags;
1297 for (int t = 0; t < num; t++) {
1298 if (flags[t] & 2) {
1299 if (!(flags[t] & 16)) {
1300 for (int i = 0; i < 4; i++) {
1301 mapIdx[t][i] = data + 1 - static_cast<uint8_t*>(map);
1302 data += 1 + data[0];
1303 }
1304 } else {
1305 data += reinterpret_cast<uintptr_t>(data) & 0x01;
1306 for (int i = 0; i < 4; i++) {
1307 mapIdx[t][i] = reinterpret_cast<uint16_t*>(data) + 1 -
1308 static_cast<uint16_t*>(map);
1309 data += 2 + 2 * read_le_u16(data);
1310 }
1311 }
1312 }
1313 }
1314 data += reinterpret_cast<uintptr_t>(data) & 0x01;
1315 }
1316
1317 for (int t = 0; t < num; t++) {
1318 ei[t].precomp->indexTable = data;
1319 data += size[t][0][0];
1320 if (split) {
1321 ei[num + t].precomp->indexTable = data;
1322 data += size[t][1][0];
1323 }
1324 }
1325
1326 for (int t = 0; t < num; t++) {
1327 ei[t].precomp->sizeTable = reinterpret_cast<uint16_t*>(data);
1328 data += size[t][0][1];
1329 if (split) {
1330 ei[num + t].precomp->sizeTable = reinterpret_cast<uint16_t*>(data);
1331 data += size[t][1][1];
1332 }
1333 }
1334
1335 for (int t = 0; t < num; t++) {
1336 data = reinterpret_cast<uint8_t*>(
1337 (reinterpret_cast<uintptr_t>(data) + 0x3f) & ~0x3f);
1338 ei[t].precomp->data = data;
1339 data += size[t][0][2];
1340 if (split) {
1341 data = reinterpret_cast<uint8_t*>(
1342 (reinterpret_cast<uintptr_t>(data) + 0x3f) & ~0x3f);
1343 ei[num + t].precomp->data = data;
1344 data += size[t][1][2];
1345 }
1346 }

1348 if (type == DTM && be->hasPawns) {
1349 PAWN(be)->dtmSwitched =

1351 }

1356 int probe_table(const ChessBoard& pos, int s, int* success, const int type) {
1357 // Obtain the position's material-signature key
1358 const Key key = calc_key_from_position(pos);
1359
1360 // Test for KvK
1361 if (type == WDL && (pos.ours() | pos.theirs()) == pos.kings()) {
1362 return 0;
1363 }
1364
1365 int hash_idx = key >> (64 - TB_HASHBITS);
1366 while (tb_hash_[hash_idx].key && tb_hash_[hash_idx].key != key) {
1367 hash_idx = (hash_idx + 1) & ((1 << TB_HASHBITS) - 1);
1368 }
1369 if (!tb_hash_[hash_idx].ptr) {
1370 *success = 0;
1371 return 0;
1372 }
1373
1374 BaseEntry* be = tb_hash_[hash_idx].ptr;
1375 if ((type == DTM && !be->hasDtm) || (type == DTZ && !be->hasDtz)) {
1376 *success = 0;
1377 return 0;
1378 }
1379
1380 // Use double-checked locking to reduce locking overhead
1381 if (!atomic_load_explicit(&be->ready[type], std::memory_order_acquire)) {
1382 Mutex::Lock lock(ready_mutex_);
1383 if (!atomic_load_explicit(&be->ready[type], std::memory_order_relaxed)) {
1384 char str[16];
1385 prt_str(pos, str, be->key != key);
1386 if (!init_table(be, str, type)) {
1387 tb_hash_[hash_idx].ptr = nullptr; // mark as deleted
1388 *success = 0;
1389 return 0;
1390 }
1391 atomic_store_explicit(&be->ready[type], true,
1392 std::memory_order_release);
1393 }
1394 }
1395
1396 bool bside, flip;
1397 if (!be->symmetric) {
1398 flip = key != be->key;
1399 bside = (!pos.flipped()) == flip;
1400 if (type == DTM && be->hasPawns && PAWN(be)->dtmSwitched) {
1401 flip = !flip;
1402 bside = !bside;
1403 }
1404 } else {
1405 flip = pos.flipped();
1406 bside = false;
1407 }
1408
1409 EncInfo* ei = first_ei(be, type);
1410 int p[TB_PIECES];
1411 size_t idx;
1412 int t = 0;
1413 uint8_t flags = 0;
1414
1415 if (!be->hasPawns) {
1416 if (type == DTZ) {
1417 flags = PIECE(be)->dtzFlags;
1418 if ((flags & 1) != bside && !be->symmetric) {
1419 *success = -1;
1420 return 0;
1421 }
1422 }
1423 ei = type != DTZ ? &ei[bside] : ei;
1424 for (int i = 0; i < be->num;) {
1425 i = fill_squares(pos, ei->pieces, flip, 0, p, i);
1426 }
1427 idx = encode_piece(p, ei, be);
1428 } else {
1429 int i = fill_squares(pos, ei->pieces, flip, flip ? 0x38 : 0, p, 0);
1430 t = leading_pawn(p, be, type != DTM ? FILE_ENC : RANK_ENC);
1431 if (type == DTZ) {
1432 flags = PAWN(be)->dtzFlags[t];
1433 if ((flags & 1) != bside && !be->symmetric) {
1434 *success = -1;
1435 return 0;
1436 }
1437 }
1438 ei = type == WDL ? &ei[t + 4 * bside]
1439 : type == DTM ? &ei[t + 6 * bside]
1440 : &ei[t];
1441 while (i < be->num) {
1442 i = fill_squares(pos, ei->pieces, flip, flip ? 0x38 : 0, p, i);
1443 }
1444 idx = type != DTM ? encode_pawn_f(p, ei, be) : encode_pawn_r(p, ei, be);
1445 }
1446
1447 uint8_t* w = decompress_pairs(ei->precomp, idx);
1448
1449 if (type == WDL) return static_cast<int>(w[0]) - 2;
1450
1451 int v = w[0] + ((w[1] & 0x0f) << 8);
1452
1453 if (type == DTM) {
1454 if (!be->dtmLossOnly) {
1455 v = from_le_u16(
1456 be->hasPawns
1457 ? PAWN(be)->dtmMap[PAWN(be)->dtmMapIdx[t][bside][s] + v]
1458 : PIECE(be)->dtmMap[PIECE(be)->dtmMapIdx[bside][s] + v]);
1459 }
1460 } else {
1461 if (flags & 2) {
1462 const int m = kWdlToMap[s + 2];
1463 if (!(flags & 16)) {
1464 v = be->hasPawns
1465 ? static_cast<uint8_t*>(
1466 PAWN(be)->dtzMap)[PAWN(be)->dtzMapIdx[t][m] + v]
1467 : static_cast<uint8_t*>(
1468 PIECE(be)->dtzMap)[PIECE(be)->dtzMapIdx[m] + v];
1469 } else {
1470 v = from_le_u16(
1471 be->hasPawns
1472 ? static_cast<uint16_t*>(
1473 PAWN(be)->dtzMap)[PAWN(be)->dtzMapIdx[t][m] + v]
1474 : static_cast<uint16_t*>(
1475 PIECE(be)->dtzMap)[PIECE(be)->dtzMapIdx[m] + v]);
1476 }
1477 }
1478 if (!(flags & kPAFlags[s + 2]) || (s & 1)) v *= 2;
1479 }
1480
1481 return v;
1482 }

1516// For a position where the side to move has a winning capture it is not
1517// necessary to store a winning value so the generator treats such positions as
1518// "don't cares" and tries to assign to it a value that improves the compression
1519// ratio. Similarly, if the side to move has a drawing capture, then the
1520// position is at least drawn. If the position is won, then the TB needs to
1521// store a win value. But if the position is drawn, the TB may store a loss
1522// value if that is better for compression. All of this means that during
1523// probing, the engine must look at captures and probe their results and must
1524// probe the position itself. The "best" result of these probes is the correct
1525// result for the position. DTZ table don't store values when a following move
1526// is a zeroing winning move (winning capture or winning pawn move). Also DTZ
1527// store wrong values for positions where the best move is an ep-move (even if
1528// losing). So in all these cases set the state to ZEROING_BEST_MOVE.
1529template <bool CheckZeroingMoves>
1530WDLScore SyzygyTablebase::search(const Position& pos, ProbeState* result) {
1531 WDLScore value;
1532 WDLScore best_value = WDL_LOSS;
1533 auto move_list = pos.GetBoard().GenerateLegalMoves();
1534 const size_t total_count = move_list.size();
1535 size_t move_count = 0;
1536 for (const Move& move : move_list) {
1537 if (!is_capture(pos.GetBoard(), move) &&
1538 (!CheckZeroingMoves || !pos.GetBoard().pawns().get(move.from()))) {
1539 continue;
1540 }
1541 move_count++;
1542 auto new_pos = Position(pos, move);
1543 value = static_cast<WDLScore>(-search(new_pos, result));
1544 if (*result == FAIL) return WDL_DRAW;
1545 if (value > best_value) {
1546 best_value = value;
1547 if (value >= WDL_WIN) {
1548 *result = ZEROING_BEST_MOVE; // Winning DTZ-zeroing move
1549 return value;
1550 }
1551 }
1552 }
1553 // In case we have already searched all the legal moves we don't have to probe
1554 // the TB because the stored score could be wrong. For instance TB tables do
1555 // not contain information on position with ep rights, so in this case the
1556 // result of probe_wdl_table is wrong. Also in case of only capture moves, for
1557 // instance here 4K3/4q3/6p1/2k5/6p1/8/8/8 w - - 0 7, we have to return with
1558 // ZEROING_BEST_MOVE set.
1559 const bool no_more_moves = (move_count && move_count == total_count);
1560 if (no_more_moves) {
1561 value = best_value;
1562 } else {
1563 int raw_result = static_cast<int>(ProbeState::OK);
1564 value = static_cast<WDLScore>(
1565 impl_->probe_wdl_table(pos.GetBoard(), &raw_result));
1566 *result = static_cast<ProbeState>(raw_result);
1567 if (*result == FAIL) return WDL_DRAW;
1568 }
1569 // DTZ stores a "don't care" value if bestValue is a win
1570 if (best_value >= value) {
1571 *result = (best_value > WDL_DRAW || no_more_moves ? ZEROING_BEST_MOVE : OK);
1572 return best_value;
1573 }
1574 *result = OK;
1575 return value;
1576}

1578// Probe the WDL table for a particular position.
1579// If *result != FAIL, the probe was successful.
1580// The return value is from the point of view of the side to move:
1581// -2 : loss
1582// -1 : loss, but draw under 50-move rule
1583// 0 : draw
1584// 1 : win, but draw under 50-move rule
1585// 2 : win
1586WDLScore SyzygyTablebase::probe_wdl(const Position& pos, ProbeState* result) {
1587 *result = OK;
1588 return search(pos, result);
1589}

1591// Probe the DTZ table for a particular position.
1592// If *result != FAIL, the probe was successful.
1593// The return value is from the point of view of the side to move:
1594// n < -100 : loss, but draw under 50-move rule
1595// -100 <= n < -1 : loss in n ply (assuming 50-move counter == 0)
1596// -1 : loss, the side to move is mated
1597// 0 : draw
1598// 1 < n <= 100 : win in n ply (assuming 50-move counter == 0)
1599// 100 < n : win, but draw under 50-move rule
1600//
1601// The return value n can be off by 1: a return value -n can mean a loss in n+1
1602// ply and a return value +n can mean a win in n+1 ply. This cannot happen for
1603// tables with positions exactly on the "edge" of the 50-move rule.
1604//
1605// This implies that if dtz > 0 is returned, the position is certainly a win if
1606// dtz + 50-move-counter <= 99. Care must be taken that the engine picks moves
1607// that preserve dtz + 50-move-counter <= 99.
1608//
1609// If n = 100 immediately after a capture or pawn move, then the position is
1610// also certainly a win, and during the whole phase until the next capture or
1611// pawn move, the inequality to be preserved is dtz
1612// + 50-movecounter <= 100.
1613//
1614// In short, if a move is available resulting in dtz + 50-move-counter <= 99,
1615// then do not accept moves leading to dtz + 50-move-counter == 100.
1616int SyzygyTablebase::probe_dtz(const Position& pos, ProbeState* result) {
1617 *result = OK;
1618 const WDLScore wdl = search<true>(pos, result);
1619 if (*result == FAIL || wdl == WDL_DRAW) { // DTZ tables don't store draws
1620 return 0;
1621 }
1622 // DTZ stores a 'don't care' value in this case, or even a plain wrong one as
1623 // in case the best move is a losing ep, so it cannot be probed.
1624 if (*result == ZEROING_BEST_MOVE) return dtz_before_zeroing(wdl);
1625 int raw_result = 1;
1626 int dtz = impl_->probe_dtz_table(pos.GetBoard(), wdl, &raw_result);
1627 *result = static_cast<ProbeState>(raw_result);
1628 if (*result == FAIL) return 0;
1629 if (*result != CHANGE_STM) {
1630 return (dtz + 1 +
1631 100 * (wdl == WDL_BLESSED_LOSS || wdl == WDL_CURSED_WIN)) *
1632 sign_of(wdl);
1633 }
1634 // DTZ stores results for the other side, so we need to do a 1-ply search and
1635 // find the winning move that minimizes DTZ.
1636 int min_DTZ = 0xFFFF;
1637 for (const Move& move : pos.GetBoard().GenerateLegalMoves()) {
1638 Position next_pos = Position(pos, move);
1639 const bool zeroing = next_pos.GetRule50Ply() == 0;
1640 // For zeroing moves we want the dtz of the move _before_ doing it,
1641 // otherwise we will get the dtz of the next move sequence. Search the
1642 // position after the move to get the score sign (because even in a winning
1643 // position we could make a losing capture or going for a draw).
1644 dtz = zeroing ? -dtz_before_zeroing(search(next_pos, result))
1645 : -probe_dtz(next_pos, result);
1646 // If the move mates, force minDTZ to 1
1647 if (dtz == 1 && next_pos.GetBoard().IsUnderCheck() &&
1648 next_pos.GetBoard().GenerateLegalMoves().empty()) {
1649 min_DTZ = 1;
1650 }
1651 // Convert result from 1-ply search. Zeroing moves are already accounted by
1652 // dtz_before_zeroing() that returns the DTZ of the previous move.
1653 if (!zeroing) dtz += sign_of(dtz);
1654 // Skip the draws and if we are winning only pick positive dtz
1655 if (dtz < min_DTZ && sign_of(dtz) == sign_of(wdl)) min_DTZ = dtz;
1656 if (*result == FAIL) return 0;
1657 }
1658 // When there are no legal moves, the position is mate: we return -1
1659 return min_DTZ == 0xFFFF ? -1 : min_DTZ;
1660}

1662// Use the DTZ tables to rank root moves.
1663//
1664// A return value false indicates that not all probes were successful.
1665bool SyzygyTablebase::root_probe(const Position& pos, bool has_repeated,
1666 bool win_only, std::vector<Move>* safe_moves) {
1667 ProbeState result;
1668 auto root_moves = pos.GetBoard().GenerateLegalMoves();
1669 // Obtain 50-move counter for the root position
1670 const int cnt50 = pos.GetRule50Ply();
1671 // Check whether a position was repeated since the last zeroing move.
1672 const bool rep = has_repeated;
1673 int dtz;
1674 std::vector<int> ranks;
1675 ranks.reserve(root_moves.size());
1676 int best_rank = (win_only ? 1 : -1000);
1677 // Probe and rank each move
1678 for (auto& m : root_moves) {
1679 Position next_pos = Position(pos, m);
1680 // Calculate dtz for the current move counting from the root position
1681 if (next_pos.GetRule50Ply() == 0) {
1682 // In case of a zeroing move, dtz is one of -101/-1/0/1/101
1683 const WDLScore wdl = static_cast<WDLScore>(-probe_wdl(next_pos, &result));
1684 dtz = dtz_before_zeroing(wdl);
1685 } else {
1686 // Otherwise, take dtz for the new position and correct by 1 ply
1687 dtz = -probe_dtz(next_pos, &result);
1688 dtz = dtz > 0 ? dtz + 1 : dtz < 0 ? dtz - 1 : dtz;
1689 }
1690 // Make sure that a mating move is assigned a dtz value of 1
1691 if (next_pos.GetBoard().IsUnderCheck() && dtz == 2 &&
1692 next_pos.GetBoard().GenerateLegalMoves().size() == 0) {
1693 dtz = 1;
1694 }
1695 if (result == FAIL) return false;
1696 // Better moves are ranked higher. Certain wins are ranked equally.
1697 // Losing moves are ranked equally unless a 50-move draw is in sight.
1698 int r = dtz > 0 ? (dtz + cnt50 <= 99 && !rep ? 1000 : 1000 - (dtz + cnt50))
1699 : dtz < 0
1700 ? (-dtz * 2 + cnt50 < 100 ? -1000 : -1000 + (-dtz + cnt50))
1701 : 0;
1702 if (r > best_rank) best_rank = r;
1703 ranks.push_back(r);
1704 }
1705 // Disable all but the equal best moves.
1706 int counter = 0;
1707 for (auto& m : root_moves) {
1708 if (ranks[counter] == best_rank) {
1709 safe_moves->push_back(m);
1710 }
1711 counter++;
1712 }
1713 return true;
1714}

1716// Use the WDL tables to rank root moves.
1717// This is a fallback for the case that some or all DTZ tables are missing.
1718//
1719// A return value false indicates that not all probes were successful.
1720bool SyzygyTablebase::root_probe_wdl(const Position& pos,
1721 std::vector<Move>* safe_moves) {
1722 static const int WDL_to_rank[] = {-1000, -899, 0, 899, 1000};
1723 auto root_moves = pos.GetBoard().GenerateLegalMoves();
1724 ProbeState result;
1725 std::vector<int> ranks;
1726 ranks.reserve(root_moves.size());
1727 int best_rank = -1000;
