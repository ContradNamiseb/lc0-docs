32// fp16-specific kernels used by certain layers //
34
35// SE layer implementation using single fused kernel.
36
37// N blocks.
38// C threads per block.
39// 'HWC' input data processed by thread block.
40// Each thread processes 8x8 elements.
41// K is the no. of outputs of first fully connected layer (same as no. of inputs
42// for second fully connected layer).
43// The kernel assumes K <= C.
44
45template <int C, int K>
46/*
47DPCT1110:20: The total declared local variable size in device function
48SE_Layer_NHWC exceeds 128 bytes and may cause high register pressure. Consult
49with your hardware vendor to find the total register size available and adjust
50the code, or use smaller sub-group size to avoid high register pressure.
51*/
52void SE_Layer_NHWC(sycl::half* output, const sycl::half* skip,
53 const sycl::half* input, const sycl::half* w1,
54 const sycl::half* b1, const sycl::half* w2,
55 const sycl::half* b2, const sycl::half* bPrev,
56 ActivationFunction activation,
57 const sycl::nd_item<3>& item_ct1, sycl::half* sharedData) {
58#if DPCT_COMPATIBILITY_TEMP >= 530
59 const int elementsPerThread = 64; // 8x8 board
60 const int se_K = K;
61
62 int n = item_ct1.get_group(2);
63 int c = item_ct1.get_local_id(2);
64
65 sycl::half2 localData[elementsPerThread];
66
67 sycl::half S = 0;
68
69 sycl::half bias = 0;
70 if (bPrev) bias = bPrev[c];
71
72// 1. Global avg (1 avg per thread).
73#pragma unroll
74 for (int i = 0; i < elementsPerThread; i++) {
75 int localIndex = i * C + c;
76 int inputIndex = n * C * elementsPerThread + localIndex;
77 localData[i].x() = input[inputIndex] + bias;
78 localData[i].y() = skip[inputIndex];
79 S += localData[i].x();
80 }
81
82 sycl::half avg = S / (sycl::half)elementsPerThread;
83 sharedData[c] = avg;
84
85 item_ct1.barrier(sycl::access::fence_space::local_space);
86
87 // 2. First fully connected layer.
88 if (c < K) {
89 S = 0;
90
91#pragma unroll
92 for (int i = 0; i < C; i++) {
93 S += sharedData[i] * readw1(i, c);
94 }
95
96 S += b1[c];
97
98 S = activate(S, activation);
99
100 sharedData[c] = S;
101 }
102 item_ct1.barrier(sycl::access::fence_space::local_space);
103
104 // 3. Second fully connected layer.
105 S = 0;
106 sycl::half B = 0;
107#pragma unroll
108 for (int i = 0; i < K; i++) {
109 sycl::half val = sharedData[i];
110 S += val * readw2(i, c);
111 B += val * readw2(i, c + C);
112 }
113 S += b2[c];
114 B += b2[c + C];
115
116 // Sigmoid (only on the scale part).
117 S = (sycl::half)(1.0f / (1.0f + sycl::exp(-(float)(S))));
118
119// 4. Scale, and add skip connection, perform relu, and write to output.
120#pragma unroll
121 for (int i = 0; i < elementsPerThread; i++) {
122 int localIndex = i * C + c;
123 int inputIndex = n * C * elementsPerThread + localIndex;
124 sycl::half val = localData[i].y() + localData[i].x() * S + B;
125
126 // Relu activation function.
127 val = (sycl::half)activate((float)val, activation);
128
129 output[inputIndex] = val;
130 }
131#endif
132}

442// Get board for this thread from shared memory.
443// We are just using shared memory to store local thread data in this kernel to
444// help reduce some register pressure and spills to local memory.
445#define BOARD(y, x) shboard[(y)*8 + (x)]
446
447// input is in transformed space (HWNC layout) --- output of GEMM
448// output is also in transformed space (HWNC layout) --- input to GEMM (for
449// next layer)
450// 'C' threads per block
451// 'N' blocks
452// Every thread generates an entire board/plane (8x8 elements).
453template <ActivationFunction activation, bool use_bias, bool use_skip>
454/*
455DPCT1110:35: The total declared local variable size in device function
456OutputInputTransformKernel_fp16_shmem_board exceeds 128 bytes and may cause high
457register pressure. Consult with your hardware vendor to find the total register
458size available and adjust the code, or use smaller sub-group size to avoid high
459register pressure.
460*/
461void OutputInputTransformKernel_fp16_shmem_board(
462 int N, int C, int se_K, sycl::half* output, const sycl::half* input,
463 sycl::half* skip, const sycl::half* bias, const sycl::half* w1,
464 const sycl::half* b1, const sycl::half* w2, const sycl::half* b2,
465 const sycl::nd_item<3>& item_ct1, uint8_t* dpct_local, float* shared_data,
466 sycl::local_accessor<float, 2> shared_sums) {
467#if DPCT_COMPATIBILITY_TEMP >= 530
468 int k = item_ct1.get_local_id(2);
469 int n = item_ct1.get_group(2);
470
471 auto _sboard = (sycl::half*)dpct_local;
472 sycl::half* shboard = &_sboard[k * 72]; // 72 instead of 64 to reduce shared
473 // memory bank conflicts.
474 sycl::half b = bias[k];
475
476#pragma unroll
477 for (int hStart = 0; hStart < 8; hStart += 4)
478#pragma unroll
479 for (int wStart = 0; wStart < 8; wStart += 4) {
480 // i) read to per thread registers (for doing output transform)
481 int shln = n * 4 + (hStart / 4) * 2 + (wStart / 4);
482 sycl::half outElTransformed[6][6];
483#pragma unroll
484 for (int y = 0; y < 6; y++)
485#pragma unroll
486 for (int x = 0; x < 6; x++)
487 outElTransformed[y][x] = input[TEMP_INDEX_HWNC(y, x, shln, k)];
488
489 // ii) transform it
490 sycl::half outEl[4][4];
491 OutputTransform4x4(&outEl[0][0], &outElTransformed[0][0]);
492
493#pragma unroll
494 for (int y = 0; y < 4; y++)
495 copyAs<sycl::uint2>(&BOARD(hStart + y, wStart), &outEl[y][0]);
496 }
497
498 // Add bias, and compute the average for SE.
499 float S = 0;
500 float B = 0;
501
502#pragma unroll
503 for (int y = 0; y < 8; y++) {
504 sycl::half boardRow[8];
505 copyAs<sycl::uint4>(&boardRow, &BOARD(y, 0));
506#pragma unroll
507 for (int x = 0; x < 8; x++) {
508 if (use_bias) boardRow[x] += b;
509 S += (float)boardRow[x];
510 }
511 if (use_bias) copyAs<sycl::uint4>(&BOARD(y, 0), &boardRow);
512 }
513
514 float avg = S / 64;
515 shared_data[k] = avg;
516
517 int lane = k & 0x1F;
518 int warp = k >> 5;
519 item_ct1.barrier(sycl::access::fence_space::local_space);
520
521 // First fully-connected layer for SE
522
523 // As se_K << C, we want to loop over se_K instead of C
524 // even if it means taking the sum across threads
525
526 // per-warp sums
527
528 for (int i = 0; i < se_K; i++) {
529 float val = shared_data[k] * float(readw1(k, i));
530 val = warpReduce(val, item_ct1);
531 if (lane == 0) shared_sums[warp][i] = val;
532 }
533 item_ct1.barrier(sycl::access::fence_space::local_space);
534 if (k < se_K) {
535 S = 0;
536 for (int i = 0; i < C / 32; i++) S += shared_sums[i][k];
537
538 S += (float)b1[k];
539 S = activate(S, activation);
540 shared_data[k] = S;
541 }
542
543 item_ct1.barrier(sycl::access::fence_space::local_space);
544
545 // Second fully-connected layer for SE
546 S = 0;
547 for (int i = 0; i < se_K; i++) {
548 float val = shared_data[i];
549 S += val * float(readw2(i, k));
550 B += val * float(readw2(i, k + C));
551 }
552 S += (float)b2[k];
553 B += (float)b2[k + C];
554
555 // Sigmoid (only on the scale part).
556 S = 1.0f / (1.0f + sycl::exp(-S));
557
558 // Scale/bias, add skip connection, perform activation, and write to output.
559 for (int h = 0; h < 8; h++) {
560 sycl::half boardRow[8];
561 copyAs<sycl::uint4>(&boardRow[0], &BOARD(h, 0));
562
563#pragma unroll
564 for (int w = 0; w < 8; w++) {
565 boardRow[w] = (sycl::half)(float(boardRow[w]) * S + B);
566 }
567
568 // residual add
569 if (use_skip) {
570 sycl::half skipInp[8];
571 copyAs<sycl::uint4>(&skipInp[0], &skip[INDEX_NHCW(n, k, h, 0)]);
572#pragma unroll
573 for (int w = 0; w < 8; w++) boardRow[w] += skipInp[w];
574 }
575
576 if (activation != ACTIVATION_NONE) {
577#pragma unroll
578 for (int w = 0; w < 8; w++)
579 boardRow[w] = (sycl::half)activate((float)boardRow[w], activation);
580 }
581
582 // write un-transformed output to 'skip' if required
583 if (use_skip) {
584 copyAs<sycl::uint4>(&skip[INDEX_NHCW(n, k, h, 0)], &boardRow[0]);
585 }
586
587 copyAs<sycl::uint4>(&BOARD(h, 0), &boardRow);
588 }
589
590 // Perform input transform.
591
592 int c = k;
593 // top-left
594 {
595 sycl::half inEl[6][6] = {};
596
597#pragma unroll
598 for (int i = 0; i < 5; i++)
599#pragma unroll
