57void ShaderWrapper::Init(ID3D12Device* device) {
58 // Create root signature - common for all shaders.
59
60 // 8+1+8 slots
61 // slot 0 to 7 -> root UAV slots 0 to 7 (all in space 0)
62 // slot 8 -> root constants (16 constants - should be enough)
63 // slot 9 to 16 -> descriptor UAVs of same allocations as slots 0-7, bound
64 // at shader slots 8-15
65
66 D3D12_ROOT_PARAMETER root_parameter[kUavSlots + 1 + kUavSlots];
67 for (int i = 0; i < kUavSlots; i++) {
68 root_parameter[i].ParameterType = D3D12_ROOT_PARAMETER_TYPE_UAV;
69 root_parameter[i].Descriptor.RegisterSpace = 0;
70 root_parameter[i].Descriptor.ShaderRegister = i;
71 root_parameter[i].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
72 }
73
74 root_parameter[kUavSlots].ParameterType =
75 D3D12_ROOT_PARAMETER_TYPE_32BIT_CONSTANTS;
76 root_parameter[kUavSlots].Constants.RegisterSpace = 0;
77 root_parameter[kUavSlots].Constants.ShaderRegister = 0;
78 root_parameter[kUavSlots].Constants.Num32BitValues = 16;
79 root_parameter[kUavSlots].ShaderVisibility = D3D12_SHADER_VISIBILITY_ALL;
80
81 D3D12_DESCRIPTOR_RANGE descRange[kUavSlots] = {};
82 for (int i = 0; i < kUavSlots; i++) {
83 descRange[i].BaseShaderRegister = i + kUavSlots;
84 descRange[i].NumDescriptors = 1;
85 descRange[i].OffsetInDescriptorsFromTableStart = 0;
86 descRange[i].RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_UAV;
87 descRange[i].RegisterSpace = 0;
88
89 root_parameter[i + kUavSlots + 1].ParameterType =
90 D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
91 root_parameter[i + kUavSlots + 1].DescriptorTable.NumDescriptorRanges = 1;
92 root_parameter[i + kUavSlots + 1].DescriptorTable.pDescriptorRanges =
93 &descRange[i];
94 root_parameter[i + kUavSlots + 1].ShaderVisibility =
95 D3D12_SHADER_VISIBILITY_ALL;
96 }
97
98 D3D12_ROOT_SIGNATURE_DESC root_sig_desc = {kUavSlots + 1 + kUavSlots,
99 root_parameter, 0, NULL,
100 D3D12_ROOT_SIGNATURE_FLAG_NONE};
101
102 ID3DBlob* serialized_layout = NULL;
103 D3D12SerializeRootSignature(&root_sig_desc, D3D_ROOT_SIGNATURE_VERSION_1_0,
104 &serialized_layout, NULL);
105
106 ReportDxErrors(device->CreateRootSignature(
107 1, serialized_layout->GetBufferPointer(),
108 serialized_layout->GetBufferSize(), IID_PPV_ARGS(&root_sign_)));
109
110 serialized_layout->Release();
111
112 // Create PSO objects for each shader.
113 // PSO basically holds the compiled shader object.
114
115 // Expand planes shaders.
116 D3D12_COMPUTE_PIPELINE_STATE_DESC state_desc = {};
117 state_desc.pRootSignature = root_sign_;
118
119 state_desc.CS = {g_ExpandPlanes_shader_fp16,
120 sizeof(g_ExpandPlanes_shader_fp16)};
121 ReportDxErrors(device->CreateComputePipelineState(
122 &state_desc, IID_PPV_ARGS(&expand_planes_fp16_)));
123
124 state_desc.CS = {g_ExpandPlanes_shader_fp32,
125 sizeof(g_ExpandPlanes_shader_fp32)};
126 ReportDxErrors(device->CreateComputePipelineState(
127 &state_desc, IID_PPV_ARGS(&expand_planes_fp32_)));
128
129 // Winograd Input Transform shader.
130 state_desc.CS = {g_input_transform_shader_fp32,
131 sizeof(g_input_transform_shader_fp32)};
132 ReportDxErrors(device->CreateComputePipelineState(
133 &state_desc, IID_PPV_ARGS(&winograd_input_transform_)));
134
135 // Winograd Output Transform shader.
136 state_desc.CS = {g_output_transform_shader_fp32,
137 sizeof(g_output_transform_shader_fp32)};
138 ReportDxErrors(device->CreateComputePipelineState(
139 &state_desc, IID_PPV_ARGS(&winograd_output_transform_)));
140
141 // 1x1 convolution shader.
142 state_desc.CS = {g_conv_1x1_shader_fp32, sizeof(g_conv_1x1_shader_fp32)};
143 ReportDxErrors(device->CreateComputePipelineState(
144 &state_desc, IID_PPV_ARGS(&conv_1x1_)));
145
146 // policy map shader.
147 state_desc.CS = {g_policy_map_shader_fp32, sizeof(g_policy_map_shader_fp32)};
148 ReportDxErrors(device->CreateComputePipelineState(
149 &state_desc, IID_PPV_ARGS(&policy_map_)));
150
151 state_desc.CS = {g_MatrixMul_Fp32, sizeof(g_MatrixMul_Fp32)};
152 ReportDxErrors(device->CreateComputePipelineState(&state_desc,
153 IID_PPV_ARGS(&gemm_)));
154
155 // Add vectors shader.
156 state_desc.CS = {g_add_vectors_shader, sizeof(g_add_vectors_shader)};
157 ReportDxErrors(device->CreateComputePipelineState(
158 &state_desc, IID_PPV_ARGS(&add_vectors_)));
159
160 // Various SE shaders.
161 CREATE_SE_PSO(128);
162 CREATE_SE_PSO(256);
163 CREATE_SE_PSO(320);
164 CREATE_SE_PSO(384);
165 CREATE_SE_PSO(512);
166 CREATE_SE_PSO(640);
167 CREATE_SE_PSO(768);
168 CREATE_SE_PSO(1024);
169
170 // Various output-transform fused with SE shaders.
171 CREATE_WINOGRAD_SE_PSO(128)
172 CREATE_WINOGRAD_SE_PSO(256)
173 CREATE_WINOGRAD_SE_PSO(320)
174 CREATE_WINOGRAD_SE_PSO(384)
175 CREATE_WINOGRAD_SE_PSO(512)
176 CREATE_WINOGRAD_SE_PSO(640)
177 CREATE_WINOGRAD_SE_PSO(768)
178 CREATE_WINOGRAD_SE_PSO(1024)
179}

213void ShaderWrapper::ExpandPlanes(ID3D12GraphicsCommandList4* command_list,
214 DXAlloc output_tensor, DXAlloc masks,
215 DXAlloc values, int batchSize, bool fp16) {
216 const int N = batchSize * kInputPlanes;
217 int consts[] = {N, kInputPlanes};
218 command_list->SetComputeRootSignature(root_sign_);
219 command_list->SetPipelineState(fp16 ? expand_planes_fp16_
220 : expand_planes_fp32_);
221 command_list->SetComputeRootUnorderedAccessView(0, output_tensor.gpu_va);
222 command_list->SetComputeRootUnorderedAccessView(1, masks.gpu_va);
223 command_list->SetComputeRootUnorderedAccessView(2, values.gpu_va);
224 command_list->SetComputeRoot32BitConstants(kUavSlots, 2, &consts, 0);
225
226 int elements = batchSize * kInputPlanes * 8 * 8;
227 int blocks = DivUp(elements, kExpandPlanesElementsPerBlock);
228 command_list->Dispatch(blocks, 1, 1);
229}

231void ShaderWrapper::InputTransform(ID3D12GraphicsCommandList4* command_list,
232 DXAlloc transformed_input, DXAlloc input,
233 int N, int C, bool /*fp16*/) {
234 int consts[] = {N, C};
235 command_list->SetComputeRootSignature(root_sign_);
236 command_list->SetPipelineState(winograd_input_transform_);
237 command_list->SetComputeRootUnorderedAccessView(0, input.gpu_va);
238 command_list->SetComputeRootUnorderedAccessView(1, transformed_input.gpu_va);
239 command_list->SetComputeRoot32BitConstants(
240 kUavSlots, ARR_ELEMENT_COUNT(consts), &consts, 0);
241 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 0,
242 input.desc_handle_vector);
243 command_list->SetComputeRootDescriptorTable(
244 kUavSlots + 1 + 1, transformed_input.desc_handle_scalar);
245
246 int blocks = DivUp(N * C, kWinogradTransformShaderBlockSize);
247 command_list->Dispatch(blocks, 1, 1);
248}

250void ShaderWrapper::Se(ID3D12GraphicsCommandList4* command_list, DXAlloc output,
251 DXAlloc input, DXAlloc skip_connection, DXAlloc bias,
252 DXAlloc se_w1, DXAlloc se_b1, DXAlloc se_w2,
253 DXAlloc se_b2, int N, int K, bool relu, bool bias_add,
254 bool skip_add, int se_k, bool /*fp16*/) {
255 int consts[] = {N, K, relu, bias_add, skip_add, se_k};
256 command_list->SetComputeRootSignature(root_sign_);
257 command_list->SetComputeRootUnorderedAccessView(0, input.gpu_va);
258 command_list->SetComputeRootUnorderedAccessView(1, output.gpu_va);
259 if (bias_add) command_list->SetComputeRootUnorderedAccessView(2, bias.gpu_va);
260 if (skip_add)
261 command_list->SetComputeRootUnorderedAccessView(3, skip_connection.gpu_va);
262 command_list->SetComputeRootUnorderedAccessView(4, se_w1.gpu_va);
263 command_list->SetComputeRootUnorderedAccessView(5, se_b1.gpu_va);
264 command_list->SetComputeRootUnorderedAccessView(6, se_w2.gpu_va);
265 command_list->SetComputeRootUnorderedAccessView(7, se_b2.gpu_va);
266
267 command_list->SetComputeRoot32BitConstants(
268 kUavSlots, ARR_ELEMENT_COUNT(consts), &consts, 0);
269
270 command_list->SetComputeRootDescriptorTable(
271 kUavSlots + 1 + 0, input.desc_handle_vector);
272 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 1,
273 output.desc_handle_vector);
274 if (bias_add)
275 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 2,
276 bias.desc_handle_scalar);
277 if (skip_add)
278 command_list->SetComputeRootDescriptorTable(
279 kUavSlots + 1 + 3, skip_connection.desc_handle_vector);
280
281 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 4,
282 se_w1.desc_handle_scalar);
283 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 5,
284 se_b1.desc_handle_scalar);
285 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 6,
286 se_w2.desc_handle_scalar);
287 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 7,
288 se_b2.desc_handle_scalar);
289
290 int blocks = N;
291 if (K <= 128)
292 command_list->SetPipelineState(se_128_);
293 else if (K <= 256)
294 command_list->SetPipelineState(se_256_);
295 else if (K <= 320)
296 command_list->SetPipelineState(se_320_);
297 else if (K <= 384)
298 command_list->SetPipelineState(se_384_);
299 else if (K <= 512)
300 command_list->SetPipelineState(se_512_);
301 else if (K <= 640)
302 command_list->SetPipelineState(se_640_);
303 else if (K <= 768)
304 command_list->SetPipelineState(se_768_);
305 else if (K <= 1024)
306 command_list->SetPipelineState(se_1024_);
307 else
308 throw Exception("Unsupported channel count for SE");
309
310 command_list->Dispatch(blocks, 1, 1);
311}

313void ShaderWrapper::OutputTransform(ID3D12GraphicsCommandList4* command_list,
314 DXAlloc output, DXAlloc transformed_output,
315 DXAlloc skip_connection, DXAlloc bias,
316 DXAlloc se_w1, DXAlloc se_b1, DXAlloc se_w2,
317 DXAlloc se_b2, int N, int K, bool relu,
318 bool bias_add, bool skip_add, bool fused_se,
319 int se_k, bool /*fp16*/) {
320 int consts[] = {N, K, relu, bias_add, skip_add, fused_se, se_k};
321 command_list->SetComputeRootSignature(root_sign_);
322 command_list->SetComputeRootUnorderedAccessView(0, transformed_output.gpu_va);
323 command_list->SetComputeRootUnorderedAccessView(1, output.gpu_va);
324 if (bias_add) command_list->SetComputeRootUnorderedAccessView(2, bias.gpu_va);
325 if (skip_add)
326 command_list->SetComputeRootUnorderedAccessView(3, skip_connection.gpu_va);
327 if (fused_se) {
328 command_list->SetComputeRootUnorderedAccessView(4, se_w1.gpu_va);
329 command_list->SetComputeRootUnorderedAccessView(5, se_b1.gpu_va);
330 command_list->SetComputeRootUnorderedAccessView(6, se_w2.gpu_va);
331 command_list->SetComputeRootUnorderedAccessView(7, se_b2.gpu_va);
332 }
333 command_list->SetComputeRoot32BitConstants(
334 kUavSlots, ARR_ELEMENT_COUNT(consts), &consts, 0);
335
336 command_list->SetComputeRootDescriptorTable(
337 kUavSlots + 1 + 0, transformed_output.desc_handle_scalar);
338 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 1,
339 output.desc_handle_vector);
340 if (bias_add)
341 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 2,
342 bias.desc_handle_scalar);
343 if (skip_add)
344 command_list->SetComputeRootDescriptorTable(
345 kUavSlots + 1 + 3, skip_connection.desc_handle_vector);
346 if (fused_se) {
347 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 4,
348 se_w1.desc_handle_scalar);
349 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 5,
350 se_b1.desc_handle_scalar);
351 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 6,
352 se_w2.desc_handle_scalar);
353 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 7,
354 se_b2.desc_handle_scalar);
355 }
356
357 int blocks = 0;
358 if (fused_se) {
359 blocks = N;
360 if (K <= 128)
361 SET_WINOGRAD_SE_PSO(128)
362 else if (K <= 256)
363 SET_WINOGRAD_SE_PSO(256)
364 else if (K <= 320)
365 SET_WINOGRAD_SE_PSO(320)
366 else if (K <= 384)
367 SET_WINOGRAD_SE_PSO(384)
368 else if (K <= 512)
369 SET_WINOGRAD_SE_PSO(512)
370 else if (K <= 640)
371 SET_WINOGRAD_SE_PSO(640)
372 else if (K <= 768)
373 SET_WINOGRAD_SE_PSO(768)
374 else if (K <= 1024)
375 SET_WINOGRAD_SE_PSO(1024)
376 else
377 throw Exception("Unsupported channel count for SE");
378
379 } else {
380 blocks = DivUp(N * K, kWinogradTransformShaderBlockSize);
381 command_list->SetPipelineState(winograd_output_transform_);
382 }
383
384 command_list->Dispatch(blocks, 1, 1);
385}

413void ShaderWrapper::AddVectors(ID3D12GraphicsCommandList4* command_list,
414 DXAlloc C, DXAlloc A, DXAlloc B, int c_size,
415 int a_size, int b_size, bool relu, bool tanh,
416 bool fp16) {
417 if (fp16) {
418 // Shader handles 2 elements per thread in fp16 mode.
419 assert(a_size % 2 == 0);
420 assert(b_size % 2 == 0);
421 assert(c_size % 2 == 0);
422 a_size /= 2;
423 b_size /= 2;
424 c_size /= 2;
425 }
426 int consts[] = {a_size, b_size, c_size, relu, tanh, fp16};
427 command_list->SetComputeRootSignature(root_sign_);
428 command_list->SetPipelineState(add_vectors_);
429 command_list->SetComputeRootUnorderedAccessView(0, A.gpu_va);
430 command_list->SetComputeRootUnorderedAccessView(1, B.gpu_va);
431 command_list->SetComputeRootUnorderedAccessView(2, C.gpu_va);
432 command_list->SetComputeRoot32BitConstants(
433 kUavSlots, ARR_ELEMENT_COUNT(consts), &consts, 0);
434
435 int blocks = DivUp(c_size, kAddVectorsBlockSize);
436 command_list->Dispatch(blocks, 1, 1);
437}

456void ShaderWrapper::MatrixMultiply(ID3D12GraphicsCommandList4* command_list,
457 DXAlloc output, DXAlloc A, DXAlloc B, int M,
458 int N, int K, int batch, bool /*fp16*/) {
459 int Consts[] = {M, N, K, batch};
460 command_list->SetComputeRootSignature(root_sign_);
461
462 command_list->SetPipelineState(gemm_);
463
464 command_list->SetComputeRootUnorderedAccessView(0, A.gpu_va);
465 command_list->SetComputeRootUnorderedAccessView(1, B.gpu_va);
466 command_list->SetComputeRootUnorderedAccessView(2, output.gpu_va);
467 command_list->SetComputeRoot32BitConstants(
468 kUavSlots, ARR_ELEMENT_COUNT(Consts), &Consts, 0);
469 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 0,
470 A.desc_handle_vector);
471 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 1,
472 B.desc_handle_vector);
473 command_list->SetComputeRootDescriptorTable(kUavSlots + 1 + 2,
474 output.desc_handle_vector);
475
476
477 int blocksX = DivUp(N, kGemmElPerBlockX);
478 int blocksY = DivUp(M, kGemmElPerBlockY);
479 int blocksZ = batch;
480
481 command_list->Dispatch(blocksX, blocksY, blocksZ);
482}
