95std::unique_ptr<Network> NetworkFactory::LoadNetwork(
96 const OptionsDict& options) {
97 std::string net_path =
98 options.Get<std::string>(SharedBackendParams::kWeightsId);
99 const std::string backend =
100 options.Get<std::string>(SharedBackendParams::kBackendId);
101 const std::string backend_options =
102 options.Get<std::string>(SharedBackendParams::kBackendOptionsId);
103
104 std::optional<WeightsFile> weights;
105 if (!net_path.empty()) weights = LoadWeights(net_path);
106 OptionsDict network_options(&options);
107 network_options.AddSubdictFromString(backend_options);
108
109 auto ptr = NetworkFactory::Get()->Create(backend, std::move(weights),
110 network_options);
111 network_options.CheckAllOptionsRead(backend);
112 return ptr;
113}
