37// Fast approximate log2(x). Does no range checking.
38// The approximation used here is log2(2^N*(1+f)) ~ N+f*(1+k-k*f) where N is the
39// exponent and f the fraction (mantissa), f>=0. The constant k is used to tune
40// the approximation accuracy. In the final version some constants were slightly
41// modified for better accuracy with 32 bit floating point math.
42inline float FastLog2(const float a) {
43 uint32_t tmp;
44 std::memcpy(&tmp, &a, sizeof(float));
45 uint32_t expb = tmp >> 23;
46 tmp = (tmp & 0x7fffff) | (0x7f << 23);
47 float out;
48 std::memcpy(&out, &tmp, sizeof(float));
49 out -= 1.0f;
50 // Minimize max absolute error.
51 return out * (1.3465552f - 0.34655523f * out) - 127 + expb;
52}

54// Fast approximate 2^x. Does only limited range checking.
55// The approximation used here is 2^(N+f) ~ 2^N*(1+f*(1-k+k*f)) where N is the
56// integer and f the fractional part, f>=0. The constant k is used to tune the
57// approximation accuracy. In the final version some constants were slightly
58// modified for better accuracy with 32 bit floating point math.
59inline float FastExp2(const float a) {
60 int32_t exp;
61 if (a < 0) {
62 if (a < -126) return 0.0;
63 // Not all compilers optimize floor, so we use (a-1) here to round down.
64 // This is obviously off-by-one for integer a, but fortunately the error
65 // correction term gives the exact value for 1 (by design, for continuity).
66 exp = static_cast<int32_t>(a - 1);
67 } else {
68 exp = static_cast<int32_t>(a);
69 }
70 float out = a - exp;
71 // Minimize max relative error.
72 out = 1.0f + out * (0.6602339f + 0.33976606f * out);
73 int32_t tmp;
74 std::memcpy(&tmp, &out, sizeof(float));
75 tmp += static_cast<int32_t>(static_cast<uint32_t>(exp) << 23);
76 std::memcpy(&out, &tmp, sizeof(float));
77 return out;
78}

80// Fast approximate ln(x). Does no range checking.
81inline float FastLog(const float a) {
82 return 0.6931471805599453f * FastLog2(a);
83}

85// Fast approximate exp(x). Does only limited range checking.
86inline float FastExp(const float a) { return FastExp2(1.442695040f * a); }

88// Safeguarded fast logistic function, based on FastExp().
89inline float FastLogistic(const float a) {
90 if (a > 20.0f) {return 1.0f;}
91 if (a < -20.0f) {return 0.0f;}
92 return 1.0f / (1.0f + FastExp(-a));
93}

95inline float FastSign(const float a) {
96 // Microsoft compiler does not have a builtin for copysign and emits a
97 // library call which is too expensive for hot paths.
98#if defined(_MSC_VER)
99 // This doesn't treat signed 0's the same way that copysign does, but it
100 // should be good enough, for our use case.
101 return a < 0 ? -1.0f : 1.0f;
102#else
103 return std::copysign(1.0f, a);
104#endif
105}
