78BaseWeights::ConvBlock::ConvBlock(const pblczero::Weights::ConvBlock& block)
79 : weights(LayerAdapter(block.weights()).as_vector()),
80 biases(LayerAdapter(block.biases()).as_vector()),
81 bn_gammas(LayerAdapter(block.bn_gammas()).as_vector()),
82 bn_betas(LayerAdapter(block.bn_betas()).as_vector()),
83 bn_means(LayerAdapter(block.bn_means()).as_vector()),
84 bn_stddivs(LayerAdapter(block.bn_stddivs()).as_vector()) {
85 if (weights.size() == 0) {
86 // Empty ConvBlock.
87 return;
88 }
89
90 if (bn_betas.size() == 0) {
91 // Old net without gamma and beta.
92 for (auto i = size_t{0}; i < bn_means.size(); i++) {
93 bn_betas.emplace_back(0.0f);
94 bn_gammas.emplace_back(1.0f);
95 }
96 }
97 if (biases.size() == 0) {
98 for (auto i = size_t{0}; i < bn_means.size(); i++) {
99 biases.emplace_back(0.0f);
100 }
101 }
102
103 if (bn_means.size() == 0) {
104 // No batch norm.
105 return;
106 }
107
108 // Fold batch norm into weights and biases.
109 // Variance to gamma.
110 for (auto i = size_t{0}; i < bn_stddivs.size(); i++) {
111 bn_gammas[i] *= 1.0f / std::sqrt(bn_stddivs[i] + kEpsilon);
112 bn_means[i] -= biases[i];
113 }
114
115 auto outputs = biases.size();
116
117 // We can treat the [inputs, filter_size, filter_size] dimensions as one.
118 auto inputs = weights.size() / outputs;
119
120 for (auto o = size_t{0}; o < outputs; o++) {
121 for (auto c = size_t{0}; c < inputs; c++) {
122 weights[o * inputs + c] *= bn_gammas[o];
123 }
124
125 biases[o] = -bn_gammas[o] * bn_means[o] + bn_betas[o];
126 }
127
128 // Batch norm weights are not needed anymore.
129 bn_stddivs.clear();
130 bn_means.clear();
131 bn_betas.clear();
132 bn_gammas.clear();
133}

232MultiHeadWeights::MultiHeadWeights(const pblczero::Weights& weights)
233 : BaseWeights(weights),
234 ip_pol_w(LayerAdapter(weights.policy_heads().has_ip_pol_w()
235 ? weights.policy_heads().ip_pol_w()
236 : weights.ip_pol_w())
237 .as_vector()),
238 ip_pol_b(LayerAdapter(weights.policy_heads().has_ip_pol_b()
239 ? weights.policy_heads().ip_pol_b()
240 : weights.ip_pol_b())
241 .as_vector()) {
242 policy_heads.emplace(std::piecewise_construct,
243 std::forward_as_tuple("vanilla"),
244 std::forward_as_tuple(weights.policy_heads().vanilla(),
245 ip_pol_w, ip_pol_b));
246 if (weights.has_policy_heads()) {
247 if (weights.policy_heads().has_optimistic_st()) {
248 policy_heads.emplace(
249 std::piecewise_construct, std::forward_as_tuple("optimistic"),
250 std::forward_as_tuple(weights.policy_heads().optimistic_st(),
251 ip_pol_w, ip_pol_b));
252 }
253 if (weights.policy_heads().has_soft()) {
254 policy_heads.emplace(std::piecewise_construct,
255 std::forward_as_tuple("soft"),
256 std::forward_as_tuple(weights.policy_heads().soft(),
257 ip_pol_w, ip_pol_b));
258 }
259 if (weights.policy_heads().has_opponent()) {
260 policy_heads.emplace(
261 std::piecewise_construct, std::forward_as_tuple("opponent"),
262 std::forward_as_tuple(weights.policy_heads().opponent(), ip_pol_w,
263 ip_pol_b));
264 }
265 } else {
266 if (weights.has_policy() || weights.has_policy1() ||
267 weights.has_ip_pol_w()) {
268 auto& vanilla = policy_heads.at("vanilla");
269 vanilla.policy1 = ConvBlock(weights.policy1());
270 vanilla.policy = ConvBlock(weights.policy());
271 vanilla.ip2_pol_w = LayerAdapter(weights.ip2_pol_w()).as_vector();
272 vanilla.ip2_pol_b = LayerAdapter(weights.ip2_pol_b()).as_vector();
273 vanilla.ip3_pol_w = LayerAdapter(weights.ip3_pol_w()).as_vector();
274 vanilla.ip3_pol_b = LayerAdapter(weights.ip3_pol_b()).as_vector();
275 vanilla.ip4_pol_w = LayerAdapter(weights.ip4_pol_w()).as_vector();
276 vanilla.pol_encoder_head_count = weights.pol_headcount();
277 for (const auto& enc : weights.pol_encoder()) {
278 vanilla.pol_encoder.emplace_back(enc);
279 }
280 } else {
281 throw Exception("Could not find valid policy head weights.");
282 }
283 }
284
285 value_heads.emplace("winner", weights.value_heads().winner());
286 if (weights.has_value_heads()) {
287 if (weights.value_heads().has_q()) {
288 value_heads.emplace("q", weights.value_heads().q());
289 }
290 if (weights.value_heads().has_st()) {
291 value_heads.emplace("st", weights.value_heads().st());
292 }
293 } else {
294 if (weights.has_value() || weights.has_ip_val_w()) {
295 auto& winner = value_heads.at("winner");
296 winner.value = ConvBlock(weights.value());
297 winner.ip_val_w = LayerAdapter(weights.ip_val_w()).as_vector();
298 winner.ip_val_b = LayerAdapter(weights.ip_val_b()).as_vector();
299 winner.ip1_val_w = LayerAdapter(weights.ip1_val_w()).as_vector();
300 winner.ip1_val_b = LayerAdapter(weights.ip1_val_b()).as_vector();
301 winner.ip2_val_w = LayerAdapter(weights.ip2_val_w()).as_vector();
302 winner.ip2_val_b = LayerAdapter(weights.ip2_val_b()).as_vector();
303 } else {
304 throw Exception("Could not find valid value head weights.");
305 }
306 }
