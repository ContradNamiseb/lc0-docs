38template <class T>
39std::string TransposeAndReturnRaw(const std::vector<int>& dims,
40 std::vector<int> order,
41 const std::vector<T>& from) {
42 if (order.empty()) {
43 return {reinterpret_cast<const char*>(from.data()),
44 reinterpret_cast<const char*>(from.data() + from.size())};
45 } else {
46 std::vector<T> dst(from.size());
47 TransposeTensor(dims, order, from, &dst[0]);
48 return {reinterpret_cast<const char*>(dst.data()),
49 reinterpret_cast<const char*>(dst.data() + dst.size())};
50 }
51}
52}
