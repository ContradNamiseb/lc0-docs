37float ComputeEstimatedMovesToGo(int ply, float midpoint, float steepness) {
38 // An analysis of chess games shows that the distribution of game lengths
39 // looks like a log-logistic distribution. The mean residual time function
40 // calculates how many more moves are expected in the game given that we are
41 // at the current ply. Given that this function can be expensive to compute,
42 // we calculate the median residual time function instead. This is derived and
43 // shown to be similar to the mean residual time in "Some Useful Properties of
44 // Log-Logistic Random Variables for Health Care Simulations" (Clark &
45 // El-Taha, 2015).
46 // midpoint: The median length of games.
47 // steepness: How quickly the function drops off from its maximum value,
48 // around the midpoint.
49 const float move = ply / 2.0f;
50 return midpoint * std::pow(1 + 2 * std::pow(move / midpoint, steepness),
51 1 / steepness) -
52 move;
53}

106std::unique_ptr<SearchStopper> LegacyTimeManager::GetStopper(
107 const GoParams& params, const Position& position, size_t /*total_memory*/,
108 size_t /*avg_node_size*/, uint32_t /*nodes*/) {
109 const bool is_black = position.IsBlackToMove();
110 const std::optional<int64_t>& time = (is_black ? params.btime : params.wtime);
111 // If no time limit is given, don't stop on this condition.
112 if (params.infinite || params.ponder || !time) return nullptr;
113
114 const std::optional<int64_t>& inc = is_black ? params.binc : params.winc;
115 const int increment = inc ? std::max(int64_t(0), *inc) : 0;
116
117 float movestogo = ComputeEstimatedMovesToGo(
118 position.GetGamePly(), time_curve_midpoint_, time_curve_steepness_);
119
120 // If the number of moves remaining until the time control are less than
121 // the estimated number of moves left in the game, then use the number of
122 // moves until the time control instead.
123 if (params.movestogo &&
124 *params.movestogo > 0 && // Ignore non-standard uci command.
125 *params.movestogo < movestogo) {
126 movestogo = *params.movestogo;
127 }
128
129 // Total time, including increments, until time control.
130 auto total_moves_time =
131 std::max(0.0f, *time + increment * (movestogo - 1) - move_overhead_);
132
133 // If there is time spared from previous searches, the `time_to_squander` part
134 // of it will be used immediately, remove that from planning.
135 int time_to_squander = 0;
136 if (time_spared_ms_ > 0) {
137 total_moves_time = std::max(0.0f, total_moves_time - time_spared_ms_);
138 time_to_squander = time_spared_ms_ * spend_saved_time_;
139 time_spared_ms_ -= time_to_squander;
140 }
141
142 // Evenly split total time between all moves.
143 float this_move_time = total_moves_time / movestogo;
144
145 // Add bonus time per ply of the opening book to compensate starting from an
146 // uncommon position without a tree to reuse.
147 // Limit the bonus to max. 12 plies, which also prevents spending too much
148 // time on the first move in resumed games.
149 if (first_move_of_game_) {
150 this_move_time *= (1.0f + first_move_bonus_ +
151 book_ply_bonus_ * std::min(12, position.GetGamePly()));
152 first_move_of_game_ = false;
153 }
154
155 // Only extend thinking time with slowmover if smart pruning can potentially
156 // reduce it.
157 constexpr int kSmartPruningToleranceMs = 200;
158 if (slowmover_ < 1.0 ||
159 this_move_time * slowmover_ > kSmartPruningToleranceMs) {
160 // If time is planned to be overused because of slowmover, remove excess
161 // of that time from spared time.
162 time_spared_ms_ -= this_move_time * (slowmover_ - 1);
163 this_move_time *= slowmover_;
164 }
165
166 LOGFILE << "Budgeted time for the move: " << this_move_time << "ms(+"
167 << time_to_squander << "ms to squander). Remaining time " << *time
168 << "ms(-" << move_overhead_ << "ms overhead)";
169 // Use `time_to_squander` time immediately.
170 this_move_time += time_to_squander;
171
172 // Make sure we don't exceed current time limit with what we calculated.
173 auto deadline =
174 std::min(static_cast<int64_t>(this_move_time), *time - move_overhead_);
175 return std::make_unique<LegacyStopper>(deadline, &time_spared_ms_);
176}
