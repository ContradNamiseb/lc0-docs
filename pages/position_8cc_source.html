75GameResult PositionHistory::ComputeGameResult() const {
76 const auto& board = Last().GetBoard();
77 auto legal_moves = board.GenerateLegalMoves();
78 if (legal_moves.empty()) {
79 if (board.IsUnderCheck()) {
80 // Checkmate.
81 return IsBlackToMove() ? GameResult::WHITE_WON : GameResult::BLACK_WON;
82 }
83 // Stalemate.
84 return GameResult::DRAW;
85 }
86
87 if (!board.HasMatingMaterial()) return GameResult::DRAW;
88 if (Last().GetRule50Ply() >= 100) return GameResult::DRAW;
89 if (Last().GetRepetitions() >= 2) return GameResult::DRAW;
90
91 return GameResult::UNDECIDED;
92}

105void PositionHistory::Append(Move m) {
106 // TODO(mooskagh) That should be emplace_back(Last(), m), but MSVS STL
107 // has a bug in implementation of emplace_back, when
108 // reallocation happens. (it also reallocates Last())
109 positions_.push_back(Position(Last(), m));
110 int cycle_length;
111 int repetitions = ComputeLastMoveRepetitions(&cycle_length);
112 positions_.back().SetRepetitions(repetitions, cycle_length);
113}

115int PositionHistory::ComputeLastMoveRepetitions(int* cycle_length) const {
116 *cycle_length = 0;
117 const auto& last = positions_.back();
118 // TODO(crem) implement hash/cache based solution.
119 if (last.GetRule50Ply() < 4) return 0;
120
121 for (int idx = positions_.size() - 5; idx >= 0; idx -= 2) {
122 const auto& pos = positions_[idx];
123 if (pos.GetBoard() == last.GetBoard()) {
124 *cycle_length = positions_.size() - 1 - idx;
125 return 1 + pos.GetRepetitions();
126 }
127 if (pos.GetRule50Ply() < 2) return 0;
128 }
129 return 0;
130}

132bool PositionHistory::DidRepeatSinceLastZeroingMove() const {

141uint64_t PositionHistory::HashLast(int positions) const {
142 uint64_t hash = positions;
143 for (auto iter = positions_.rbegin(), end = positions_.rend(); iter != end;
144 ++iter) {
145 if (!positions--) break;
146 hash = HashCat(hash, iter->Hash());
147 }
148 return HashCat(hash, Last().GetRule50Ply());
149}
