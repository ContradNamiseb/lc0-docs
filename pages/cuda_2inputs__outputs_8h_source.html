77template <typename DataType>
78struct InputsOutputs {
79 InputsOutputs(unsigned maxBatchSize, bool wdl, bool moves_left,
80 size_t tensor_mem_size = 0, size_t scratch_size = 0,
81 bool cublasDisableTensorCores = false) {
82 ReportCUDAErrors(cudaHostAlloc(
83 &input_masks_mem_, maxBatchSize * kInputPlanes * sizeof(uint64_t),
84 cudaHostAllocMapped));
85 ReportCUDAErrors(cudaMalloc(
86 &input_masks_mem_gpu_, maxBatchSize * kInputPlanes * sizeof(uint64_t)));
87
88 ReportCUDAErrors(
89 cudaHostAlloc(&input_val_mem_,
90 maxBatchSize * kInputPlanes * sizeof(input_val_mem_[0]),
91 cudaHostAllocMapped));
92 ReportCUDAErrors(cudaMalloc(
93 &input_val_mem_gpu_,
94 maxBatchSize * kInputPlanes * sizeof(input_val_mem_gpu_[0])));
95
96 ReportCUDAErrors(cudaHostAlloc(
97 &op_policy_mem_,
98 maxBatchSize * kNumOutputPolicy * sizeof(op_policy_mem_[0]), 0));
99
100 // Seperate device memory copy for policy output.
101 // It's faster to write to device memory and then copy to host memory
102 // than having the kernel write directly to it.
103 ReportCUDAErrors(cudaMalloc(
104 &op_policy_mem_gpu_,
105 maxBatchSize * kNumOutputPolicy * sizeof(op_policy_mem_[0])));
106 ReportCUDAErrors(cudaHostAlloc(
107 &op_value_mem_, maxBatchSize * (wdl ? 3 : 1) * sizeof(op_value_mem_[0]),
108 cudaHostAllocMapped));
109 ReportCUDAErrors(cudaMalloc(
110 &op_value_mem_gpu_,
111 maxBatchSize * (wdl ? 3 : 1) * sizeof(op_value_mem_gpu_[0])));
112 if (wdl && sizeof(DataType) != sizeof(float)) {
113 wdl_cpu_softmax_ = std::make_unique<float[]>(maxBatchSize * 2);
114 }
115 ReportCUDAErrors(
116 cudaEventCreateWithFlags(&upload_done_event_, cudaEventDisableTiming));
117 ReportCUDAErrors(
118 cudaEventCreateWithFlags(&policy_done_event_, cudaEventDisableTiming));
119 ReportCUDAErrors(
120 cudaEventCreateWithFlags(&value_done_event_, cudaEventDisableTiming));
121 ReportCUDAErrors(cudaEventCreateWithFlags(&wdl_download_done_event_,
122 cudaEventDisableTiming));
123 ReportCUDAErrors(cudaEventCreateWithFlags(&download_done_event_,
124 cudaEventDisableTiming));
125 if (moves_left) {
126 ReportCUDAErrors(cudaHostAlloc(
127 &op_moves_left_mem_, maxBatchSize * sizeof(op_moves_left_mem_[0]),
128 cudaHostAllocMapped));
129 ReportCUDAErrors(
130 cudaMalloc(&op_moves_left_mem_gpu_,
131 maxBatchSize * sizeof(op_moves_left_mem_gpu_[0])));
132 ReportCUDAErrors(cudaEventCreateWithFlags(&moves_left_done_event_,
133 cudaEventDisableTiming));
134 }
135
136 ReportCUDAErrors(
137 cudaStreamCreateWithFlags(&exec_stream_, cudaStreamNonBlocking));
138 ReportCUDAErrors(
139 cudaEventCreateWithFlags(&join_capture_event_, cudaEventDisableTiming));
140 cuda_graphs_ = std::make_unique<CudaGraphExec<DataType>[]>(maxBatchSize);
141
142 // memory for network execution managed inside this structure
143 if (tensor_mem_size) {
144 multi_stream_ = true;
145 ReportCUDAErrors(
146 cudaStreamCreateWithFlags(&compute_stream_, cudaStreamNonBlocking));
147 ReportCUDAErrors(
148 cudaStreamCreateWithFlags(&upload_stream_, cudaStreamNonBlocking));
149 ReportCUDAErrors(
150 cudaStreamCreateWithFlags(&download_stream_, cudaStreamNonBlocking));
151 ReportCUDAErrors(cudaMalloc(&scratch_mem_, scratch_size));
152 for (auto& mem : tensor_mem_) {
153 ReportCUDAErrors(cudaMalloc(&mem, tensor_mem_size));
154 ReportCUDAErrors(
155 cudaMemsetAsync(mem, 0, tensor_mem_size, compute_stream_));
156 }
157 ReportCUBLASErrors(cublasCreate(&cublas_));
158 ReportCUBLASErrors(cublasSetMathMode(
159 cublas_, cublasDisableTensorCores ? CUBLAS_PEDANTIC_MATH
160 : CUBLAS_TENSOR_OP_MATH));
161 ReportCUBLASErrors(cublasSetStream(cublas_, compute_stream_));
162 } else {
163 multi_stream_ = false;
164 }
165 }
166 ~InputsOutputs() {
167 ReportCUDAErrors(cudaFreeHost(input_masks_mem_));
168 ReportCUDAErrors(cudaFree(input_masks_mem_gpu_));
169 ReportCUDAErrors(cudaFreeHost(input_val_mem_));
170 ReportCUDAErrors(cudaFree(input_val_mem_gpu_));
171 ReportCUDAErrors(cudaFreeHost(op_policy_mem_));
172 ReportCUDAErrors(cudaFree(op_policy_mem_gpu_));
173 ReportCUDAErrors(cudaFreeHost(op_value_mem_));
174 ReportCUDAErrors(cudaFree(op_value_mem_gpu_));
175 ReportCUDAErrors(cudaEventDestroy(upload_done_event_));
176 ReportCUDAErrors(cudaEventDestroy(policy_done_event_));
177 ReportCUDAErrors(cudaEventDestroy(value_done_event_));
178 ReportCUDAErrors(cudaEventDestroy(wdl_download_done_event_));
179 ReportCUDAErrors(cudaEventDestroy(download_done_event_));
180 if (op_moves_left_mem_ != nullptr) {
181 ReportCUDAErrors(cudaFreeHost(op_moves_left_mem_));
182 ReportCUDAErrors(cudaFree(op_moves_left_mem_gpu_));
183 ReportCUDAErrors(cudaEventDestroy(moves_left_done_event_));
184 }
185 ReportCUDAErrors(cudaEventDestroy(join_capture_event_));
186 ReportCUDAErrors(cudaStreamDestroy(exec_stream_));
187
188 if (multi_stream_) {
189 for (auto mem : tensor_mem_) {
190 if (mem) ReportCUDAErrors(cudaFree(mem));
191 }
192 if (scratch_mem_) ReportCUDAErrors(cudaFree(scratch_mem_));
193 if (offset_pointers_) ReportCUDAErrors(cudaFree(offset_pointers_));
194 if (head_offset_pointers_) {
195 ReportCUDAErrors(cudaFree(head_offset_pointers_));
196 }
197 ReportCUDAErrors(cudaStreamDestroy(compute_stream_));
198 ReportCUDAErrors(cudaStreamDestroy(upload_stream_));
199 ReportCUDAErrors(cudaStreamDestroy(download_stream_));
200 ReportCUBLASErrors(cublasDestroy(cublas_));
201 }
202 }
203 uint64_t* input_masks_mem_;
204 DataType* input_val_mem_;
205 DataType* op_policy_mem_;
206 DataType* op_value_mem_;
207 DataType* op_moves_left_mem_ = nullptr;
208
209 // Copies in VRAM.
210 uint64_t* input_masks_mem_gpu_;
211 DataType* input_val_mem_gpu_;
212 DataType* op_policy_mem_gpu_;
213 DataType* op_value_mem_gpu_;
214 DataType* op_moves_left_mem_gpu_ = nullptr;
215
216 std::unique_ptr<float[]> wdl_cpu_softmax_;
217
218 // memory needed to run the network owned by InputsOutputs when multi_stream
219 // is enabled
220 bool multi_stream_;
221 void* tensor_mem_[3];
222 void* scratch_mem_;
223 void** offset_pointers_ = nullptr;
224 void** head_offset_pointers_ = nullptr;
225
226 // cuda stream used to run the network
227 cudaStream_t compute_stream_ = nullptr;
228 cudaStream_t upload_stream_ = nullptr;
229 cudaStream_t download_stream_ = nullptr;
230
231 // cuda events to synchronize between streams
232 cudaEvent_t upload_done_event_ = nullptr;
233 cudaEvent_t policy_done_event_ = nullptr;
234 cudaEvent_t value_done_event_ = nullptr;
235 cudaEvent_t moves_left_done_event_ = nullptr;
236 cudaEvent_t wdl_download_done_event_ = nullptr;
237 cudaEvent_t download_done_event_ = nullptr;
238
239 // cuda graph support
240 cudaStream_t exec_stream_ = nullptr;
241 std::unique_ptr<CudaGraphExec<DataType>[]> cuda_graphs_;
242 cudaEvent_t join_capture_event_ = nullptr;
243
244 // cublas handle used to run the network
245 cublasHandle_t cublas_ = nullptr;
246};

249template <typename DataType>
250struct CudaGraphCapture {
251 static constexpr int kMinimumFreeMemory = 100 * 1024 * 1024;
252
253 CudaGraphCapture(InputsOutputs<DataType>& io, cudaStream_t upload_stream,
254 cudaStream_t download_stream)
255 : io_(io),
256 upload_stream_(upload_stream),
257 download_stream_(download_stream) {
258 ReportCUDAErrors(cudaStreamBeginCapture(upload_stream_,
259 cudaStreamCaptureModeThreadLocal));
260 }
261
262 ~CudaGraphCapture() {
263 if (graph_ != nullptr) {
264 ReportCUDAErrors(cudaGraphDestroy(graph_));
265 }
266 }
267
268 static bool EnsureEnoughFreeMemory() {
269 size_t free_mem = 0;
270 size_t total_mem = 0;
271 ReportCUDAErrors(cudaMemGetInfo(&free_mem, &total_mem));
272 return free_mem > kMinimumFreeMemory;
273 }
274
275 void EndCapture() {
276 ReportCUDAErrors(
277 cudaEventRecord(io_.join_capture_event_, download_stream_));
278 ReportCUDAErrors(
279 cudaStreamWaitEvent(upload_stream_, io_.join_capture_event_, 0));
280 ReportCUDAErrors(cudaStreamEndCapture(upload_stream_, &graph_));
281 }
282
283 InputsOutputs<DataType>& io_;
284 cudaStream_t upload_stream_;
285 cudaStream_t download_stream_;
286
287 cudaGraph_t graph_ = nullptr;
288};

290template <typename DataType>
291inline CudaGraphExec<DataType>& CudaGraphExec<DataType>::operator=(
292 const CudaGraphCapture<DataType>& graph) {
293 assert(graph_exec_ == nullptr);
294 if (graph.graph_ == nullptr) {
295 throw Exception("Trying to instantiate an nullptr cuda graph");
296 }
297 ReportCUDAErrors(
298 cudaGraphInstantiate(&graph_exec_, graph.graph_, nullptr, nullptr, 0));
299#if CUDART_VERSION >= 11010
300 ReportCUDAErrors(cudaGraphUpload(graph_exec_, graph.io_.exec_stream_));
301#endif
302 return *this;
303}
