40static void global_avg_pooling(size_t batches, size_t channels,
41 const float* input, const float* bias,
42 float* output) {
43 for (auto b = size_t{0}; b < batches; b++) {
44 for (auto ch = size_t{0}; ch < channels; ch++) {
45 auto c = b * channels + ch;
46 auto acc = 0.0f;
47 for (auto i = size_t{0}; i < kSquares; i++) {
48 acc += input[c * kSquares + i];
49 }
50 output[c] = acc / kSquares + bias[ch];
51 }
52 }
53}

65static void apply_se(const size_t channels, const size_t batch_size,
66 const float* input, const float* bias, const float* res,
67 const float* scale, float* output,
68 const ActivationFunction activation) {
69 const auto lambda_sigmoid = [](const auto val) {
70 return 1.0f / (1.0f + std::exp(-val));
71 };
72
73 for (auto batch = size_t{0}; batch < batch_size; batch++) {
74 for (auto ch = size_t{0}; ch < channels; ch++) {
75 auto c = channels * batch + ch;
76 auto gamma = lambda_sigmoid(scale[c + batch * channels]);
77 auto beta = scale[c + batch * channels + channels] + gamma * bias[ch];
78 Activate(kSquares, gamma, &input[c * kSquares], &res[c * kSquares], beta,
79 &output[c * kSquares], activation);
80 }
81 }
82}

84template <bool use_eigen>
85void ApplySEUnit(const size_t batch_size, const size_t channels,
86 const size_t se_fc_outputs, const float* input,
87 const float* ch_bias, const float* residual,
88 const float* weights_w1, const float* weights_b1,
89 const float* weights_w2, const float* weights_b2,
90 float* output, const ActivationFunction activation) {
91 std::vector<float> pool(2 * channels * batch_size);
92 std::vector<float> fc_out1(batch_size * se_fc_outputs);
93
94 global_avg_pooling(batch_size, channels, input, ch_bias, pool.data());
95
96 FullyConnectedLayer<use_eigen>::Forward1D(batch_size, channels, se_fc_outputs,
97 pool.data(), weights_w1, weights_b1,
98 activation, // Activation On
99 fc_out1.data());
100
101 FullyConnectedLayer<use_eigen>::Forward1D(batch_size, se_fc_outputs,
102 2 * channels, fc_out1.data(),
103 weights_w2, weights_b2,
104 ACTIVATION_NONE, // Activation Off
105 pool.data());
106
107 // Sigmoid, scale and add residual
108 apply_se(channels, batch_size, input, ch_bias, residual, pool.data(), output,
109 activation);
110}
