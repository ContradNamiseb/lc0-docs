48TrainingDataWriter::TrainingDataWriter(int game_id) {
49 static std::string directory =
50 GetLc0CacheDirectory() + "data-" + Random::Get().GetString(12);
51 // It's fine if it already exists.
52 CreateDirectory(directory.c_str());
53
54 std::ostringstream oss;
55 oss << directory << '/' << "game_" << std::setfill('0') << std::setw(6)
56 << game_id << ".gz";
57
58 filename_ = oss.str();
59 fout_ = gzopen(filename_.c_str(), "wb");
60 if (!fout_) throw Exception("Cannot create gzip file " + filename_);
61}

69void TrainingDataWriter::WriteChunk(const V6TrainingData& data) {
70 auto bytes_written =
71 gzwrite(fout_, reinterpret_cast<const char*>(&data), sizeof(data));
72 if (bytes_written != sizeof(data)) {
73 throw Exception("Unable to write into " + filename_);
74 }
