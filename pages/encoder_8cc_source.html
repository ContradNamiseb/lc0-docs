36int CompareTransposing(BitBoard board, int initial_transform) {
37 uint64_t value = board.as_int();
38 if ((initial_transform & FlipTransform) != 0) {
39 value = ReverseBitsInBytes(value);
40 }
41 if ((initial_transform & MirrorTransform) != 0) {
42 value = ReverseBytesInBytes(value);
43 }
44 auto alternative = TransposeBitsInBytes(value);
45 if (value < alternative) return -1;
46 if (value > alternative) return 1;
47 return 0;
48}

50int ChooseTransform(const ChessBoard& board) {
51 // If there are any castling options no transform is valid.
52 // Even using FRC rules, king and queen side castle moves are not symmetrical.
53 if (!board.castlings().no_legal_castle()) {
54 return 0;
55 }
56 auto our_king = (board.kings() & board.ours()).as_int();
57 int transform = NoTransform;
58 if ((our_king & 0x0F0F0F0F0F0F0F0FULL) != 0) {
59 transform |= FlipTransform;
60 our_king = ReverseBitsInBytes(our_king);
61 }
62 // If there are any pawns only horizontal flip is valid.
63 if (board.pawns().as_int() != 0) {
64 return transform;
65 }
66 if ((our_king & 0xFFFFFFFF00000000ULL) != 0) {
67 transform |= MirrorTransform;
68 our_king = ReverseBytesInBytes(our_king);
69 }
70 // Our king is now always in bottom right quadrant.
71 // Transpose for king in top right triangle, or if on diagonal whichever has
72 // the smaller integer value for each test scenario.
73 if ((our_king & 0xE0C08000ULL) != 0) {
74 transform |= TransposeTransform;
75 } else if ((our_king & 0x10204080ULL) != 0) {
76 auto outcome = CompareTransposing(board.ours() | board.theirs(), transform);
77 if (outcome == -1) return transform;
78 if (outcome == 1) return transform | TransposeTransform;
79 outcome = CompareTransposing(board.ours(), transform);
80 if (outcome == -1) return transform;
81 if (outcome == 1) return transform | TransposeTransform;
82 outcome = CompareTransposing(board.kings(), transform);
83 if (outcome == -1) return transform;
84 if (outcome == 1) return transform | TransposeTransform;
85 outcome = CompareTransposing(board.queens(), transform);
86 if (outcome == -1) return transform;
87 if (outcome == 1) return transform | TransposeTransform;
88 outcome = CompareTransposing(board.rooks(), transform);
89 if (outcome == -1) return transform;
90 if (outcome == 1) return transform | TransposeTransform;
91 outcome = CompareTransposing(board.knights(), transform);
92 if (outcome == -1) return transform;
93 if (outcome == 1) return transform | TransposeTransform;
94 outcome = CompareTransposing(board.bishops(), transform);
95 if (outcome == -1) return transform;
96 if (outcome == 1) return transform | TransposeTransform;
97 // If all piece types are symmetrical and ours is symmetrical and
98 // ours+theirs is symmetrical, everything is symmetrical, so transpose is a
99 // no-op.
100 }
101 return transform;
102}

105bool IsCanonicalFormat(pblczero::NetworkFormat::InputFormat input_format) {
106 return input_format >=
107 pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION;
108}
109bool IsCanonicalArmageddonFormat(
110 pblczero::NetworkFormat::InputFormat input_format) {
111 return input_format ==
112 pblczero::NetworkFormat::
113 INPUT_112_WITH_CANONICALIZATION_HECTOPLIES_ARMAGEDDON ||
114 input_format == pblczero::NetworkFormat::
115 INPUT_112_WITH_CANONICALIZATION_V2_ARMAGEDDON;
116}
117bool IsHectopliesFormat(pblczero::NetworkFormat::InputFormat input_format) {
118 return input_format >=
119 pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_HECTOPLIES;
120}
121bool Is960CastlingFormat(pblczero::NetworkFormat::InputFormat input_format) {
122 return input_format >= pblczero::NetworkFormat::INPUT_112_WITH_CASTLING_PLANE;
123}

125int TransformForPosition(pblczero::NetworkFormat::InputFormat input_format,
126 const PositionHistory& history) {
127 if (!IsCanonicalFormat(input_format)) {
128 return 0;
129 }
130 const ChessBoard& board = history.Last().GetBoard();
131 return ChooseTransform(board);
132}

134InputPlanes EncodePositionForNN(
135 pblczero::NetworkFormat::InputFormat input_format,
136 std::span<const Position> history, int history_planes,
137 FillEmptyHistory fill_empty_history, int* transform_out) {
138 InputPlanes result(kAuxPlaneBase + 8);
139
140 int transform = 0;
141 // Canonicalization format needs to stop early to avoid applying transform in
142 // history across incompatible transitions. It is also more canonical since
143 // history before these points is not relevant to the final result.
144 bool stop_early = IsCanonicalFormat(input_format);
145 // When stopping early, we want to know if castlings has changed, so capture
146 // it for the first board.
147 ChessBoard::Castlings castlings;
148 {
149 const ChessBoard& board = history.back().GetBoard();
150 const bool we_are_black = board.flipped();
151 if (IsCanonicalFormat(input_format)) {
152 transform = ChooseTransform(board);
153 }
154 switch (input_format) {
155 case pblczero::NetworkFormat::INPUT_CLASSICAL_112_PLANE: {
156 // "Legacy" input planes with:
157 // - Plane 104 (0-based) filled with 1 if we can castle queenside.
158 // - Plane 105 filled with ones if we can castle kingside.
159 // - Plane 106 filled with ones if they can castle queenside.
160 // - Plane 107 filled with ones if they can castle kingside.
161 if (board.castlings().we_can_000()) result[kAuxPlaneBase + 0].SetAll();
162 if (board.castlings().we_can_00()) result[kAuxPlaneBase + 1].SetAll();
163 if (board.castlings().they_can_000()) {
164 result[kAuxPlaneBase + 2].SetAll();
165 }
166 if (board.castlings().they_can_00()) result[kAuxPlaneBase + 3].SetAll();
167 break;
168 }
169
170 case pblczero::NetworkFormat::INPUT_112_WITH_CASTLING_PLANE:
171 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION:
172 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_HECTOPLIES:
173 case pblczero::NetworkFormat::
174 INPUT_112_WITH_CANONICALIZATION_HECTOPLIES_ARMAGEDDON:
175 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_V2:
176 case pblczero::NetworkFormat::
177 INPUT_112_WITH_CANONICALIZATION_V2_ARMAGEDDON: {
178 // - Plane 104 for positions of rooks (both white and black) which
179 // have
180 // a-side (queenside) castling right.
181 // - Plane 105 for positions of rooks (both white and black) which have
182 // h-side (kingside) castling right.
183 const auto& cast = board.castlings();
184 result[kAuxPlaneBase + 0].mask =
185 (cast.we_can_000()
186 ? BitBoard::FromSquare(Square(cast.our_queenside_rook, kRank1))
187 .as_int()
188 : 0) |
189 (cast.they_can_000()
190 ? BitBoard::FromSquare(
191 Square(cast.their_queenside_rook, kRank8))
192 .as_int()
193 : 0);
194 result[kAuxPlaneBase + 1].mask =
195 (cast.we_can_00()
196 ? BitBoard::FromSquare(Square(cast.our_kingside_rook, kRank1))
197 .as_int()
198 : 0) |
199 (cast.they_can_00() ? BitBoard::FromSquare(
200 Square(cast.their_kingside_rook, kRank8))
201 .as_int()
202 : 0);
203 break;
204 }
205 default:
206 throw Exception("Unsupported input plane encoding " +
207 std::to_string(input_format));
208 };
209 if (IsCanonicalFormat(input_format)) {
210 result[kAuxPlaneBase + 4].mask = board.en_passant().as_int();
211 } else {
212 if (we_are_black) result[kAuxPlaneBase + 4].SetAll();
213 }
214 if (IsHectopliesFormat(input_format)) {
215 result[kAuxPlaneBase + 5].Fill(history.back().GetRule50Ply() / 100.0f);
216 } else {
217 result[kAuxPlaneBase + 5].Fill(history.back().GetRule50Ply());
218 }
219 // Plane kAuxPlaneBase + 6 used to be movecount plane, now it's all zeros
220 // unless we need it for canonical armageddon side to move.
221 if (IsCanonicalArmageddonFormat(input_format)) {
222 if (we_are_black) result[kAuxPlaneBase + 6].SetAll();
223 }
224 // Plane kAuxPlaneBase + 7 is all ones to help NN find board edges.
225 result[kAuxPlaneBase + 7].SetAll();
226 if (stop_early) {
227 castlings = board.castlings();
228 }
229 }
230 bool skip_non_repeats =
231 input_format ==
232 pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_V2 ||
233 input_format == pblczero::NetworkFormat::
234 INPUT_112_WITH_CANONICALIZATION_V2_ARMAGEDDON;
235 bool flip = false;
236 int history_idx = history.size() - 1;
237 for (int i = 0; i < std::min(history_planes, kMoveHistory);
238 ++i, --history_idx) {
239 const Position& position = history[history_idx < 0 ? 0 : history_idx];
240 ChessBoard board = position.GetBoard();
241 if (flip) board.Mirror();
242 // Castling changes can't be repeated, so we can stop early.
243 if (stop_early && board.castlings().as_int() != castlings.as_int()) break;
244 // Enpassants can't be repeated, but we do need to always send the current
245 // position.
246 if (stop_early && history_idx != static_cast<int>(history.size()) - 1 &&
247 !board.en_passant().empty()) {
248 break;
249 }
250 // If en-passant is possible we know the previous move.
251 if (fill_empty_history == FillEmptyHistory::NO &&
252 (history_idx < -1 ||
253 (history_idx == -1 && board.en_passant().empty()))) {
254 break;
255 }
256 // Board may be flipped so compare with position.GetBoard().
257 if (history_idx < 0 && fill_empty_history == FillEmptyHistory::FEN_ONLY &&
258 position.GetBoard() == ChessBoard::kStartposBoard) {
259 break;
260 }
261 const int repetitions = position.GetRepetitions();
262 // Canonical v2 only writes an item if it is a repeat, unless its the most
263 // recent position.
264 if (skip_non_repeats && repetitions == 0 && i > 0) {
265 if (history_idx > 0) flip = !flip;
266 // If no capture no pawn is 0, the previous was start of game, capture or
267 // pawn push, so there can't be any more repeats that are worth
268 // considering.
269 if (position.GetRule50Ply() == 0) break;
270 // Decrement i so it remains the same as the history_idx decrements.
271 --i;
272 continue;
273 }
274
275 const int base = i * kPlanesPerBoard;
276 result[base + 0].mask = (board.ours() & board.pawns()).as_int();
277 result[base + 1].mask = (board.ours() & board.knights()).as_int();
278 result[base + 2].mask = (board.ours() & board.bishops()).as_int();
279 result[base + 3].mask = (board.ours() & board.rooks()).as_int();
280 result[base + 4].mask = (board.ours() & board.queens()).as_int();
281 result[base + 5].mask = (board.ours() & board.kings()).as_int();
282
283 result[base + 6].mask = (board.theirs() & board.pawns()).as_int();
284 result[base + 7].mask = (board.theirs() & board.knights()).as_int();
285 result[base + 8].mask = (board.theirs() & board.bishops()).as_int();
286 result[base + 9].mask = (board.theirs() & board.rooks()).as_int();
287 result[base + 10].mask = (board.theirs() & board.queens()).as_int();
288 result[base + 11].mask = (board.theirs() & board.kings()).as_int();
289
290 if (repetitions >= 1) result[base + 12].SetAll();
291
292 // If en passant flag is set, undo last pawn move by removing the pawn from
293 // the new square and putting into pre-move square.
294 if (history_idx < 0 && !board.en_passant().empty()) {
295 const auto idx = GetLowestBit(board.en_passant().as_int());
296 if (idx < 8) { // "Us" board
297 result[base + 0].mask +=
298 ((0x0000000000000100ULL - 0x0000000001000000ULL) << idx);
299 } else {
300 result[base + 6].mask +=
301 ((0x0001000000000000ULL - 0x0000000100000000ULL) << (idx - 56));
302 }
303 }
304 if (history_idx > 0) flip = !flip;
305 // If no capture no pawn is 0, the previous was start of game, capture or
306 // pawn push, so no need to go back further if stopping early.
307 if (stop_early && position.GetRule50Ply() == 0) break;
308 }
309 if (transform != NoTransform) {
310 // Transform all masks.
311 for (int i = 0; i <= kAuxPlaneBase + 4; i++) {
312 auto v = result[i].mask;
313 if (v == 0 || v == ~0ULL) continue;
314 if ((transform & FlipTransform) != 0) {
315 v = ReverseBitsInBytes(v);
316 }
317 if ((transform & MirrorTransform) != 0) {
318 v = ReverseBytesInBytes(v);
319 }
320 if ((transform & TransposeTransform) != 0) {
321 v = TransposeBitsInBytes(v);
322 }
323 result[i].mask = v;
324 }
325 }
326 if (transform_out) *transform_out = transform;
327 return result;
328}

330InputPlanes EncodePositionForNN(
331 pblczero::NetworkFormat::InputFormat input_format,
332 const PositionHistory& history, int history_planes,
333 FillEmptyHistory fill_empty_history, int* transform_out) {
334 return EncodePositionForNN(input_format, history.GetPositions(),
335 history_planes, fill_empty_history, transform_out);
336}

574const std::array kPackedIdxToNNIdx = []() {
575 std::array<uint16_t, 64 * 64 * 4> indices;
576 size_t idx = 0;
577 for (const char* move_str : kMoveStrs) {
578 std::string move(move_str);
579 uint16_t from = Square::Parse(move.substr(0, 2)).as_idx();
580 uint16_t to = Square::Parse(move.substr(2, 2)).as_idx();
581 uint16_t promotion = move.size() == 5 ? PieceType::Parse(move[4]).idx : 0;
582 uint16_t packed_idx = promotion * 64 * 64 + from * 64 + to;
583 indices[packed_idx] = idx++;
584 }
585 return indices;
586}();

588uint16_t MoveAsPackedInt(Move move) {
589 enum Masks : uint16_t {
590 // clang-format off
591 kFromToMask = 0b0000111111111111,
592 kPromotion = 0b0100000000000000,
593 kPieceMask = 0b0011000000000000,
594 // clang-format on
595 };
596 uint16_t val = move.raw_data();
597 return (val & kPromotion) ? (val & (kFromToMask | kPieceMask))
598 : (val & kFromToMask);
599}

601Square Transform(Square sq, int transform) {
602 File file = sq.file();
603 Rank rank = sq.rank();
604 if ((transform & (MirrorTransform | TransposeTransform)) != 0) rank.Flip();
605 if ((transform & (FlipTransform | TransposeTransform)) != 0) file.Flop();
606 return Square(file, rank);
607}

611uint16_t MoveToNNIndex(Move move, int transform) {
612 if (transform == 0) return kPackedIdxToNNIdx[MoveAsPackedInt(move)];
613 const Square from = Transform(move.from(), transform);
614 const Square to = Transform(move.to(), transform);
615 const Move transformed =
616 move.is_promotion() ? Move::WhitePromotion(from, to, move.promotion())
617 : Move::White(from, to);
618 return kPackedIdxToNNIdx[MoveAsPackedInt(transformed)];
619}

621Move MoveFromNNIndex(int idx, int transform) {
622 std::string m_str = kMoveStrs[idx];
623 auto from = Square::Parse(m_str.substr(0, 2));
624 auto to = Square::Parse(m_str.substr(2, 2));
625 if (transform != 0) {
626 int inv_transform;
627 if (transform & TransposeTransform) {
628 inv_transform = TransposeTransform;
629 if (transform & FlipTransform) inv_transform |= MirrorTransform;
630 if (transform & MirrorTransform) inv_transform |= FlipTransform;
631 } else {
632 inv_transform = transform;
633 }
634 to = Transform(to, inv_transform);
635 from = Transform(from, inv_transform);
636 }
637 return m_str.size() == 5
638 ? Move::WhitePromotion(from, to, PieceType::Parse(m_str[4]))
639 : Move::White(from, to);
640}
