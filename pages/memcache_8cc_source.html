45struct CachedValue {
46 float q;
47 float d;
48 float m;
49 uint8_t num_moves;
50 std::unique_ptr<float[]> p;
51};

53void CachedValueToEvalResult(const CachedValue& cv, const EvalResultPtr& ptr) {
54 if (ptr.d) *ptr.d = cv.d;
55 if (ptr.q) *ptr.q = cv.q;
56 if (ptr.m) *ptr.m = cv.m;
57 std::copy(cv.p.get(), cv.p.get() + ptr.p.size(), ptr.p.begin());
58}

100class MemCacheComputation : public BackendComputation {
101 public:
102 MemCacheComputation(std::unique_ptr<BackendComputation> wrapped_computation,
103 MemCache* memcache)
104 : wrapped_computation_(std::move(wrapped_computation)),
105 memcache_(memcache),
106 entries_(memcache->max_batch_size_) {}
107
108 private:
109 size_t UsedBatchSize() const override {
110 return wrapped_computation_->UsedBatchSize();
111 }
112 virtual AddInputResult AddInput(const EvalPosition& pos,
113 EvalResultPtr result) override {
114 assert(pos.legal_moves.size() == result.p.size() || result.p.empty());
115 const uint64_t hash = ComputeEvalPositionHash(pos);
116 {
117 HashKeyedCacheLock<CachedValue> lock(&memcache_->cache_, hash);
118 // Sometimes search queries NN without passing the legal moves. It is
119 // still cached in this case, but in subsequent queries we only return it
120 // if legal moves are not passed again. Otherwise check the size to guard
121 // against hash collisions.
122 if (lock.holds_value() &&
123 (pos.legal_moves.empty() ||
124 (lock->p && lock->num_moves == pos.legal_moves.size()))) {
125 CachedValueToEvalResult(**lock, result);
126 return AddInputResult::FETCHED_IMMEDIATELY;
127 }
128 }
129 size_t entry_idx = entries_.emplace_back(
130 Entry{hash, std::make_unique<CachedValue>(), result});
131 auto& value = entries_[entry_idx].value;
132 value->p.reset(pos.legal_moves.empty() ? nullptr
133 : new float[pos.legal_moves.size()]);
134 value->num_moves = pos.legal_moves.size();
135 return wrapped_computation_->AddInput(
136 pos, EvalResultPtr{&value->q, &value->d, &value->m,
137 value->p ? std::span<float>{value->p.get(),
138 pos.legal_moves.size()}
139 : std::span<float>{}});
140 }
141
142 virtual void ComputeBlocking() override {
143 wrapped_computation_->ComputeBlocking();
144 for (auto& entry : entries_) {
145 CachedValueToEvalResult(*entry.value, entry.result_ptr);
146 memcache_->cache_.Insert(entry.key, std::move(entry.value));
147 }
148 }
149
150 struct Entry {
151 uint64_t key;
152 std::unique_ptr<CachedValue> value;
153 EvalResultPtr result_ptr;
154 };
155
156 std::unique_ptr<BackendComputation> wrapped_computation_;
157 MemCache* memcache_;
158 AtomicVector<Entry> entries_;
159};

165std::optional<EvalResult> MemCache::GetCachedEvaluation(
166 const EvalPosition& pos) {
167 const uint64_t hash = ComputeEvalPositionHash(pos);
168 HashKeyedCacheLock<CachedValue> lock(&cache_, hash);
169 if (!lock.holds_value() ||
170 (!pos.legal_moves.empty() &&
171 !(lock->p && lock->num_moves == pos.legal_moves.size()))) {
172 return std::nullopt;
173 }
174 EvalResult result;
175 result.d = lock->d;
176 result.q = lock->q;
177 result.m = lock->m;
178 if (lock->p) {
179 result.p.reserve(pos.legal_moves.size());
180 std::copy(lock->p.get(), lock->p.get() + pos.legal_moves.size(),
181 std::back_inserter(result.p));
182 }
183 return result;
184}
