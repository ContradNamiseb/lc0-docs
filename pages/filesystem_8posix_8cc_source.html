44std::vector<std::string> GetFileList(const std::string& directory) {
45 std::vector<std::string> result;
46 DIR* dir = opendir(directory.c_str());
47 if (!dir) return result;
48 while (auto* entry = readdir(dir)) {
49 bool exists = false;
50 switch (entry->d_type) {
51 case DT_REG:
52 exists = true;
53 break;
54 case DT_LNK:
55 // check that the soft link actually points to a regular file.
56 const std::string filename = directory + "/" + entry->d_name;
57 struct stat s;
58 exists =
59 stat(filename.c_str(), &s) == 0 && (s.st_mode & S_IFMT) == S_IFREG;
60 break;
61 }
62 if (exists) result.push_back(entry->d_name);
63 }
64 closedir(dir);
65 return result;
66}

96std::string GetUserCacheDirectory() {
97#ifdef __APPLE__
98 constexpr auto kLocalDir = "Library/Caches/";
99#else
100 constexpr auto kLocalDir = ".cache/";
101 const char *xdg_cache_home = std::getenv("XDG_CACHE_HOME");
102 if (xdg_cache_home != NULL && CheckDir(xdg_cache_home)) {
103 return std::string(xdg_cache_home) + "/";
104 }
105#endif
106 const char *home = std::getenv("HOME");
107 if (home == NULL) return std::string();
108 const std::string path = std::string(home) + "/" + kLocalDir;
109 if (!CheckDir(path)) return std::string();
110 return path;
111}
