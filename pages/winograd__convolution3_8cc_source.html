59template <bool use_eigen>
60void WinogradConvolution3<use_eigen>::TransformIn(const size_t batch_size,
61 const float* input,
62 const size_t channels) {
63#ifndef USE_ISPC
64
65 static const size_t kCacheSize = 128;
66 float x[kWinogradAlpha][kWinogradAlpha];
67 float T1[kWinogradAlpha][kWinogradAlpha];
68 float R[16][kCacheSize];
69 for (size_t batch_index = 0; batch_index < batch_size; batch_index++) {
70 size_t channels_rem = channels;
71 const float* input_batch =
72 input + batch_index * kWidth * kHeight * channels;
73 float* V_batch = &V_[channels * kTiles * batch_index];
74 for (size_t channel_long = 0; channel_long < channels;
75 channel_long += kCacheSize) {
76 const size_t channel_step = std::min<size_t>(kCacheSize, channels_rem);
77 channels_rem -= channel_step;
78 for (int block_y = 0; block_y < kWtiles; block_y++) {
79 for (int block_x = 0; block_x < kWtiles; block_x++) {
80 // Tiles overlap by 2
81 const int yin = 2 * block_y - 1;
82 const int xin = 2 * block_x - 1;
83
84 for (size_t ch = 0; ch < channel_step; ++ch) {
85 const size_t channel = channel_long + ch;
86
87 const float* input_channel =
88 input_batch + channel * (kWidth * kHeight);
89 for (int i = 0; i < kWinogradAlpha; i++) {
90 for (int j = 0; j < kWinogradAlpha; j++) {
91 if ((yin + i) >= 0 && (xin + j) >= 0 && (yin + i) < kHeight &&
92 (xin + j) < kWidth) {
93 x[i][j] = input_channel[(yin + i) * kWidth + (xin + j)];
94 } else {
95 x[i][j] = 0.0f;
96 }
97 }
98 }
99
100 // Calculates transpose(B).x.B
101 // B = [[ 1.0, 0.0, 0.0, 0.0],
102 // [ 0.0, 1.0, -1.0, 1.0],
103 // [-1.0, 1.0, 1.0, 0.0],
104 // [ 0.0, 0.0, 0.0, -1.0]]
105
106 T1[0][0] = x[0][0] - x[2][0];
107 T1[0][1] = x[0][1] - x[2][1];
108 T1[0][2] = x[0][2] - x[2][2];
109 T1[0][3] = x[0][3] - x[2][3];
110 T1[1][0] = x[1][0] + x[2][0];
111 T1[1][1] = x[1][1] + x[2][1];
112 T1[1][2] = x[1][2] + x[2][2];
113 T1[1][3] = x[1][3] + x[2][3];
114 T1[2][0] = x[2][0] - x[1][0];
115 T1[2][1] = x[2][1] - x[1][1];
116 T1[2][2] = x[2][2] - x[1][2];
117 T1[2][3] = x[2][3] - x[1][3];
118 T1[3][0] = x[1][0] - x[3][0];
119 T1[3][1] = x[1][1] - x[3][1];
120 T1[3][2] = x[1][2] - x[3][2];
121 T1[3][3] = x[1][3] - x[3][3];
122
123 R[0][ch] = T1[0][0] - T1[0][2];
124 R[1][ch] = T1[0][1] + T1[0][2];
125 R[2][ch] = T1[0][2] - T1[0][1];
126 R[3][ch] = T1[0][1] - T1[0][3];
127 R[4][ch] = T1[1][0] - T1[1][2];
128 R[5][ch] = T1[1][1] + T1[1][2];
129 R[6][ch] = T1[1][2] - T1[1][1];
130 R[7][ch] = T1[1][1] - T1[1][3];
131 R[8][ch] = T1[2][0] - T1[2][2];
132 R[9][ch] = T1[2][1] + T1[2][2];
133 R[10][ch] = T1[2][2] - T1[2][1];
134 R[11][ch] = T1[2][1] - T1[2][3];
135 R[12][ch] = T1[3][0] - T1[3][2];
136 R[13][ch] = T1[3][1] + T1[3][2];
137 R[14][ch] = T1[3][2] - T1[3][1];
138 R[15][ch] = T1[3][1] - T1[3][3];
139 }
140 float* V_channel = V_batch + channel_long;
141 const auto V_incr = channels * kTiles * batch_size;
142 float* wTile_V = V_channel + channels * (block_y * kWtiles + block_x);
143 for (size_t i = 0; i < 16; ++i) {
144 for (size_t ch = 0; ch < channel_step; ++ch) {
145 wTile_V[ch] = R[i][ch];
146 }
147 wTile_V += V_incr;
148 }
149 }
150 }
151 }
152 }
153
154#else // USE_ISPC
155
156 ispc::winograd_TransformIn_ispc(batch_size, input, channels, &V_[0]);
157
158#endif // USE_ISPC

162template <>
163void WinogradConvolution3<false>::Sgemm(const size_t batch_size,
164 const float* weights,
165 const size_t input_channels,
166 const size_t output_channels) {
167#ifdef USE_MKL
168
169 /*
170 void cblas_sgemm_batch (const CBLAS_LAYOUT Layout, const CBLAS_TRANSPOSE*
171 transa_array, const CBLAS_TRANSPOSE* transb_array, const MKL_INT* m_array,
172 const MKL_INT* n_array, const MKL_INT* k_array, const float* alpha_array,
173 const float **a_array, const MKL_INT* lda_array, const float **b_array, const
174 MKL_INT* ldb_array, const float* beta_array, float **c_array, const MKL_INT*
175 ldc_array, const MKL_INT group_count, const MKL_INT* group_size);
176 */
177
178 CBLAS_TRANSPOSE transA = CblasNoTrans;
179 CBLAS_TRANSPOSE transB = CblasNoTrans;
180 MKL_INT m_array = output_channels;
181 MKL_INT n_array = batch_size * kTiles;
182 MKL_INT k_array = input_channels;
183 float alpha_array = 1.0;
184 const float* a_array[kWinogradTile];
185 MKL_INT lda_array = output_channels;
186 const float* b_array[kWinogradTile];
187 MKL_INT ldb_array = input_channels;
188 float* c_array[kWinogradTile];
189 MKL_INT ldc_array = output_channels;
190 float beta_array = 0.0;
191 MKL_INT groupSize = kWinogradTile;
192
193 for (auto b = 0; b < kWinogradTile; b++) {
194 auto offset_u = b * output_channels * input_channels;
195 auto offset_v = b * batch_size * input_channels * kTiles;
196 auto offset_m = b * batch_size * output_channels * kTiles;
197
198 a_array[b] = &weights[offset_u];
199 b_array[b] = &V_[offset_v];
200 c_array[b] = &M_[offset_m];
201 }
202
203 cblas_sgemm_batch(CblasColMajor, &transA, &transB, &m_array, &n_array,
204 &k_array, &alpha_array, a_array, &lda_array, b_array,
205 &ldb_array, &beta_array, c_array, &ldc_array, 1,
206 &groupSize);
207
208#else
209
210 for (size_t b = 0; b < kWinogradTile; b++) {
211 auto offset_u = b * output_channels * input_channels;
212
213 // In col major
214 //
215 // M = weights(T) x V
216 //
217 // cols tiles input_channels tiles
218 // rows output_channels output_channels input_channels
219
220 auto offset_v = b * batch_size * input_channels * kTiles;
221 auto offset_m = b * batch_size * output_channels * kTiles;
222 cblas_sgemm(CblasColMajor, // Row major format
223 CblasNoTrans, // A no trans
224 CblasNoTrans, // B no trans
225 (int)output_channels, // rows W, M
226 (int)(batch_size * kTiles), // cols V, M
227 (int)input_channels, // cols W, rows V
228 1.0f, // alpha
229 &weights[offset_u], // W
230 (int)output_channels, // ldW
231 &V_[offset_v], // V
232 (int)input_channels, 0.0f, // ldV
233 &M_[offset_m], // M
234 (int)output_channels); // ldM
235 }
236#endif
237}
238#endif

240template <>
241void WinogradConvolution3<true>::Sgemm(const size_t batch_size,
242 const float* weights,
243 const size_t input_channels,
244 const size_t output_channels) {
245 for (size_t b = 0; b < kWinogradTile; b++) {
246 auto offset_u = b * output_channels * input_channels;
247 auto offset_v = b * batch_size * input_channels * kTiles;
248 auto offset_m = b * batch_size * output_channels * kTiles;
249 auto C_mat = EigenMatrixMap<float>(&M_[offset_m], output_channels,
250 batch_size * kTiles);
251 C_mat.noalias() = ConstEigenMatrixMap<float>(
252 &weights[offset_u], output_channels, input_channels) *
253 ConstEigenMatrixMap<float>(&V_[offset_v], input_channels,
254 batch_size * kTiles);
255 }

258template <bool use_eigen>
259void WinogradConvolution3<use_eigen>::TransformOut(const size_t batch_size,
260 float* output,
261 const size_t channels) {
262#ifndef USE_ISPC
263
264 float m[kWinogradTile];
265
266 for (size_t batch_index = 0; batch_index < batch_size; batch_index++) {
267 const float* M_batch = &M_[channels * kTiles * batch_index];
268 float* output_batch = output + batch_index * kWidth * kHeight * channels;
269
270 for (size_t channel = 0; channel < channels; channel++) {
271 const float* M_channel = M_batch + channel;
272 float* output_channel = output_batch + channel * (kHeight * kWidth);
273
274 for (int block_x = 0; block_x < kWtiles; block_x++) {
275 for (int block_y = 0; block_y < kWtiles; block_y++) {
276 const auto x = 2 * block_x;
277 const auto y = 2 * block_y;
278
279 const auto b = block_y * kWtiles + block_x;
280 const float* M_wtile = M_channel + channels * b;
281 const auto M_incr = channels * kTiles * batch_size;
282
283 for (int wTile = 0; wTile < kWinogradTile; wTile++) {
284 m[wTile] = *M_wtile;
285 M_wtile += M_incr;
286 }
287
288 // Calculates transpose(A).temp_m.A
289 // A = [1.0, 0.0],
290 // [1.0, 1.0],
291 // [1.0, -1.0],
292 // [0.0, -1.0]]
293
294 auto o11 = m[0 * 4 + 0] + m[0 * 4 + 1] + m[0 * 4 + 2] + m[1 * 4 + 0] +
295 m[1 * 4 + 1] + m[1 * 4 + 2] + m[2 * 4 + 0] + m[2 * 4 + 1] +
296 m[2 * 4 + 2];
297
298 auto o12 = m[0 * 4 + 1] - m[0 * 4 + 2] - m[0 * 4 + 3] + m[1 * 4 + 1] -
299 m[1 * 4 + 2] - m[1 * 4 + 3] + m[2 * 4 + 1] - m[2 * 4 + 2] -
300 m[2 * 4 + 3];
301
302 auto o21 = m[1 * 4 + 0] + m[1 * 4 + 1] + m[1 * 4 + 2] - m[2 * 4 + 0] -
303 m[2 * 4 + 1] - m[2 * 4 + 2] - m[3 * 4 + 0] - m[3 * 4 + 1] -
304 m[3 * 4 + 2];
305
306 auto o22 = m[1 * 4 + 1] - m[1 * 4 + 2] - m[1 * 4 + 3] - m[2 * 4 + 1] +
307 m[2 * 4 + 2] + m[2 * 4 + 3] - m[3 * 4 + 1] + m[3 * 4 + 2] +
308 m[3 * 4 + 3];
309
310 output_channel[(y)*kWidth + (x)] = o11;
311 output_channel[(y)*kWidth + (x + 1)] = o12;
312 output_channel[(y + 1) * kWidth + (x)] = o21;
313 output_channel[(y + 1) * kWidth + (x + 1)] = o22;
314 }
315 }
316 }
317 }
318
319#else // USE_ISPC
320
321 ispc::winograd_TransformOut_ispc(batch_size, &M_[0], channels, output);
322
323#endif // USE_ISPC
324}
