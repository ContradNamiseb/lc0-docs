46static void CopyFloatToHalf(dx_half* out, const float* in, size_t elements) {
47 for (int i = 0; i < elements; i++) {
48 out[i] = FP32toFP16(in[i]);
49 }

52static void CpuTranspose(float* op, float* ip, size_t rows, size_t cols) {
53 for (size_t i = 0; i < rows; i++)
54 for (size_t j = 0; j < cols; j++) op[j * rows + i] = ip[i * cols + j];
55}

57template <int M, int N, int K>
58static void MatrixMulCPU(float* c, const float* a, const float* b) {
59 for (int i = 0; i < M; ++i)
60 for (int j = 0; j < N; ++j) {
61 float S = 0;
62 for (int k = 0; k < K; ++k) S += a[i * K + k] * b[k * N + j];
63 c[i * N + j] = S;
64 }
65}

67static void FilterTransform4x4(float* transformed_filter, const float* filter) {
68 // transform applied to filter (of size 3x3)
69 float G[6 * 3] = {1.0f / 4, 0, 0, -1.0f / 6, -1.0f / 6,
70 -1.0f / 6, -1.0f / 6, 1.0f / 6, -1.0f / 6, 1.0f / 24,
71 1.0f / 12, 1.0f / 6, 1.0f / 24, -1.0f / 12, 1.0f / 6,
72 0, 0, 1};
73
74 float Gt[3 * 6] = {1.0f / 4, -1.0f / 6, -1.0f / 6, 1.0f / 24, 1.0f / 24, 0,
75 0, -1.0f / 6, 1.0f / 6, 1.0f / 12, -1.0f / 12, 0,
76 0, -1.0f / 6, -1.0f / 6, 1.0f / 6, 1.0f / 6, 1};
77
78 float temp_filter[6 * 3];
79 MatrixMulCPU<6, 3, 3>(temp_filter, G, filter);
80 MatrixMulCPU<6, 6, 3>(transformed_filter, temp_filter, Gt);
81}

83#define FILTER_IDX_NCHW(k, c, h, w) ((k)*C * S * R + (c)*S * R + (h)*R + w)
84
85// Transform filter for winograd.
86// (e.g: for K C H W - 256x256x3x3, filter output is 6x6x256x256 - H W K C)
87static void TransformFilterTensor_Winograd4x4(int K, int C,
88 float* transformed_filter,
89 const float* weight) {
90 constexpr int S = 3;
91 constexpr int R = 3;
92
93 for (int k = 0; k < K; k++) {
94 for (int c = 0; c < C; c++) {
95 // 1. read single filter from memory
96 float filter_tile[3][3];
97 for (int s = 0; s < S; s++)
98 for (int r = 0; r < R; r++) {
99 filter_tile[s][r] = weight[FILTER_IDX_NCHW(k, c, s, r)];
100 }
101
102 // 2. transform it
103 float transformed_tile[6][6];
104 FilterTransform4x4(&(transformed_tile[0][0]), &(filter_tile[0][0]));
105
106 // 3. write it back to memory (in HWCK layout)
107 for (int i = 0; i < 6; i++)
108 for (int j = 0; j < 6; j++) {
109 transformed_filter[i * 6 * C * K + j * C * K + c * K + k] =
110 transformed_tile[i][j];
111 }
112 }
113 }

116static void GetGemmTensorDesc(TensorDesc* out_desc, int batch_size, int rows,
117 int cols, bool fp16) {
118 memset(out_desc, 0, sizeof(TensorDesc));
119 out_desc->DimensionCount = 4;
120 out_desc->DataType = fp16 ? 1 : 0;
121
122 out_desc->Size[0] = batch_size;
123 out_desc->Size[1] = 1;
124 out_desc->Size[2] = rows;
125 out_desc->Size[3] = cols;
126
127 // row-major by default
128 out_desc->Stride[3] = 1;
129 out_desc->Stride[2] = cols;
130 out_desc->Stride[1] = rows * cols;
131 out_desc->Stride[0] = rows * cols;
132
133 for (int i = 0; i < 4; i++) out_desc->StrideAlignment[i] = 1;
134
135 out_desc->BaseAlignmentInBytes = 4096; // arbitary
136 out_desc->PhysicalSizeInElements = batch_size * rows * cols;
137}

139static void GetConvTensorDesc(TensorDesc* out_desc, int N, int C, int H, int W,
140 bool fp16) {
141 memset(out_desc, 0, sizeof(TensorDesc));
142 out_desc->DimensionCount = 4;
143 out_desc->DataType = fp16 ? 1 : 0;
144
145 out_desc->Size[0] = N;
146 out_desc->Size[1] = C;
147 out_desc->Size[2] = H;
148 out_desc->Size[3] = W;
149
150 // NCHW layout
151 out_desc->Stride[3] = 1;
152 out_desc->Stride[2] = W;
153 out_desc->Stride[1] = H * W;
154 out_desc->Stride[0] = C * H * W;
155
156 for (int i = 0; i < 4; i++) out_desc->StrideAlignment[i] = 1;
157
158 out_desc->BaseAlignmentInBytes = 4096; // arbitary
159 out_desc->PhysicalSizeInElements = N * C * H * W;
160}

164GemmMetaCommand::GemmMetaCommand(DxContext* dx_context, int rows, int cols,
165 int K, int gemm_batch, bool fp16,
166 bool a_transpose, bool b_transpose) {
167 memset(scratch_data_persistent_, 0, sizeof(scratch_data_persistent_));
168 memset(scratch_data_temporary_, 0, sizeof(scratch_data_temporary_));
169 memset(meta_commands_, 0, sizeof(meta_commands_));
170
171 // Note: the way GEMM is used, the 'rows'/M - dimension is a function of
172 // batch size. gemm_batch is different and unrelated (either 36 for Winograd,
173 // or 1 for other FC layers)
174 int num_meta_commands = 1;
175 if (rows == 0) {
176 // Create metacommands for each 'rows' that is multiple of 8.
177 num_meta_commands = kMaxMetacommands;
178 rows_known_ = false;
179 } else {
180 rows_known_ = true;
181 }
182
183 for (int i = 0; i < num_meta_commands; i++) {
184 int num_rows = rows ? rows : (i + 1) * kMetacommandGranulity;
185
186 GemmCreateDesc createDesc = {};
187 GetGemmTensorDesc(&createDesc.DescOut, gemm_batch, num_rows, cols, fp16);
188 GetGemmTensorDesc(&createDesc.DescA, gemm_batch, a_transpose ? K : num_rows,
189 a_transpose ? num_rows : K, fp16);
190 GetGemmTensorDesc(&createDesc.DescB, gemm_batch, b_transpose ? cols : K,
191 b_transpose ? K : cols, fp16);
192 createDesc.cMatrixNull = 1;
193 createDesc.ActivationIsNull = 1;
194 createDesc.Alpha = 1.0;
195 createDesc.Beta = 0.0;
196 createDesc.Precision = fp16 ? 1 : 0; // 0 - fp32, 1 - fp16
197 createDesc.TransA = a_transpose;
198 createDesc.TransB = b_transpose;
199
200 ID3D12MetaCommand* pMetacommand = nullptr;
201 HRESULT hr = dx_context->getDevice()->CreateMetaCommand(
202 GemmGuid, 1, &createDesc, sizeof(createDesc),
203 IID_PPV_ARGS(&pMetacommand));
204
205 if (hr != S_OK) {
206#ifdef DEBUG_DUMP_PER_LAYER_DATA
207 printf(
208 "\nCan't create GEMM Metacommand for "
209 "rows: %d, cols: %d, K: %d, batch: "
210 "%d\n",
211 num_rows, cols, K, gemm_batch);
212#endif
213 create_succeeded_ = false;
214 return;
215 }
216
217 meta_commands_[i] = pMetacommand;
218
219 size_t persistent_size = pMetacommand->GetRequiredParameterResourceSize(
220 D3D12_META_COMMAND_PARAMETER_STAGE_EXECUTION, 4);
221 size_t temp_size = pMetacommand->GetRequiredParameterResourceSize(
222 D3D12_META_COMMAND_PARAMETER_STAGE_EXECUTION, 5);
223
224 if (persistent_size) {
225 dx_context->CreateAlloc(persistent_size, D3D12_HEAP_TYPE_DEFAULT,
226 scratch_data_persistent_[i], fp16);
227 }
228
229 if (temp_size) {
230 dx_context->CreateAlloc(temp_size, D3D12_HEAP_TYPE_DEFAULT,
231 scratch_data_temporary_[i], fp16);
232 }
233
234 GemmInitDesc initDesc = {};
235 initDesc.PersistentResource =
236 scratch_data_persistent_[i].desc_handle_scalar;
237 initDesc.TemporaryResource = scratch_data_temporary_[i].desc_handle_scalar;
238
239 dx_context->getCommandList()->InitializeMetaCommand(
240 meta_commands_[i], &initDesc, sizeof(initDesc));
241 }
242
243 create_succeeded_ = true;
244}

246void GemmMetaCommand::PerformGemm(int rows, DXAlloc A, DXAlloc B,
247 DXAlloc output,
248 ID3D12GraphicsCommandList4* command_list) {
249 if (!create_succeeded_) throw Exception("Metacommand not created");
250
251 int index = 0;
252 if (!rows_known_) {
253 index = DivUp(rows, 8) - 1;
254 }
255
256 ID3D12MetaCommand* meta_command = meta_commands_[index];
257 DXAlloc& scratch_persistent = scratch_data_persistent_[index];
258 DXAlloc& scratch_temporary = scratch_data_temporary_[index];
259
260 GemmExecuteDesc exec_desc = {};
261 exec_desc.AResource = A.desc_handle_scalar;
262 exec_desc.BResource = B.desc_handle_scalar;
263 exec_desc.OutputResource = output.desc_handle_scalar;
264 exec_desc.PersistentResource = scratch_persistent.desc_handle_scalar;
265 exec_desc.TemporaryResource = scratch_temporary.desc_handle_scalar;
266
267 command_list->ExecuteMetaCommand(meta_command, &exec_desc, sizeof(exec_desc));
268}

280ConvMetaCommand::ConvMetaCommand(DxContext* dx_context, int C, int K, int H,
281 int W, int F, bool relu, bool bias,
282 bool fp16) {
283 memset(scratch_data_persistent_, 0, sizeof(scratch_data_persistent_));
284 memset(scratch_data_temporary_, 0, sizeof(scratch_data_temporary_));
285 memset(meta_commands_, 0, sizeof(meta_commands_));
286
287 for (int i = 0; i < kMaxMetacommands; i++) {
288 int n = (i + 1) * kMetacommandGranulity;
289
290 ConvCreateDesc createDesc = {};
291 GetConvTensorDesc(&createDesc.InputDesc, n, C, H, W, fp16);
292 GetConvTensorDesc(&createDesc.OutputDesc, n, K, H, W, fp16);
293 GetConvTensorDesc(&createDesc.FilterDesc, K, C, F, F, fp16);
294 GetConvTensorDesc(&createDesc.BiasDesc, K, 1, 1, 1, fp16);
295 createDesc.BiasNull = bias ? 0 : 1;
296 createDesc.Mode = 1; // 1 is for cross-correlation (0 - conv)
297
298 createDesc.Direction = 0; // forward
299 createDesc.DimensionCount = 2; // 2D conv
300 createDesc.Stride[0] = 1;
301 createDesc.Stride[1] = 1;
302 createDesc.Dilation[0] = 1;
303 createDesc.Dilation[1] = 1;
304
305 int pad = (F - 1) / 2;
306 createDesc.StartPadding[0] = pad;
307 createDesc.StartPadding[1] = pad;
308 createDesc.EndPadding[0] = pad;
309 createDesc.EndPadding[1] = pad;
310 createDesc.GroupCount = 1;
311 if (relu) {
312 createDesc.ActivationFunction = 9; // relu (guess?)
313 createDesc.ActivationIsNull = 0;
314 } else {
315 createDesc.ActivationIsNull = 1;
316 }
317 createDesc.Precision = fp16 ? 1 : 0; // 0 - fp32, 1 - fp16
318
319 ID3D12MetaCommand* pMetacommand = nullptr;
320 HRESULT hr = dx_context->getDevice()->CreateMetaCommand(
321 ConvGuid, 1, &createDesc, sizeof(createDesc),
322 IID_PPV_ARGS(&pMetacommand));
323
324 if (hr != S_OK) {
325#ifdef DEBUG_DUMP_PER_LAYER_DATA
326 printf(
327 "\nCan't create Conv Metacommand for "
328 "N, C, K, H, W, f: %d %d %d %d %d %d\n",
329 n, C, K, H, W, F);
330#endif
331 create_succeeded_ = false;
332 return;
333 }
334
335 meta_commands_[i] = pMetacommand;
336
337 size_t persistent_size = pMetacommand->GetRequiredParameterResourceSize(
338 D3D12_META_COMMAND_PARAMETER_STAGE_EXECUTION, 4);
339 size_t temp_size = pMetacommand->GetRequiredParameterResourceSize(
340 D3D12_META_COMMAND_PARAMETER_STAGE_EXECUTION, 5);
341
342 if (persistent_size) {
343 dx_context->CreateAlloc(persistent_size, D3D12_HEAP_TYPE_DEFAULT,
344 scratch_data_persistent_[i], fp16);
345 }
346
347 if (temp_size) {
348 dx_context->CreateAlloc(temp_size, D3D12_HEAP_TYPE_DEFAULT,
349 scratch_data_temporary_[i], fp16);
350 }
351
352 InitConvDesc initDesc = {};
353 initDesc.PersistentResource =
354 scratch_data_persistent_[i].desc_handle_scalar;
355 initDesc.TemporaryResource = scratch_data_temporary_[i].desc_handle_scalar;
356
357 dx_context->getCommandList()->InitializeMetaCommand(
358 meta_commands_[i], &initDesc, sizeof(initDesc));
359 }
360 use_bias_ = bias;
361 create_succeeded_ = true;
362}

402 BaseLayer* ip, int C, int H, int W, int filter, int Cin,

419 b1_(),
420 b2_() {

435 }
436
437 if (has_se_) {
438 const size_t num_weights1 = C * se_k_;
439 const size_t num_weights2 = num_weights1 * 2;
440 const size_t num_biases1 = se_k_;
441 const size_t num_biases2 = 2 * C;
442
443 const size_t weight_size1 = element_size * num_weights1;
444 const size_t weight_size2 = element_size * num_weights2;
445 const size_t biases_size1 = element_size * num_biases1;
446 const size_t biases_size2 = element_size * num_biases2;
447
448 dx_context->CreateAlloc(weight_size1, D3D12_HEAP_TYPE_DEFAULT, w1_, fp16);
449 dx_context->CreateAlloc(weight_size2, D3D12_HEAP_TYPE_DEFAULT, w2_, fp16);
450 dx_context->CreateAlloc(biases_size1, D3D12_HEAP_TYPE_DEFAULT, b1_, fp16);
451 dx_context->CreateAlloc(biases_size2, D3D12_HEAP_TYPE_DEFAULT, b2_, fp16);
452 }
453
454 shader_wrapper_ = dx_context->getShaderWrapper();
455}

457void ConvLayer::LoadWeights(float* cpu_filter, float* cpu_bias,
458 DxContext* dx_context) {
459 int num_weights = c_input_ * C * filter_size_ * filter_size_;
460 size_t element_size = fp16_ ? sizeof(dx_half) : sizeof(float);
461 size_t weight_size = element_size * num_weights;
462 size_t bias_size = element_size * C;
463
464 std::vector<dx_half> temp(num_weights);
465 if (fp16_) {
466 CopyFloatToHalf(temp.data(), cpu_filter, num_weights);
467 dx_context->ScheduleUpload(weights_, temp.data(), weight_size);
468 } else {
469 dx_context->ScheduleUpload(weights_, cpu_filter, weight_size);
470 }
471
472 if (cpu_bias) {
473 if (fp16_) {
474 CopyFloatToHalf(temp.data(), cpu_bias, C);
475 dx_context->ScheduleUpload(biases_, temp.data(), bias_size);
476 } else {
477 dx_context->ScheduleUpload(biases_, cpu_bias, bias_size);
478 }
479 }
480
481 if (filter_size_ == 3) {
482 std::vector<float> temp_transformed(num_weights * 4);
483 TransformFilterTensor_Winograd4x4(C, c_input_, temp_transformed.data(),
484 cpu_filter);
485 if (fp16_) {
486 std::vector<dx_half> temp_transformed_half(num_weights * 4);
487 CopyFloatToHalf(temp_transformed_half.data(), temp_transformed.data(),
488 num_weights * 4);
489 dx_context->ScheduleUpload(transformed_weights_,
490 temp_transformed_half.data(), weight_size * 4);
491 } else {
492 dx_context->ScheduleUpload(transformed_weights_, temp_transformed.data(),
493 weight_size * 4);
494 }
495 }
496}

498void ConvLayer::LoadSEWeights(float* w1, float* b1, float* w2, float* b2) {
499 size_t element_size = fp16_ ? sizeof(dx_half) : sizeof(float);
500 const size_t num_weights1 = C * se_k_;
501 const size_t num_weights2 = num_weights1 * 2;
502 const size_t weight_size1 = element_size * num_weights1;
503 const size_t weight_size2 = element_size * num_weights2;
504
505 const size_t num_biases1 = se_k_;
506 const size_t biases_size1 = element_size * num_biases1;
507 const size_t num_biases2 = 2 * C;
508 const size_t biases_size2 = element_size * num_biases2;
509
510 // The shader uses transposed weight matrices.
511
512 std::vector<float> temp_transposed(num_weights2);
513 std::vector<dx_half> temp_half(num_weights2);
514
515 CpuTranspose(temp_transposed.data(), w1, se_k_, C);
516 if (fp16_) {
517 CopyFloatToHalf(temp_half.data(), temp_transposed.data(), num_weights1);
518 dx_context_->ScheduleUpload(w1_, temp_half.data(), weight_size1);
519 } else {
520 dx_context_->ScheduleUpload(w1_, temp_transposed.data(), weight_size1);
521 }
522
523 CpuTranspose(temp_transposed.data(), w2, 2 * C, se_k_);
524 if (fp16_) {
525 CopyFloatToHalf(temp_half.data(), temp_transposed.data(), num_weights2);
526 dx_context_->ScheduleUpload(w2_, temp_half.data(), weight_size2);
527 } else {
528 dx_context_->ScheduleUpload(w2_, temp_transposed.data(), weight_size2);
529 }
530
531 if (fp16_) {
532 CopyFloatToHalf(temp_half.data(), b1, num_biases1);
533 dx_context_->ScheduleUpload(b1_, temp_half.data(), biases_size1);
534 } else {
535 dx_context_->ScheduleUpload(b1_, b1, biases_size1);
536 }
537
538 if (fp16_) {
539 CopyFloatToHalf(temp_half.data(), b2, num_biases2);
540 dx_context_->ScheduleUpload(b2_, temp_half.data(), biases_size2);
541 } else {
542 dx_context_->ScheduleUpload(b2_, b2, biases_size2);
543 }
544}

546void ConvLayer::Eval(int N, DXAlloc output, DXAlloc input, DXAlloc input2,
547 DXAlloc scratch, DXAlloc scratch2,
548 ID3D12GraphicsCommandList4* command_list) {
549 // Use winograd for filter size of 3, when GEMM metacommand is available,
550 // Or when GEMM metacommand isn't available but Convolution metacommand is
551 // also not available (compute shader matrix multiply path).
552 bool useWinograd =
553 (filter_size_ == 3) &&
554 ((meta_command_gemm_ && meta_command_gemm_->IsAvailable()) ||
555 !meta_command_conv_ || !meta_command_conv_->IsAvailable());
556
557 if (useWinograd) {
558 // Need to pad up the input to gemm too (i.e, the transformed Input tensor)!
559 // It's in HWNC layout, and 'N'/GemmN needs to be padded up (HW = 6x6)
560 // to make it simple, just pad up N to multiple of 2 here (so that gemmN is
561 // multiple of 8).
562 // TODO: figure out why padding up by 4 is needed (instead of 2!)
563 // N = ((N + 1) / 2) * 2;
564 N = ((N + 3) / 4) * 4;
565
566 // 1. Input transform (input->scratch)
567 shader_wrapper_->InputTransform(command_list, scratch, input, N, c_input_,
568 fp16_);
569
570 dx_context_->UavBarrier(command_list);
571
572 // 2. Gemm (scratch -> scratch2)
573 if (meta_command_gemm_ && meta_command_gemm_->IsAvailable())
574 meta_command_gemm_->PerformGemm(N * 4, scratch, transformed_weights_,
575 scratch2, command_list);
576 else
577 shader_wrapper_->MatrixMultiply(command_list, scratch2, scratch,
578 transformed_weights_, N * 4, C, c_input_,
579 36, fp16_);
580
581 dx_context_->UavBarrier(command_list);
582
583 // 3. Output transform (scratch2 -> output)
584 shader_wrapper_->OutputTransform(
585 command_list, output, scratch2, input2, biases_, w1_, b1_, w2_, b2_, N,
586 C, use_relu_, use_bias_, skip_add_, has_se_, se_k_, fp16_);
587
588 } else if (meta_command_conv_ && meta_command_conv_->IsAvailable()) {
589 if (skip_add_ || has_se_)
590 meta_command_conv_->PerformConv(N, input, weights_, biases_, scratch,
591 command_list);
592 else
593 meta_command_conv_->PerformConv(N, input, weights_, biases_, output,
594 command_list);
595 if (has_se_) {
596 dx_context_->UavBarrier(command_list);
597 shader_wrapper_->Se(command_list, output, scratch, input2, biases_, w1_,
598 b1_, w2_, b2_, N, C, use_relu_, false, skip_add_,
599 se_k_, fp16_);
600 } else if (skip_add_) {
601 // Need seperate pass for skip connection addition as Metacommand API
602 // doesn't allow it to be fused with convolution.
603 dx_context_->UavBarrier(command_list);
604 shader_wrapper_->AddVectors(command_list, output, scratch, input2,
605 N * C * H * W, N * C * H * W, N * C * H * W,
606 use_relu_, false, fp16_);
607 }
608 } else if (filter_size_ == 1) {
609 shader_wrapper_->Conv1x1(command_list, output, input, weights_, biases_, N,
610 c_input_, C, use_relu_, use_bias_, fp16_);
611 } else {
612 throw Exception("Unsupported filter shape for convolution! ");
613 }
614}

627FCLayer::FCLayer(bool fp16, DxContext* dx_context, BaseLayer* ip, int C, int H,
628 int W, bool bias, bool relu, bool tanh)
629 : BaseLayer(C, H, W, ip, dx_context, fp16),
630 use_bias_(bias),
631 use_relu_(relu),
632 meta_command_(),
633 use_tanh_(tanh) {
634 size_t element_size = fp16_ ? sizeof(dx_half) : sizeof(float);
635 size_t weight_size =
636 element_size * C * H * W * ip->GetC() * ip->GetH() * ip->GetW();
637 size_t blas_size = element_size * C * H * W;
638
639 dx_context->CreateAlloc(weight_size, D3D12_HEAP_TYPE_DEFAULT, weights_, fp16);
640 if (use_bias_)
641 dx_context->CreateAlloc(blas_size, D3D12_HEAP_TYPE_DEFAULT, biases_, fp16);
642
643 shader_wrapper_ = dx_context->getShaderWrapper();
644
645 // Create metacommand object
646 int rows = 0; // batch size
647 int cols = C * H * W; // cols of the output matrix
648 int K = ip->GetC() * ip->GetH() * ip->GetW(); // cols of input matrix
649 // We do Out = A * weight.
650 // The weight matrix need to be transpsoed before it can be multiplied.
651 // The transpose is done on CPU when loading weights
652 meta_command_ = std::make_unique<GemmMetaCommand>(dx_context, rows, cols, K,
653 1, fp16, false, false);
654}

656void FCLayer::LoadWeights(float* cpuWeight, float* cpuBias,
657 DxContext* dx_context) {
658 size_t rows = C * H * W;
659 size_t cols = input_->GetC() * input_->GetH() * input_->GetW();
660 size_t num_weights = rows * cols;
661
662 size_t element_size = fp16_ ? sizeof(dx_half) : sizeof(float);
663 size_t weight_size = element_size * num_weights;
664 size_t num_biases = C * H * W;
665 size_t bias_size = element_size * num_biases;
666
667 std::vector<float> temp_transposed(num_weights);
668 CpuTranspose(temp_transposed.data(), cpuWeight, rows, cols);
669 std::vector<dx_half> temp(num_weights);
670 if (fp16_) {
671 CopyFloatToHalf(temp.data(), temp_transposed.data(), num_weights);
672 dx_context->ScheduleUpload(weights_, temp.data(), weight_size);
673 } else {
674 dx_context->ScheduleUpload(weights_, temp_transposed.data(), weight_size);
675 }
676
677 if (cpuBias) {
678 if (fp16_) {
679 CopyFloatToHalf(temp.data(), cpuBias, C);
680 dx_context->ScheduleUpload(biases_, temp.data(), bias_size);
681 } else {
682 dx_context->ScheduleUpload(biases_, cpuBias, bias_size);
683 }
684 }
685}

687void FCLayer::Eval(int N, DXAlloc output, DXAlloc input, DXAlloc /*input2*/,
688 DXAlloc /*scratch*/, DXAlloc /*scratch2*/,
689 ID3D12GraphicsCommandList4* command_list) {
690 int num_outputs = C * H * W;
691 int num_inputs = input_->GetC() * input_->GetH() * input_->GetW();
692
693 if (meta_command_->IsAvailable())
694 meta_command_->PerformGemm(N, input, weights_, output, command_list);
695 else
696 shader_wrapper_->MatrixMultiply(command_list, output, input, weights_,
697 DivUp(N, 8) * 8, num_outputs, num_inputs, 1,
698 fp16_);
699
700 if (use_bias_ || use_relu_ || use_tanh_) {
701 dx_context_->UavBarrier(command_list);
702 shader_wrapper_->AddVectors(command_list, output, output, biases_,
703 N * num_outputs, N * num_outputs, num_outputs,
704 use_relu_, use_tanh_, fp16_);
705 }
706}
