46uint64_t DxContext::FlushCL(ID3D12GraphicsCommandList4* cl) {
47 if (!cl) cl = command_list_;
48 cl->Close();
49 command_queue_->ExecuteCommandLists(1, (ID3D12CommandList**)&cl);
50 command_queue_->Signal(fence_, ++fence_val_);
51 return fence_val_;
52}

54void DxContext::WaitForGpu(uint64_t fence_val) {
55 if (!fence_val) fence_val = fence_val_;
56 // Wait for commands to finish on GPU.
57 // (spinloop has lowest latency, we can try event based signal if CPU
58 // overhead becomes a bottleneck).
59 while (fence_->GetCompletedValue() < fence_val)
60 ;
61 upload_scratch_mem_.offset = 0;
62}

64void DxContext::ResetCL(ID3D12GraphicsCommandList4* cl,
65 ID3D12CommandAllocator* ca, bool reset) {
66 if (!cl) cl = command_list_;
67 if (!ca) ca = command_allocator_;
68 if (reset) {
69 ca->Reset();
70 cl->Reset(ca, NULL);
71 }
72 cl->SetDescriptorHeaps(1, &desc_heap_);
73}

75void DxContext::FlushAndWait() {
76 FlushCL();
77 WaitForGpu();
78 ResetCL();

96void DxContext::CopyTensor(DXAlloc dst, DXAlloc src, int bytes) {
97 CD3DX12_RESOURCE_BARRIER barrier;
98
99 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
100 src.resource, D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
101 D3D12_RESOURCE_STATE_COPY_SOURCE);
102 command_list_->ResourceBarrier(1, &barrier);
103
104 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
105 dst.resource, D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
106 D3D12_RESOURCE_STATE_COPY_DEST);
107 command_list_->ResourceBarrier(1, &barrier);
108
109 command_list_->CopyBufferRegion(dst.resource, dst.offset, src.resource,
110 src.offset, bytes);
111
112 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
113 src.resource, D3D12_RESOURCE_STATE_COPY_SOURCE,
114 D3D12_RESOURCE_STATE_UNORDERED_ACCESS);
115 command_list_->ResourceBarrier(1, &barrier);
116
117 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
118 dst.resource, D3D12_RESOURCE_STATE_COPY_DEST,
119 D3D12_RESOURCE_STATE_UNORDERED_ACCESS);
120 command_list_->ResourceBarrier(1, &barrier);
121}

166DxContext::DxContext(const OptionsDict& options) {
167 gpu_id_ = options.GetOrDefault<int>("gpu", 0);
168
169 IDXGIFactory4* pFactory = nullptr;
170 IDXGIAdapter* pAdapter = nullptr;
171 ReportDxErrors(CreateDXGIFactory2(0, IID_PPV_ARGS(&pFactory)));
172 ReportDxErrors(pFactory->EnumAdapters(gpu_id_, &pAdapter));
173 pFactory->Release();
174
175 if (!pAdapter) throw Exception("Invalid GPU Id: " + std::to_string(gpu_id_));
176
177 ReportDxErrors(D3D12CreateDevice(pAdapter, D3D_FEATURE_LEVEL_11_0,
178 IID_PPV_ARGS(&device_)));
179 pAdapter->Release();
180
181 D3D12_COMMAND_QUEUE_DESC commandqueueDesc;
182 commandqueueDesc.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
183 commandqueueDesc.NodeMask = 0;
184 commandqueueDesc.Priority = 0;
185 commandqueueDesc.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
186 ReportDxErrors(device_->CreateCommandQueue(&commandqueueDesc,
187 IID_PPV_ARGS(&command_queue_)));
188
189 ReportDxErrors(device_->CreateCommandAllocator(
190 D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&command_allocator_)));
191
192 ReportDxErrors(device_->CreateCommandList(1, D3D12_COMMAND_LIST_TYPE_DIRECT,
193 command_allocator_, NULL,
194 IID_PPV_ARGS(&command_list_)));
195
196 D3D12_DESCRIPTOR_HEAP_DESC heapDesc = {};
197 heapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
198 heapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
199 heapDesc.NumDescriptors = kNumDescHeapSlots;
200 ReportDxErrors(
201 device_->CreateDescriptorHeap(&heapDesc, IID_PPV_ARGS(&desc_heap_)));
202
203 command_list_->SetDescriptorHeaps(1, &desc_heap_);
204
205 next_slot_in_desc_heap_ = 0;
206
207 fence_val_ = 0ull;
208 ReportDxErrors(device_->CreateFence(fence_val_, D3D12_FENCE_FLAG_NONE,
209 IID_PPV_ARGS(&fence_)));
210
211 shader_wrapper_.Init(device_);
212
213 // Allocate scratch space for uploads and read-back.
214 CreateAlloc(kUploadDownloadScratchSize, D3D12_HEAP_TYPE_UPLOAD,
215 upload_scratch_mem_, false);
216 CreateAlloc(kUploadDownloadScratchSize, D3D12_HEAP_TYPE_READBACK,
217 readback_scratch_mem_, false);
218}

236void DxContext::CreateAlloc(size_t size, D3D12_HEAP_TYPE type, DXAlloc& alloc,
237 bool fp16) {
238 // some alignment
239 int factor = DivUp((int)size, 4);
240 size = factor * 4;
241
242 D3D12_HEAP_PROPERTIES heapDesc = {};
243 heapDesc.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
244 heapDesc.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
245 heapDesc.CreationNodeMask = 1;
246 heapDesc.VisibleNodeMask = 1;
247
248 if (type == D3D12_HEAP_TYPE_CUSTOM) {
249 // Use custom heap type to allow GPU writing to system memory directly
250 heapDesc.MemoryPoolPreference = D3D12_MEMORY_POOL_L0;
251 heapDesc.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_WRITE_BACK;
252 }
253
254 heapDesc.Type = type;
255
256 D3D12_RESOURCE_DESC bufferDesc = {};
257 bufferDesc.MipLevels = 1;
258 bufferDesc.Format = DXGI_FORMAT_UNKNOWN;
259 bufferDesc.Height = 1;
260 if (type == D3D12_HEAP_TYPE_DEFAULT || type == D3D12_HEAP_TYPE_CUSTOM)
261 bufferDesc.Flags = D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS;
262 bufferDesc.DepthOrArraySize = 1;
263 bufferDesc.SampleDesc.Count = 1;
264 bufferDesc.SampleDesc.Quality = 0;
265 bufferDesc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
266 bufferDesc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
267
268 D3D12_RESOURCE_STATES resourceState = D3D12_RESOURCE_STATE_UNORDERED_ACCESS;
269 if (type == D3D12_HEAP_TYPE_UPLOAD)
270 resourceState = D3D12_RESOURCE_STATE_GENERIC_READ;
271 else if (type == D3D12_HEAP_TYPE_READBACK)
272 resourceState = D3D12_RESOURCE_STATE_COPY_DEST;
273
274 bufferDesc.Width = size;
275 ReportDxErrors(device_->CreateCommittedResource(
276 &heapDesc, D3D12_HEAP_FLAG_NONE, &bufferDesc, resourceState, nullptr,
277 IID_PPV_ARGS(&alloc.resource)));
278
279 alloc.offset = 0;
280 alloc.gpu_va = alloc.resource->GetGPUVirtualAddress();
281
282 // Create desc heap entries for UAV resources.
283 if (resourceState == D3D12_RESOURCE_STATE_UNORDERED_ACCESS) {
284 int handleIncrementSize = device_->GetDescriptorHandleIncrementSize(
285 D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
286
287 size_t element_size = fp16 ? sizeof(dx_half) : sizeof(float);
288
289 // Scalar UAV.
290 {
291 int slot = next_slot_in_desc_heap_++;
292
293 CD3DX12_CPU_DESCRIPTOR_HANDLE cpuDescHandle(
294 desc_heap_->GetCPUDescriptorHandleForHeapStart(), slot,
295 handleIncrementSize);
296
297 CD3DX12_GPU_DESCRIPTOR_HANDLE gpuDescHandle(
298 desc_heap_->GetGPUDescriptorHandleForHeapStart(), slot,
299 handleIncrementSize);
300
301 D3D12_UNORDERED_ACCESS_VIEW_DESC uavDesc = {};
302 uavDesc.ViewDimension = D3D12_UAV_DIMENSION_BUFFER;
303 uavDesc.Format = fp16 ? DXGI_FORMAT_R16_FLOAT : DXGI_FORMAT_R32_FLOAT;
304 uavDesc.Buffer.FirstElement = 0;
305 uavDesc.Buffer.NumElements = (UINT)(size / element_size);
306
307 device_->CreateUnorderedAccessView(alloc.resource, nullptr, &uavDesc,
308 cpuDescHandle);
309
310 alloc.desc_handle_scalar = gpuDescHandle;
311 }
312
313 // 4-component vector UAV.
314 {
315 int slot = next_slot_in_desc_heap_++;
316
317 CD3DX12_CPU_DESCRIPTOR_HANDLE cpuDescHandle(
318 desc_heap_->GetCPUDescriptorHandleForHeapStart(), slot,
319 handleIncrementSize);
320
321 CD3DX12_GPU_DESCRIPTOR_HANDLE gpuDescHandle(
322 desc_heap_->GetGPUDescriptorHandleForHeapStart(), slot,
323 handleIncrementSize);
324
325 D3D12_UNORDERED_ACCESS_VIEW_DESC uavDesc = {};
326 uavDesc.ViewDimension = D3D12_UAV_DIMENSION_BUFFER;
327 uavDesc.Format = fp16 ? DXGI_FORMAT_R16G16B16A16_FLOAT
328 : DXGI_FORMAT_R32G32B32A32_FLOAT;
329 uavDesc.Buffer.FirstElement = 0;
330 uavDesc.Buffer.NumElements = (UINT)(size / (4 * element_size));
331
332 device_->CreateUnorderedAccessView(alloc.resource, nullptr, &uavDesc,
333 cpuDescHandle);
334
335 alloc.desc_handle_vector = gpuDescHandle;
336 }
337 }
338}

340void DxContext::ScheduleUpload(DXAlloc alloc, const void* data, size_t size) {
341 // Make sure enough space is available in the upload scratch buffer
342 assert(size <= kUploadDownloadScratchSize);
343 if (upload_scratch_mem_.offset + size > kUploadDownloadScratchSize)
344 FlushAndWait();
345
346 uint8_t* temp;
347 upload_scratch_mem_.resource->Map(0, nullptr, (void**)&temp);
348
349 dx_half* cpuPtr = (dx_half*)(temp + upload_scratch_mem_.offset);
350 memcpy(cpuPtr, data, size);
351
352 CD3DX12_RESOURCE_BARRIER barrier;
353
354 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
355 alloc.resource, D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
356 D3D12_RESOURCE_STATE_COPY_DEST);
357 command_list_->ResourceBarrier(1, &barrier);
358
359 command_list_->CopyBufferRegion(alloc.resource, alloc.offset,
360 upload_scratch_mem_.resource,
361 upload_scratch_mem_.offset, size);
362
363 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
364 alloc.resource, D3D12_RESOURCE_STATE_COPY_DEST,
365 D3D12_RESOURCE_STATE_UNORDERED_ACCESS);
366 command_list_->ResourceBarrier(1, &barrier);
367
368 upload_scratch_mem_.resource->Unmap(0, nullptr);
369
370 // reset at flush and wait
371 upload_scratch_mem_.offset += (uint32_t)size;
372}

374DxNetwork::DxNetwork(const WeightsFile& file, const OptionsDict& options)
375 : dx_context_(options),
376 capabilities_{file.format().network_format().input(),
377 file.format().network_format().output(),
378 file.format().network_format().moves_left()} {
379 LegacyWeights weights(file.weights());
380
381 has_conv_policy_ = file.format().network_format().policy() ==
382 pblczero::NetworkFormat::POLICY_CONVOLUTION;
383 max_batch_size_ =
384 options.GetOrDefault<int>("max_batch", kMaxSupportedBatchSize);
385
386 // Default is fp16, to use fp32: --backend-opts=fp16=false.
387 fp16_ = options.GetOrDefault<bool>("fp16", DEFAULT_FP16);
388
389 // Default is to attempt using Winograd algorithm for Convolutions using GEMM
390 // Metacommand first, if not available - attempt using Convolution Metacommand
391 // directly (whatever algorithm HW vendor is providing), and if neither is
392 // available use winograd algorithm with our own GEMM compute shader.
393 // The below backend options can be used to override this for testing.
394 bool enable_gemm_metacommand =
395 options.GetOrDefault<bool>("enable-gemm-metacommand", true);
396 bool enable_conv_metacommand =
397 options.GetOrDefault<bool>("enable-conv-metacommand", true);
398
399 const int kNumFilters = (int)weights.input.biases.size();
400
401 num_blocks_ = (int)weights.residual.size();
402 has_se_ = weights.residual[0].has_se;
403 int pol_channels = (int)weights.policy.biases.size();
404
405 // Build the network, and copy the weights to GPU memory.
406
407 // Unique GEMMs for winograd required by the network.
408 if (enable_gemm_metacommand) {
409 input_conv_gemm_metacommand_ = std::make_unique<GemmMetaCommand>(
410 &dx_context_, 0, kNumFilters, kInputPlanes, 36, fp16_, false, false);
411
412 residual_block_gemm_metacommand_ = std::make_unique<GemmMetaCommand>(
413 &dx_context_, 0, kNumFilters, kNumFilters, 36, fp16_, false, false);
414
415 if (has_conv_policy_) {
416 policy_conv_gemm_metacommand_ = std::make_unique<GemmMetaCommand>(
417 &dx_context_, 0, pol_channels, kNumFilters, 36, fp16_, false, false);
418 }
419 }
420
421 // Unique Conv metacommands required by the network.
422 if (enable_conv_metacommand) {
423 // Create only if we were not able to create GEMM metacommands for some
424 // reason 3x3, 112 channels -> kNumFilters channels, relu, bias.
425 if (!input_conv_gemm_metacommand_ ||
426 !input_conv_gemm_metacommand_->IsAvailable())
427 input_conv_metacommand_ = std::make_unique<ConvMetaCommand>(
428 &dx_context_, kInputPlanes, kNumFilters, 8, 8, 3, true, true, fp16_);
429
430 if (!residual_block_gemm_metacommand_ ||
431 !residual_block_gemm_metacommand_->IsAvailable()) {
432 // 3x3, kNumFilters channels -> kNumFilters channels, relu, bias.
433 resi_block_conv_1_metacommand_ = std::make_unique<ConvMetaCommand>(
434 &dx_context_, kNumFilters, kNumFilters, 8, 8, 3, true, true, fp16_);
435
436 // 3x3, kNumFilters channels -> kNumFilters channels, no relu
437 // relu needs to be done after SE and/or skip connection add.
438 resi_block_conv_2_metacommand_ = std::make_unique<ConvMetaCommand>(
439 &dx_context_, kNumFilters, kNumFilters, 8, 8, 3, false, true, fp16_);
440 }
441
442 if (has_conv_policy_ && (!policy_conv_gemm_metacommand_ ||
443 !policy_conv_gemm_metacommand_->IsAvailable()))
444 policy_conv_metacommand_ = std::make_unique<ConvMetaCommand>(
445 &dx_context_, kNumFilters, pol_channels, 8, 8, 3, false, true, fp16_);
446 }
447
448 // input
449 {
450 auto inputConv = std::make_unique<ConvLayer>(
451 fp16_, input_conv_gemm_metacommand_.get(),
452 input_conv_metacommand_.get(), &dx_context_, nullptr, kNumFilters, 8, 8,
453 3, kInputPlanes, true, true);
454
455 inputConv->LoadWeights(&weights.input.weights[0], &weights.input.biases[0],
456 &dx_context_);
457
458 network_.emplace_back(std::move(inputConv));
459 }
460
461 // residual block
462 for (size_t block = 0; block < weights.residual.size(); block++) {
463 auto conv1 = std::make_unique<ConvLayer>(
464 fp16_, residual_block_gemm_metacommand_.get(),
465 resi_block_conv_1_metacommand_.get(), &dx_context_, getLastLayer(),
466 kNumFilters, 8, 8, 3, kNumFilters, true, true);
467
468 conv1->LoadWeights(&weights.residual[block].conv1.weights[0],
469 &weights.residual[block].conv1.biases[0], &dx_context_);
470
471 network_.emplace_back(std::move(conv1));
472
473 int se_k = 0;
474 if (has_se_) se_k = (int)weights.residual[block].se.b1.size();
475
476 auto conv2 = std::make_unique<ConvLayer>(
477 fp16_, residual_block_gemm_metacommand_.get(),
478 resi_block_conv_2_metacommand_.get(), &dx_context_, getLastLayer(),
479 kNumFilters, 8, 8, 3, kNumFilters, true, true, true, has_se_, se_k);
480
481 conv2->LoadWeights(&weights.residual[block].conv2.weights[0],
482 &weights.residual[block].conv2.biases[0], &dx_context_);
483
484 if (has_se_) {
485 conv2->LoadSEWeights(
486 &weights.residual[block].se.w1[0], &weights.residual[block].se.b1[0],
487 &weights.residual[block].se.w2[0], &weights.residual[block].se.b2[0]);
488 }
489 network_.emplace_back(std::move(conv2));
490 }
491
492 BaseLayer* resi_last = getLastLayer();
493
494 // policy head
495 if (has_conv_policy_) {
496 // conv1 is same as residual block convolution.
497 auto conv1 = std::make_unique<ConvLayer>(
498 fp16_, residual_block_gemm_metacommand_.get(),
499 resi_block_conv_1_metacommand_.get(), &dx_context_, getLastLayer(),
500 kNumFilters, 8, 8, 3, kNumFilters, true, true);
501 conv1->LoadWeights(&weights.policy1.weights[0], &weights.policy1.biases[0],
502 &dx_context_);
503 network_.emplace_back(std::move(conv1));
504
505 // conv2 has different no. of output filters (pol_channels). No relu.
506 auto conv2 = std::make_unique<ConvLayer>(
507 fp16_, policy_conv_gemm_metacommand_.get(),
508 policy_conv_metacommand_.get(), &dx_context_, getLastLayer(),
509 pol_channels, 8, 8, 3, kNumFilters, true, false);
510
511 conv2->LoadWeights(&weights.policy.weights[0], &weights.policy.biases[0],
512 &dx_context_);
513 network_.emplace_back(std::move(conv2));
514
515 // Policy map layer
516 auto policyMap =
517 std::make_unique<PolicyMapLayer>(fp16_, &dx_context_, getLastLayer(),
518 kNumOutputPolicy, 1, 1, 73 * 8 * 8);
519 policyMap->LoadWeights(kConvPolicyMap);
520 network_.emplace_back(std::move(policyMap));
521
522 } else {
523 // 1x1 convolution, pol_channels output filters
524 auto convPol = std::make_unique<ConvLayer>(
525 fp16_, nullptr, nullptr, &dx_context_, getLastLayer(), pol_channels, 8,
526 8, 1, kNumFilters, true, true);
527 convPol->LoadWeights(&weights.policy.weights[0], &weights.policy.biases[0],
528 &dx_context_);
529 network_.emplace_back(std::move(convPol));
530
531 // FC with bias, no activation
532 // pad up kNumOutputPolicy to be a multiple of 8
533 assert(weights.ip_pol_b.size() == kNumOutputPolicy);
534 auto FCPol = std::make_unique<FCLayer>(fp16_, &dx_context_, getLastLayer(),
535 kNumOutputPolicyPadded8, 1, 1, true,
536 false, false);
537 // Copy weights to temp space which is padded in size.
538 std::vector<float> tempBias(kNumOutputPolicyPadded8);
539 std::vector<float> tempWeight(kNumOutputPolicyPadded8 *
540 weights.ip_pol_w.size() / kNumOutputPolicy);
541 memcpy(tempBias.data(), weights.ip_pol_b.data(),
542 weights.ip_pol_b.size() * sizeof(float));
543 memcpy(tempWeight.data(), weights.ip_pol_w.data(),
544 weights.ip_pol_w.size() * sizeof(float));
545
546 FCPol->LoadWeights(tempWeight.data(), tempBias.data(), &dx_context_);
547 network_.emplace_back(std::move(FCPol));
548 }
549
550 // value head
551 {
552 int val_channels = (int)weights.value.biases.size();
553
554 // 1x1 convolution, val_channels output filters
555 auto convVal = std::make_unique<ConvLayer>(
556 fp16_, nullptr, nullptr, &dx_context_, resi_last, val_channels, 8, 8, 1,
557 kNumFilters, true, true);
558 convVal->LoadWeights(&weights.value.weights[0], &weights.value.biases[0],
559 &dx_context_);
560 network_.emplace_back(std::move(convVal));
561
562 // Bias and relu activation.
563 auto FCVal1 = std::make_unique<FCLayer>(fp16_, &dx_context_, getLastLayer(),
564 (int)weights.ip1_val_b.size(), 1, 1,
565 true, true, false);
566 FCVal1->LoadWeights(&weights.ip1_val_w[0], &weights.ip1_val_b[0],
567 &dx_context_);
568 network_.emplace_back(std::move(FCVal1));
569
570 has_wdl_ = file.format().network_format().value() ==
571 pblczero::NetworkFormat::VALUE_WDL;
572
573 // Fully connected layer with Bias.
574 // tanh activation for non wdl nets, no activation for wdl.
575 auto fc2_tanh = !has_wdl_;
576 auto FCVal2 = std::make_unique<FCLayer>(fp16_, &dx_context_, getLastLayer(),
577 kNumOutputValuePadded8, 1, 1, true,
578 false, fc2_tanh);
579 // Pad up the weights
580 std::vector<float> tempBias(kNumOutputValuePadded8);
581 std::vector<float> tempWeight(kNumOutputValuePadded8 *
582 weights.ip2_val_w.size() /
583 weights.ip2_val_b.size());
584 memcpy(tempBias.data(), weights.ip2_val_b.data(),
585 weights.ip2_val_b.size() * sizeof(float));
586 memcpy(tempWeight.data(), weights.ip2_val_w.data(),
587 weights.ip2_val_w.size() * sizeof(float));
588 FCVal2->LoadWeights(tempWeight.data(), tempBias.data(), &dx_context_);
589 network_.emplace_back(std::move(FCVal2));
590 }
591
592 // Moves left head
593 moves_left_ = (file.format().network_format().moves_left() ==
594 pblczero::NetworkFormat::MOVES_LEFT_V1) &&
595 options.GetOrDefault<bool>("mlh", true);
596 if (moves_left_) {
597 // 1x1 convolution, moves_left biases output filters
598 auto convMov = std::make_unique<ConvLayer>(
599 fp16_, nullptr, nullptr, &dx_context_, resi_last,
600 weights.moves_left.biases.size(), 8, 8, 1, kNumFilters, true, true);
601 convMov->LoadWeights(&weights.moves_left.weights[0],
602 &weights.moves_left.biases[0], &dx_context_);
603 network_.emplace_back(std::move(convMov));
604
605 // Bias and relu activation.
606 auto FCMov1 = std::make_unique<FCLayer>(fp16_, &dx_context_, getLastLayer(),
607 (int)weights.ip1_mov_b.size(), 1, 1,
608 true, true, false);
609 FCMov1->LoadWeights(&weights.ip1_mov_w[0], &weights.ip1_mov_b[0],
610 &dx_context_);
611 network_.emplace_back(std::move(FCMov1));
612
613 // Fully connected layer with Bias and relu.
614 auto FCMov2 = std::make_unique<FCLayer>(fp16_, &dx_context_, getLastLayer(),
615 kNumOutputMovesLeftPadded8, 1, 1,
616 true, true, false);
617 // Pad up the weights
618 std::vector<float> tempBias(kNumOutputMovesLeftPadded8);
619 std::vector<float> tempWeight(kNumOutputMovesLeftPadded8 *
620 weights.ip2_mov_w.size() /
621 weights.ip2_mov_b.size());
622 memcpy(tempBias.data(), weights.ip2_mov_b.data(),
623 weights.ip2_mov_b.size() * sizeof(float));
624 memcpy(tempWeight.data(), weights.ip2_mov_w.data(),
625 weights.ip2_mov_w.size() * sizeof(float));
626 FCMov2->LoadWeights(tempWeight.data(), tempBias.data(), &dx_context_);
627 network_.emplace_back(std::move(FCMov2));
628 }
629
630 dx_context_.FlushAndWait();
631
632 // Allocate GPU memory for running the network
633 // 4 buffers of max size are enough:
634 // * one to hold input,
635 // * second to hold output
636 // * third to hold skip connection's input
637 // * and fourth to act as scratch space needed by some layers.
638 size_t max_size = resi_last->GetOutputSize(max_batch_size_);
639
640 // Winograd transformed inputs/outputs need more space.
641 // Every 4x4 block of input/output is transfored to 6x6 block.
642 max_size *= (size_t)ceil(36.0 / 16.0);
643
644 for (auto& mem : tensor_mem_) {
645 dx_context_.CreateAlloc(max_size, D3D12_HEAP_TYPE_DEFAULT, mem, fp16_);
646 }

649void DxNetwork::Eval(InputsOutputsDx* io, int batch_size) {
650 if (batch_size > kMaxSupportedBatchSize)
651 throw Exception("Unsupported batch size: " + std::to_string(batch_size));
652
653#ifdef DEBUG_DUMP_PER_LAYER_DATA
654 lock_.lock();
655 ID3D12GraphicsCommandList4* cl = dx_context_.getCommandList();
656#else
657 ID3D12GraphicsCommandList4* cl = io->command_list_;
658 dx_context_.ResetCL(cl, io->command_allocator_, io->needs_reset_);
659#endif
660
661// Expand packed board representation into full planes.
662
663#ifdef COPY_BEFORE_SHADER_READ
664 // First copy from upload heap to scratch mem
665 CD3DX12_RESOURCE_BARRIER barrier;
666
667 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
668 tensor_mem_[1].resource, D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
669 D3D12_RESOURCE_STATE_COPY_DEST);
670 cl->ResourceBarrier(1, &barrier);
671
672 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
673 tensor_mem_[2].resource, D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
674 D3D12_RESOURCE_STATE_COPY_DEST);
675 cl->ResourceBarrier(1, &barrier);
676
677 cl->CopyBufferRegion(tensor_mem_[1].resource, 0,
678 io->input_masks_mem_gpu_.resource, 0,
679 sizeof(uint64_t) * batch_size * kInputPlanes);
680 cl->CopyBufferRegion(tensor_mem_[2].resource, 0,
681 io->input_val_mem_gpu_.resource, 0,
682 sizeof(float) * batch_size * kInputPlanes);
683
684 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
685 tensor_mem_[1].resource, D3D12_RESOURCE_STATE_COPY_DEST,
686 D3D12_RESOURCE_STATE_UNORDERED_ACCESS);
687 cl->ResourceBarrier(1, &barrier);
688
689 barrier = CD3DX12_RESOURCE_BARRIER::Transition(
690 tensor_mem_[2].resource, D3D12_RESOURCE_STATE_COPY_DEST,
691 D3D12_RESOURCE_STATE_UNORDERED_ACCESS);
692 cl->ResourceBarrier(1, &barrier);
693
694 dx_context_.UavBarrier(cl);
695
696 dx_context_.getShaderWrapper()->ExpandPlanes(
697 cl, tensor_mem_[0], tensor_mem_[1], tensor_mem_[2], batch_size, fp16_);
698
699#else
700 dx_context_.getShaderWrapper()->ExpandPlanes(
701 cl, tensor_mem_[0], io->input_masks_mem_gpu_, io->input_val_mem_gpu_,
702 batch_size, fp16_);
703#endif
704
705 dx_context_.UavBarrier(cl);
706
707 // Debug logging (not compiled by default)
708 dx_context_.DumpTensor("After expand planes", tensor_mem_[0], 1024, fp16_);
709
710 int l = 0;
711
712 //-----------------------------------///---------------------------------------
713 // Input Conv
714 network_[l++]->Eval(batch_size, tensor_mem_[2], tensor_mem_[0], DXAlloc(),
715 tensor_mem_[1], tensor_mem_[3], cl);
716 dx_context_.UavBarrier(cl);
717
718 dx_context_.DumpTensor("After input conv", tensor_mem_[2], 1024, fp16_);
719
720 //-----------------------------------///---------------------------------------
721
722 // Residual tower.
723 for (int block = 0; block < num_blocks_; block++) {
724 // conv1
725 network_[l++]->Eval(batch_size, tensor_mem_[0], tensor_mem_[2], DXAlloc(),
726 tensor_mem_[1], tensor_mem_[3], cl);
727 dx_context_.UavBarrier(cl);
728
729 // conv2
730 network_[l++]->Eval(batch_size, tensor_mem_[2], tensor_mem_[0],
731 tensor_mem_[2], tensor_mem_[1], tensor_mem_[3], cl);
732 dx_context_.UavBarrier(cl);
733 }
734
735 dx_context_.DumpTensor("After Residual tower", tensor_mem_[2], 1024, fp16_);
736
737 //-----------------------------------///---------------------------------------
738
739 // Policy head.
740 if (has_conv_policy_) {
741 // Policy conv1.
742 network_[l++]->Eval(batch_size, tensor_mem_[0], tensor_mem_[2], DXAlloc(),
743 tensor_mem_[1], tensor_mem_[3], cl);
744 dx_context_.UavBarrier(cl);
745
746 dx_context_.DumpTensor("After policy conv1", tensor_mem_[0], 1024, fp16_);
747
748 // Policy conv2
749 network_[l++]->Eval(batch_size, tensor_mem_[1], tensor_mem_[0], DXAlloc(),
750 tensor_mem_[1], tensor_mem_[3], cl);
751
752 dx_context_.UavBarrier(cl);
753
754 dx_context_.DumpTensor("After policy conv2", tensor_mem_[1], 1024, fp16_);
755
756 // Policy Map layer (writes directly to system memory).
757 network_[l++]->Eval(batch_size, io->op_policy_mem_gpu_, tensor_mem_[1],
758 DXAlloc(), DXAlloc(), DXAlloc(), cl);
759
760 // Output of policy map layer is always FP32.
761 dx_context_.DumpTensor("After policy map", io->op_policy_mem_gpu_, 1024,
762 false);
763
764 } else {
765 // Policy conv.
766 network_[l++]->Eval(batch_size, tensor_mem_[0], tensor_mem_[2], DXAlloc(),
767 tensor_mem_[1], tensor_mem_[3], cl);
768 dx_context_.UavBarrier(cl);
769
770 // Policy FC (writes directly to system memory).
771 network_[l++]->Eval(batch_size, io->op_policy_mem_gpu_, tensor_mem_[0],
772 DXAlloc(), tensor_mem_[1], tensor_mem_[3], cl);
773 }
774
775 //-----------------------------------///---------------------------------------
776
777 // Value head.
778
779 // Value conv.
780 network_[l++]->Eval(batch_size, tensor_mem_[0], tensor_mem_[2], DXAlloc(),
781 tensor_mem_[1], tensor_mem_[3], cl);
782 dx_context_.UavBarrier(cl);
783
784 dx_context_.DumpTensor("After value conv", tensor_mem_[0], 1024, fp16_);
785
786 // value FC1.
787 network_[l++]->Eval(batch_size, tensor_mem_[1], tensor_mem_[0], DXAlloc(),
788 DXAlloc(), DXAlloc(), cl);
789 dx_context_.UavBarrier(cl);
790
791 dx_context_.DumpTensor("After value fc1", tensor_mem_[1], 128, fp16_);
792
793 // value FC2.
794 network_[l++]->Eval(batch_size, io->op_value_mem_gpu_, tensor_mem_[1],
795 DXAlloc(), DXAlloc(), DXAlloc(), cl);
796
797 dx_context_.DumpTensor("After value fc2", io->op_value_mem_gpu_, 8, fp16_);
798
799 //-----------------------------------///---------------------------------------
800
801 // Moves left head.
802 if (moves_left_) {
803 // Moves left conv.
804 network_[l++]->Eval(batch_size, tensor_mem_[0], tensor_mem_[2], DXAlloc(),
805 tensor_mem_[1], tensor_mem_[3], cl);
806 dx_context_.UavBarrier(cl);
807
808 dx_context_.DumpTensor("After moves left conv", tensor_mem_[0], 1024,
809 fp16_);
810
811 // Moves left FC1.
812 network_[l++]->Eval(batch_size, tensor_mem_[1], tensor_mem_[0], DXAlloc(),
813 DXAlloc(), DXAlloc(), cl);
814 dx_context_.UavBarrier(cl);
815
816 dx_context_.DumpTensor("After moves left fc1", tensor_mem_[1], 512, fp16_);
817
818 // Moves left FC2.
819 network_[l++]->Eval(batch_size, io->op_moves_left_mem_gpu_, tensor_mem_[1],
820 DXAlloc(), DXAlloc(), DXAlloc(), cl);
821
822 dx_context_.DumpTensor("After moves left fc2", io->op_moves_left_mem_gpu_,
823 8, fp16_);
824 }
825
826//-----------------------------------///---------------------------------------
827#ifdef DEBUG_DUMP_PER_LAYER_DATA
828 dx_context_.FlushAndWait();
829 lock_.unlock();
830#else
831 // TODO: Get rid of this lock once we move the Command Queue also to
832 // InputsOutputs structure This isn't a bottleneck anyway (for CPU side perf).
833 // The hope is that we will get some GPU side parallelism with multiple async
834 // compute queues.
835 lock_.lock();
836 uint64_t fence = dx_context_.FlushCL(cl);
837 lock_.unlock();
838
839 dx_context_.WaitForGpu(fence);
840 io->needs_reset_ = true;
841#endif
842
843 // Do some simple post-processing operations on CPU:
844 // - un-padding of policy and value heads.
845 // - value head softmax (for wdl enabled nets)
846 // We do them outside the lock to get some more parallelism.
847 int val_vector_size = has_wdl_ ? 3 : 1;
848 if (fp16_) {
849 // Policy:
850 // Un-pad policy output, and convert to fp32.
851 if (!has_conv_policy_) {
852 dx_half* padded_pol_fp16 = (dx_half*)io->op_policy_mem_;
853 for (int n = 0; n < batch_size; n++)
854 for (int i = 0; i < kNumOutputPolicy; i++)
855 io->op_policy_mem_final_[n * kNumOutputPolicy + i] =
856 FP16toFP32(padded_pol_fp16[n * kNumOutputPolicyPadded8 + i]);
857 }
858 // Value:
859 // Un-pad value output, converting it to fp32.
860 dx_half* padded_val_fp16 = (dx_half*)io->op_value_mem_;
861 for (int n = 0; n < batch_size; n++)
862 for (int i = 0; i < val_vector_size; i++)
863 io->op_value_mem_final_[n * val_vector_size + i] =
864 FP16toFP32(padded_val_fp16[n * kNumOutputValuePadded8 + i]);
865 if (moves_left_) {
866 // Moves left:
867 // Un-pad moves left output, converting it to fp32.
868 dx_half* padded_moves_left_fp16 = (dx_half*)io->op_moves_left_mem_;
869 for (int n = 0; n < batch_size; n++)
870 io->op_moves_left_mem_final_[n] =
871 FP16toFP32(padded_moves_left_fp16[n * kNumOutputMovesLeftPadded8]);
872 }
873 } else {
874 // Policy:
875 // Un-pad policy output.
876 if (!has_conv_policy_) {
877 for (int i = 0; i < batch_size; i++)
878 memcpy(io->op_policy_mem_final_ + kNumOutputPolicy * i,
879 io->op_policy_mem_ + kNumOutputPolicyPadded8 * i,
880 kNumOutputPolicy * sizeof(float));
881 }
882
883 // Value:
884 // Un-pad value output.
885 for (int i = 0; i < batch_size; i++)
886 memcpy(io->op_value_mem_final_ + val_vector_size * i,
887 io->op_value_mem_ + kNumOutputValuePadded8 * i,
888 val_vector_size * sizeof(float));
889 if (moves_left_) {
890 // Moves left:
891 // Un-pad moves left output.
892 for (int i = 0; i < batch_size; i++)
893 memcpy(io->op_moves_left_mem_final_ + i,
894 io->op_moves_left_mem_ + kNumOutputMovesLeftPadded8 * i,
895 sizeof(float));
896 }
897 }
898
899 // Softmax on value head for wdl enabled networks.
900 if (has_wdl_) {
901 for (int i = 0; i < batch_size; i++) {
902 float w_val = io->op_value_mem_final_[i * 3 + 0];
903 float d_val = io->op_value_mem_final_[i * 3 + 1];
904 float l_val = io->op_value_mem_final_[i * 3 + 2];
905 float max_val = std::max({w_val, d_val, l_val});
906 w_val = exp(w_val - max_val);
907 d_val = exp(d_val - max_val);
908 l_val = exp(l_val - max_val);
909 float S = w_val + d_val + l_val;
910 w_val /= S;
911 d_val /= S;
912 l_val /= S;
913
914 io->op_value_mem_final_[i * 3 + 0] = w_val;
915 io->op_value_mem_final_[i * 3 + 1] = d_val;
916 io->op_value_mem_final_[i * 3 + 2] = l_val;
917 }
918 }
