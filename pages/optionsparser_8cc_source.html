121bool OptionsParser::ProcessFlags(const std::vector<std::string>& args) {
122 auto show_help = false;
123 if (CommandLine::BinaryName().find("simple") != std::string::npos) {
124 visibility_mode_ = OptionId::kSimpleMode;
125 } else if (CommandLine::BinaryName().find("pro") != std::string::npos) {
126 visibility_mode_ = OptionId::kProMode;
127 }
128 for (auto iter = args.begin(), end = args.end(); iter != end; ++iter) {
129 std::string param = *iter;
130 if (param == "--show-hidden") {
131 visibility_mode_ = OptionId::kProMode;
132 continue;
133 }
134 if (param == "-h" || param == "--help") {
135 // Set a flag so that --show-hidden after --help works.
136 show_help = true;
137 continue;
138 }
139
140 if (param.substr(0, 2) == "--") {
141 std::string context;
142 param = param.substr(2);
143 std::string value;
144 auto pos = param.find('=');
145 if (pos != std::string::npos) {
146 value = param.substr(pos + 1);
147 param = param.substr(0, pos);
148 }
149 pos = param.rfind('.');
150 if (pos != std::string::npos) {
151 context = param.substr(0, pos);
152 param = param.substr(pos + 1);
153 }
154 bool processed = false;
155 Option* option = FindOptionByLongFlag(param);
156 if (option &&
157 option->ProcessLongFlag(param, value, GetMutableOptions(context))) {
158 processed = true;
159 }
160 if (!processed) {
161 CERR << "Unknown command line flag: " << *iter << ".";
162 CERR << "For help run:\n " << CommandLine::BinaryName() << " --help";
163 return false;
164 }
165 continue;
166 }
167 if (param.size() == 2 && param[0] == '-') {
168 std::string value;
169 bool processed = false;
170 if (iter + 1 != end) {
171 value = *(iter + 1);
172 }
173 for (auto& option : options_) {
174 if (option->ProcessShortFlag(param[1], GetMutableOptions())) {
175 processed = true;
176 break;
177 } else if (option->ProcessShortFlagWithValue(param[1], value,
178 GetMutableOptions())) {
179 if (!value.empty()) ++iter;
180 processed = true;
181 break;
182 }
183 }
184 if (!processed) {
185 CERR << "Unknown command line flag: " << *iter << ".";
186 CERR << "For help run:\n " << CommandLine::BinaryName() << " --help";
187 return false;
188 }
189 continue;
190 }
191
192 CERR << "Unknown command line argument: " << *iter << ".\n";
193 CERR << "For help run:\n " << CommandLine::BinaryName() << " --help";
194 return false;
195 }
196 if (show_help) {
197 ShowHelp();
198 return false;
199 }
200 return true;
201}

208std ::string FormatFlag(char short_flag, const std::string& long_flag,
209 const std::string& help,
210 const std::string& uci_option = {},
211 const std::string& def = {}) {
212 std::ostringstream oss;
213 oss << " ";
214 if (short_flag) {
215 oss << '-' << short_flag;
216 } else {
217 oss << " ";
218 }
219 if (short_flag && !long_flag.empty()) {
220 oss << ", ";
221 } else {
222 oss << " ";
223 }
224 std::string long_flag_str = "";
225 if (!short_flag && long_flag.empty()) {
226 long_flag_str = "(uci parameter)";
227 } else {
228 long_flag_str = long_flag.empty() ? "" : "--" + long_flag;
229 }
230 oss << long_flag_str;
231 auto help_lines = FlowText(help, kHelpWidth);
232 bool is_first_line = true;
233 for (const auto& line : help_lines) {
234 if (is_first_line) {
235 is_first_line = false;
236 if (long_flag_str.size() < kHelpIndent - 7) {
237 oss << std::string(kHelpIndent - 7 - long_flag_str.size(), ' ') << line
238 << "\n";
239 continue;
240 }
241 oss << "\n";
242 }
243 oss << std::string(kHelpIndent, ' ') << line << "\n";
244 }
245 if (!def.empty() || !uci_option.empty()) {
246 oss << std::string(kUciLineIndent, ' ') << '[';
247 if (!uci_option.empty()) oss << "UCI: " << uci_option;
248 if (!uci_option.empty() && !def.empty()) oss << " ";
249 if (!def.empty()) oss << "DEFAULT: " << def;
250 oss << "]\n";
251 }
252 oss << '\n';
253 return oss.str();
254}

257void OptionsParser::ShowHelp() const {
258 std::cout << "Usage: " << CommandLine::BinaryName() << " [<mode>] [flags...]"
259 << std::endl;
260
261 std::cout << "\nAvailable modes. A help for a mode: "
262 << CommandLine::BinaryName() << " <mode> --help\n";
263 for (const auto& mode : CommandLine::GetModes()) {
264 std::cout << " " << std::setw(10) << std::left << mode.first << " "
265 << mode.second << std::endl;
266 }
267
268 std::cout << "\nAllowed command line flags for current mode:\n";
269 std::cout << FormatFlag('h', "help", "Show help and exit.");
270 std::cout << FormatFlag('\0', "show-hidden",
271 "Show hidden options. Use with --help.");
272 for (const auto& option : options_) {
273 if ((option->GetId().visibility_mask() & visibility_mode_)) {
274 std::cout << option->GetHelp(values_);
275 }
276 }
277
278 auto contexts = values_.ListSubdicts();
279 if (!contexts.empty()) {
280 std::cout << "\nFlags can be defined per context (one of: "
281 << StrJoin(contexts, ", ") << "), for example:\n";
282 std::cout << " --" << contexts[0] << '.'
283 << options_.back()->GetLongFlag() << "=(value)\n";
284 }
285}

397int IntOption::ValidateIntString(const std::string& val) const {
398 int result;
399 const auto end = val.data() + val.size();
400 auto [ptr, err] = std::from_chars(val.data(), end, result);
401 if (err == std::errc::invalid_argument) {
402 throw Exception("Flag '--" + GetLongFlag() + "' has an invalid format.");
403 } else if (err == std::errc::result_out_of_range) {
404 throw Exception("Flag '--" + GetLongFlag() + "' is out of range.");
405 } else if (ptr != end) {
406 throw Exception("Flag '--" + GetLongFlag() + "' has trailing characters.");
407 } else {
408 return result;
409 }
410}
