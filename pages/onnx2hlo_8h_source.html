40struct Onnx2HloOptions {
41 // Constants larger that this size in bytes will be passed as parameters
42 // instead. This allows them to be shared between different modules.
43 size_t max_inline_constant_size = 1024;
44 // It set, ensures that the input/output tensors have given type (does not
45 // affect constants passed as parameters).
46 std::optional<pblczero::XlaShapeProto::Type> io_type = std::nullopt;
47 // If error occurs during conversion, return a partial result instead of
48 // failing. Only to be used for debugging.
