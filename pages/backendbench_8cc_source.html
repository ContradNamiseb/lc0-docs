80void BackendBenchmark::Run() {
81 OptionsParser options;
82 SharedBackendParams::Populate(&options);
83 options.Add<IntOption>(kThreadsOptionId, 1, 128) = kDefaultThreads;
84
85 options.Add<IntOption>(kBatchesId, 1, 999999999) = 100;
86 options.Add<IntOption>(kStartBatchSizeId, 1, 1024) = 1;
87 options.Add<IntOption>(kMaxBatchSizeId, 1, 1024) = 256;
88 options.Add<IntOption>(kBatchStepId, 1, 256) = 1;
89 options.Add<BoolOption>(kHeaderOnlyOnceId) = false;
90 options.Add<StringOption>(kFenId) = ChessBoard::kStartposFen;
91 options.Add<BoolOption>(kClippyId) = false;
92
93 if (!options.ProcessAllFlags()) return;
94
95 try {
96 auto option_dict = options.GetOptionsDict();
97
98 auto backend = BackendManager::Get()->CreateFromParams(option_dict);
99 const int threads = option_dict.Get<int>(kThreadsOptionId);
100
101 classic::NodeTree tree;
102 tree.ResetToPosition(option_dict.Get<std::string>(kFenId), {});
103 EvalPosition pos{tree.GetPositionHistory().GetPositions(), {}};
104 std::vector<std::thread> handles;
105
106 // Do any backend initialization outside the loop.
107 auto warm = [&]() {
108 // Give GPU enough work to make it go from idle clocks to max clocks.
109 for (int i = 0; i < 2; i++) {
110 auto warmup = backend->CreateComputation();
111 for (int j = 0; j < option_dict.Get<int>(kMaxBatchSizeId); ++j) {
112 warmup->AddInput(pos, {});
113 }
114 warmup->ComputeBlocking();
115 }
116 };
117 for (int t = 1; t < threads; t++) {
118 handles.emplace_back(warm);
119 }
120 warm();
121 for (auto& handle : handles) {
122 handle.join();
123 }
124 handles.clear();
125
126 const int batches = option_dict.Get<int>(kBatchesId);
127
128 int best = 1;
129 int best2 = 1;
130 int best3 = 1;
131 float best_nps = 0.0f;
132 float best_nps2 = 0.0f;
133 float best_nps3 = 0.0f;
134 std::optional<std::chrono::time_point<std::chrono::steady_clock>> pending;
135 using tp = std::chrono::time_point<std::chrono::steady_clock>;
136 std::vector<std::vector<tp>> ends(threads);
137 for (auto& vend : ends) {
138 vend.resize(batches + 1);
139 }
140 std::vector<std::chrono::duration<double>> times(batches);
141 std::vector<int> thread_counts(threads);
142 for (int i = option_dict.Get<int>(kStartBatchSizeId);
143 i <= option_dict.Get<int>(kMaxBatchSizeId);
144 i += option_dict.Get<int>(kBatchStepId)) {
145 handles.reserve(threads);
146 std::atomic<int> j{0};
147
148 auto compute = [&](int tid = 0) {
149 int count = 0;
150 auto& end = ends[tid];
151 // Ignore the first batch to let GPU queue fill for stable measurements.
152 while (j++ < batches) {
153 // Put i copies of tree root node into computation and compute.
154 auto computation = backend->CreateComputation();
155 for (int k = 0; k < i; k++) {
156 computation->AddInput(pos, {});
157 }
158 computation->ComputeBlocking();
159 end[count++] = std::chrono::steady_clock::now();
160 }
161 thread_counts[tid] = count;
162 };
163
164 for (int t = 1; t < threads; t++) {
165 handles.emplace_back(compute, t);
166 }
167
168 compute(0);
169 for (auto& handle : handles) {
170 handle.join();
171 }
172
173 handles.clear();
174
175 double stddev = 0;
176 double total = 0;
177 int batches_done = 0;
178 for (int t = 0; t < threads; t++) {
179 for (int j = 1; j < thread_counts[t]; j++) {
180 times[batches_done] = (ends[t][j] - ends[t][j - 1]) / threads;
181 total += times[batches_done].count();
182 batches_done++;
183 }
184 }
185
186 double mean = total / batches_done;
187
188 for (int j = 0; j < batches_done; j++) {
189 double diff = times[j].count() - mean;
190 stddev += diff * diff;
191 }
192 stddev = std::sqrt(stddev / (batches_done - 1));
193 double cv = stddev / mean;
194
195 std::sort(times.begin(), times.begin() + batches_done);
196
197 mean *= 1000;
198
199 const auto nps = i * batches_done / total;
200 const auto median = batches_done % 2 == 0
201 ? 2 * i /
202 (times[batches_done / 2 - 1].count() +
203 times[batches_done / 2].count())
204 : i / times[batches_done / 2].count();
205 if (option_dict.Get<bool>(kHeaderOnlyOnceId)
206 ? i == option_dict.Get<int>(kStartBatchSizeId)
207 : ((i - option_dict.Get<int>(kStartBatchSizeId)) /
208 option_dict.Get<int>(kBatchStepId) % 32 ==
209 0)) {
210 std::cout << "size,"
211 " mean nps,"
212 " mean ms,"
213 " sdev,"
214 " cv,"
215 " max nps,"
216 " median,"
217 " min nps,"
218 << std::endl;
219 }
220 // clang-format off
221 std::cout << std::setw(4) << i << ","
222 << std::fixed << std::setprecision(0)
223 << std::setw(9) << nps << ","
224 << std::defaultfloat << std::setprecision(4)
225 << std::setw(8) << mean << ","
226 << std::fixed << std::setprecision(4)
227 << std::setw(7) << stddev * 1000 << ","
228 << std::setw(7) << cv << ","
229 << std::fixed << std::setprecision(0)
230 << std::setw(8) << i / times[0].count() << ","
231 << std::setw(8) << median << ","
232 << std::setw(8) << i / times[batches_done - 1].count()
233 << std::endl;
234 // clang-format on
235
236 if (option_dict.Get<bool>(kClippyId)) {
237 float nps_ingame = std::pow((nps + best_nps) / 2, 1.085);
238 float nps_ingame2 = std::pow((nps + best_nps2) / 2, 1.085);
239 float nps_ingame3 = std::pow((nps + best_nps3) / 2, 1.085);
240 float threshold = 0.16947 * exp(-4.1695e-6 * nps_ingame * 180) + 0.02;
241 float threshold2 = 0.16947 * exp(-4.1695e-6 * nps_ingame2 * 15) + 0.02;
242 float threshold3 = 0.16947 * exp(-4.1695e-6 * nps_ingame3 * 1) + 0.02;
243
244 if (nps > best_nps &&
245 threshold * (i - best) * best_nps < (nps - best_nps) * best) {
246 best_nps = nps;
247 best = i;
248 if (threshold2 * (i - best2) * best_nps2 <
249 (nps - best_nps2) * best2) {
250 best_nps2 = nps;
251 best2 = i;
252 if (threshold3 * (i - best3) * best_nps3 <
253 (nps - best_nps3) * best3) {
254 best_nps3 = nps;
255 best3 = i;
256 }
257 }
258 if (!pending) {
259 pending = std::chrono::steady_clock::now();
260 }
261 }
262 if (pending) {
263 std::chrono::duration<double> time =
264 std::chrono::steady_clock::now() - *pending;
265 if (time.count() > 10) {
266 Clippy("Recommended minibatch-size for this net (so far):",
267 "1s/move (Bullet): ", std::to_string(best3),
268 "15s/move (Rapid): ", std::to_string(best2),
269 "3min/move (Tournament): ", std::to_string(best));
270 pending.reset();
271 }
272 }
273 }
274 }
275 if (option_dict.Get<bool>(kClippyId)) {
276 Clippy("Recommended minibatch-size for this net:",
277 "1s/move (Bullet): ", std::to_string(best3),
278 "15s/move (Rapid): ", std::to_string(best2),
279 "3min/move (Tournament): ", std::to_string(best));
280 }
281 } catch (Exception& ex) {
282 std::cerr << ex.what() << std::endl;
283 }
284}
