60// Policy priors (P) are stored in a compressed 16-bit format.
61//
62// Source values are 32-bit floats:
63// * bit 31 is sign (zero means positive)
64// * bit 30 is sign of exponent (zero means nonpositive)
65// * bits 29..23 are value bits of exponent
66// * bits 22..0 are significand bits (plus a "virtual" always-on bit: s âˆˆ [1,2))
67// The number is then sign * 2^exponent * significand, usually.
68// See https://www.h-schmidt.net/FloatConverter/IEEE754.html for details.
69//
70// In compressed 16-bit value we store bits 27..12:
71// * bit 31 is always off as values are always >= 0
72// * bit 30 is always off as values are always < 2
73// * bits 29..28 are only off for values < 4.6566e-10, assume they are always on
74// * bits 11..0 are for higher precision, they are dropped leaving only 11 bits
75// of precision
76//
77// When converting to compressed format, bit 11 is added to in order to make it
78// a rounding rather than truncation.
79//
80// Out of 65556 possible values, 2047 are outside of [0,1] interval (they are in
81// interval (1,2)). This is fine because the values in [0,1] are skewed towards
82// 0, which is also exactly how the components of policy tend to behave (since
83// they add up to 1).
84
85// If the two assumed-on exponent bits (3<<28) are in fact off, the input is
86// rounded up to the smallest value with them on. We accomplish this by
87// subtracting the two bits from the input and checking for a negative result
88// (the subtraction works despite crossing from exponent to significand). This
89// is combined with the round-to-nearest addition (1<<11) into one op.
92void Edge::SetP(float p) {
93 assert(0.0f <= p && p <= 1.0f);
94 constexpr int32_t roundings = (1 << 11) - (3 << 28);
95 int32_t tmp;
96 std::memcpy(&tmp, &p, sizeof(float));
97 tmp += roundings;
98 p_ = (tmp < 0) ? 0 : static_cast<uint16_t>(tmp >> 12);
99}
100
101float Edge::GetP() const {
102 // Reshift into place and set the assumed-set exponent bits.
103 uint32_t tmp = (static_cast<uint32_t>(p_) << 12) | (3 << 28);
104 float ret;
105 std::memcpy(&ret, &tmp, sizeof(uint32_t));
106 return ret;
107}

215void LowNode::MakeTerminal(GameResult result, float plies_left, Terminal type) {
216 SetBounds(result, result);
217 terminal_type_ = type;
218 m_ = plies_left;
219 if (result == GameResult::DRAW) {
220 wl_ = 0.0f;
221 d_ = 1.0f;
222 } else if (result == GameResult::WHITE_WON) {
223 wl_ = 1.0f;
224 d_ = 0.0f;
225 } else if (result == GameResult::BLACK_WON) {
226 wl_ = -1.0f;
227 d_ = 0.0f;
228 }
229
230 assert(WLDMInvariantsHold());
231}

234void LowNode::MakeNotTerminal(const Node* node) {
235 assert(edges_);
236 if (!IsTerminal()) return;
237
238 terminal_type_ = Terminal::NonTerminal;
239 lower_bound_ = GameResult::BLACK_WON;
240 upper_bound_ = GameResult::WHITE_WON;
241 n_ = 0;
242 wl_ = 0.0;
243 d_ = 0.0;
244 m_ = 0.0;
245
246 // Include children too.
247 if (node->GetNumEdges() > 0) {
248 for (const auto& child : node->Edges()) {
249 const auto n = child.GetN();
250 if (n > 0) {
251 n_ += n;
252 // Flip Q for opponent.
253 // Default values don't matter as n is > 0.
254 wl_ += child.GetWL(0.0f) * n;
255 d_ += child.GetD(0.0f) * n;
256 m_ += child.GetM(0.0f) * n;
257 }
258 }
259
260 // Recompute with current eval (instead of network's) and children's eval.
261 wl_ /= n_;
262 d_ /= n_;
263 m_ /= n_;
264 }
265
266 assert(WLDMInvariantsHold());
267}

301void Node::MakeNotTerminal(bool also_low_node) {
302 // At least one of node and low node pair needs to be a terminal.
303 if (!IsTerminal() &&
304 (!also_low_node || !low_node_ || !low_node_->IsTerminal()))
305 return;
306
307 terminal_type_ = Terminal::NonTerminal;
308 repetition_ = false;
309 if (low_node_) { // Two-fold or derived terminal.
310 // Revert low node first.
311 if (also_low_node && low_node_) low_node_->MakeNotTerminal(this);
312
313 auto [lower_bound, upper_bound] = low_node_->GetBounds();
314 lower_bound_ = -upper_bound;
315 upper_bound_ = -lower_bound;
316 n_ = low_node_->GetN();
317 wl_ = -low_node_->GetWL();
318 d_ = low_node_->GetD();
319 m_ = low_node_->GetM() + 1;
320 } else { // Real terminal.
321 lower_bound_ = GameResult::BLACK_WON;
322 upper_bound_ = GameResult::WHITE_WON;
323 n_ = 0.0f;
324 wl_ = 0.0f;
325 d_ = 0.0f;
326 m_ = 0.0f;
327 }
328
329 assert(WLDMInvariantsHold());
330}

337bool Node::TryStartScoreUpdate() {
338 if (n_ > 0) {
339 n_in_flight_.fetch_add(1, std::memory_order_acq_rel);
340 return true;
341 } else {
342 uint32_t expected_n_if_flight_ = 0;
343 return n_in_flight_.compare_exchange_strong(expected_n_if_flight_, 1,
344 std::memory_order_acq_rel);
345 }

355void LowNode::FinalizeScoreUpdate(float v, float d, float m,
356 uint32_t multivisit) {
357 assert(edges_);
358 // Recompute Q.
359 wl_ += multivisit * (v - wl_) / (n_ + multivisit);
360 d_ += multivisit * (d - d_) / (n_ + multivisit);
361 m_ += multivisit * (m - m_) / (n_ + multivisit);
362
363 assert(WLDMInvariantsHold());
364
365 // Increment N.
366 n_ += multivisit;
367}

383void Node::FinalizeScoreUpdate(float v, float d, float m, uint32_t multivisit) {
384 // Recompute Q.
385 wl_ += multivisit * (v - wl_) / (n_ + multivisit);
386 d_ += multivisit * (d - d_) / (n_ + multivisit);
387 m_ += multivisit * (m - m_) / (n_ + multivisit);
388
389 assert(WLDMInvariantsHold());
390
391 // Increment N.
392 n_ += multivisit;
393 // Decrement virtual loss.
394 assert(GetNInFlight() >= (uint32_t)multivisit);
395 n_in_flight_.fetch_sub(multivisit, std::memory_order_acq_rel);
396}

421void LowNode::ReleaseChildrenExceptOne(Node* node_to_save) {
422 auto& ngc = NodeGarbageCollector::Instance();
423 // Stores node which will have to survive (or nullptr if it's not found).
424 std::unique_ptr<Node> saved_node;
425 // Pointer to atomic_unique_ptr, so that we could move from it.
426 for (auto node = &child_; *node; node = (*node)->GetSibling()) {
427 // If current node is the one that we have to save.
428 if (node->get() == node_to_save) {
429 // Kill all remaining siblings.
430 ngc.AddToGcQueue(*(*node)->GetSibling());
431 // Save the node, and take the ownership from the unique_ptr.
432 saved_node.reset(node->release());
433 break;
434 }
435 }
436 // Make saved node the only child. (kills previous siblings).
437 ngc.AddToGcQueue(child_);
438 child_ = std::move(saved_node);
439}

480template <typename VisitorType, typename EdgeVisitorType>
481static void TreeWalk(const Node* node, bool as_opponent,
482 Node::VisitorId::type id, VisitorType visitor,
483 EdgeVisitorType edge) {
484 const std::shared_ptr<LowNode>& low_node = node->GetLowNode();
485 if (!low_node || !low_node->Visit(id)) {
486 return;
487 }
488
489 visitor(low_node.get(), as_opponent);
490
491 for (auto& child_edge : node->Edges()) {
492 auto child = child_edge.node();
493 if (child == nullptr) {
494 break;
495 }
496 edge(child, as_opponent, low_node.get());
497 }
498
499 for (auto& child_edge : node->Edges()) {
500 auto child = child_edge.node();
501 if (child == nullptr) {
502 return;
503 }
504 TreeWalk(child, !as_opponent, id, visitor, edge);
505 }
506}

627static constexpr float wld_tolerance = 0.000001f;
628static constexpr float m_tolerance = 0.000001f;
629
630static bool WLDMInvariantsHold(float wl, float d, float m) {
631 return -(1.0f + wld_tolerance) < wl && wl < (1.0f + wld_tolerance) && //
632 -(0.0f + wld_tolerance) < d && d < (1.0f + wld_tolerance) && //
633 -(0.0f + m_tolerance) < m && //
634 std::abs(wl) + std::abs(d) < (1.0f + wld_tolerance);
635}
636
637bool Node::WLDMInvariantsHold() const {
638 if (dag_classic::WLDMInvariantsHold(GetWL(), GetD(), GetM())) return true;
639
640 std::cerr << DebugString() << std::endl;
641
642 return false;
643}

648 std::cerr << DebugString() << std::endl;
649
650 return false;

676void NodeTree::MakeMove(Move move) {
677 Node* new_head = nullptr;
678 for (auto& n : current_head_->Edges()) {
679 if (n.GetMove() == move) {
680 new_head = n.GetOrSpawnNode(current_head_);
681 // Ensure head is not terminal, so search can extend or visit children of
682 // "terminal" positions, e.g., WDL hits, converted terminals, 3-fold draw.
683 if (new_head->IsTerminal()) new_head->MakeNotTerminal();
684 break;
685 }
686 }
687 // Release nodes from last move if any.
688 current_head_->ReleaseChildrenExceptOne(new_head);
689 new_head = current_head_->GetChild();
690 current_head_ =
691 new_head ? new_head : current_head_->CreateSingleChildNode(move);
692 history_.Append(move);
693 moves_.push_back(move);
694}

698void NodeTree::TrimTreeAtHead() {
699 current_head_->Trim();
700 // Flush the thread local destruction queue.
701 NodeGarbageCollector::Instance().NotifyThreadGoingSleep();

705bool NodeTree::ResetToPosition(const GameState& pos) {
706 if (gamebegin_node_ && (history_.Starting() != pos.startpos)) {
707 // Completely different position.
708 DeallocateTree();
709 }
710
711 if (!gamebegin_node_) {
712 gamebegin_node_ = std::make_unique<Node>(0);
713 }
714
715 history_.Reset(pos.startpos);
716 moves_.clear();
717
718 Node* old_head = current_head_;
719 current_head_ = gamebegin_node_.get();
720 bool seen_old_head = (gamebegin_node_.get() == old_head);
721 for (const Move m : pos.moves) {
722 MakeMove(m);
723 if (old_head == current_head_) seen_old_head = true;
724 }
725
726 // MakeMove guarantees that no siblings exist; but, if we didn't see the old
727 // head, it means we might have a position that was an ancestor to a
728 // previously searched position, which means that the current_head_ might
729 // retain old n_ and q_ (etc) data, even though its old children were
730 // previously trimmed; we need to reset current_head_ in that case.
731 if (!seen_old_head) TrimTreeAtHead();
732 NodeGarbageCollector::Instance().NotifyThreadGoingSleep();
733 return seen_old_head;
734}
735
736bool NodeTree::ResetToPosition(const std::string& starting_fen,
737 const std::vector<std::string>& moves) {
738 GameState state;
739 state.startpos = Position::FromFen(starting_fen);
740 ChessBoard cur_board = state.startpos.GetBoard();
741 state.moves.reserve(moves.size());
742 for (const auto& move : moves) {
743 Move m = cur_board.ParseMove(move);
744 state.moves.push_back(m);
745 cur_board.ApplyMove(m);
746 cur_board.Mirror();
747 }
748 return ResetToPosition(state);
749}

837bool NodeGarbageCollector::ShouldQueue(std::unique_ptr<Node>& node) const {
838 // We don't want to queue null pointers.
839 if (!node) {
840 return false;
841 }
842
843 // If state is exit, it means thread local queues have been destroyed.
844 State s = state_.load(std::memory_order_acquire);
845 if (s == Exit) {
846 return false;
847 }
848
849 // We directly free the node, if queue is running and we are in the GC thread.
850 // All other queue request should be pushed to the thread local batch.
851 return s != Running || !LocalWork().IsWorker();
852}
853
855void NodeGarbageCollector::GCThread() {
856 auto& shared_work = LocalWork(true);
857 assert(shared_work.IsWorker());
858 State s;
859 while ((s = state_.load(std::memory_order_acquire)) != Exit) {
860 if (s == GoToSleep) {
861 // Signal other threads that we have stopped destruction work.
862 if (SetState(s, Sleeping)) {
863 s = Sleeping;
864 } else {
865 continue;
866 }
867 }
868 if (s == Sleeping) {
869#ifndef NO_STD_ATOMIC_WAIT
870 state_.wait(Sleeping, std::memory_order_acquire);
871#else
872 Mutex::Lock lock(state_mutex_);
873 state_signal_.wait(lock.get_raw(),
874 [this]() { return Sleeping != state_; });
875#endif
876 if (!shared_work.empty()) {
877 // Check for early exit from previous free. The work can be freed
878 // before the batch is full.
879 ReleaseNodesWork new_work(true);
880 new_work.swap(shared_work);
881 }
882 continue;
883 }
884
885 assert(s == Running);
886
887 bool empty = true;
888 std::vector<std::unique_ptr<Node>> nodes;
889 {
890 SpinMutex::Lock lock(mutex_);
891 if (!released_nodes_.empty()) {
892 empty = false;
893 nodes = std::move(released_nodes_.front());
894 released_nodes_.pop_front();
895 }
896 }
897
898 if (!empty) {
899 LOGFILE << "Garbage collection starting.";
900 }
901
902 // Free nodes one by one. LowNode destructor calls AddToGcQueue which allows
903 // recursive destruction terminate before freeing a whole branch.
904 while (!nodes.empty()) {
905 if (!IsActive()) {
906 break;
907 }
908 nodes.pop_back();
909 }
910
911 if (!empty) {
912 LOGFILE << "Garbage collection ending.";
913 }
914
915 // Go to sleep if empty or search stopped.
916 if (empty || !IsActive()) {
917 // Lock is requrired to avoid race between other thread queueing work and
918 // calling Start().
919 SpinMutex::Lock lock(mutex_);
920 // There wasn't enough time to free all nodes. They must go back to the
921 // list.
922 if (!nodes.empty()) {
923 released_nodes_.emplace_front(std::move(nodes));
924 }
925
926 // Going to sleep if the queue is empty.
927 if (released_nodes_.empty()) {
928 State old = Running;
929 SetState(old, Sleeping);
930 }
931 }
932 }
933}

958void ReleaseNodesWork::Submit() {
959 if (released_nodes_.empty()) {
960 return;
961 }
962 auto& worker = NodeGarbageCollector::Instance();
963 SpinMutex::Lock lock(worker.mutex_);
964 // If this is worker, we have oldest nodes. Keep them at front of the queue.
965 if (IsWorker()) {
966 worker.released_nodes_.emplace_front(std::move(released_nodes_));
967 } else {
968 worker.released_nodes_.emplace_back(std::move(released_nodes_));
969 }
970}
