88MoveList StringsToMovelist(const std::vector<std::string>& moves,
89 const ChessBoard& board) {
90 MoveList result;
91 if (moves.size()) {
92 result.reserve(moves.size());
93 const auto legal_moves = board.GenerateLegalMoves();
94 for (const auto& move : moves) {
95 const Move m = board.ParseMove(move);
96 if (std::find(legal_moves.begin(), legal_moves.end(), m) !=
97 legal_moves.end()) {
98 result.emplace_back(m);
99 }
100 }
101 if (result.empty()) throw Exception("No legal searchmoves.");
102 }
103 return result;
104}

123void DagClassicSearch::StartSearch(const GoParams& params) {
124 LCTRACE_FUNCTION_SCOPE;
125 auto forwarder =
126 std::make_unique<NonOwningUciRespondForwarder>(uci_responder_);
127 if (options_->Get<Button>(kClearTree).TestAndReset()) {
128 tree_->TrimTreeAtHead();
129 LOGFILE << "Tree cleared.";
130 }
131
132 const auto cache_size =
133 options_->Get<int>(SharedBackendParams::kNNCacheSizeId);
134 // FIXME: This is too conservative.
135 const size_t kAvgNodeSize =
136 sizeof(Node) + sizeof(LowNode) +
137 classic::MemoryWatchingStopper::kAvgMovesPerPosition * sizeof(Edge);
138 const size_t kAvgCacheItemSize =
139 3 * sizeof(float) + sizeof(std::unique_ptr<float[]>) +
140 sizeof(float[classic::MemoryWatchingStopper::kAvgMovesPerPosition]);
141 size_t total_memory =
142 tree_.get()->GetCurrentHead()->GetN() * kAvgNodeSize +
143 (sizeof(TranspositionTable::value_type) + 1) * tt_.bucket_count() +
144 cache_size * kAvgCacheItemSize;
145 auto stopper = time_manager_->GetStopper(
146 params, tree_.get()->HeadPosition(), total_memory, kAvgNodeSize,
147 tree_.get()->GetCurrentHead()->GetN());
148 search_ = std::make_unique<Search>(
149 *tree_, backend_, std::move(forwarder),
150 StringsToMovelist(params.searchmoves, tree_->HeadPosition().GetBoard()),
151 *move_start_time_, std::move(stopper), params.infinite, params.ponder,
152 *options_, &tt_, syzygy_tb_);
153
154 LOGFILE << "Timer started at "
155 << FormatTime(SteadyClockToSystemClock(*move_start_time_));
156 search_->StartThreads(options_->Get<int>(kThreadsOptionId));
157}
