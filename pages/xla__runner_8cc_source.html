122void XlaRunner::AddModule(size_t minibatch_size,
123 const pblczero::HloModuleProto& module) {
124 pblczero::CompileOptionsProto options;
125 options.mutable_executable_build_options()->set_num_replicas(1);
126 options.mutable_executable_build_options()->set_num_partitions(1);
127 options.mutable_executable_build_options()
128 ->mutable_device_assignment()
129 ->set_replica_count(1);
130 options.mutable_executable_build_options()
131 ->mutable_device_assignment()
132 ->set_computation_count(1);
133 options.mutable_executable_build_options()
134 ->mutable_device_assignment()
135 ->add_computation_devices()
136 ->add_replica_device_ids(device_);
137 auto executable = pjrt_client_->CompileHlo(module.OutputAsString(),
138 options.OutputAsString());
139 executables_.push_back({minibatch_size, std::move(executable)});
140 std::sort(executables_.begin(), executables_.end());
141}

143void XlaRunner::SetFrozenInputs(
144 const std::vector<std::unique_ptr<XlaTensor>> inputs) {
145 param_idxs_.clear();
146 std::vector<std::unique_ptr<PjrtHostToDeviceTransfer>> transfers_;
147 for (size_t i = 0; i < inputs.size(); ++i) {
148 const auto* input = inputs[i].get();
149 if (!input) {
150 param_idxs_.push_back(i);
151 continue;
152 }
153 transfers_.push_back(pjrt_client_->HostToDevice(
154 {static_cast<const char*>(input->data()), input->size()},
155 XlaTypeToPjrtType(input->type()), input->shape(),
156 devices_.at(device_).get()));
157 }
158
159 owned_buffers_.clear();
160 buffers_.clear();
161 buffers_.resize(inputs.size());
162 size_t transfer_idx = 0;
163 for (size_t i = 0; i < inputs.size(); ++i) {
164 if (inputs[i]) {
165 owned_buffers_.push_back(
166 transfers_[transfer_idx++]->AwaitAndReleaseBuffer());
167 buffers_[i] = owned_buffers_.back().get();
168 }
169 }
170}

175std::vector<std::unique_ptr<XlaMutableTensor>> XlaRunner::ExecuteBlocking(
176 const std::vector<XlaMutableTensor*>& inputs) {
177 if (inputs.size() != 1) {
178 throw Exception("Only one input is kinda supported.");
179 }
180 // Find the smallest batch size that fits the input.
181 auto iter = std::find_if(
182 executables_.begin(), executables_.end(), [&](const auto& e) {
183 return e.first >= static_cast<size_t>(inputs[0]->shape()[0]);
184 });
185 if (iter == executables_.end()) {
186 throw Exception("No executable found for batch size " +
187 std::to_string(inputs[0]->shape()[0]));
188 }
189 const size_t batch_size = iter->first;
190 // Update the shape to match the rounded up batch size. After growing, the
191 // batch size must fit within tensor buffer capacity (it's fine to have
192 // garbage in the tail of that buffer).
193 std::vector<int64_t> new_shape = inputs[0]->shape();
194 new_shape[0] = batch_size;
195 inputs[0]->Reshape(new_shape);
196 // Transfer the input to the device.
197 auto input_buffer =
198 pjrt_client_
199 ->HostToDevice(
200 {static_cast<const char*>(inputs[0]->data()), inputs[0]->size()},
201 XlaTypeToPjrtType(inputs[0]->type()), new_shape,
202 devices_.at(device_).get())
203 ->AwaitAndReleaseBuffer();
204 // Make a copy to support multiple concurrent calls, not sure if it's needed.
205 auto input_buffers = buffers_;
206 input_buffers[param_idxs_[0]] = input_buffer.get();
207 // Execute!
208 auto outputs = iter->second->ExecuteBlocking(input_buffers);
209
210 // Now we need to transfer the outputs back to the host.
211 std::vector<std::unique_ptr<XlaMutableTensor>> result;
212 result.reserve(outputs.size());
213 std::vector<std::unique_ptr<PjrtEvent>> done_events;
214 done_events.reserve(outputs.size());
215 // Initialte transfers from device to host.
216 for (size_t i = 0; i < outputs.size(); ++i) {
217 const auto& output = outputs[i];
218 auto new_tensor = std::make_unique<XlaMutableTensor>(
219 PjrtTypeToXlaType(output->GetType()), output->GetDimensions());
220 done_events.push_back(
221 output->DeviceToHost(new_tensor->mutable_data(), new_tensor->size()));
222 result.push_back(std::move(new_tensor));
223 }
224 // Wait for the transfers to complete.
225 for (size_t i = 0; i < outputs.size(); ++i) done_events[i]->Await();
226 return result;
227}
