55 // Starts the games and blocks until all games are finished.
56 void Play();

59 void Abort();

61 GameResult GetGameResult(int index) const { return results_[index]; }
62
63 std::vector<Move> GetMoves(int index) const {
64 std::vector<Move> moves;
65 bool flip = !trees_[index]->IsBlackToMove();
66 for (classic::Node* node = trees_[index]->GetCurrentHead();
67 node != trees_[index]->GetGameBeginNode(); node = node->GetParent()) {
68 moves.push_back(node->GetParent()->GetEdgeToNode(node)->GetMove(flip));
69 flip = !flip;
70 }
71 std::reverse(moves.begin(), moves.end());
72 return moves;
73 }
74
75 private:
76 // options_[0] is for white player, [1] for black.
77 PlayerOptions options_[2];
78 // Node tree for player1 and player2. If the tree is shared between players,
79 // tree_[0] == tree_[1].
80 std::vector<std::shared_ptr<classic::NodeTree>> trees_;
81 std::vector<GameResult> results_;
82 bool abort_ = false;
83 std::mutex mutex_;
84 SyzygyTablebase* syzygy_tb_;
85 std::unique_ptr<Evaluator> eval_;
86};
