50inline bool GzGetLine(gzFile file, std::string& line) {
51 bool flag = false;
52 char s[2000];
53 line.clear();
54 while (gzgets(file, s, sizeof(s))) {
55 flag = true;
56 line += s;
57 auto r = line.find_last_of('\n');
58 if (r != std::string::npos) {
59 line.erase(r);
60 break;
61 }
62 }
63 return flag;
64}

66class PgnReader {
67 public:
68 void AddPgnFile(const std::string& filepath) {
69 const gzFile file = gzopen(filepath.c_str(), "r");
70 if (!file) {
71 throw Exception(errno == ENOENT ? "Opening book file not found."
72 : "Error opening opening book file.");
73 }
74
75 std::string line;
76 bool in_comment = false;
77 bool started = false;
78 while (GzGetLine(file, line)) {
79 // Check if we have a UTF-8 BOM. If so, just ignore it.
80 // Only supposed to exist in the first line, but should not matter.
81 if (line.substr(0, 3) == "\xEF\xBB\xBF") line = line.substr(3);
82 if (!line.empty() && line.back() == '\r') line.pop_back();
83 // TODO: support line breaks in tags to ensure they are properly ignored.
84 if (line.empty() || line[0] == '[') {
85 if (started) {
86 Flush();
87 started = false;
88 }
89 auto uc_line = line;
90 std::transform(
91 uc_line.begin(), uc_line.end(), uc_line.begin(),
92 [](unsigned char c) { return std::toupper(c); } // correct
93 );
94 if (uc_line.find("[FEN \"", 0) == 0) {
95 auto start_trimmed = line.substr(6);
96 cur_startpos_ = start_trimmed.substr(0, start_trimmed.find('"'));
97 cur_board_.SetFromFen(cur_startpos_);
98 }
99 continue;
100 }
101 // Must have at least one non-tag non-empty line in order to be considered
102 // a game.
103 started = true;
104 // Handle braced comments.
105 int cur_offset = 0;
106 while ((in_comment && line.find('}', cur_offset) != std::string::npos) ||
107 (!in_comment && line.find('{', cur_offset) != std::string::npos)) {
108 if (in_comment && line.find('}', cur_offset) != std::string::npos) {
109 line = line.substr(0, cur_offset) +
110 line.substr(line.find('}', cur_offset) + 1);
111 in_comment = false;
112 } else {
113 cur_offset = line.find('{', cur_offset);
114 in_comment = true;
115 }
116 }
117 if (in_comment) {
118 line = line.substr(0, cur_offset);
119 }
120 // Trim trailing comment.
121 if (line.find(';') != std::string::npos) {
122 line = line.substr(0, line.find(';'));
123 }
124 if (line.empty()) continue;
125 std::istringstream iss(line);
126 std::string word;
127 while (!iss.eof()) {
128 word.clear();
129 iss >> word;
130 if (word.size() < 2) continue;
131 // Trim move numbers from front.
132 const auto idx = word.find('.');
133 if (idx != std::string::npos) {
134 bool all_nums = true;
135 for (size_t i = 0; i < idx; i++) {
136 if (word[i] < '0' || word[i] > '9') {
137 all_nums = false;
138 break;
139 }
140 }
141 if (all_nums) {
142 word = word.substr(idx + 1);
143 }
144 }
145 // Pure move numbers can be skipped.
146 if (word.size() < 2) continue;
147 // Ignore score line.
148 if (word == "1/2-1/2" || word == "1-0" || word == "0-1" || word == "*")
149 continue;
150 cur_game_.push_back(SanToMove(word, cur_board_));
151 cur_board_.ApplyMove(cur_game_.back());
152 // Board ApplyMove wants mirrored for black, but outside code wants
153 // normal, so mirror it back again.
154 // Check equal to 0 since we've already added the position.
155 if ((cur_game_.size() % 2) == 0) cur_game_.back().Flip();
156 cur_board_.Mirror();
157 }
158 }
159 if (started) {
160 Flush();
161 }
162 gzclose(file);
163 }
164 std::vector<Opening> GetGames() const { return games_; }
165 std::vector<Opening>&& ReleaseGames() { return std::move(games_); }

167 private:
168 void Flush() {
169 games_.push_back({cur_startpos_, cur_game_});
170 cur_game_.clear();
171 cur_board_.SetFromFen(ChessBoard::kStartposFen);
172 cur_startpos_ = ChessBoard::kStartposFen;
173 }

175 static std::optional<PieceType> PieceToPieceType(int p) {
176 switch (p) {
177 case -1:
178 return std::nullopt;
179 case 2:
180 return kQueen;
181 case 3:
182 return kBishop;
183 case 4:
184 return kKnight;
185 case 5:
186 return kRook;
187 default:
188 // 0 and 1 are pawn and king, which are not legal promotions, other
189 // numbers don't correspond to a known piece type.
190 CERR << "Unexpected promotion!!";
191 throw Exception("Trying to create a move with illegal promotion.");
192 }
193 }

195 Move SanToMove(const std::string& san, const ChessBoard& board) {
196 int p = 0;
197 size_t idx = 0;
198 if (san[0] == 'K') {
199 p = 1;
200 } else if (san[0] == 'Q') {
201 p = 2;
202 } else if (san[0] == 'B') {
203 p = 3;
204 } else if (san[0] == 'N') {
205 p = 4;
206 } else if (san[0] == 'R') {
207 p = 5;
208 } else if (san.substr(0, 3) == "O-O") {
209 Move m;
210 auto king_board = board.kings() & board.ours();
211 Square king_sq(File::FromIdx(GetLowestBit(king_board.as_int())), kRank1);
212 m = Move::WhiteCastling(king_sq.file(),
213 san.substr(3, 2) == "-O"
214 ? board.castlings().our_queenside_rook
215 : board.castlings().our_kingside_rook);
216 return m;
217 }
218 if (p != 0) idx++;
219 // Formats e4 1e5 de5 d1e5 - with optional x's - followed by =Q for
220 // promotions, and even more characters after that also optional.
221 int r1 = -1;
222 int c1 = -1;
223 int r2 = -1;
224 int c2 = -1;
225 int p2 = -1;
226 bool pPending = false;
227 for (; idx < san.size(); idx++) {
228 if (san[idx] == 'x') continue;
229 if (san[idx] == '=') {
230 pPending = true;
231 continue;
232 }
233 if (san[idx] >= '1' && san[idx] <= '8') {
234 r1 = r2;
235 r2 = san[idx] - '1';
236 continue;
237 }
238 if (san[idx] >= 'a' && san[idx] <= 'h') {
239 c1 = c2;
240 c2 = san[idx] - 'a';
241 continue;
242 }
243 if (pPending) {
244 if (san[idx] == 'Q') {
245 p2 = 2;
246 } else if (san[idx] == 'B') {
247 p2 = 3;
248 } else if (san[idx] == 'N') {
249 p2 = 4;
250 } else if (san[idx] == 'R') {
251 p2 = 5;
252 }
253 pPending = false;
254 break;
255 }
256 break;
257 }
258 if (r1 == -1 || c1 == -1) {
259 // Need to find the from cell based on piece.
260 int sr1 = r1;
261 int sr2 = r2;
262 if (board.flipped()) {
263 if (sr1 != -1) sr1 = 7 - sr1;
264 sr2 = 7 - sr2;
265 }
266 BitBoard searchBits;
267 if (p == 0) {
268 searchBits = (board.pawns() & board.ours());
269 } else if (p == 1) {
270 searchBits = (board.kings() & board.ours());
271 } else if (p == 2) {
272 searchBits = (board.queens() & board.ours());
273 } else if (p == 3) {
274 searchBits = (board.bishops() & board.ours());
275 } else if (p == 4) {
276 searchBits = (board.knights() & board.ours());
277 } else if (p == 5) {
278 searchBits = (board.rooks() & board.ours());
279 }
280 auto plm = board.GenerateLegalMoves();
281 int pr1 = -1;
282 int pc1 = -1;
283 for (Square sq : searchBits) {
284 if (sr1 != -1 && sq.rank().idx != sr1) continue;
285 if (c1 != -1 && sq.file().idx != c1) continue;
286 std::optional<PieceType> promotion = PieceToPieceType(p2);
287 std::optional<Square> enpassant = std::nullopt;
288 if (!board.en_passant().empty()) {
289 auto sq = *board.en_passant().begin();
290 enpassant = Square(sq.file(), kRank6);
291 }
292 Square to(File::FromIdx(c2), Rank::FromIdx(sr2));
293 Move move_to_find = promotion ? Move::WhitePromotion(sq, to, *promotion)
294 : enpassant && *enpassant == to
295 ? Move::WhiteEnPassant(sq, to)
296 : Move::White(sq, to);
297 if (std::find(plm.begin(), plm.end(), move_to_find) == plm.end()) {
298 continue;
299 }
300 if (pc1 != -1) {
301 CERR << "Ambiguous!!";
302 throw Exception("Opening book move seems ambiguous.");
303 }
304 pr1 = sq.rank().idx;
305 pc1 = sq.file().idx;
306 }
307 if (pc1 == -1) {
308 CERR << "No Match!!";
309 throw Exception("Opening book move seems illegal.");
310 }
311 r1 = pr1;
312 c1 = pc1;
313 if (board.flipped()) {
314 r1 = 7 - r1;
315 }
316 }
317 std::optional<PieceType> promotion = PieceToPieceType(p2);
318
319 std::optional<Square> enpassant = std::nullopt;
320 if (!board.en_passant().empty()) {
321 auto sq = *board.en_passant().begin();
322 enpassant = Square(sq.file(), board.flipped() ? kRank3 : kRank6);
323 }
324 Square from(File::FromIdx(c1), Rank::FromIdx(r1));
325 Square to(File::FromIdx(c2), Rank::FromIdx(r2));
326 Move m = promotion ? Move::WhitePromotion(from, to, *promotion)
327 : enpassant && *enpassant == to ? Move::WhiteEnPassant(from, to)
328 : Move::White(from, to);
329 if (board.flipped()) m.Flip();
330 return m;
331 }

336 std::vector<Opening> games_;
337};
