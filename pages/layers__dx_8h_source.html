42// The Layer objects only hold memory for weights, biases, etc
43// memory for input and output tensors is provided by caller of Eval.
44
45class BaseLayer {
46 public:
47 int GetC() const { return C; }
48 int GetH() const { return H; }
49 int GetW() const { return W; }
50
51 BaseLayer(int c, int h, int w, BaseLayer* ip, DxContext* dx_context,
52 bool fp16);
53 virtual ~BaseLayer() = default;
54 size_t GetOutputSize(int N) const {
55 return (fp16_ ? sizeof(dx_half) : sizeof(float)) * N * C * H * W;
56 }
57
58 // input2 is optional (skip connection).
59 virtual void Eval(int N, DXAlloc output, DXAlloc input, DXAlloc input2,
60 DXAlloc scratch, DXAlloc scratch2,
61 ID3D12GraphicsCommandList4* command_list) = 0;
62
63 protected:
64 BaseLayer* input_;
65 DxContext* dx_context_;
66
67 bool fp16_;
68
69 // Output tensor dimensions.
70 int C;
71 int H;
72 int W;
73};

75// Holds Metacommand objects and their scratch space for all allowed batch
76// sizes.
77class GemmMetaCommand {
78 private:
79 // Need to create a Metacommand object for each batch size unfortunately!
80 // Some hw vendors don't support arbitary sizes anyway, so we create only
81 // multiples of 8 in no. of rows (when M is 0).
82
83 static constexpr int kMetacommandGranulity = 8;
84 static constexpr int kMaxMetacommands =
85 (kMaxSupportedBatchSize * 4) / kMetacommandGranulity;
86 ID3D12MetaCommand* meta_commands_[kMaxMetacommands];
87
88 DXAlloc scratch_data_persistent_[kMaxMetacommands];
89 DXAlloc scratch_data_temporary_[kMaxMetacommands];
90
91 bool rows_known_;
92 bool create_succeeded_;
93
94 public:
95 GemmMetaCommand(DxContext* dx_context, int M, int N, int K, int gemm_batch,
96 bool fp16, bool a_transpose, bool b_transpose);
97 ~GemmMetaCommand();
98
99 void PerformGemm(int rows, DXAlloc A, DXAlloc B, DXAlloc Output,
100 ID3D12GraphicsCommandList4* command_list);
101
102 bool IsAvailable() { return create_succeeded_; }
103};
