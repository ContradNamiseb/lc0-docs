39class BaseLayer {
40 public:

48 BaseLayer(int c, int h, int w, BaseLayer* ip, sycl::queue& sycl_queue);
49 BaseLayer(int c, int h, int w, BaseLayer* ip, bool nhwc, sycl::queue& sycl_queue);
50 virtual ~BaseLayer() = default;
51 size_t GetOutputSize(int N) const { return sizeof(DataType) * N * C * H * W; }
52
53 // Input2 is optional (skip connection).
54 //virtual void Eval(int N, DataType* output, const DataType* input,
55 // const DataType* input2, void* scratch, size_t scratch_size,
56 // cudnnHandle_t cudnn, dpct::queue_ptr cublas,
57 // dpct::queue_ptr stream, DataType*** = nullptr) = 0;

60 const DataType* input2, void* scratch, size_t scratch_size,

63 protected:
64 BaseLayer* input_;

67 int C; // Output tensor dimensions.

69 int W;
70
71 bool nhwc_; // tensor layout
72
73 void cublasRowMajorMatrixMul(const DataType* A, const DataType* B,
