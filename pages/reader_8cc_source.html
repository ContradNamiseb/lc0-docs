32InputPlanes PlanesFromTrainingData(const V6TrainingData& data) {
33 InputPlanes result;
34 for (int i = 0; i < 104; i++) {
35 result.emplace_back();
36 result.back().mask = ReverseBitsInBytes(data.planes[i]);
37 }
38 switch (data.input_format) {
39 case pblczero::NetworkFormat::INPUT_CLASSICAL_112_PLANE: {
40 result.emplace_back();
41 result.back().mask = data.castling_us_ooo != 0 ? ~0LL : 0LL;
42 result.emplace_back();
43 result.back().mask = data.castling_us_oo != 0 ? ~0LL : 0LL;
44 result.emplace_back();
45 result.back().mask = data.castling_them_ooo != 0 ? ~0LL : 0LL;
46 result.emplace_back();
47 result.back().mask = data.castling_them_oo != 0 ? ~0LL : 0LL;
48 break;
49 }
50 case pblczero::NetworkFormat::INPUT_112_WITH_CASTLING_PLANE:
51 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION:
52 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_HECTOPLIES:
53 case pblczero::NetworkFormat::
54 INPUT_112_WITH_CANONICALIZATION_HECTOPLIES_ARMAGEDDON:
55 case pblczero::NetworkFormat::INPUT_112_WITH_CANONICALIZATION_V2:
56 case pblczero::NetworkFormat::
57 INPUT_112_WITH_CANONICALIZATION_V2_ARMAGEDDON: {
58 result.emplace_back();
59 result.back().mask =
60 data.castling_us_ooo |
61 (static_cast<uint64_t>(data.castling_them_ooo) << 56);
62 result.emplace_back();
63 result.back().mask = data.castling_us_oo |
64 (static_cast<uint64_t>(data.castling_them_oo) << 56);
65 // 2 empty planes in this format.
66 result.emplace_back();
67 result.emplace_back();
68 break;
69 }
70
71 default:
72 throw Exception("Unsupported input plane encoding " +
73 std::to_string(data.input_format));
74 }
75 result.emplace_back();
76 auto typed_format =
77 static_cast<pblczero::NetworkFormat::InputFormat>(data.input_format);
78 if (IsCanonicalFormat(typed_format)) {
79 result.back().mask = static_cast<uint64_t>(data.side_to_move_or_enpassant)
80 << 56;
81 } else {
82 result.back().mask = data.side_to_move_or_enpassant != 0 ? ~0LL : 0LL;
83 }
84 result.emplace_back();
85 if (IsHectopliesFormat(typed_format)) {
86 result.back().Fill(data.rule50_count / 100.0f);
87 } else {
88 result.back().Fill(data.rule50_count);
89 }
90 result.emplace_back();
91 // Empty plane, except for canonical armageddon.
92 if (IsCanonicalArmageddonFormat(typed_format) &&
93 data.invariance_info >= 128) {
94 result.back().SetAll();
95 }
96 result.emplace_back();
97 // All ones plane.
98 result.back().SetAll();
99 if (IsCanonicalFormat(typed_format) && data.invariance_info != 0) {
100 // Undo transformation here as it makes the calling code simpler.
101 int transform = data.invariance_info;
102 for (size_t i = 0; i <= result.size(); i++) {
103 auto v = result[i].mask;
104 if (v == 0 || v == ~0ULL) continue;
105 if ((transform & TransposeTransform) != 0) {
106 v = TransposeBitsInBytes(v);
107 }
108 if ((transform & MirrorTransform) != 0) {
109 v = ReverseBytesInBytes(v);
110 }
111 if ((transform & FlipTransform) != 0) {
112 v = ReverseBitsInBytes(v);
113 }
114 result[i].mask = v;
115 }
116 }
117 return result;
118}

130bool TrainingDataReader::ReadChunk(V6TrainingData* data) {
131 if (format_v6) {
132 int read_size = gzread(fin_, reinterpret_cast<void*>(data), sizeof(*data));
133 if (read_size < 0) throw Exception("Corrupt read.");
134 return read_size == sizeof(*data);
135 } else {
136 int v6_extra = 48;
137 int v5_extra = 16;
138 int v4_extra = 16;
139 int v3_size = sizeof(*data) - v4_extra - v5_extra - v6_extra;
140 int read_size = gzread(fin_, reinterpret_cast<void*>(data), v3_size);
141 if (read_size < 0) throw Exception("Corrupt read.");
142 if (read_size != v3_size) return false;
143 auto orig_version = data->version;
144 switch (data->version) {
145 case 3: {
146 data->version = 4;
147 // First convert 3 to 4 to reduce code duplication.
148 char* v4_extra_start = reinterpret_cast<char*>(data) + v3_size;
149 // Write 0 bytes for 16 extra bytes - corresponding to 4 floats of 0.0f.
150 for (int i = 0; i < v4_extra; i++) {
151 v4_extra_start[i] = 0;
152 }
153 [[fallthrough]];
154 }
155 case 4: {
156 // If actually 4, we need to read the additional data first.
157 if (orig_version == 4) {
158 read_size = gzread(
159 fin_,
160 reinterpret_cast<void*>(reinterpret_cast<char*>(data) + v3_size),
161 v4_extra);
162 if (read_size < 0) throw Exception("Corrupt read.");
163 if (read_size != v4_extra) return false;
164 }
165 data->version = 5;
166 char* data_ptr = reinterpret_cast<char*>(data);
167 // Shift data after version back 4 bytes.
168 memmove(data_ptr + 2 * sizeof(uint32_t), data_ptr + sizeof(uint32_t),
169 v3_size + v4_extra - sizeof(uint32_t));
170 data->input_format = pblczero::NetworkFormat::INPUT_CLASSICAL_112_PLANE;
171 data->root_m = 0.0f;
172 data->best_m = 0.0f;
173 data->plies_left = 0.0f;
174 [[fallthrough]];
175 }
176 case 5: {
177 // If actually 5, we need to read the additional data first.
178 if (orig_version == 5) {
179 read_size = gzread(
180 fin_,
181 reinterpret_cast<void*>(reinterpret_cast<char*>(data) + v3_size),
182 v4_extra + v5_extra);
183 if (read_size < 0) throw Exception("Corrupt read.");
184 if (read_size != v4_extra + v5_extra) return false;
185 }
186 data->version = 6;
187 // Type of dummy was changed from signed to unsigned - which means -1 on
188 // disk is read in as 255.
189 if (data->dummy > 1 && data->dummy < 255) {
190 throw Exception("Invalid result read in v5 data before upgrade.");
191 }
192 data->result_q =
193 data->dummy == 255 ? -1.0f : (data->dummy == 0 ? 0.0f : 1.0f);
194 data->result_d = data->dummy == 0 ? 1.0f : 0.0f;
195 data->dummy = 0;
196 data->played_q = 0.0f;
197 data->played_d = 0.0f;
198 data->played_m = 0.0f;
199 // Mark orig as NaN since scripts further downstream already have to
200 // handle that case.
201 data->orig_q = std::numeric_limits<float>::quiet_NaN();
202 data->orig_d = std::numeric_limits<float>::quiet_NaN();
203 data->orig_m = std::numeric_limits<float>::quiet_NaN();
204 data->visits = 0;
205 data->played_idx = 0;
206 data->best_idx = 0;
207 data->policy_kld = 0.0f;
208 data->reserved = 0;
209 return true;
210 }
211 case 6: {
212 format_v6 = true;
213 read_size = gzread(
214 fin_,
215 reinterpret_cast<void*>(reinterpret_cast<char*>(data) + v3_size),
216 v4_extra + v5_extra + v6_extra);
217 if (read_size < 0) throw Exception("Corrupt read.");
218 return read_size == v4_extra + v5_extra + v6_extra;
219 }
220 default:
221 throw Exception("Unknown format.");
222 }
223 }
