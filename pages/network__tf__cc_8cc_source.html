53Output MakeConst(const Scope& scope, TensorShape shape,
54 const std::vector<float>& values,
55 const std::vector<int>& order = {}) {
56 auto tensor = Tensor(DataType::DT_FLOAT, shape);
57 CHECK_EQ(tensor.NumElements(), static_cast<int>(values.size()))
58 << shape.DebugString();
59
60 std::vector<int> dims;
61 for (const auto& x : shape) {
62 dims.push_back(x.size);
63 }
64 TransposeTensor(dims, order, values, tensor.flat<float>().data());
65
66 return Const(scope, tensor);
67}
68
69Output MakeIntConst(const Scope& scope, TensorShape shape,
70 const std::vector<int32_t>& values) {
71 auto tensor = Tensor(DataType::DT_INT32, shape);
72 CHECK_EQ(tensor.NumElements(), static_cast<int>(values.size()))
73 << shape.DebugString();
74 memcpy(tensor.flat<int32_t>().data(), values.data(),
75 values.size() * sizeof(values[0]));
76 return Const(scope, tensor);
77}

79template <bool CPU>
80Output SqueezeAndExcite(const Scope& scope, Input input, int channels,
81 const LegacyWeights::SEunit& weights,
82 const std::string& basename) {
83 const int se_channels = weights.b1.size();
84 // NCHW ("NHWC" for CPU case) format reduced to NC.
85 auto pooled = Mean(scope, input, CPU ? Input({1, 2}) : Input({2, 3}));
86 auto w1 = MakeConst(scope, {channels, se_channels}, weights.w1);
87 auto b1 = MakeConst(scope, {se_channels}, weights.b1);
88 auto dense1_mul = MatMul(scope, pooled, w1);
89 auto dense1_add = Add(scope, dense1_mul, b1);
90 auto relu = Relu(scope, dense1_add);
91 auto w2 = MakeConst(scope, {se_channels, 2 * channels}, weights.w2);
92 auto b2 = MakeConst(scope, {2 * channels}, weights.b2);
93 auto dense2_mul = MatMul(scope, relu, w2);
94 auto dense2_add = Add(scope, dense2_mul, b2);
95 auto reshape = Reshape(
96 scope, dense2_add,
97 CPU ? Input({-1, 1, 1, 2 * channels}) : Input({-1, 2 * channels, 1, 1}));
98 auto outputs = Split(scope, CPU ? 3 : 1, reshape, 2);
99 auto sigmoid = Sigmoid(scope, outputs[0]);
100 auto out_mul = Mul(scope, sigmoid, input);
101 auto out_add = Add(scope, out_mul, outputs[1]);
102
103 pooled.node()->set_name(basename + "/pooled");
104 w1.node()->set_name(basename + "/w1");
105 b1.node()->set_name(basename + "/b1");
106 dense1_mul.node()->set_name(basename + "/dense1_mul");
107 dense1_add.node()->set_name(basename + "/dense1_add");
108 relu.node()->set_name(basename + "/relu");
109 w2.node()->set_name(basename + "/w2");
110 b2.node()->set_name(basename + "/b2");
111 dense2_mul.node()->set_name(basename + "/dense2_mul");
112 dense2_add.node()->set_name(basename + "/dense2_add");
113 reshape.node()->set_name(basename + "/reshape");
114 outputs[0].node()->set_name(basename + "/outputs/1");
115 outputs[0].node()->set_name(basename + "/outputs/2");
116 sigmoid.node()->set_name(basename + "/sigmoid");
117 out_mul.node()->set_name(basename + "/out_mul");
118 out_add.node()->set_name(basename + "/out_add");
119
120 return out_add;
121}

123template <bool CPU>
124Output MakeConvBlock(const Scope& scope, Input input, int channels,
125 int input_channels, int output_channels,
126 const LegacyWeights::ConvBlock& weights,
127 const std::string& basename,
128 const LegacyWeights::SEunit* const seunit = nullptr,
129 Input* mixin = nullptr, bool relu = true) {
130 // CPU only supports "NHWC", while for GPU "NCHW" is better.
131 const char* const kDataFormat = CPU ? "NHWC" : "NCHW";
132 auto w_conv =
133 MakeConst(scope, {channels, channels, input_channels, output_channels},
134 weights.weights, {3, 2, 0, 1});
135 w_conv.node()->set_name(basename + "/weights");
136 auto conv2d = Conv2D(scope, input, w_conv, {1, 1, 1, 1}, "SAME",
137 Conv2D::DataFormat(kDataFormat).Dilations({1, 1, 1, 1}));
138 conv2d.node()->set_name(basename + "/conv");
139 auto b_conv = MakeConst(scope, {output_channels}, weights.biases);
140 b_conv.node()->set_name(basename + "/weights_biases");
141 Output conv_b =
142 BiasAdd(scope, conv2d, b_conv, BiasAdd::DataFormat(kDataFormat));
143 conv_b.node()->set_name(basename + "/bias_add");
144 if (seunit) {
145 conv_b = SqueezeAndExcite<CPU>(scope, conv_b, output_channels, *seunit,
146 basename + "/se");
147 }
148 if (mixin) {
149 conv_b = Add(scope, conv_b, *mixin);
150 conv_b.node()->set_name(basename + "/mixin");
151 }
152 if (relu) {
153 auto out = Relu(scope, conv_b);
154 out.node()->set_name(basename + "/relu");
155 return out;
156 } else {
157 return conv_b;
158 }
159}

161template <bool CPU>
162Output MakeResidualBlock(const Scope& scope, Input input, int channels,
163 const LegacyWeights::Residual& weights,
164 const std::string& basename) {
165 auto block1 = MakeConvBlock<CPU>(scope, input, 3, channels, channels,
166 weights.conv1, basename + "/conv1");
167 auto block2 = MakeConvBlock<CPU>(
168 scope, block1, 3, channels, channels, weights.conv2, basename + "/conv2",
169 weights.has_se ? &weights.se : nullptr, &input);
170 return block2;
171}

173template <bool CPU>
174std::tuple<Output, Output, Output> MakeNetwork(const Scope& scope, Input input,
175 const LegacyWeights& weights,
176 bool wdl, bool moves_left) {
177 const int filters = weights.input.weights.size() / kInputPlanes / 9;
178
179 // Input convolution.
180 auto flow = MakeConvBlock<CPU>(scope, input, 3, kInputPlanes, filters,
181 weights.input, "input/conv");
182
183 // Residual tower
184 for (size_t i = 0; i < weights.residual.size(); ++i) {
185 const auto& block = weights.residual[i];
186 flow = MakeResidualBlock<CPU>(scope, flow, filters, block,
187 "block_" + std::to_string(i));
188 }
189
190 // Policy head
191 Output policy_head;
192 if (!weights.policy1.weights.empty()) {
193 // Conv policy head.
194 auto conv_pol1 = MakeConvBlock<CPU>(scope, flow, 3, filters, filters,
195 weights.policy1, "policy/conv1");
196 auto conv_pol =
197 MakeConvBlock<CPU>(scope, conv_pol1, 3, filters, 80, weights.policy,
198 "policy/conv2", nullptr, nullptr, /* relu= */ false);
199
200 // [1858 -> HWC or CHW]
201 std::vector<int> policy_map(1858);
202 for (const auto& mapping : kConvPolicyMap) {
203 if (mapping == -1) continue;
204 const auto index = &mapping - kConvPolicyMap;
205 const auto displacement = index / 64;
206 const auto square = index % 64;
207 const auto row = square / 8;
208 const auto col = square % 8;
209 if (CPU) {
210 policy_map[mapping] = ((row * 8) + col) * 80 + displacement;
211 } else {
212 policy_map[mapping] = ((displacement * 8) + row) * 8 + col;
213 }
214 }
215 auto mapping = MakeIntConst(scope, {1858}, policy_map);
216 auto flattened_conv =
217 Reshape(scope, conv_pol, Const(scope, {-1, 80 * 8 * 8}));
218 policy_head = GatherV2(scope, flattened_conv, mapping, 1);
219
220 mapping.node()->set_name("policy/mapping_table");
221 flattened_conv.node()->set_name("policy/flatten");
222 } else {
223 const int policy_conv_size = weights.policy.biases.size();
224 auto conv_pol =
225 MakeConvBlock<CPU>(scope, flow, 1, filters, policy_conv_size,
226 weights.policy, "policy/conv");
227 conv_pol =
228 Reshape(scope, conv_pol, Const(scope, {-1, policy_conv_size * 8 * 8}));
229 auto ip_pol_w = CPU ? MakeConst(scope, {8, 8, policy_conv_size, 1858},
230 weights.ip_pol_w, {3, 2, 0, 1})
231 : MakeConst(scope, {policy_conv_size, 8, 8, 1858},
232 weights.ip_pol_w, {3, 0, 1, 2});
233 ip_pol_w = Reshape(scope, ip_pol_w,
234 Const(scope, {policy_conv_size * 8 * 8, 1858}));
235 auto ip_pol_b = MakeConst(scope, {1858}, weights.ip_pol_b);
236 policy_head = Add(scope, MatMul(scope, conv_pol, ip_pol_w), ip_pol_b);
237 }
238
239 // Value head
240 auto conv_val = MakeConvBlock<CPU>(scope, flow, 1, filters, 32, weights.value,
241 "value/conv");
242 conv_val = Reshape(scope, conv_val, Const(scope, {-1, 32 * 8 * 8}));
243
244 auto ip1_val_w =
245 CPU ? MakeConst(scope, {8, 8, 32, 128}, weights.ip1_val_w, {3, 2, 0, 1})
246 : MakeConst(scope, {32, 8, 8, 128}, weights.ip1_val_w, {3, 0, 1, 2});
247 ip1_val_w = Reshape(scope, ip1_val_w, Const(scope, {32 * 8 * 8, 128}));
248 auto ip1_val_b = MakeConst(scope, {128}, weights.ip1_val_b);
249 auto value_flow =
250 Relu(scope, Add(scope, MatMul(scope, conv_val, ip1_val_w), ip1_val_b));
251 Output value_head;
252 if (wdl) {
253 auto ip2_val_w = MakeConst(scope, {128, 3}, weights.ip2_val_w);
254 auto ip2_val_b = MakeConst(scope, {3}, weights.ip2_val_b);
255 ip2_val_w.node()->set_name("value/w2");
256 ip2_val_b.node()->set_name("value/b2");
257 auto ip_fc = Add(scope, MatMul(scope, value_flow, ip2_val_w), ip2_val_b);
258 value_head = Softmax(scope, ip_fc);
259 } else {
260 auto ip2_val_w = MakeConst(scope, {128, 1}, weights.ip2_val_w);
261 auto ip2_val_b = MakeConst(scope, {1}, weights.ip2_val_b);
262 ip2_val_w.node()->set_name("value/w2");
263 ip2_val_b.node()->set_name("value/b2");
264 auto ip_fc = Add(scope, MatMul(scope, value_flow, ip2_val_w), ip2_val_b);
265 value_head = Tanh(scope, ip_fc);
266 }
267 ip1_val_w.node()->set_name("value/w1");
268 ip1_val_b.node()->set_name("value/b1");
269 value_flow.node()->set_name("value/relu");
270
271 // Moves left head
272 Output moves_left_head;
273 if (moves_left) {
274 const int mlh_channels = weights.moves_left.biases.size();
275 auto conv_mov = MakeConvBlock<CPU>(scope, flow, 1, filters, mlh_channels,
276 weights.moves_left, "mlh/conv");
277 conv_mov =
278 Reshape(scope, conv_mov, Const(scope, {-1, mlh_channels * 8 * 8}));
279
280 const int mlh_fc1_outputs = weights.ip1_mov_b.size();
281 auto ip1_mov_w =
282 CPU ? MakeConst(scope, {8, 8, mlh_channels, mlh_fc1_outputs},
283 weights.ip1_mov_w, {3, 2, 0, 1})
284 : MakeConst(scope, {mlh_channels, 8, 8, mlh_fc1_outputs},
285 weights.ip1_mov_w, {3, 0, 1, 2});
286 ip1_mov_w = Reshape(scope, ip1_mov_w,
287 Const(scope, {mlh_channels * 8 * 8, mlh_fc1_outputs}));
288 auto ip1_mov_b = MakeConst(scope, {mlh_fc1_outputs}, weights.ip1_mov_b);
289 auto mov_flow =
290 Relu(scope, Add(scope, MatMul(scope, conv_mov, ip1_mov_w), ip1_mov_b));
291 auto ip2_mov_w = MakeConst(scope, {mlh_fc1_outputs, 1}, weights.ip2_mov_w);
292 auto ip2_mov_b = MakeConst(scope, {1}, weights.ip2_mov_b);
293 auto ip_fc = Add(scope, MatMul(scope, mov_flow, ip2_mov_w), ip2_mov_b);
294 moves_left_head = Relu(scope, ip_fc);
295 }
296
297 policy_head.node()->set_name("policy/out");
298 value_head.node()->set_name("value/out");
299 moves_left_head.node()->set_name("mlh/out");
300 return {policy_head, value_head, moves_left_head};
301}

346template <bool CPU>
347class TFNetworkComputation : public NetworkComputation {
348 public:
349 TFNetworkComputation(const TFNetwork<CPU>* network) : network_(network) {}
350 void AddInput(InputPlanes&& input) override {
351 raw_input_.emplace_back(input);
352 }
353 void ComputeBlocking() override {
354 PrepareInput();
355 status_ = network_->Compute(input_, &output_);
356 CHECK(status_.ok()) << status_.ToString();
357 }
358
359 int GetBatchSize() const override { return raw_input_.size(); }
360 float GetQVal(int sample) const override {
361 if (network_->IsWdl()) {
362 const auto w = output_[0].template matrix<float>()(sample, 0);
363 const auto l = output_[0].template matrix<float>()(sample, 2);
364 return w - l;
365 } else {
366 return output_[0].template matrix<float>()(sample, 0);
367 }
368 }
369 float GetDVal(int sample) const override {
370 if (network_->IsWdl()) {
371 const auto d = output_[0].template matrix<float>()(sample, 1);
372 return d;
373 } else {
374 return 0.0f;
375 }
376 }
377 float GetPVal(int sample, int move_id) const override {
378 return output_[1].template matrix<float>()(sample, move_id);
379 }
380 float GetMVal(int sample) const override {
381 if (network_->IsMlh()) {
382 return output_[2].template matrix<float>()(sample, 0);
383 } else {
384 return 0.0f;
385 }
386 }
387
388 private:
392 void PrepareInput();
393
394 const TFNetwork<CPU>* network_;
395 std::vector<InputPlanes> raw_input_;
396
397 tensorflow::Tensor input_;
398 std::vector<tensorflow::Tensor> output_;
399 tensorflow::Status status_;
400};

402// Version for GPU.
403template <>
404void TFNetworkComputation<false>::PrepareInput() {
405 input_ = tensorflow::Tensor(
406 tensorflow::DataType::DT_FLOAT,
407 {static_cast<int>(raw_input_.size()), kInputPlanes, 8, 8});
408
409 auto flat = input_.flat<float>();
410 memset(flat.data(), 0, flat.size() * sizeof(*flat.data()));
411 auto iter = flat.data();
412 for (const auto& sample : raw_input_) {
413 CHECK_EQ(sample.size(), kInputPlanes);
414 for (const auto& plane : sample) {
415 for (auto bit : IterateBits(plane.mask)) {
416 *(iter + bit) = plane.value;
417 }
418 iter += 64;
419 }
420 }
421}
422
423// Version for CPU.
424template <>
425void TFNetworkComputation<true>::PrepareInput() {
426 input_ = tensorflow::Tensor(
427 tensorflow::DataType::DT_FLOAT,
428 {static_cast<int>(raw_input_.size()), 8, 8, kInputPlanes});
429
430 auto flat = input_.flat<float>();
431 memset(flat.data(), 0, flat.size() * sizeof(*flat.data()));
432 auto* data = flat.data();
433 for (size_t input_idx = 0; input_idx < raw_input_.size(); ++input_idx) {
434 const auto& sample = raw_input_[input_idx];
435 int base = kInputPlanes * 8 * 8 * input_idx;
436
437 CHECK_EQ(sample.size(), kInputPlanes);
438 for (int plane_idx = 0; plane_idx < kInputPlanes; ++plane_idx) {
439 const auto& plane = sample[plane_idx];
440 for (auto bit : IterateBits(plane.mask)) {
441 data[base + bit * kInputPlanes + plane_idx] = plane.value;
442 }
443 }
444 }

447template <bool CPU>
448TFNetwork<CPU>::TFNetwork(const WeightsFile& file, const OptionsDict& options,
449 bool wdl)
450 : scope_(Scope::NewRootScope()),
451 capabilities_{file.format().network_format().input(),
452 file.format().network_format().output(),
453 file.format().network_format().moves_left()},
454 wdl_(wdl) {
455 const LegacyWeights weights(file.weights());
456 tensorflow::SessionOptions session_options;
457 if (CPU) (*session_options.config.mutable_device_count())["GPU"] = 0;
458 session_ =
459 std::make_unique<tensorflow::ClientSession>(scope_, session_options);
460
461 if (CPU) {
462 input_ = std::make_unique<Placeholder>(
463 scope_, DataType::DT_FLOAT,
464 Placeholder::Shape({-1, 8, 8, kInputPlanes}));
465 } else {
466 input_ = std::make_unique<Placeholder>(
467 scope_, DataType::DT_FLOAT,
468 Placeholder::Shape({-1, kInputPlanes, 8, 8}));
469 }
470 input_->node()->set_name("input_planes");
471
472 auto output = MakeNetwork<CPU>(scope_, *input_, weights, wdl, IsMlh());
473 CHECK(scope_.ok()) << scope_.status().ToString();
474 policy_head_ = std::make_unique<Output>(std::get<0>(output));
475 value_head_ = std::make_unique<Output>(std::get<1>(output));
476 moves_left_head_ = std::make_unique<Output>(std::get<2>(output));
477
478 if (options.Exists<std::string>("dump-graphdef") ||
479 options.Exists<std::string>("dump-graphdef-txt")) {
480 GraphDef gdef;
481 CHECK(scope_.ToGraphDef(&gdef).ok());
482 if (options.Exists<std::string>("dump-graphdef")) {
483 std::ofstream f(options.Get<std::string>("dump-graphdef").c_str());
484 f.exceptions(std::ifstream::failbit);
485 f << gdef.SerializeAsString();
486 }
487 if (options.Exists<std::string>("dump-graphdef-txt")) {
488 std::ofstream f(options.Get<std::string>("dump-graphdef-txt").c_str());
489 f.exceptions(std::ifstream::failbit);
490 f << gdef.DebugString();
491 }
492 }
493
494 // First request to tensorflow is slow (0.6s), so doing an empty request for
495 // preheating.
496 auto fake_request = NewComputation();
497 fake_request->AddInput(InputPlanes(kInputPlanes));
498 fake_request->ComputeBlocking();
499}

514template <bool CPU>
515std::unique_ptr<Network> MakeTFNetwork(const std::optional<WeightsFile>& w,
516 const OptionsDict& options) {
517 if (!w) {
518 throw Exception("The " +
519 std::string(CPU ? "tensorflow-cc-cpu" : "tensorflow-cc") +
520 " backend requires a network file.");
521 }
522 const WeightsFile& weights = *w;
523 if (weights.format().network_format().network() !=
524 pblczero::NetworkFormat::NETWORK_CLASSICAL_WITH_HEADFORMAT &&
525 weights.format().network_format().network() !=
526 pblczero::NetworkFormat::NETWORK_SE_WITH_HEADFORMAT) {
527 throw Exception("Network format " +
528 pblczero::NetworkFormat::NetworkStructure_Name(
529 weights.format().network_format().network()) +
530 " is not supported by Tensorflow C++ backend.");
531 }
532 if (weights.format().network_format().policy() !=
533 pblczero::NetworkFormat::POLICY_CLASSICAL &&
534 weights.format().network_format().policy() !=
535 pblczero::NetworkFormat::POLICY_CONVOLUTION) {
536 throw Exception("Policy format " +
537 pblczero::NetworkFormat::PolicyFormat_Name(
538 weights.format().network_format().policy()) +
539 " is not supported by Tensorflow C++ backend.");
540 }
541 if (weights.format().network_format().value() !=
542 pblczero::NetworkFormat::VALUE_CLASSICAL &&
543 weights.format().network_format().value() !=
544 pblczero::NetworkFormat::VALUE_WDL) {
545 throw Exception("Value format " +
546 pblczero::NetworkFormat::ValueFormat_Name(
547 weights.format().network_format().value()) +
548 " is not supported by Tensorflow C++ backend.");
549 }
550 if (weights.format().network_format().default_activation() !=
551 pblczero::NetworkFormat::DEFAULT_ACTIVATION_RELU) {
552 throw Exception(
553 "Default activation " +
554 pblczero::NetworkFormat::DefaultActivation_Name(
555 weights.format().network_format().default_activation()) +
556 " is not supported by Tensorflow C++ backend.");
557 }
558 return std::make_unique<TFNetwork<CPU>>(
559 weights, options,
560 weights.format().network_format().value() ==
561 pblczero::NetworkFormat::VALUE_WDL);
562}
