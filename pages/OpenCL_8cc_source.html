94void OpenCL_Network::add_weights(size_t layer, size_t size,
95 const float* weights) {
96 if (layer >= m_layers.size()) {
97 m_layers.push_back(Layer());
98 }
99
100 auto converted_weights = std::vector<net_t>();
101 for (auto i = size_t{0}; i < size; i++) {
102 converted_weights.emplace_back(weights[i]);
103 }
104
105 auto weightSize = size * sizeof(decltype(converted_weights)::value_type);
106 m_layers.back().weights.emplace_back(
107 m_opencl.m_context, CL_MEM_COPY_HOST_PTR | CL_MEM_READ_ONLY, weightSize,
108 const_cast<net_t*>(converted_weights.data()));
109}

125 if (type == CL_DEVICE_TYPE_CPU) {

141void OpenCL::process_tuners(std::string tuners) {
142 std::string buf;
143 std::stringstream ss(tuners);
144 std::size_t found;
145
146 auto mwg = false;
147 auto nwg = false;
148 auto kwg = false;
149 auto ndimc = false;
150 auto mdimc = false;
151 auto vwm = false;
152 auto vwn = false;
153 while (ss >> buf) {
154 found = buf.find("=");
155 if (found == std::string::npos) {
156 CERR << "Invalid tuner string: " << tuners << std::endl;
157 std::exit(-1);
158 }
159 std::string name = buf.substr(0, found);
160 auto value = std::stoi(buf.substr(found + 1, std::string::npos));
161 if (name == "-DMWG") {
162 m_sgemm_tuners.mwg = value;
163 mwg = true;
164 }
165 if (name == "-DNWG") {
166 m_sgemm_tuners.nwg = value;
167 nwg = true;
168 }
169 if (name == "-DKWG") {
170 m_sgemm_tuners.kwg = value;
171 kwg = true;
172 }
173 if (name == "-DMDIMC") {
174 m_sgemm_tuners.mdimc = value;
175 mdimc = true;
176 }
177 if (name == "-DNDIMC") {
178 m_sgemm_tuners.ndimc = value;
179 ndimc = true;
180 }
181 if (name == "-DVWM") {
182 m_sgemm_tuners.vwm = value;
183 vwm = true;
184 }
185 if (name == "-DVWN") {
186 m_sgemm_tuners.vwn = value;
187 vwn = true;
188 }
189 }
190 if (!mwg || !nwg || !kwg || !mdimc || !ndimc || !vwm || !vwn) {
191 CERR << "Missing tuner parameters";
192 if (!mwg) {
193 CERR << " MWG";
194 }
195 if (!nwg) {
196 CERR << " NWG";
197 }
198 if (!kwg) {
199 CERR << " KWG";
200 }
201 if (!mdimc) {
202 CERR << " MDIMC";
203 }
204 if (!ndimc) {
205 CERR << " NDIMC";
206 }
207 if (!vwm) {
208 CERR << " VWM";
209 }
210 if (!vwn) {
211 CERR << " VWN";
212 }
213 CERR << std::endl;
214 std::exit(-1);
215 }

232void OpenCL::initialize(const int channels, const OpenCLParams& params) {
233 CERR << "Initializing OpenCL.";
234 std::vector<cl::Platform> platforms;
235 try {
236 cl::Platform::get(&platforms);
237 } catch (const cl::Error& e) {
238 CERR << "OpenCL: " << e.what();
239 throw;
240 }
241
242 auto best_version = 0.0f;
243 cl::Platform best_platform;
244 cl::Device best_device;
245 std::string best_vendor;
246 auto best_score = 0;
247 auto found_device = false;
248 auto id = 0;
249
250 CERR << "Detected " << platforms.size() << " OpenCL platforms.";
251
252 for (const auto& p : platforms) {
253 std::string platvers = p.getInfo<CL_PLATFORM_VERSION>();
254 std::string platprof = p.getInfo<CL_PLATFORM_PROFILE>();
255 std::string platname = p.getInfo<CL_PLATFORM_NAME>();
256 std::string platvend = p.getInfo<CL_PLATFORM_VENDOR>();
257 CERR << "Platform version: " << platvers;
258 CERR << "Platform profile: " << platprof;
259 CERR << "Platform name: " << platname;
260 CERR << "Platform vendor: " << platvend;
261
262 std::istringstream versstream(platvers);
263 std::string tmp;
264 float opencl_version;
265 versstream >> tmp >> opencl_version;
266
267 std::vector<cl::Device> devices;
268 try {
269 p.getDevices(CL_DEVICE_TYPE_ALL, &devices);
270 } catch (const cl::Error& e) {
271 CERR << "Error getting device(s): " << e.what() << ": " << e.err()
272 << std::endl;
273 devices.clear();
274 }
275 for (auto& d : devices) {
276 CERR << "Device ID: " << id;
277 CERR << "Device name: "
278 << trim_left(d.getInfo<CL_DEVICE_NAME>().c_str());
279 CERR << "Device type: "
280 << opencl_dev_type_to_string(d.getInfo<CL_DEVICE_TYPE>());
281 CERR << "Device vendor: " << d.getInfo<CL_DEVICE_VENDOR>();
282 CERR << "Device driver: " << d.getInfo<CL_DRIVER_VERSION>();
283 CERR << "Device speed: " << d.getInfo<CL_DEVICE_MAX_CLOCK_FREQUENCY>()
284 << " MHZ";
285 CERR << "Device cores: " << d.getInfo<CL_DEVICE_MAX_COMPUTE_UNITS>()
286 << " CU";
287
288 // assign score, try to find best device
289 int this_score = 0;
290 std::string this_vendor = d.getInfo<CL_DEVICE_VENDOR>();
291 std::transform(this_vendor.begin(), this_vendor.end(),
292 this_vendor.begin(), ::tolower);
293 this_score += 1000 * (this_vendor.find("advanced micro devices") !=
294 std::string::npos);
295 this_score += 1000 * (this_vendor.find("amd") != std::string::npos);
296 this_score += 1000 * (this_vendor.find("nvidia") != std::string::npos);
297 this_score += 500 * (this_vendor.find("intel") != std::string::npos);
298 this_score += 100 * (d.getInfo<CL_DEVICE_TYPE>() == CL_DEVICE_TYPE_GPU);
299 this_score += opencl_version * 10;
300 CERR << "Device score: " << this_score;
301
302 bool preferred = params.gpuId == id;
303
304 if ((this_score > best_score) || preferred) {
305 best_version = opencl_version;
306 best_platform = p;
307 best_device = d;
308 best_vendor = this_vendor;
309 if (preferred) {
310 best_score = std::numeric_limits<decltype(best_score)>::max();
311 } else {
312 best_score = this_score;
313 }
314 found_device = true;
315 }
316 id++;
317 }
318 }
319
320 if (!found_device) {
321 throw std::runtime_error("No suitable OpenCL device found.");
322 }
323
324 CERR << "Selected platform: " << best_platform.getInfo<CL_PLATFORM_NAME>();
325 CERR << "Selected device: "
326 << trim_left(best_device.getInfo<CL_DEVICE_NAME>().c_str());
327 CERR << "with OpenCL " << std::fixed << std::setprecision(1) << best_version
328 << " capability.";
329 cl::Context context;
330 try {
331 context = cl::Context(best_device);
332 } catch (const cl::Error& e) {
333 CERR << "Error creating OpenCL context: " << e.what() << ": " << e.err();
334 throw std::runtime_error("Error creating OpenCL context.");
335 }
336 m_context = context;
337 m_device = best_device;
338
339 // Make program of the source code in the context.
340 try {
341 m_program = cl::Program(
342 m_context, sourceCode_config + sourceCode_convolve1 +
343 sourceCode_convolve3 + sourceCode_se + sourceCode_sgemm +
344 sourceCode_sgemv + sourceCode_policymap);
345 } catch (const cl::Error& e) {
346 CERR << "Error getting kernels: " << e.what() << ": " << e.err();
347 throw std::runtime_error("Error getting OpenCL kernels.");
348 }
349
350 m_cl_args = cl_args;
351
352 auto t = Tuner(*this, params, m_context, m_device);
353 auto sgemm_tuners = t.load_sgemm_tuners(
354 channels, params.tune_batch_size * WINOGRAD_P, channels, WINOGRAD_TILE);
355
356 // Build program for these specific devices.
357 try {
358 std::string args = cl_args;
359 args += sgemm_tuners;
360 m_program.build(args.c_str());
361 } catch (const cl::Error&) {
362 CERR << "Error building kernels: "
363 << m_program.getBuildInfo<CL_PROGRAM_BUILD_LOG>(m_device) << ".";
364 throw std::runtime_error("Error building OpenCL kernels.");
365 }
366
367 process_tuners(sgemm_tuners);
368
369 auto sgemm_kernel = cl::Kernel(m_program, "XgemmBatched");
370
371 m_wavefront_size =
372 sgemm_kernel
373 .getWorkGroupInfo<CL_KERNEL_PREFERRED_WORK_GROUP_SIZE_MULTIPLE>(
374 best_device);
375 CERR << "Wavefront/Warp size: " << m_wavefront_size << std::endl;
376
377 m_max_workgroup_size = best_device.getInfo<CL_DEVICE_MAX_WORK_GROUP_SIZE>();
378 m_max_workgroup_dims = best_device.getInfo<CL_DEVICE_MAX_WORK_ITEM_SIZES>();
379
380 CERR << "Max workgroup size: " << m_max_workgroup_size;
381 std::ostringstream ss;
382 for (auto d : m_max_workgroup_dims) ss << ' ' << d;
383 CERR << "Max workgroup dimensions:" << ss.str();
384 m_init_ok = true;
385}

387std::unique_ptr<OpenCLBuffers> OpenCL_Network::acquire_buffers() const {
388 std::lock_guard<std::mutex> lock(m_pool_mutex);
389 if (m_buffers_pool.empty()) return std::make_unique<OpenCLBuffers>(*this);
390 auto result = std::move(m_buffers_pool.back());
391 m_buffers_pool.pop_back();
392 return result;
393}
