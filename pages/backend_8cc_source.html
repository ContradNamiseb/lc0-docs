37std::vector<EvalResult> Backend::EvaluateBatch(
38 std::span<const EvalPosition> positions) {
39 std::vector<EvalResult> results;
40 results.reserve(positions.size());
41 std::unique_ptr<BackendComputation> computation = CreateComputation();
42 for (const EvalPosition& pos : positions) {
43 results.emplace_back();
44 EvalResult& result = results.back();
45 result.p.resize(pos.legal_moves.size());
46 computation->AddInput(
47 pos, EvalResultPtr{&result.q, &result.d, &result.m,
48 std::span<float>(result.p.data(), result.p.size())});
49 }
50 computation->ComputeBlocking();
51 return results;
52}

54uint64_t Backend::ConfigurationHash(const OptionsDict& options) const {
55 uint64_t hash = std::hash<std::string>{}(
56 options.Get<std::string>(SharedBackendParams::kBackendId));
57 hash = HashCat(hash, std::hash<std::string>{}(options.Get<std::string>(
58 SharedBackendParams::kBackendOptionsId)));
59 hash = HashCat(hash, std::hash<std::string>{}(options.Get<std::string>(
60 SharedBackendParams::kWeightsId)));
61 hash = HashCat(hash, std::hash<float>{}(options.Get<float>(
62 SharedBackendParams::kPolicySoftmaxTemp)));
63 hash = HashCat(hash, std::hash<std::string>{}(options.Get<std::string>(
64 SharedBackendParams::kHistoryFill)));
65 return hash;
66}
