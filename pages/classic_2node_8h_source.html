47// Children of a node are stored the following way:
48// * Edges and Nodes edges point to are stored separately.
49// * There may be dangling edges (which don't yet point to any Node object yet)
50// * Edges are stored are a simple array on heap.
51// * Nodes are stored as a linked list, and contain index_ field which shows
52// which edge of a parent that node points to.
53// Or they are stored a contiguous array of Node objects on the heap if
54// solid_children_ is true. If the children have been 'solidified' their
55// sibling links are unused and left empty. In this state there are no
56// dangling edges, but the nodes may not have ever received any visits.
57//
58// Example:
59// Parent Node
60// |
61// +-------------+-------------+----------------+--------------+
62// | | | | |
63// Edge 0(Nf3) Edge 1(Bc5) Edge 2(a4) Edge 3(Qxf7) Edge 4(a3)
64// (dangling) | (dangling) | (dangling)
65// Node, Q=0.5 Node, Q=-0.2
66//
67// Is represented as:
68// +--------------+
69// | Parent Node |
70// +--------------+ +--------+
71// | edges_ | -------------------------------------> | Edge[] |
72// | | +------------+ +--------+
73// | child_ | -> | Node | | Nf3 |
74// +--------------+ +------------+ | Bc5 |
75// | index_ = 1 | | a4 |
76// | q_ = 0.5 | +------------+ | Qxf7 |
77// | sibling_ | -> | Node | | a3 |
78// +------------+ +------------+ +--------+
79// | index_ = 3 |
80// | q_ = -0.2 |
81// | sibling_ | -> nullptr
82// +------------+

630// TODO(crem) Replace this with less hacky iterator once we support C++17.
631// This class has multiple hypostases within one class:
632// * Range (begin() and end() functions)
633// * Iterator (operator++() and operator*())
634// * Element, pointed by iterator (EdgeAndNode class mainly, but Edge_Iterator
635// is useful too when client wants to call GetOrSpawnNode).
636// It's safe to slice EdgeAndNode off Edge_Iterator.
637// It's more customary to have those as three classes, but
638// creating zoo of classes and copying them around while iterating seems
639// excessive.
640//
641// All functions are not thread safe (must be externally synchronized), but
642// it's fine if GetOrSpawnNode is called between calls to functions of the
643// iterator (e.g. advancing the iterator). Other functions that manipulate
644// child_ of parent or the sibling chain are not safe to call while iterating.
645template <bool is_const>
646class Edge_Iterator : public EdgeAndNode {
647 public:
648 using Ptr = std::conditional_t<is_const, const std::unique_ptr<Node>*,
649 std::unique_ptr<Node>*>;
650 using value_type = Edge_Iterator;
651 using iterator_category = std::forward_iterator_tag;
652 using difference_type = std::ptrdiff_t;
653 using pointer = Edge_Iterator*;
654 using reference = Edge_Iterator&;
655
656 // Creates "end()" iterator.
657 Edge_Iterator() {}
658
659 // Creates "begin()" iterator. Also happens to be a range constructor.
660 // child_ptr will be nullptr if parent_node is solid children.
661 Edge_Iterator(const Node& parent_node, Ptr child_ptr)
662 : EdgeAndNode(parent_node.edges_.get(), nullptr),
663 node_ptr_(child_ptr),
664 total_count_(parent_node.num_edges_) {
665 if (edge_ && child_ptr != nullptr) Actualize();
666 if (edge_ && child_ptr == nullptr) {
667 node_ = parent_node.child_.get();
668 }
669 }
670
671 // Function to support range interface.
672 Edge_Iterator<is_const> begin() { return *this; }
673 Edge_Iterator<is_const> end() { return {}; }
674
675 // Functions to support iterator interface.
676 // Equality comparison operators are inherited from EdgeAndNode.
677 void operator++() {
678 // If it was the last edge in array, become end(), otherwise advance.
679 if (++current_idx_ == total_count_) {
680 edge_ = nullptr;
681 } else {
682 ++edge_;
683 if (node_ptr_ != nullptr) {
684 Actualize();
685 } else {
686 ++node_;
687 }
688 }
689 }
690 Edge_Iterator& operator*() { return *this; }
691
692 // If there is node, return it. Otherwise spawn a new one and return it.
693 Node* GetOrSpawnNode(Node* parent) {
694 if (node_) return node_; // If there is already a node, return it.
695 // Should never reach here in solid mode.
696 assert(node_ptr_ != nullptr);
697 Actualize(); // But maybe other thread already did that.
698 if (node_) return node_; // If it did, return.
699 // Now we are sure we have to create a new node.
700 // Suppose there are nodes with idx 3 and 7, and we want to insert one with
701 // idx 5. Here is how it looks like:
702 // node_ptr_ -> &Node(idx_.3).sibling_ -> Node(idx_.7)
703 // Here is how we do that:
704 // 1. Store pointer to a node idx_.7:
705 // node_ptr_ -> &Node(idx_.3).sibling_ -> nullptr
706 // tmp -> Node(idx_.7)
707 std::unique_ptr<Node> tmp = std::move(*node_ptr_);
708 // 2. Create fresh Node(idx_.5):
709 // node_ptr_ -> &Node(idx_.3).sibling_ -> Node(idx_.5)
710 // tmp -> Node(idx_.7)
711 *node_ptr_ = std::make_unique<Node>(parent, current_idx_);
712 // 3. Attach stored pointer back to a list:
713 // node_ptr_ ->
714 // &Node(idx_.3).sibling_ -> Node(idx_.5).sibling_ -> Node(idx_.7)
715 (*node_ptr_)->sibling_ = std::move(tmp);
716 // 4. Actualize:
717 // node_ -> &Node(idx_.5)
718 // node_ptr_ -> &Node(idx_.5).sibling_ -> Node(idx_.7)
719 Actualize();
720 return node_;
721 }
722
723 private:
724 void Actualize() {
725 // This must never be called in solid mode.
726 assert(node_ptr_ != nullptr);
727 // If node_ptr_ is behind, advance it.
728 // This is needed (and has to be 'while' rather than 'if') as other threads
729 // could spawn new nodes between &node_ptr_ and *node_ptr_ while we didn't
730 // see.
731 while (*node_ptr_ && (*node_ptr_)->index_ < current_idx_) {
732 node_ptr_ = &(*node_ptr_)->sibling_;
733 }
734 // If in the end node_ptr_ points to the node that we need, populate node_
735 // and advance node_ptr_.
736 if (*node_ptr_ && (*node_ptr_)->index_ == current_idx_) {
737 node_ = (*node_ptr_).get();
738 node_ptr_ = &node_->sibling_;
739 } else {
740 node_ = nullptr;
741 }
742 }
743
744 // Pointer to a pointer to the next node. Has to be a pointer to pointer
745 // as we'd like to update it when spawning a new node.
746 Ptr node_ptr_;
747 uint16_t current_idx_ = 0;
748 uint16_t total_count_ = 0;
749};

751// TODO(crem) Replace this with less hacky iterator once we support C++17.
752// This class has multiple hypostases within one class:
753// * Range (begin() and end() functions)
754// * Iterator (operator++() and operator*())
755// It's more customary to have those as two classes, but
756// creating zoo of classes and copying them around while iterating seems
757// excessive.
758//
759// All functions are not thread safe (must be externally synchronized).
760template <bool is_const>
761class VisitedNode_Iterator {
762 public:
763 // Creates "end()" iterator.
764 VisitedNode_Iterator() {}
765
766 // Creates "begin()" iterator. Also happens to be a range constructor.
767 // child_ptr will be nullptr if parent_node is solid children.
768 VisitedNode_Iterator(const Node& parent_node, Node* child_ptr)
769 : node_ptr_(child_ptr),
770 total_count_(parent_node.num_edges_),
771 solid_(parent_node.solid_children_) {
772 if (node_ptr_ != nullptr && node_ptr_->GetN() == 0) {
773 operator++();
774 }
775 }
776 // These are technically wrong, but are usable to compare with end().
777 bool operator==(const VisitedNode_Iterator<is_const>& other) const {
778 return node_ptr_ == other.node_ptr_;
779 }
780 bool operator!=(const VisitedNode_Iterator<is_const>& other) const {
781 return node_ptr_ != other.node_ptr_;
782 }
783
784 // Function to support range interface.
785 VisitedNode_Iterator<is_const> begin() { return *this; }
786 VisitedNode_Iterator<is_const> end() { return {}; }
787
788 // Functions to support iterator interface.
789 // Equality comparison operators are inherited from EdgeAndNode.
790 void operator++() {
791 if (solid_) {
792 while (++current_idx_ != total_count_ &&
793 node_ptr_[current_idx_].GetN() == 0) {
794 if (node_ptr_[current_idx_].GetNInFlight() == 0) {
795 // Once there is not even n in flight, we can skip to the end. This is
796 // due to policy being in sorted order meaning that additional n in
797 // flight are always selected from the front of the section with no n
798 // in flight or visited.
799 current_idx_ = total_count_;
800 break;
801 }
802 }
803 if (current_idx_ == total_count_) {
804 node_ptr_ = nullptr;
805 }
806 } else {
807 do {
808 node_ptr_ = node_ptr_->sibling_.get();
809 // If n started is 0, can jump direct to end due to sorted policy
810 // ensuring that each time a new edge becomes best for the first time,
811 // it is always the first of the section at the end that has NStarted of
812 // 0.
813 if (node_ptr_ != nullptr && node_ptr_->GetN() == 0 &&
814 node_ptr_->GetNInFlight() == 0) {
815 node_ptr_ = nullptr;
816 break;
817 }
818 } while (node_ptr_ != nullptr && node_ptr_->GetN() == 0);
819 }
820 }
821 Node* operator*() {
822 if (solid_) {
823 return &(node_ptr_[current_idx_]);
824 } else {
825 return node_ptr_;
826 }
827 }
828
829 private:
830 // Pointer to current node.
831 Node* node_ptr_ = nullptr;
832 uint16_t current_idx_ = 0;
833 uint16_t total_count_ = 0;
834 bool solid_ = false;
835};
