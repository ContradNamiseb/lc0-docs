48 void AddInput(InputPlanes&& input) override {
49 const auto iter_mask =
50 &inputs_outputs_->input_masks_mem_[batch_size_ * kInputPlanes];
51 const auto iter_val =
52 &inputs_outputs_->input_val_mem_[batch_size_ * kInputPlanes];
53
54 int i = 0;
55 for (const auto& plane : input) {
56 iter_mask[i] = plane.mask;
57 iter_val[i] = plane.value;
58 i++;
59 }
60
61 batch_size_++;
62 }
63
64 void ComputeBlocking() override;
65
66 int GetBatchSize() const override { return batch_size_; }
67
68 float GetQVal(int sample) const override {
69 if (wdl_) {
70 auto w = inputs_outputs_->op_value_mem_[3 * sample + 0];
71 auto l = inputs_outputs_->op_value_mem_[3 * sample + 2];
72 return w - l;
73 } else {
74 return inputs_outputs_->op_value_mem_[sample];
75 }
76 }
77
78 float GetDVal(int sample) const override {
79 if (wdl_) {
80 auto d = inputs_outputs_->op_value_mem_[3 * sample + 1];
81 return d;
82 } else {
83 return 0.0f;
84 }
85 }
86
87 float GetPVal(int sample, int move_id) const override {
88 return inputs_outputs_->op_policy_mem_[sample * kNumOutputPolicy + move_id];
89 }
90
91 float GetMVal(int sample) const override {
92 if (moves_left_) {
93 return inputs_outputs_->op_moves_left_mem_[sample];
94 }
95 return 0.0f;
96 }
97
98 private:
99 // Memory holding inputs, outputs.
100 std::unique_ptr<InputsOutputs> inputs_outputs_;
101 int batch_size_;
102 bool wdl_;
103 bool moves_left_;
104
105 MetalNetwork* network_;
106};
