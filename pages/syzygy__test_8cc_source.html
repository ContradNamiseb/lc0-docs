39void TestValidRootExpectation(
40 SyzygyTablebase* tablebase, const std::string& fen,
41 const std::vector<std::string>& valid_moves,
42 const std::vector<std::string>& invalid_moves,
43 const std::vector<std::string>& invalid_dtz_only = {},
44 bool has_repeated = false) {
45 ChessBoard board;
46 PositionHistory history;
47 int rule50ply;
48 int gameply;
49 board.SetFromFen(fen, &rule50ply, &gameply);
50 history.Reset(board, rule50ply, gameply);
51 MoveList allowed_moves_dtz;
52 tablebase->root_probe(history.Last(), has_repeated, false,
53 &allowed_moves_dtz);
54 MoveList allowed_moves_wdl;
55 tablebase->root_probe_wdl(history.Last(), &allowed_moves_wdl);
56 for (auto move_str : valid_moves) {
57 Move move = board.ParseMove(move_str);
58 EXPECT_TRUE(std::find(allowed_moves_dtz.begin(), allowed_moves_dtz.end(),
59 move) != allowed_moves_dtz.end());
60 EXPECT_TRUE(std::find(allowed_moves_wdl.begin(), allowed_moves_wdl.end(),
61 move) != allowed_moves_wdl.end());
62 }
63 for (auto move_str : invalid_moves) {
64 Move move = board.ParseMove(move_str);
65 EXPECT_FALSE(std::find(allowed_moves_dtz.begin(), allowed_moves_dtz.end(),
66 move) != allowed_moves_dtz.end());
67 EXPECT_FALSE(std::find(allowed_moves_wdl.begin(), allowed_moves_wdl.end(),
68 move) != allowed_moves_wdl.end());
69 }
70 for (auto move_str : invalid_dtz_only) {
71 Move move = board.ParseMove(move_str);
72 EXPECT_FALSE(std::find(allowed_moves_dtz.begin(), allowed_moves_dtz.end(),
73 move) != allowed_moves_dtz.end());
74 EXPECT_TRUE(std::find(allowed_moves_wdl.begin(), allowed_moves_wdl.end(),
75 move) != allowed_moves_wdl.end());
76 }
77}

79void TestValidExpectation(SyzygyTablebase* tablebase, const std::string& fen,
80 WDLScore expected, int expected_dtz) {
81 ChessBoard board;
82 PositionHistory history;
83 board.SetFromFen(fen);
84 history.Reset(board, 0, 1);
85 ProbeState result;
86 WDLScore score = tablebase->probe_wdl(history.Last(), &result);
87 EXPECT_NE(result, FAIL);
88 EXPECT_EQ(score, expected);
89 int moves = tablebase->probe_dtz(history.Last(), &result);
90 EXPECT_NE(result, FAIL);
91 EXPECT_EQ(moves, expected_dtz);
92}

94TEST(Syzygy, Simple3PieceProbes) {
95 SyzygyTablebase tablebase;
96 tablebase.init(kPaths);
97 if (tablebase.max_cardinality() < 3) {
98 // These probes require 3 piece tablebase.
99 return;
100 }
101 // Longest 3 piece position.
102 TestValidExpectation(&tablebase, "8/8/8/8/8/8/2Rk4/1K6 b - - 0 1", WDL_LOSS,
103 -32);
104 // Invert color of above, no change.
105 TestValidExpectation(&tablebase, "8/8/8/8/8/8/2rK4/1k6 w - - 0 1", WDL_LOSS,
106 -32);
107 // Horizontal mirror.
108 TestValidExpectation(&tablebase, "8/8/8/8/8/8/4kR2/6K1 b - - 0 1", WDL_LOSS,
109 -32);
110 // Vertical mirror.
111 TestValidExpectation(&tablebase, "6K1/4kR2/8/8/8/8/8/8 b - - 0 1", WDL_LOSS,
112 -32);
113 // Horizontal mirror again.
114 TestValidExpectation(&tablebase, "1K6/2Rk4/8/8/8/8/8/8 b - - 0 1", WDL_LOSS,
115 -32);
116
117 // A draw by capture position, leaving KvK.
118 TestValidExpectation(&tablebase, "5Qk1/8/8/8/8/8/8/4K3 b - - 0 1", WDL_DRAW,
119 0);
120
121 // A position with a pawn which is to move and win from there.
122 TestValidExpectation(&tablebase, "6k1/8/8/8/8/5p2/8/2K5 b - - 0 1", WDL_WIN,
123 1);
124
125 // A position with a pawn that needs a king move first to win.
126 TestValidExpectation(&tablebase, "8/8/8/8/8/k1p5/8/3K4 b - - 0 1", WDL_WIN,
127 3);
128
129 // A position with a pawn that needs a few king moves before its a loss.
130 TestValidExpectation(&tablebase, "8/2p5/8/8/8/5k2/8/2K5 w - - 0 1", WDL_LOSS,
131 -8);
132}
