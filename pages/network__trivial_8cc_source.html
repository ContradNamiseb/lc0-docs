420 float q = 0.0f;
421 q += DotProduct(input[0].mask, kPawns);
422 q -= DotProduct(ReverseBytesInBytes(input[6].mask), kPawns);
423 q += DotProduct(input[1].mask, kKnights);
424 q -= DotProduct(ReverseBytesInBytes(input[7].mask), kKnights);
425 q += DotProduct(input[2].mask, kBishops);
426 q -= DotProduct(ReverseBytesInBytes(input[8].mask), kBishops);
427 q += DotProduct(input[3].mask, kRooks);
428 q -= DotProduct(ReverseBytesInBytes(input[9].mask), kRooks);
429 q += DotProduct(input[4].mask, kQueens);
430 q -= DotProduct(ReverseBytesInBytes(input[10].mask), kQueens);
431 const bool endgame =
432 (input[4].mask == 0 ||
433 (input[3].mask == 0 && NumBits(input[1].mask | input[2].mask) <= 1)) &&
434 (input[10].mask == 0 ||
435 (input[9].mask == 0 && NumBits(input[7].mask | input[8].mask) <= 1));
436 if (endgame) {
437 q += DotProduct(input[5].mask, kKingsEndgame);
438 q -= DotProduct(ReverseBytesInBytes(input[11].mask), kKingsEndgame);
439 } else {
440 q += DotProduct(input[5].mask, kKings);
441 q -= DotProduct(ReverseBytesInBytes(input[11].mask), kKings);
442 }
443 // Multiply Q by 10, otherwise evals too low. :-/
444 q_.push_back(2.0f / (1.0f + std::exp(q * -10.0f)) - 1.0f);
