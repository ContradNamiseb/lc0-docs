84PjrtKeyValue MakeKeyValue(const PJRT_NamedValue* kv) {
85 PjrtKeyValue result;
86 result.set_key({kv->name, kv->name_size});
87 switch (kv->type) {
88 case PJRT_NamedValue_kString:
89 result.set_value(std::string(kv->string_value, kv->value_size));
90 break;
91 case PJRT_NamedValue_kInt64:
92 result.set_value(kv->int64_value);
93 break;
94 case PJRT_NamedValue_kInt64List:
95 result.set_value(std::vector<int64_t>(
96 kv->int64_array_value, kv->int64_array_value + kv->value_size));
97 break;
98 case PJRT_NamedValue_kFloat:
99 result.set_value(kv->float_value);
100 break;
101 case PJRT_NamedValue_kBool:
102 result.set_value(kv->bool_value);
103 break;
104 }
105 return result;
106}

150std::vector<std::unique_ptr<PjrtDeviceBuffer>> PjrtExecutable::ExecuteBlocking(
151 const std::vector<PjrtDeviceBuffer*>& inputs) {
152 auto options = MakeStruct<PJRT_ExecuteOptions>();
153 options.num_non_donatable_input_indices = inputs.size();
154 std::vector<int64_t> non_donatable_indices(inputs.size());
155 // TODO the buffer 0 is actually donatable.
156 std::iota(non_donatable_indices.begin(), non_donatable_indices.end(), 0);
157 options.non_donatable_input_indices = non_donatable_indices.data();
158
159 auto args = MakeStruct<PJRT_LoadedExecutable_Execute_Args>();
160 args.executable = executable_;
161 args.options = &options;
162 args.num_devices = 1;
163 std::vector<PJRT_Buffer*> buffers(inputs.size());
164 for (size_t i = 0; i < inputs.size(); ++i) buffers[i] = inputs[i]->buffer_;
165 PJRT_Buffer* const* buffers_ptr = buffers.data();
166 args.num_args = inputs.size();
167 args.argument_lists = &buffers_ptr;
168
169 std::vector<PJRT_Buffer*> outputs(num_outputs_);
170 PJRT_Buffer** outputs_ptr = outputs.data();
171 PJRT_Event* event_ptr;
172 args.output_lists = &outputs_ptr;
173 args.device_complete_events = &event_ptr;
174 CheckError(api_->PJRT_LoadedExecutable_Execute(&args));
175
176 PjrtEvent event(api_, event_ptr);
177 event.Await();
178
179 std::vector<std::unique_ptr<PjrtDeviceBuffer>> output_buffers;
180 output_buffers.reserve(num_outputs_);
181 for (size_t i = 0; i < num_outputs_; ++i) {
182 output_buffers.push_back(
183 std::make_unique<PjrtDeviceBuffer>(api_, outputs[i]));
184 }
185 return output_buffers;
186}

342Pjrt::Pjrt(const char* library_path) : PjrtCommon(nullptr) {
343 // TODO factor out the dlopen/dlsym code into a separate function, and
344 // implement for other OSes.
345 void* handle = dlopen(library_path, RTLD_LAZY);
346 if (!handle) {
347 throw PjrtException(PjrtErrorCode::INVALID_ARGUMENT,
348 "Unable to load PJRT library " +
349 std::string(library_path) + ": " + dlerror());
350 }
351 typedef const PJRT_Api* (*PjrtApiFunc)();
352 auto func = reinterpret_cast<PjrtApiFunc>(dlsym(handle, "GetPjrtApi"));
353 if (!func) {
354 throw PjrtException(PjrtErrorCode::INVALID_ARGUMENT,
355 "Unable to find GetPjrtApi() in PJRT library " +
356 std::string(library_path) + ": " + dlerror());
357 }
358 api_ = func();
359 if (!api_) {
360 throw PjrtException(PjrtErrorCode::INVALID_ARGUMENT,
361 "GetPjrtApi() returned nullptr in PJRT library " +
362 std::string(library_path));
363 }
364 auto [major, minor] = ApiVersion();
365 if (major != PJRT_API_MAJOR || minor < PJRT_API_MINOR) {
366 throw PjrtException(
367 PjrtErrorCode::INVALID_ARGUMENT,
368 "PJRT library " + std::string(library_path) +
369 " has incompatible API version: " + std::to_string(major) + "." +
370 std::to_string(minor) + " vs " + std::to_string(PJRT_API_MAJOR) +
371 "." + std::to_string(PJRT_API_MINOR));
372 }
373 Initialize();
374}
