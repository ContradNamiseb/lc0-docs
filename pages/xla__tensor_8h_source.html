43inline size_t GetXlaTypeSize(pblczero::XlaShapeProto::Type type) {
44 switch (type) {
45 case pblczero::XlaShapeProto::F16:
46 return sizeof(uint16_t);
47 case pblczero::XlaShapeProto::BF16:
48 return sizeof(uint16_t);
49 case pblczero::XlaShapeProto::F8E5M2:
50 return sizeof(uint8_t);
51 case pblczero::XlaShapeProto::F32:
52 return sizeof(float);
53 case pblczero::XlaShapeProto::F64:
54 return sizeof(double);
55 case pblczero::XlaShapeProto::S32:
56 return sizeof(int32_t);
57 case pblczero::XlaShapeProto::S64:
58 return sizeof(int64_t);
59 default:
60 throw Exception("Add size for type " +
61 pblczero::XlaShapeProto::Type_Name(type));
62 }
63}

65// An interface for in-host-memory tensor in XLA format.
66class XlaTensor {
67 public:
68 virtual ~XlaTensor() = default;
69 virtual const std::vector<int64_t>& shape() const = 0;
70 virtual const void* data() const = 0;
71 // Returns amount of valid data in bytes.
72 virtual size_t size() const = 0;
73 // Returns amount of memory that are allowed to address in bytes. This is
74 // useful when the size of the buffer has to be increased to match the
75 // supported batch size.
76 virtual size_t capacity() const = 0;
77 virtual pblczero::XlaShapeProto::Type type() const = 0;
78
79 std::string DebugString();
80};

102// Tensor that owns data, used e.g. for XLA output.
103class XlaMutableTensor : public XlaTensor {
104 public:
105 XlaMutableTensor(pblczero::XlaShapeProto::Type type,
106 const std::vector<int64_t>& shape, size_t capacity = 0)
107 : shape_(shape),
108 type_(type),
109 size_(GetBufferSize(type, shape)),
110 capacity_(std::max(size_, capacity)),
111 // TODO replace with make_unique_for_overwrite() once C++20 is
112 // available.
113 data_(new char[capacity_]) {}
114
115 const std::vector<int64_t>& shape() const override { return shape_; }
116 const void* data() const override { return data_.get(); }
117 // TODO replace with std::span once C++20 is available.
118 void* mutable_data() { return data_.get(); }
119 size_t size() const override { return size_; }
120 size_t capacity() const override { return capacity_; }
121 pblczero::XlaShapeProto::Type type() const override { return type_; }
122 void Reshape(const std::vector<int64_t>& new_shape);
123 void Cast(pblczero::XlaShapeProto::Type new_type);
124
125 static size_t GetBufferSize(pblczero::XlaShapeProto::Type type,
126 const std::vector<int64_t>& shape) {
127 return GetXlaTypeSize(type) * std::accumulate(shape.begin(), shape.end(), 1,
128 std::multiplies<int64_t>());
129 }
130
131 private:
132 std::vector<int64_t> shape_;
133 pblczero::XlaShapeProto::Type type_;
134 size_t size_;
135 size_t capacity_;
136 std::unique_ptr<char[]> data_;
137};
