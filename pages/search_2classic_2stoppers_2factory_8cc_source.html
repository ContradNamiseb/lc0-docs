68// Defines options for time management strategies.
69void PopulateTimeManagementOptions(RunType for_what, OptionsParser* options) {
70 PopulateCommonStopperOptions(for_what, options);
71 options->Add<IntOption>(kMoveOverheadId, 0, 100000000) = 200;
72 options->Add<StringOption>(kTimeManagerId) = "legacy";
73 options->Add<FloatOption>(kSlowMoverId, 0.0f, 100.0f) = 1.0f;
74}

76// Instantiates the requested time manager.
77std::unique_ptr<TimeManager> MakeTimeManager(const OptionsDict& options) {
78 const int64_t move_overhead = options.Get<int>(kMoveOverheadId);
79
80 OptionsDict tm_options;
81 tm_options.AddSubdictFromString(options.Get<std::string>(kTimeManagerId));
82 if (!options.IsDefault<float>(kSlowMoverId)) {
83 // Assume that default behavior of simple and normal mode is the same.
84 if (!options.IsDefault<std::string>(kTimeManagerId)) {
85 throw Exception("You can't set both time manager and slowmover value");
86 }
87 float slowmover = options.Get<float>(kSlowMoverId);
88 tm_options.GetMutableSubdict("legacy")->Set("slowmover", slowmover);
89 }
90 const auto managers = tm_options.ListSubdicts();
91
92 std::unique_ptr<TimeManager> time_manager;
93 if (managers.size() != 1) {
94 throw Exception("Exactly one time manager should be specified, " +
95 std::to_string(managers.size()) + " specified instead.");
96 }
97
98 // Create the specific time manager based on the name.
99 if (managers[0] == "legacy") {
100 time_manager =
101 MakeLegacyTimeManager(move_overhead, tm_options.GetSubdict("legacy"));
102 } else if (managers[0] == "alphazero") {
103 time_manager = MakeAlphazeroTimeManager(move_overhead,
104 tm_options.GetSubdict("alphazero"));
105 } else if (managers[0] == "smooth") {
106 time_manager =
107 MakeSmoothTimeManager(move_overhead, tm_options.GetSubdict("smooth"));
108 } else if (managers[0] == "simple") {
109 time_manager =
110 MakeSimpleTimeManager(move_overhead, tm_options.GetSubdict("simple"));
111 }
112
113 if (!time_manager) {
114 throw Exception("Unknown time manager: [" + managers[0] + "]");
115 }
116 tm_options.CheckAllOptionsRead("");
117
118 // Wrap the time manager with common stoppers (time overhead, memory limit,
119 // etc.).
120 return MakeCommonTimeManager(std::move(time_manager), options, move_overhead);
121}
