44template <typename T, size_t N>
45std::vector<std::string> GetAllEnumValues(const std::array<T, N>& vals,
46 std::string (*func)(T)) {
47 std::vector<std::string> res;
48 std::transform(vals.begin(), vals.end(), std::back_inserter(res), func);
49 return res;
50}

52template <typename T, size_t N>
53T GetEnumValueFromString(const std::string& str_value,
54 const std::array<T, N>& vals, std::string (*func)(T)) {
55 auto iter = std::find_if(vals.begin(), vals.end(),
56 [&](T val) { return func(val) == str_value; });
57 if (iter == vals.end()) {
58 throw Exception("Enum value " + str_value + " is unknown.");
59 }
60 return *iter;
61}

106bool ProcessParameters(OptionsParser* options) {
107 using pblczero::NetworkFormat;
108 using pblczero::OnnxModel;
109 options->Add<StringOption>(kInputFilenameId);
110 options->Add<StringOption>(kOutputFilenameId);
111 // Data format options.
112 options->Add<ChoiceOption>(
113 kInputFormatId, GetAllEnumValues(NetworkFormat::InputFormat_AllValues,
114 NetworkFormat::InputFormat_Name)) =
115 NetworkFormat::InputFormat_Name(NetworkFormat::INPUT_CLASSICAL_112_PLANE);
116 options->Add<ChoiceOption>(
117 kPolicyFormatId, GetAllEnumValues(NetworkFormat::PolicyFormat_AllValues,
118 NetworkFormat::PolicyFormat_Name)) =
119 NetworkFormat::PolicyFormat_Name(NetworkFormat::POLICY_CLASSICAL);
120 options->Add<ChoiceOption>(
121 kValueFormatId, GetAllEnumValues(NetworkFormat::ValueFormat_AllValues,
122 NetworkFormat::ValueFormat_Name)) =
123 NetworkFormat::ValueFormat_Name(NetworkFormat::VALUE_WDL);
124 options->Add<ChoiceOption>(
125 kMovesLeftFormatId,
126 GetAllEnumValues(NetworkFormat::MovesLeftFormat_AllValues,
127 NetworkFormat::MovesLeftFormat_Name)) =
128 NetworkFormat::MovesLeftFormat_Name(NetworkFormat::MOVES_LEFT_V1);
129 // Onnx options.
130 options->Add<StringOption>(kOnnxInputId);
131 options->Add<StringOption>(kOnnxOutputPolicyId);
132 options->Add<StringOption>(kOnnxOutputValueId);
133 options->Add<StringOption>(kOnnxOutputWdlId);
134 options->Add<StringOption>(kOnnxOutputMlhId);
135
136 options->Add<BoolOption>(kValidateModelId) = true;
137 options->Add<BoolOption>(kFixRule50Id) = false;
138 options->Add<BoolOption>(kFixWdlSoftmaxId) = false;
139
140 if (!options->ProcessAllFlags()) return false;
141
142 const OptionsDict& dict = options->GetOptionsDict();
143 dict.EnsureExists<std::string>(kInputFilenameId);
144 dict.EnsureExists<std::string>(kOutputFilenameId);
145 return true;
146}

148bool ValidateNetwork(const pblczero::Net& weights, pblczero::ModelProto& onnx) {
149 const auto& onnx_model = weights.onnx_model();
150
151 if (!onnx.has_ir_version()) {
152 CERR << "ONNX file doesn't appear to have version specified. Likely not an "
153 "ONNX file.";
154 return false;
155 }
156 const auto& onnx_inputs = onnx.graph().input();
157 std::set<std::string> inputs;
158 std::transform(onnx_inputs.begin(), onnx_inputs.end(),
159 std::inserter(inputs, inputs.end()),
160 [](const auto& x) { return std::string(x.name()); });
161
162 const auto& onnx_outputs = onnx.graph().output();
163 std::set<std::string> outputs;
164 std::transform(onnx_outputs.begin(), onnx_outputs.end(),
165 std::inserter(outputs, outputs.end()),
166 [](const auto& x) { return std::string(x.name()); });
167
168 auto check_exists = [](std::string_view n, std::set<std::string>* nodes) {
169 std::string name(n);
170 if (!nodes->contains(name)) {
171 CERR << "Node '" << name << "' doesn't exist in ONNX.";
172 return false;
173 }
174 nodes->erase(name);
175 return true;
176 };
177
178 if (onnx_model.has_input_planes() &&
179 !check_exists(onnx_model.input_planes(), &inputs)) {
180 return false;
181 }
182 if (onnx_model.has_output_value() &&
183 !check_exists(onnx_model.output_value(), &outputs)) {
184 return false;
185 }
186 if (onnx_model.has_output_wdl() &&
187 !check_exists(onnx_model.output_wdl(), &outputs)) {
188 return false;
189 }
190 if (onnx_model.has_output_policy() &&
191 !check_exists(onnx_model.output_policy(), &outputs)) {
192 return false;
193 }
194 if (onnx_model.has_output_mlh() &&
195 !check_exists(onnx_model.output_mlh(), &outputs)) {
196 return false;
197 }
198 for (const auto& input : inputs) {
199 CERR << "Warning: ONNX input node '" << input << "' not used.";
200 }
201 for (const auto& output : outputs) {
202 CERR << "Warning: ONNX output node '" << output << "' not used.";
203 }
204
205 if (!onnx_model.has_input_planes()) {
206 CERR << "The --" << kOnnxInputId.long_flag()
207 << " must be defined. Typical value for the ONNX networks exported "
208 "from Leela is /input/planes.";
209 return false;
210 }
211 if (!onnx_model.has_output_policy()) {
212 CERR << "The --" << kOnnxOutputPolicyId.long_flag()
213 << " must be defined. Typical value for the ONNX networks exported "
214 "from Leela is /output/policy.";
215 return false;
216 }
217 if (!onnx_model.has_output_value() && !onnx_model.has_output_wdl()) {
218 CERR << "Either --" << kOnnxOutputValueId.long_flag() << " or --"
219 << kOnnxOutputWdlId.long_flag()
220 << " must be defined. Typical values for the ONNX networks exported "
221 "from Leela are /output/value and /output/wdl.";
222 return false;
223 }
224 if (onnx_model.has_output_value() && onnx_model.has_output_wdl()) {
225 CERR << "Both --" << kOnnxOutputValueId.long_flag() << " and --"
226 << kOnnxOutputWdlId.long_flag()
227 << " are set. Only one of them has to be set.";
228 return false;
229 }
230
231 return true;
232}

234void FixRule50(pblczero::ModelProto& model, const std::string& in, bool fp16) {
235 std::string name = "rule50fix";
236
237 for (size_t i = 0; i < model.graph().node_size(); i++) {
238 auto node = model.graph().node(i);
239 for (size_t j = 0; j < node.input_size(); j++) {
240 if (node.input(j) == in) {
241 CERR << "Inerting scaling between " << in << " and " << node.name();
242 model.mutable_graph()->mutable_node(i)->mutable_input()->at(j) =
243 std::string(name);
244 }
245 }
246 }
247
248 auto* init = model.mutable_graph()->add_initializer();
249 init->set_name(name + "_weights");
250 init->add_dims(112);
251 init->add_dims(1);
252 init->add_dims(1);
253 if (fp16) {
254 init->set_data_type(pblczero::TensorProto::FLOAT16);
255 std::vector<uint16_t> rule50weights(112, FP32toFP16(1.0f));
256 rule50weights[109] = FP32toFP16(1.0f / 99);
257 init->set_raw_data(
258 std::string(reinterpret_cast<const char*>(rule50weights.data()),
259 rule50weights.size() * sizeof(uint16_t)));
260 } else {
261 init->set_data_type(pblczero::TensorProto::FLOAT);
262 std::vector<float> rule50weights(112, 1.0f);
263 rule50weights[109] = 1.0f / 99;
264 init->set_raw_data(
265 std::string(reinterpret_cast<const char*>(rule50weights.data()),
266 rule50weights.size() * sizeof(float)));
267 }
268 auto* new_node = model.mutable_graph()->add_node();
269 new_node->set_name(name);
270 new_node->set_op_type("Mul");
271 new_node->add_input(in);
272 new_node->add_output(name);
273
274 new_node->add_input(name + "_weights");
275}

277void FixWdlSoftmax(pblczero::ModelProto& model, const std::string& out) {
278 std::string name = "softmax_fix";
279
280 for (size_t i = 0; i < model.graph().node_size(); i++) {
281 auto node = model.graph().node(i);
282 for (size_t j = 0; j < node.output_size(); j++) {
283 if (node.output(j) == out) {
284 CERR << "Inserting softmax between " << node.name() << " and " << out;
285 model.mutable_graph()->mutable_node(i)->mutable_output()->at(j) =
286 std::string(name);
287 break;
288 }
289 }
290 }
291
292 auto* new_node = model.mutable_graph()->add_node();
293 new_node->set_name(name);
294 new_node->set_op_type("Softmax");
295 new_node->add_input(name);
296 new_node->add_output(out);
297}

299pblczero::OnnxModel_DataType GetDataType(pblczero::ModelProto& model,
300 const std::string& name) {
301 using pblczero::TensorProto;
302 using pblczero::OnnxModel;
303 for (auto& in : model.graph().input()) {
304 if (in.name() == name && in.has_type() && in.type().has_tensor_type() &&
305 in.type().tensor_type().has_elem_type()) {
306 auto data_type = in.type().tensor_type().elem_type();
307 switch (data_type) {
308 case TensorProto::FLOAT:
309 return OnnxModel::FLOAT;
310 case TensorProto::FLOAT16:
311 return OnnxModel::FLOAT16;
312 default:
313 throw Exception("Unsupported data type: " +
314 TensorProto::DataType_Name(data_type));
315 }
316 }
317 }
318 return OnnxModel::FLOAT;
319}

321bool EnsureOutDataType(pblczero::ModelProto& model, const std::string& name,
322 pblczero::OnnxModel_DataType data_type) {
323 // Check if output has the correct data type and set it if not.
324 for (size_t i = 0; i < model.graph().output_size(); i++) {
325 auto out = model.graph().output(i);
326 if (out.name() == name) {
327 if (!out.has_type()) {
328 model.mutable_graph()->mutable_output(i)->mutable_type();
329 }
330 if (!out.type().has_tensor_type()) {
331 model.mutable_graph()
332 ->mutable_output(i)
333 ->mutable_type()
334 ->mutable_tensor_type();
335 }
336 if (!out.type().tensor_type().has_elem_type() ||
337 out.type().tensor_type().elem_type() !=
338 static_cast<pblczero::TensorProto_DataType>(data_type)) {
339 model.mutable_graph()
340 ->mutable_output(i)
341 ->mutable_type()
342 ->mutable_tensor_type()
343 ->set_elem_type(
344 static_cast<pblczero::TensorProto_DataType>(data_type));
345 break;
346 }
347 return false;
348 }
349 }
350
351 // Insert a cast to the correct data type.
352 for (size_t i = 0; i < model.graph().node_size(); i++) {
353 auto node = model.graph().node(i);
354 for (size_t j = 0; j < node.output_size(); j++) {
355 if (node.output(j) == name) {
356 CERR << "Inserting cast between " << node.name() << " and " << name;
357 model.mutable_graph()->mutable_node(i)->mutable_output()->at(j) =
358 std::string(name + "/cast");
359 break;
360 }
361 }
362 }
363
364 auto* new_node = model.mutable_graph()->add_node();
365 new_node->set_name(name + "/cast");
366 new_node->set_op_type("Cast");
367 new_node->add_input(name + "/cast");
368 new_node->add_output(name);
369 auto* attr = new_node->add_attribute();
370 attr->set_name("to");
371 attr->set_type(pblczero::AttributeProto::INT);
372 attr->set_i(data_type);
373 return true;
374}

376bool MaybeFixOnnx(pblczero::ModelProto& model, const OptionsDict& dict,
377 pblczero::OnnxModel_DataType data_type) {
378 bool updated = false;
379
380 // Input.
381 if (dict.OwnExists<std::string>(kOnnxInputId)) {
382 if (dict.Get<bool>(kFixRule50Id)) {
383 FixRule50(model, dict.Get<std::string>(kOnnxInputId),
384 data_type == pblczero::OnnxModel::FLOAT16);
385 updated = true;
386 }
387 }
388
389 // Policy.
390 if (dict.OwnExists<std::string>(kOnnxOutputPolicyId)) {
391 updated |= EnsureOutDataType(
392 model, dict.Get<std::string>(kOnnxOutputPolicyId), data_type);
393 }
394
395 // Value.
396 if (dict.OwnExists<std::string>(kOnnxOutputValueId)) {
397 updated |= EnsureOutDataType(
398 model, dict.Get<std::string>(kOnnxOutputValueId), data_type);
399 }
400 if (dict.OwnExists<std::string>(kOnnxOutputWdlId)) {
401 auto out = dict.Get<std::string>(kOnnxOutputWdlId);
402 if (dict.Get<bool>(kFixWdlSoftmaxId)) {
403 FixWdlSoftmax(model, out);
404 updated = true;
405 }
406 updated |= EnsureOutDataType(model, out, data_type);
407 }
408
409 // Mlh.
410 if (dict.OwnExists<std::string>(kOnnxOutputMlhId)) {
411 updated |= EnsureOutDataType(model, dict.Get<std::string>(kOnnxOutputMlhId),
412 data_type);
413 }
414
415 return updated;
416}

420void ConvertOnnxToLeela() {
421 using pblczero::NetworkFormat;
422 using pblczero::OnnxModel;
423 OptionsParser options_parser;
424 if (!ProcessParameters(&options_parser)) return;
425
426 const OptionsDict& dict = options_parser.GetOptionsDict();
427
428 auto onnx_model = ReadFileToString(dict.Get<std::string>(kInputFilenameId));
429 pblczero::ModelProto model;
430 model.ParseFromString(onnx_model);
431
432 pblczero::Net out_weights;
433 out_weights.set_magic(0x1c0);
434 // ONNX networks appeared in v0.28.
435 out_weights.mutable_min_version()->set_major(0);
436 out_weights.mutable_min_version()->set_minor(28);
437
438 auto format = out_weights.mutable_format()->mutable_network_format();
439 format->set_network(NetworkFormat::NETWORK_ONNX);
440 auto onnx = out_weights.mutable_onnx_model();
441 auto data_type = OnnxModel::FLOAT;
442
443 // Input.
444 format->set_input(GetEnumValueFromString(
445 dict.Get<std::string>(kInputFormatId),
446 NetworkFormat::InputFormat_AllValues, NetworkFormat::InputFormat_Name));
447 if (dict.OwnExists<std::string>(kOnnxInputId)) {
448 auto in = dict.Get<std::string>(kOnnxInputId);
449 onnx->set_input_planes(in);
450 data_type = GetDataType(model, in);
451 }
452 onnx->set_data_type(data_type);
453
454 // Policy.
455 format->set_policy(GetEnumValueFromString(
456 dict.Get<std::string>(kPolicyFormatId),
457 NetworkFormat::PolicyFormat_AllValues, NetworkFormat::PolicyFormat_Name));
458 if (dict.OwnExists<std::string>(kOnnxOutputPolicyId)) {
459 onnx->set_output_policy(dict.Get<std::string>(kOnnxOutputPolicyId));
460 }
461
462 // Value.
463 format->set_value(GetEnumValueFromString(
464 dict.Get<std::string>(kValueFormatId),
465 NetworkFormat::ValueFormat_AllValues, NetworkFormat::ValueFormat_Name));
466 if (dict.OwnExists<std::string>(kOnnxOutputValueId)) {
467 onnx->set_output_value(dict.Get<std::string>(kOnnxOutputValueId));
468 format->set_output(NetworkFormat::OUTPUT_CLASSICAL);
469 }
470 if (dict.OwnExists<std::string>(kOnnxOutputWdlId)) {
471 onnx->set_output_wdl(dict.Get<std::string>(kOnnxOutputWdlId));
472 format->set_output(NetworkFormat::OUTPUT_WDL);
473 }
474
475 // Mlh.
476 if (dict.OwnExists<std::string>(kOnnxOutputMlhId)) {
477 format->set_moves_left(
478 GetEnumValueFromString(dict.Get<std::string>(kMovesLeftFormatId),
479 NetworkFormat::MovesLeftFormat_AllValues,
480 NetworkFormat::MovesLeftFormat_Name));
481 onnx->set_output_mlh(dict.Get<std::string>(kOnnxOutputMlhId));
482 }
483
484 if (MaybeFixOnnx(model, dict, data_type)) {
485 onnx->set_model(model.OutputAsString());
486 } else {
487 onnx->set_model(onnx_model);
488 }
489 if (dict.Get<bool>(kValidateModelId) &&
490 !ValidateNetwork(out_weights, model)) {
491 return;
492 }
493 WriteStringToGzFile(dict.Get<std::string>(kOutputFilenameId),
494 out_weights.OutputAsString());
495 ShowNetworkFormatInfo(out_weights);
496 ShowNetworkOnnxInfo(out_weights, dict.Get<bool>(kValidateModelId));
497 COUT << "Done.";
498}
