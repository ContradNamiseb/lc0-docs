294template <typename T>
295using EigenMatrixMap =
296 Eigen::Map<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>>;
297template <typename T>
298using ConstEigenMatrixMap =
299 Eigen::Map<const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>>;
300template <typename T>
301using EigenStridedMatrixMap =
302 Eigen::Map<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>, 0,
303 Eigen::OuterStride<>>;
304template <typename T>
305using ConstEigenStridedMatrixMap =
306 Eigen::Map<const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>, 0,
307 Eigen::OuterStride<>>;

311 vec.clear();
312 vec.resize(size);

316template <bool use_eigen>
317void BlasComputation<use_eigen>::ForwardEncoderLayer(
318 std::vector<float>& encoder_buffer, std::vector<float>& encoder_buffer2,
319 std::vector<float>& encoder_buffer3, std::vector<float>& encoder_buffer4,
320 size_t batch_size, const MultiHeadWeights::EncoderLayer& layer,
321 int embedding_size, int heads, ActivationFunction smolgen_activation,
322 ActivationFunction ffn_activation, float alpha, float default_eps) {
323 const int d_model = layer.mha.q_b.size();
324 const int dff_size = layer.ffn.dense1_b.size();
325 const int hidden_channels =
326 layer.mha.has_smolgen ? layer.mha.smolgen.compress.size() / embedding_size
327 : 0;
328 const int hidden_sz =
329 layer.mha.has_smolgen ? layer.mha.smolgen.dense1_b.size() : 0;
330 const int gen_sz_outputs =
331 layer.mha.has_smolgen ? layer.mha.smolgen.dense2_b.size() : 0;
332
333 const int largest_batch_size = std::min(max_batch_size_, planes_.size());
334
335 vec_adjust(encoder_buffer, largest_batch_size * d_model * kSquares);
336 vec_adjust(encoder_buffer2,
337 largest_batch_size *
338 std::max(std::max(d_model, hidden_channels) * kSquares,
339 gen_sz_outputs));
340 vec_adjust(encoder_buffer3,
341 largest_batch_size * std::max(d_model * kSquares, hidden_sz));
342 vec_adjust(encoder_buffer4,
343 batch_size * kSquares * std::max(kSquares * heads, dff_size));
344
345 // Smolgen.
346 if (layer.mha.has_smolgen) {
347 const float* input = &encoder_buffer[0];
348 float* QK = &encoder_buffer4[0];
349
350 // Compress.
351 FullyConnectedLayer<use_eigen>::Forward1D(
352 batch_size * kSquares, embedding_size, hidden_channels, input,
353 layer.mha.smolgen.compress.data(), (const float*)nullptr,
354 ACTIVATION_NONE, encoder_buffer2.data());
355
356 // Dense 1.
357 FullyConnectedLayer<use_eigen>::Forward1D(
358 batch_size, kSquares * hidden_channels, hidden_sz,
359 encoder_buffer2.data(), layer.mha.smolgen.dense1_w.data(),
360 layer.mha.smolgen.dense1_b.data(), smolgen_activation,
361 encoder_buffer3.data());
362 // Layer Norm.
363 LayerNorm2DWithSkipConnection(batch_size, hidden_sz, encoder_buffer3.data(),
364 1.0f, (const float*)nullptr,
365 layer.mha.smolgen.ln1_gammas.data(),
366 layer.mha.smolgen.ln1_betas.data(), 1e-3);
367
368 // Dense 2.
369 FullyConnectedLayer<use_eigen>::Forward1D(
370 batch_size, hidden_sz, gen_sz_outputs, encoder_buffer3.data(),
371 layer.mha.smolgen.dense2_w.data(), layer.mha.smolgen.dense2_b.data(),
372 smolgen_activation, encoder_buffer2.data());
373 // Layer Norm.
374 LayerNorm2DWithSkipConnection(
375 batch_size, gen_sz_outputs, encoder_buffer2.data(), 1.0f,
376 (const float*)nullptr, layer.mha.smolgen.ln2_gammas.data(),
377 layer.mha.smolgen.ln2_betas.data(), 1e-3);
378
379 // Global smolgen weights.
380 FullyConnectedLayer<use_eigen>::Forward1D(
381 batch_size * heads, gen_sz_outputs / heads, kSquares * kSquares,
382 encoder_buffer2.data(), weights_.smolgen_w.data(),
383 (const float*)nullptr, ACTIVATION_NONE, QK);
384 }
385
386 // Q
387 FullyConnectedLayer<use_eigen>::Forward1D(
388 batch_size * kSquares, embedding_size, d_model, encoder_buffer.data(),
389 layer.mha.q_w.data(), layer.mha.q_b.data(), ACTIVATION_NONE,
390 encoder_buffer2.data());
391 // K
392 FullyConnectedLayer<use_eigen>::Forward1D(
393 batch_size * kSquares, embedding_size, d_model, encoder_buffer.data(),
394 layer.mha.k_w.data(), layer.mha.k_b.data(), ACTIVATION_NONE,
395 encoder_buffer3.data());
396
397 // MHA (Q, K, V)
398 const int depth = d_model / heads;
399 const float scaling = 1.0f / sqrtf(depth);
400
401 // MHA is done per batch since there's a fourth dimension introduced.
402 for (auto batch = size_t{0}; batch < batch_size; batch++) {
403 auto batchStart = batch * kSquares * d_model;
404
405 float* QK = &encoder_buffer4[batch * kSquares * kSquares * heads];
406
407 const float* Q = &encoder_buffer2[batchStart];
408 const float* K = &encoder_buffer3[batchStart];
409
410 // matmul(Q, K) for all heads per batch.
411
412 for (auto h = 0; h < heads; h++) {
413 const float* A = &Q[h * depth];
414 const float* B = &K[h * depth];
415 float* C = &QK[h * kSquares * kSquares];
416 const float beta = layer.mha.has_smolgen ? 1.0f : 0.0f;
417 if (use_eigen) {
418 auto C_mat = EigenMatrixMap<float>(C, kSquares, kSquares);
419 C_mat.noalias() =
420 beta * C_mat +
421 scaling *
422 ConstEigenStridedMatrixMap<float>(
423 B, depth, kSquares, Eigen::OuterStride<>(heads * depth))
424 .transpose() *
425 ConstEigenStridedMatrixMap<float>(
426 A, depth, kSquares, Eigen::OuterStride<>(heads * depth));
427 } else {
428#ifdef USE_BLAS
429 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, kSquares, kSquares,
430 depth, scaling, A, heads * depth, B, heads * depth, beta, C,
431 kSquares);
432#else
433 // Should never get here.
434 throw Exception("Blas backend internal error");
435#endif
436 }
437 }
438 }
439
440 // Apply Softmax.
441 float* QK = &encoder_buffer4[0];
442 for (size_t h = 0; h < batch_size * heads * kSquares * kSquares;
443 h += kSquares) {
444#if defined(USE_ISPC)
445 if (!use_eigen) {
446 ispc::SoftmaxActivation(kSquares, QK + h, QK + h);
447 continue;
448 }
449#endif
450 SoftmaxActivation(kSquares, QK + h, QK + h);
451 }
452
453 // V
454 FullyConnectedLayer<use_eigen>::Forward1D(
455 batch_size * kSquares, embedding_size, d_model, encoder_buffer.data(),
456 layer.mha.v_w.data(), layer.mha.v_b.data(), ACTIVATION_NONE,
457 encoder_buffer3.data());
458
459 for (auto batch = size_t{0}; batch < batch_size; batch++) {
460 auto batchStart = batch * kSquares * d_model;
461 // matmul(softmax(QK), V) for all heads per batch.
462 float* attn = &encoder_buffer2[batchStart];
463 const float* V = &encoder_buffer3[batchStart];
464 const float* QK = &encoder_buffer4[batch * kSquares * kSquares * heads];
465 for (auto h = 0; h < heads; h++) {
466 const float* A = &QK[h * kSquares * kSquares];
467 const float* B = &V[h * depth];
468 float* C = &attn[h * depth];
469 if (use_eigen) {
470 auto C_mat = EigenStridedMatrixMap<float>(
471 C, depth, kSquares, Eigen::OuterStride<>(heads * depth));
472 C_mat.noalias() =
473 ConstEigenStridedMatrixMap<float>(
474 B, depth, kSquares, Eigen::OuterStride<>(heads * depth)) *
475 ConstEigenMatrixMap<float>(A, kSquares, kSquares);
476 } else {
477#ifdef USE_BLAS
478 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, kSquares, depth,
479 kSquares, 1.0f, A, kSquares, B, heads * depth, 0.0f, C,
480 heads * depth);
481#endif
482 }
483 }
484 }
485
486 // Fully connected final MHA layer.
487 FullyConnectedLayer<use_eigen>::Forward1D(
488 batch_size * kSquares, d_model, embedding_size, encoder_buffer2.data(),
489 layer.mha.dense_w.data(), layer.mha.dense_b.data(), ACTIVATION_NONE,
490 encoder_buffer3.data());
491
492 // Layer Norm + skip connection.
493 LayerNorm2DWithSkipConnection(batch_size * kSquares, embedding_size,
494 encoder_buffer3.data(), alpha,
495 encoder_buffer.data(), layer.ln1_gammas.data(),
496 layer.ln1_betas.data(), default_eps);
497 std::swap(encoder_buffer3, encoder_buffer);
498
499 // FFN.
500 FullyConnectedLayer<use_eigen>::Forward1D(
501 batch_size * kSquares, embedding_size, dff_size, encoder_buffer.data(),
502 layer.ffn.dense1_w.data(), layer.ffn.dense1_b.data(), ffn_activation,
503 encoder_buffer4.data());
504
505 FullyConnectedLayer<use_eigen>::Forward1D(
506 batch_size * kSquares, dff_size, layer.ffn.dense2_b.size(),
507 encoder_buffer4.data(), layer.ffn.dense2_w.data(),
508 layer.ffn.dense2_b.data(), ACTIVATION_NONE, encoder_buffer3.data());
509
510 // Layer Norm + skip connection.
511 LayerNorm2DWithSkipConnection(batch_size * kSquares, embedding_size,
512 encoder_buffer3.data(), alpha,
513 encoder_buffer.data(), layer.ln2_gammas.data(),
514 layer.ln2_betas.data(), default_eps);
515 std::swap(encoder_buffer3, encoder_buffer);
516}

518template <bool use_eigen>
519void BlasComputation<use_eigen>::ComputeBlocking() {
520 const auto& value_head = weights_.value_heads.at(value_head_);
521 const auto& policy_head = weights_.policy_heads.at(policy_head_);
522 // Retrieve network key dimensions from the weights structure.
523 const auto num_value_channels = value_head.ip1_val_b.size();
524 const auto num_moves_channels = weights_.ip1_mov_b.size();
525 const auto num_value_input_planes =
526 attn_body_ ? value_head.ip_val_b.size() : value_head.value.biases.size();
527 const auto num_policy_input_planes = policy_head.policy.biases.size();
528 const auto num_moves_input_planes =
529 attn_body_ ? weights_.ip_mov_b.size() : weights_.moves_left.biases.size();
530 const auto num_output_policy = static_cast<size_t>(kPolicyOutputs);
531 const auto output_channels =
532 attn_body_ ? weights_.ip_emb_b.size() : weights_.input.biases.size();
533 const auto num_res_blocks = weights_.residual.size();
534
535 // max_channels is the maximum number of input channels of any
536 // convolution.
537 // Residual blocks are identical, but the first convolution might be bigger
538 // when the network has very few filters
539 // For attention body nets, input channel size is increased by positional
540 // encoding.
541 const auto enc_channels = is_pe_dense_embedding_
542 ? weights_.ip_emb_preproc_b.size() / 64
543 : kNumPosEncodingChannels;
544 const auto input_channels =
545 static_cast<size_t>(kInputPlanes + (attn_body_ ? enc_channels : 0));
546 const auto input_embed_dff = weights_.ip_emb_ffn.dense1_b.size();
547
548 const auto max_channels =
549 std::max(std::max(output_channels, input_channels), input_embed_dff);
550
551 // The policy head may increase convolution max output size.
552 const auto max_output_channels =
553 (conv_policy_ && policy_head.policy.biases.size() > output_channels)
554 ? policy_head.policy.biases.size()
555 : output_channels;
556
557 // Determine the largest batch for allocations.
558 const auto total_batches = planes_.size();
559 const auto largest_batch_size = std::min(max_batch_size_, total_batches);
560
561 /* Typically
562 input_channels = 112
563 position encoding = 64 (512 for new encoding)
564 output_channels = 192
565 max_channels = 192
566 num_value_input_planes = 32
567 num_policy_input_planes = 32
568 num_value_channels = 128
569 num_output_policy = 1858
570 */
571
572 size_t max_fc_channels = std::max(
573 num_value_channels, std::max(num_output_policy, num_moves_channels));
574 size_t max_head_planes =
575 std::max(num_policy_input_planes,
576 std::max(num_value_input_planes, num_moves_input_planes));
577 if (attn_policy_) {
578 max_head_planes = std::max(std::max(max_head_planes, size_t{67}),
579 policy_head.ip_pol_b.size());
580 }
581
582 std::unique_ptr<Buffers> buffers = network_->GetBuffers();
583
584 // Allocate data for the whole batch.
585 std::vector<float>& buffer1 = buffers->buffer1;
586 vec_adjust(buffer1, largest_batch_size * max_channels * kSquares);
587 std::vector<float>& buffer2 = buffers->buffer2;
588 vec_adjust(buffer2, largest_batch_size * max_channels * kSquares);
589 std::vector<float>& buffer3 = buffers->buffer3;
590 vec_adjust(buffer3, largest_batch_size *
591 std::max(max_channels * kSquares, max_fc_channels));
592 std::vector<float>& head_buffer = buffers->buffer4;
593 vec_adjust(head_buffer, largest_batch_size * max_head_planes * kSquares);
594
595 // Output values.
596 q_values_.reserve(wdl_ ? 3 * total_batches : total_batches);
597 policies_.reserve(total_batches);
598 if (moves_left_) m_values_.resize(total_batches);
599
600 WinogradConvolution3<use_eigen> convolve3(largest_batch_size, max_channels,
601 max_output_channels);
602
603 for (size_t start = 0; start < total_batches; start += largest_batch_size) {
604 const auto batch_size = std::min(total_batches - start, largest_batch_size);
605 for (size_t j = 0; j < batch_size; j++) {
606 EncodePlanes(planes_[start + j], &buffer1[j * kSquares * kInputPlanes]);
607 }
608
609 if (num_res_blocks > 0) {
610 // Input convolution
611 convolve3.Forward(batch_size, kInputPlanes, output_channels,
612 buffer1.data(), weights_.input.weights.data(),
613 buffer2.data());
614
615 BiasActivate(batch_size, output_channels, buffer2.data(),
616 weights_.input.biases.data(), default_activation_);
617
618 // Residual tower
619 for (auto& residual : weights_.residual) {
620 const auto& conv1 = residual.conv1;
621 const auto& conv2 = residual.conv2;
622 const auto& se = residual.se;
623
624 convolve3.Forward(batch_size, output_channels, output_channels,
625 buffer2.data(), conv1.weights.data(), buffer1.data());
626
627 BiasActivate(batch_size, output_channels, buffer1.data(),
628 conv1.biases.data(), default_activation_);
629
630 convolve3.Forward(batch_size, output_channels, output_channels,
631 buffer1.data(), conv2.weights.data(), buffer3.data());
632
633 if (residual.has_se) {
634 // No relu if followed by SE-unit and residual/bias is added later
635 auto se_fc_outputs = se.b1.size();
636 ApplySEUnit<use_eigen>(
637 batch_size, output_channels, se_fc_outputs, buffer3.data(),
638 conv2.biases.data(), buffer2.data(), se.w1.data(), se.b1.data(),
639 se.w2.data(), se.b2.data(), buffer2.data(), default_activation_);
640 } else {
641 BiasResidual(batch_size, output_channels, buffer2.data(),
642 conv2.biases.data(), buffer3.data(),
643 default_activation_);
644 }
645 }
646 }
647
648 if (attn_body_) {
649 const auto embedding_size = weights_.ip_emb_b.size();
650 assert(embedding_size > 0);
651 const auto input_size =
652 num_res_blocks == 0 ? input_channels : weights_.input.biases.size();
653
654 if (num_res_blocks == 0) {
655 // No residual means pure transformer, so process input position
656 // encoding.
657 if (is_pe_dense_embedding_) {
658 // NCHW to NHWC conversion of 12-channel slice of input.
659 for (auto batch = size_t{0}; batch < batch_size; batch++) {
660 for (auto i = 0; i < kSquares; i++) {
661 for (size_t j = 0; j < 12; j++) {
662 buffer3[batch * kSquares * 12 + i * 12 + j] =
663 buffer1[batch * kSquares * kInputPlanes + j * kSquares + i];
664 }
665 }
666 }
667 // Dense embedding preprocess layer.
668 FullyConnectedLayer<use_eigen>::Forward1D(
669 batch_size, kSquares * 12, weights_.ip_emb_preproc_b.size(),
670 buffer3.data(), weights_.ip_emb_preproc_w.data(),
671 weights_.ip_emb_preproc_b.data(), ACTIVATION_NONE,
672 buffer2.data());
673 }
674
675 // Preprocess for attention body.
676 for (auto batch = size_t{0}; batch < batch_size; batch++) {
677 for (auto i = 0; i < kSquares; i++) {
678 // Input NCHW to NHWC conversion.
679 for (size_t j = 0; j < kInputPlanes; j++) {
680 buffer3[batch * kSquares * input_size + i * input_size + j] =
681 buffer1[batch * kSquares * kInputPlanes + j * kSquares + i];
682 }
683 // Position encoding concat.
684 if (is_pe_dense_embedding_) {
685 for (size_t j = kInputPlanes; j < input_size; j++) {
686 buffer3[batch * kSquares * input_size + i * input_size + j] =
687 buffer2[batch * kSquares * enc_channels + i * enc_channels +
688 j - kInputPlanes];
689 }
690 } else {
691 for (size_t j = kInputPlanes; j < input_size; j++) {
692 buffer3[batch * kSquares * input_size + i * input_size + j] =
693 kPosEncoding[i][j - kInputPlanes];
694 }
695 }
696 }
697 }
698 }
699
700 // Input embedding.
701 FullyConnectedLayer<use_eigen>::Forward1D(
702 batch_size * kSquares, input_size, embedding_size, buffer3.data(),
703 weights_.ip_emb_w.data(), weights_.ip_emb_b.data(),
704 default_activation_, buffer1.data());
705
706 // Layer norm for new encoding.
707 if (is_pe_dense_embedding_) {
708 LayerNorm2DWithSkipConnection(
709 batch_size * kSquares, embedding_size, buffer1.data(), 1.0f,
710 (const float*)nullptr, weights_.ip_emb_ln_gammas.data(),
711 weights_.ip_emb_ln_betas.data(), 1e-3);
712 }
713
714 // Input gating
715 if (weights_.ip_mult_gate.size() > 0 && weights_.ip_add_gate.size() > 0) {
716 int idx;
717 for (auto batch = size_t{0}; batch < batch_size; batch++) {
718 for (auto i = 0; i < kSquares; i++) {
719 for (size_t j = 0; j < embedding_size; j++) {
720 idx = batch * kSquares * embedding_size + i * embedding_size + j;
721 buffer1[idx] =
722 buffer1[idx] * weights_.ip_mult_gate[j * kSquares + i] +
723 weights_.ip_add_gate[j * kSquares + i];
724 }
725 }
726 };
727 }
728
729 float alpha = (float)pow(2.0 * weights_.encoder.size(), -0.25);
730
731 // FFN in embedding for new encoding.
732 if (is_pe_dense_embedding_) {
733 const auto dff_size = weights_.ip_emb_ffn.dense1_b.size();
734 // FFN dense 1.
735 FullyConnectedLayer<use_eigen>::Forward1D(
736 batch_size * kSquares, embedding_size, dff_size, buffer1.data(),
737 weights_.ip_emb_ffn.dense1_w.data(),
738 weights_.ip_emb_ffn.dense1_b.data(), ffn_activation_,
739 buffer3.data());
740
741 // FFN dense 2.
742 FullyConnectedLayer<use_eigen>::Forward1D(
743 batch_size * kSquares, dff_size,
744 weights_.ip_emb_ffn.dense2_b.size(), buffer3.data(),
745 weights_.ip_emb_ffn.dense2_w.data(),
746 weights_.ip_emb_ffn.dense2_b.data(), ACTIVATION_NONE,
747 buffer2.data());
748
749 // Layer Norm.
750 LayerNorm2DWithSkipConnection(
751 batch_size * kSquares, weights_.ip_emb_ffn.dense2_b.size(),
752 buffer2.data(), alpha, buffer1.data(),
753 weights_.ip_emb_ffn_ln_gammas.data(),
754 weights_.ip_emb_ffn_ln_betas.data(), 1e-3);
755
756 std::swap(buffer1, buffer2);
757 }
758
759 // Attention body encoders.
760 for (auto& layer : weights_.encoder) {
761 ForwardEncoderLayer(buffer1, buffer2, buffer3, head_buffer, batch_size,
762 layer, embedding_size, weights_.encoder_head_count,
763 smolgen_activation_, ffn_activation_, alpha,
764 is_pe_dense_embedding_ ? 1e-3 : 1e-6);
765 }
766 }
767
768 // Preserve buffer1 and buffer2, used for policy and moves left heads.
769 // Value head
770 if (attn_body_) {
771 FullyConnectedLayer<use_eigen>::Forward1D(
772 batch_size * kSquares, weights_.ip_emb_b.size(),
773 num_value_input_planes, buffer1.data(), value_head.ip_val_w.data(),
774 value_head.ip_val_b.data(), default_activation_, head_buffer.data());
775 } else {
776 Convolution1<use_eigen>::Forward(
777 batch_size, output_channels, num_value_input_planes, buffer2.data(),
778 value_head.value.weights.data(), head_buffer.data());
779
780 BiasActivate(batch_size, num_value_input_planes, &head_buffer[0],
781 value_head.value.biases.data(), default_activation_);
782 }
783
784 FullyConnectedLayer<use_eigen>::Forward1D(
785 batch_size, num_value_input_planes * kSquares, num_value_channels,
786 head_buffer.data(), value_head.ip1_val_w.data(),
787 value_head.ip1_val_b.data(),
788 default_activation_, // Activation On
789 buffer3.data());
790
791 // Now get the score
792 if (wdl_) {
793 std::vector<float> wdl(3 * batch_size);
794 FullyConnectedLayer<use_eigen>::Forward1D(
795 batch_size, num_value_channels, 3, buffer3.data(),
796 value_head.ip2_val_w.data(), value_head.ip2_val_b.data(),
797 ACTIVATION_NONE, // Activation Off
798 wdl.data());
799
800 for (size_t j = 0; j < batch_size; j++) {
801 std::vector<float> wdl_softmax(3);
802 SoftmaxActivation(3, &wdl[j * 3], wdl_softmax.data());
803
804 q_values_.emplace_back(wdl_softmax[0]);
805 q_values_.emplace_back(wdl_softmax[1]);
806 q_values_.emplace_back(wdl_softmax[2]);
807 }
808 } else {
809 for (size_t j = 0; j < batch_size; j++) {
810 double winrate = FullyConnectedLayer<use_eigen>::Forward0D(
811 num_value_channels, value_head.ip2_val_w.data(),
812 &buffer3[j * num_value_channels]) +
813 value_head.ip2_val_b[0];
814
815 q_values_.emplace_back(std::tanh(winrate));
816 }
817 }
818
819 // Moves left head.
820 if (moves_left_) {
821 if (attn_body_) {
822 FullyConnectedLayer<use_eigen>::Forward1D(
823 batch_size * kSquares, weights_.ip_emb_b.size(),
824 num_moves_input_planes, buffer1.data(), weights_.ip_mov_w.data(),
825 weights_.ip_mov_b.data(), default_activation_, head_buffer.data());
826 } else {
827 Convolution1<use_eigen>::Forward(
828 batch_size, output_channels, num_moves_input_planes, buffer2.data(),
829 weights_.moves_left.weights.data(), head_buffer.data());
830
831 BiasActivate(batch_size, num_moves_input_planes, &head_buffer[0],
832 weights_.moves_left.biases.data(), default_activation_);
833 }
834
835 FullyConnectedLayer<use_eigen>::Forward1D(
836 batch_size, num_moves_input_planes * kSquares, num_moves_channels,
837 head_buffer.data(), weights_.ip1_mov_w.data(),
838 weights_.ip1_mov_b.data(),
839 default_activation_, // Activation On
840 buffer3.data());
841
842 std::vector<float> output_moves_left(batch_size);
843 FullyConnectedLayer<use_eigen>::Forward1D(
844 batch_size, num_moves_channels, 1, buffer3.data(),
845 weights_.ip2_mov_w.data(), weights_.ip2_mov_b.data(),
846 ACTIVATION_RELU, // Specifically Relu
847 &m_values_[start]);
848 }
849
850 // Policy head.
851 if (attn_policy_) {
852 if (!attn_body_) {
853 // NCHW to NHWC conversion.
854 for (auto batch = size_t{0}; batch < batch_size; batch++) {
855 for (auto i = 0; i < kSquares; i++) {
856 for (size_t j = 0; j < output_channels; j++) {
857 buffer1[batch * kSquares * output_channels + i * output_channels +
858 j] = buffer2[batch * kSquares * output_channels +
859 j * kSquares + i];
860 }
861 }
862 }
863 }
864 const size_t policy_embedding_size = policy_head.ip_pol_b.size();
865 // Policy Embedding.
866 FullyConnectedLayer<use_eigen>::Forward1D(
867 batch_size * kSquares, output_channels, policy_embedding_size,
868 buffer1.data(), policy_head.ip_pol_w.data(),
869 policy_head.ip_pol_b.data(),
870 attn_body_
871 ? default_activation_
872 : ACTIVATION_SELU, // SELU activation hardcoded for apmish nets.
873 buffer2.data());
874
875 const size_t policy_d_model = policy_head.ip2_pol_b.size();
876
877 for (auto& layer : policy_head.pol_encoder) {
878 ForwardEncoderLayer(
879 buffer2, buffer1, buffer3, head_buffer, batch_size, layer,
880 policy_embedding_size, policy_head.pol_encoder_head_count,
881 attn_body_ ? smolgen_activation_ : ACTIVATION_NONE,
882 attn_body_ ? ffn_activation_ : ACTIVATION_SELU, 1.0f, 1e-6);
883 }
884
885 // Q
886 FullyConnectedLayer<use_eigen>::Forward1D(
887 batch_size * kSquares, policy_embedding_size, policy_d_model,
888 buffer2.data(), policy_head.ip2_pol_w.data(),
889 policy_head.ip2_pol_b.data(), ACTIVATION_NONE, buffer1.data());
890 // K
891 FullyConnectedLayer<use_eigen>::Forward1D(
892 batch_size * kSquares, policy_embedding_size, policy_d_model,
893 buffer2.data(), policy_head.ip3_pol_w.data(),
894 policy_head.ip3_pol_b.data(), ACTIVATION_NONE, buffer3.data());
895 const float scaling = 1.0f / sqrtf(policy_d_model);
896 for (auto batch = size_t{0}; batch < batch_size; batch++) {
897 const float* A = &buffer1[batch * 64 * policy_d_model];
898 const float* B = &buffer3[batch * 64 * policy_d_model];
899 float* C = &head_buffer[batch * (64 * 64 + 8 * 24)];
900 if (use_eigen) {
901 auto C_mat = EigenMatrixMap<float>(C, kSquares, kSquares);
902 C_mat.noalias() =
903 scaling *
904 ConstEigenMatrixMap<float>(B, policy_d_model, kSquares)
905 .transpose() *
906 ConstEigenMatrixMap<float>(A, policy_d_model, kSquares);
907 } else {
908#ifdef USE_BLAS
909 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, kSquares,
910 kSquares, policy_d_model, scaling, A, policy_d_model, B,
911 policy_d_model, 0.0f, C, 64);
912#else
913 // Should never get here.
914 throw Exception("Blas backend internal error");
915#endif
916 }
917 }
918 // Promotion offset calculation.
919 for (auto batch = size_t{0}; batch < batch_size; batch++) {
920 float promotion_offsets[4][8];
921 // This is so small that SGEMM seems slower.
922 for (int i = 0; i < 4; i++) {
923 for (int j = 0; j < 8; j++) {
924 float sum = 0;
925 for (size_t k = 0; k < policy_d_model; k++) {
926 sum += buffer3[batch * kSquares * policy_d_model +
927 (56 + j) * policy_d_model + k] *
928 policy_head.ip4_pol_w[i * policy_d_model + k];
929 }
930 promotion_offsets[i][j] = sum;
931 }
932 }
933 for (int i = 0; i < 3; i++) {
934 for (int j = 0; j < 8; j++) {
935 promotion_offsets[i][j] += promotion_offsets[3][j];
936 }
937 }
938 for (int k = 0; k < 8; k++) { // y in cuda
939 for (int j = 0; j < 8; j++) { // w in cuda
940 for (int i = 0; i < 3; i++) { // c in cuda
941 head_buffer[batch * (64 * 64 + 8 * 24) + 64 * 64 + 24 * k +
942 3 * j + i] = head_buffer[batch * (64 * 64 + 8 * 24) +
943 (48 + k) * 64 + 56 + j] +
944 promotion_offsets[i][j];
945 }
946 }
947 }
948 }
949 // Mapping from attention policy to lc0 policy
950 for (auto batch = size_t{0}; batch < batch_size; batch++) {
951 std::vector<float> policy(num_output_policy);
952 for (auto i = 0; i < 64 * 64 + 8 * 24; i++) {
953 auto j = kAttnPolicyMap[i];
954 if (j >= 0) {
955 policy[j] = head_buffer[batch * (64 * 64 + 8 * 24) + i];
956 }
957 }
958 policies_.emplace_back(std::move(policy));
959 }
960 } else if (conv_policy_) {
961 assert(!attn_body_); // not supported with attention body
962 convolve3.Forward(batch_size, output_channels, output_channels,
963 buffer2.data(), policy_head.policy1.weights.data(),
964 buffer1.data());
965
966 BiasActivate(batch_size, output_channels, buffer1.data(),
967 policy_head.policy1.biases.data(), default_activation_);
968
969 convolve3.Forward(batch_size, output_channels, num_policy_input_planes,
970 buffer1.data(), policy_head.policy.weights.data(),
971 head_buffer.data());
972
973 BiasActivate(batch_size, num_policy_input_planes, head_buffer.data(),
974 policy_head.policy.biases.data(), ACTIVATION_NONE);
975
976 // Mapping from convolutional policy to lc0 policy
977 for (auto batch = size_t{0}; batch < batch_size; batch++) {
978 std::vector<float> policy(num_output_policy);
979 for (auto i = 0; i < kPolicyUsedPlanes * kSquares; i++) {
980 auto j = kConvPolicyMap[i];
981 if (j >= 0) {
982 policy[j] =
983 head_buffer[batch * num_policy_input_planes * kSquares + i];
984 }
985 }
986 policies_.emplace_back(std::move(policy));
987 }
988
989 } else {
990 assert(!attn_body_); // not supported with attention body
991 Convolution1<use_eigen>::Forward(
992 batch_size, output_channels, num_policy_input_planes, buffer2.data(),
993 policy_head.policy.weights.data(), head_buffer.data());
994
995 BiasActivate(batch_size, num_policy_input_planes, &head_buffer[0],
996 policy_head.policy.biases.data(), default_activation_);
997
998 FullyConnectedLayer<use_eigen>::Forward1D(
999 batch_size, num_policy_input_planes * kSquares, num_output_policy,
1000 head_buffer.data(), policy_head.ip_pol_w.data(),
1001 policy_head.ip_pol_b.data(),
1002 ACTIVATION_NONE, // Activation Off
1003 buffer3.data());
1004
1005 for (size_t j = 0; j < batch_size; j++) {
1006 std::vector<float> policy(num_output_policy);
1007
1008 // Get the moves
1009 policy.assign(buffer3.begin() + j * num_output_policy,
1010 buffer3.begin() + (j + 1) * num_output_policy);
1011 policies_.emplace_back(std::move(policy));
1012 }
1013 }
1014 }
1015 network_->ReleaseBuffers(std::move(buffers));
1016}

1018template <bool use_eigen>
1019void BlasComputation<use_eigen>::EncodePlanes(const InputPlanes& sample,
1020 float* buffer) {
1021 for (const InputPlane& plane : sample) {
1022 const float value = plane.value;
1023 for (auto i = 0; i < kSquares; i++)
1024 *(buffer++) = (plane.mask & (((uint64_t)1) << i)) != 0 ? value : 0;
1025 }

1028template <bool use_eigen>
1029BlasNetwork<use_eigen>::BlasNetwork(const WeightsFile& file,
1030 const OptionsDict& options)
1031 : capabilities_{file.format().network_format().input(),
1032 file.format().network_format().output(),
1033 file.format().network_format().moves_left()},
1034 weights_(file.weights()) {
1035 Numa::Init();
1036
1037 max_batch_size_ =
1038 static_cast<size_t>(options.GetOrDefault<int>("batch_size", 256));
1039 threads_ = options.GetOrDefault<int>("threads", 1);
1040
1041 auto nf = file.format().network_format();
1042 using NF = pblczero::NetworkFormat;
1043 wdl_ = nf.value() == NF::VALUE_WDL;
1044
1045 moves_left_ = (nf.moves_left() == NF::MOVES_LEFT_V1) &&
1046 options.GetOrDefault<bool>("mlh", true);
1047
1048 conv_policy_ = nf.policy() == NF::POLICY_CONVOLUTION;
1049
1050 attn_policy_ = nf.policy() == NF::POLICY_ATTENTION;
1051
1052 attn_body_ = nf.network() == NF::NETWORK_ATTENTIONBODY_WITH_HEADFORMAT ||
1053 nf.network() == NF::NETWORK_ATTENTIONBODY_WITH_MULTIHEADFORMAT;
1054
1055 default_activation_ = nf.default_activation() == NF::DEFAULT_ACTIVATION_MISH
1056 ? ACTIVATION_MISH
1057 : ACTIVATION_RELU;
1058
1059 is_pe_dense_embedding_ =
1060 static_cast<InputEmbedding>(
1061 file.format().network_format().input_embedding()) ==
1062 InputEmbedding::INPUT_EMBEDDING_PE_DENSE;
1063
1064 if (attn_body_) {
1065 const auto smol_act = nf.smolgen_activation();
1066 smolgen_activation_ = smol_act == NF::ACTIVATION_DEFAULT
1067 ? default_activation_
1068 : static_cast<ActivationFunction>(smol_act);
1069 const auto ffn_act = nf.ffn_activation();
1070 ffn_activation_ = ffn_act == NF::ACTIVATION_DEFAULT
1071 ? default_activation_
1072 : static_cast<ActivationFunction>(ffn_act);
1073 }
1074
1075 if (max_batch_size_ > kHardMaxBatchSize) {
1076 max_batch_size_ = kHardMaxBatchSize;
1077 }
1078
1079 const auto inputChannels = kInputPlanes;
1080 const auto channels = static_cast<int>(weights_.input.biases.size());
1081 const auto residual_blocks = weights_.residual.size();
1082
1083 weights_.input.weights =
1084 WinogradFilterTransformF(weights_.input.weights, channels, inputChannels);
1085
1086 // residual blocks
1087 for (size_t i = 0; i < residual_blocks; i++) {
1088 auto& residual = weights_.residual[i];
1089 auto& conv1 = residual.conv1;
1090 auto& conv2 = residual.conv2;
1091
1092 conv1.weights = WinogradFilterTransformF(conv1.weights, channels, channels);
1093 conv2.weights = WinogradFilterTransformF(conv2.weights, channels, channels);
1094 }
1095
1096 policy_head_ = options.GetOrDefault<std::string>("policy_head", "vanilla");
1097 // Check that selected policy head exists.
1098 if (weights_.policy_heads.count(policy_head_) == 0) {
1099 throw Exception("The policy head you specified '" + policy_head_ +
1100 "' does not exist in this net.");
1101 }
1102
1103 value_head_ = options.GetOrDefault<std::string>("value_head", "winner");
1104 // Check that selected value head exists.
1105 if (weights_.value_heads.count(value_head_) == 0) {
1106 throw Exception("The value head you specified '" + value_head_ +
1107 "' does not exist in this net.");
1108 }
1109
1110 if (conv_policy_) {
1111 auto& policy_head = weights_.policy_heads.at("vanilla");
1112 policy_head.policy1.weights = WinogradFilterTransformF(
1113 policy_head.policy1.weights, channels, channels);
1114 auto pol_channels = policy_head.policy.biases.size();
1115 policy_head.policy.weights = WinogradFilterTransformF(
1116 policy_head.policy.weights, pol_channels, channels);
1117 }
1118
1119 if (use_eigen) {
1120 CERR << "Using Eigen version " << EIGEN_WORLD_VERSION << "."
1121 << EIGEN_MAJOR_VERSION << "." << EIGEN_MINOR_VERSION;
1122 CERR << "Eigen max batch size is " << max_batch_size_ << ".";
1123 } else {
1124#ifdef USE_OPENBLAS
1125 int num_procs = openblas_get_num_procs();
1126 openblas_set_num_threads(threads_);
1127 const char* core_name = openblas_get_corename();
1128 const char* config = openblas_get_config();
1129 CERR << "BLAS vendor: OpenBLAS.";
1130 CERR << "OpenBLAS [" << config << "].";
1131 CERR << "OpenBLAS found " << num_procs << " " << core_name << " core(s).";
1132#endif
1133
1134#ifdef USE_MKL
1135 mkl_set_num_threads(threads_);
1136 CERR << "BLAS vendor: MKL.";
1137 constexpr int len = 256;
1138 char versionbuf[len];
1139 mkl_get_version_string(versionbuf, len);
1140 CERR << "MKL " << versionbuf << ".";
1141 MKLVersion version;
1142 mkl_get_version(&version);
1143 CERR << "MKL platform: " << version.Platform
1144 << ", processor: " << version.Processor << ".";
1145#endif
1146
1147#ifdef USE_DNNL
1148 const dnnl_version_t* ver = dnnl_version();
1149 CERR << "BLAS functions from DNNL version " << ver->major << "."
1150 << ver->minor << "." << ver->patch;
1151#endif
1152
1153#ifdef USE_ACCELERATE
1154 CERR << "BLAS vendor: Apple vecLib.";
1155#endif
1156 CERR << "BLAS max batch size is " << max_batch_size_ << ".";
1157 }
1158}

1160template <bool use_eigen>
1161std::unique_ptr<Network> MakeBlasNetwork(const std::optional<WeightsFile>& w,
1162 const OptionsDict& options) {
1163 if (!w) {
1164 throw Exception("The " + std::string(use_eigen ? "eigen" : "blas") +
1165 " backend requires a network file.");
1166 }
1167 const WeightsFile& weights = *w;
1168 auto nf = weights.format().network_format();
1169 using NF = pblczero::NetworkFormat;
1170 switch (nf.network()) {
1171 case NF::NETWORK_CLASSICAL_WITH_HEADFORMAT:
1172 case NF::NETWORK_SE_WITH_HEADFORMAT:
1173 case NF::NETWORK_ATTENTIONBODY_WITH_HEADFORMAT:
1174 case NF::NETWORK_ATTENTIONBODY_WITH_MULTIHEADFORMAT:
1175 break;
1176 default:
1177 throw Exception("Network format " +
1178 NF::NetworkStructure_Name(nf.network()) +
1179 " is not supported by the BLAS backend.");
1180 }
1181 switch (nf.policy()) {
1182 case NF::POLICY_CLASSICAL:
1183 case NF::POLICY_CONVOLUTION:
1184 case NF::POLICY_ATTENTION:
1185 break;
1186 default:
1187 throw Exception("Policy format " + NF::PolicyFormat_Name(nf.policy()) +
1188 " is not supported by the BLAS backend.");
1189 }
1190 switch (nf.value()) {
1191 case NF::VALUE_CLASSICAL:
1192 case NF::VALUE_WDL:
1193 break;
1194 default:
1195 throw Exception("Value format " + NF::ValueFormat_Name(nf.value()) +
1196 " is not supported by the BLAS backend.");
1197 }
1198 switch (nf.moves_left()) {
1199 case NF::MOVES_LEFT_NONE:
1200 case NF::MOVES_LEFT_V1:
1201 break;
1202 default:
1203 throw Exception("Moves left head format " +
1204 NF::MovesLeftFormat_Name(nf.moves_left()) +
1205 " is not supported by the BLAS backend.");
1206 }
1207 switch (nf.default_activation()) {
1208 case NF::DEFAULT_ACTIVATION_RELU:
1209 case NF::DEFAULT_ACTIVATION_MISH:
1210 break;
1211 default:
1212 throw Exception("Default activation " +
1213 NF::DefaultActivation_Name(nf.default_activation()) +
1214 " is not supported by the BLAS backend.");
1215 }
1216 switch (nf.input_embedding()) {
1217 case NF::INPUT_EMBEDDING_NONE:
1218 case NF::INPUT_EMBEDDING_PE_MAP:
1219 case NF::INPUT_EMBEDDING_PE_DENSE:
1220 break;
1221 default:
1222 throw Exception("Input embedding " +
1223 NF::InputEmbeddingFormat_Name(nf.input_embedding()) +
1224 " is not supported by the BLAS backend.");
1225 }
1226 return std::make_unique<BlasNetwork<use_eigen>>(weights, options);
1227}
