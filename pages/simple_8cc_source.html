38class SimpleTimeManager : public TimeManager {

40 SimpleTimeManager(int64_t move_overhead, const OptionsDict& params)
41 : move_overhead_(move_overhead),
42 base_pct_(params.GetOrDefault<float>("base-pct", 1.4f)),
43 ply_pct_(params.GetOrDefault<float>("ply-pct", 0.049f)),
44 time_factor_(params.GetOrDefault<float>("time-factor", 1.5f)),
45 opening_bonus_pct_(
46 params.GetOrDefault<float>("opening-bonus-pct", 0.0f)) {
47 if (base_pct_ <= 0.0f || base_pct_ > 100.0f) {
48 throw Exception("base-pct value to be in range [0.0, 100.0]");
49 }
50 if (ply_pct_ < 0.0f || ply_pct_ > 1.0f) {
51 throw Exception("ply-pct value to be in range [0.0, 1.0]");
52 }
53 if (time_factor_ < 0.0f || time_factor_ > 100.0f) {
54 throw Exception("time-factor value to be in range [0.0, 100.0]");
55 }
56 if (opening_bonus_pct_ < 0.0f || opening_bonus_pct_ > 1000.0f) {
57 throw Exception("opening-bonus-pct value to be in range [0.0, 1000.0]");
58 }
59 }

77std::unique_ptr<SearchStopper> SimpleTimeManager::GetStopper(
78 const GoParams& params, const Position& position, size_t /*total_memory*/,
79 size_t /*avg_node_size*/, uint32_t /*nodes*/) {
80 const bool is_black = position.IsBlackToMove();
81 const std::optional<int64_t>& time = (is_black ? params.btime : params.wtime);
82
83 // If no time limit is given, don't stop on this condition.
84 if (params.infinite || params.ponder || !time) return nullptr;
85
86 const std::optional<int64_t>& inc = is_black ? params.binc : params.winc;
87 const int increment = inc ? std::max(int64_t(0), *inc) : 0;
88
89 const float time_available =
90 static_cast<float>(*time) - static_cast<float>(move_overhead_);
91
92 const float time_ratio =
93 static_cast<float>(increment) / static_cast<float>(*time);
94
95 // increase percentage as ply count increases
96 float pct = (base_pct_ + position.GetGamePly() * ply_pct_) * 0.01f;
97
98 // increase percentage as ratio of increment to total time reaches equality
99 pct += time_ratio * time_factor_;
100
101 float time_budgeted = time_available * pct;
102
103 // apply any opening bonus and note the next move will also benefit
104 // from an increased time_saved as a result
105 if (!bonus_applied_) {
106 time_budgeted += time_budgeted * opening_bonus_pct_ * 0.01f;
107 bonus_applied_ = true;
108 }
109
110 // immediately spend time saved from smart pruning during previous move
111 if (prev_time_budgeted_ > 0.0f) {
112 const float time_saved = prev_time_budgeted_ -
113 (prev_time_available_ -
114 (time_available - static_cast<float>(increment)));
115
116 time_budgeted += std::max(time_saved, 0.0f);
117 }
118
119 time_budgeted = std::min(time_budgeted, time_available);
120
121 LOGFILE << "Budgeted time for the move: " << time_budgeted << "ms "
122 << "Remaining time " << *time << "ms(-" << move_overhead_
123 << "ms overhead)";
124
125 prev_time_budgeted_ = time_budgeted;
126 prev_time_available_ = time_available;
127
128 return std::make_unique<TimeLimitStopper>(time_budgeted);
129}
