72struct InputsOutputs {
73 InputsOutputs(OnnxNetwork* network);
74 ~InputsOutputs() {
75 switch (provider_) {
76 case OnnxProvider::CUDA:
77 case OnnxProvider::TRT:
78#ifdef USE_ONNX_CUDART
79 ReportCUDAErrors(cudaEventDestroy(inputs_uploaded_event_));
80 ReportCUDAErrors(cudaEventDestroy(inputs_processed_event_));
81 ReportCUDAErrors(cudaEventDestroy(evaluation_done_event_));
82 ReportCUDAErrors(cudaEventDestroy(outputs_download_event_));
83 ReportCUDAErrors(cudaFree(input_tensor_upload_device_));
84 ReportCUDAErrors(cudaFree(input_tensor_data_device_));
85 for (void* ptr : output_tensors_data_device_) {
86 ReportCUDAErrors(cudaFree(ptr));
87 }
88 ReportCUDAErrors(cudaFreeHost(input_tensor_data_));
89 for (void* ptr : output_tensors_data_) {
90 ReportCUDAErrors(cudaFreeHost(ptr));
91 }
92 break;
93#endif
94 default:
95 free(input_tensor_data_);
96 for (void* ptr : output_tensors_data_) {
97 free(ptr);
98 }
99 }
100 }

102 void* input_tensor_data_;
103 void* input_tensor_upload_device_;
104 void* input_tensor_data_device_;
105 std::vector<void*> output_tensors_data_;
106 std::vector<void*> output_tensors_data_device_;
107 std::vector<size_t> output_tensors_step_;

109 std::vector<float> wdl_output_data_;
110 Ort::MemoryInfo memory_info_{nullptr};
111#ifdef USE_ONNX_CUDART
112 cudaEvent_t inputs_uploaded_event_ = nullptr;

236InputsOutputs::InputsOutputs(OnnxNetwork* network)
237 : provider_(network->provider_) {
238 int max_batch_size = network->max_batch_size_;
239 int value_head = network->value_head_;
240 int wdl_head = network->wdl_head_;
241 int policy_head = network->policy_head_;
242 int mlh_head = network->mlh_head_;
243 int data_size = (network->fp16_ | network->bf16_) ? 2 : 4;
244 int outputs_size =
245 std::max({value_head, wdl_head, policy_head, mlh_head}) + 1;
246 output_tensors_data_.resize(outputs_size);
247 output_tensors_data_device_.resize(outputs_size);
248 output_tensors_step_.resize(outputs_size);
249 if (wdl_head != -1) {
250 wdl_output_data_.resize(3 * max_batch_size);
251 }
252 output_tensors_step_[policy_head] = kNumOutputPolicy;
253 if (wdl_head != -1) {
254 output_tensors_step_[wdl_head] = 3;
255 }
256 if (value_head != -1) {
257 output_tensors_step_[value_head] = 1;
258 }
259 if (mlh_head != -1) {
260 output_tensors_step_[mlh_head] = 1;
261 }
262
263 switch (provider_) {
264 case OnnxProvider::CUDA:
265 case OnnxProvider::TRT:
266#ifdef USE_ONNX_CUDART
267 ReportCUDAErrors(
268 cudaEventCreate(&inputs_processed_event_, cudaEventDisableTiming));
269 ReportCUDAErrors(
270 cudaEventCreate(&inputs_uploaded_event_, cudaEventDisableTiming));
271 ReportCUDAErrors(
272 cudaEventCreate(&evaluation_done_event_, cudaEventDisableTiming));
273 ReportCUDAErrors(
274 cudaEventCreate(&outputs_download_event_, cudaEventDisableTiming));
275 ReportCUDAErrors(
276 cudaHostAlloc(&input_tensor_data_,
277 max_batch_size * kInputPlanes * sizeof(InputPlane), 0));
278 for (int i = 0; i < outputs_size; i++) {
279 ReportCUDAErrors(cudaHostAlloc(
280 &output_tensors_data_[i],
281 max_batch_size * output_tensors_step_[i] * data_size, 0));
282 }
283
284 output_tensors_data_device_.resize(outputs_size);
285 ReportCUDAErrors(
286 cudaMalloc(&input_tensor_upload_device_,
287 max_batch_size * kInputPlanes * sizeof(InputPlane)));
288 ReportCUDAErrors(
289 cudaMalloc(&input_tensor_data_device_,
290 max_batch_size * kInputPlanes * 8 * 8 * data_size));
291 for (int i = 0; i < outputs_size; i++) {
292 ReportCUDAErrors(
293 cudaMalloc(&output_tensors_data_device_[i],
294 max_batch_size * output_tensors_step_[i] * data_size));
295 }
296 memory_info_ = Ort::MemoryInfo{"Cuda", OrtDeviceAllocator, network->gpu_,
297 OrtMemTypeDefault};
298 break;
299#endif
300 default:
301 input_tensor_data_ =
302 malloc(max_batch_size * kInputPlanes * 8 * 8 * data_size);
303 for (int i = 0; i < outputs_size; i++) {
304 output_tensors_data_[i] =
305 malloc(max_batch_size * output_tensors_step_[i] * data_size);
306 }
307 input_tensor_data_device_ = input_tensor_data_;
308 for (int i = 0; i < outputs_size; i++) {
309 output_tensors_data_device_[i] = output_tensors_data_[i];
310 }
311 memory_info_ =
312 Ort::MemoryInfo::CreateCpu(OrtDeviceAllocator, OrtMemTypeDefault);
313 }
314}

347template <typename DataType>
348void OnnxComputation<DataType>::AddInput(InputPlanes&& input) {
349 if (input_size_ >= network_->max_batch_size_) {
350 throw Exception("NN input exceeds max batch size of " +
351 std::to_string(network_->max_batch_size_) + ".");
352 }
353#ifdef USE_ONNX_CUDART
354 if (network_->provider_ == OnnxProvider::CUDA ||
355 network_->provider_ == OnnxProvider::TRT) {
356 assert(input.size() == kInputPlanes);
357 uint64_t* masks =
358 static_cast<uint64_t*>(inputs_outputs_->input_tensor_data_) +
359 input_size_ * kInputPlanes;
360 uint64_t* mask_end =
361 static_cast<uint64_t*>(inputs_outputs_->input_tensor_data_) +
362 network_->max_batch_size_ * kInputPlanes;
363 DataType* values =
364 reinterpret_cast<DataType*>(mask_end) + input_size_ * kInputPlanes;
365 for (size_t i = 0; i < kInputPlanes; i++) {
366 masks[i] = input[i].mask;
367 DataType value;
368 AsDataType(input[i].value, &value);
369 values[i] = value;
370 }
371 input_size_++;
372 if (input_size_ > network_->max_batch_size_) {
373 throw Exception("NN input exceeds max batch size of " +
374 std::to_string(network_->max_batch_size_) + ".");
375 }
376 return;
377 }
378#endif
379 raw_input_.emplace_back(std::move(input));
380 input_size_++;
381}

399template <typename DataType>
400float OnnxComputation<DataType>::GetQVal(int sample) const {
401 if (network_->wdl_head_ != -1) {
402 return inputs_outputs_->wdl_output_data_[sample * 3 + 0] -
403 inputs_outputs_->wdl_output_data_[sample * 3 + 2];
404 } else {
405 DataType* data = static_cast<DataType*>(
406 inputs_outputs_->output_tensors_data_[network_->value_head_]);
407 return AsFloat(data[sample]);
408 }
409}

411template <typename DataType>
412float OnnxComputation<DataType>::GetDVal(int sample) const {
413 if (network_->wdl_head_ == -1) return 0.0f;
414 return inputs_outputs_->wdl_output_data_[sample * 3 + 1];
415}

417template <typename DataType>
418float OnnxComputation<DataType>::GetPVal(int sample, int move_id) const {
419 DataType* data = static_cast<DataType*>(
420 inputs_outputs_->output_tensors_data_[network_->policy_head_]);
421 return AsFloat(data[sample * kNumOutputPolicy + move_id]);

424template <typename DataType>
425float OnnxComputation<DataType>::GetMVal(int sample) const {
426 if (network_->mlh_head_ == -1) return 0.0f;
427 DataType* data = static_cast<DataType*>(
428 inputs_outputs_->output_tensors_data_[network_->mlh_head_]);
429 return AsFloat(data[sample]);
430}

432template <typename DataType>
433Ort::IoBinding OnnxComputation<DataType>::PrepareInputs(int start,
434 int batch_size,
435 int step) {
436#ifdef USE_ONNX_CUDART
437 if (network_->provider_ != OnnxProvider::CUDA &&
438 network_->provider_ != OnnxProvider::TRT)
439#endif
440 {
441 DataType* iter =
442 static_cast<DataType*>(inputs_outputs_->input_tensor_data_);
443 iter += start * kInputPlanes * 8 * 8;
444 std::memset(static_cast<void*>(iter), 0,
445 batch_size * kInputPlanes * 8 * 8 * sizeof(DataType));
446 int end = std::min(start + batch_size, static_cast<int>(input_size_));
447 for (int i = start; i < end; i++) {
448 for (const auto& plane : raw_input_[i]) {
449 DataType value;
450 AsDataType(plane.value, &value);
451 for (auto bit : IterateBits(plane.mask)) {
452 *(iter + bit) = value;
453 }
454 iter += 64;
455 }
456 }
457 }
458
459 Ort::IoBinding binding{network_->session_[step - 1]};
460 for (size_t i = 0; i < inputs_outputs_->output_tensors_step_.size(); i++) {
461 int size = inputs_outputs_->output_tensors_step_[i];
462 int64_t dims[] = {batch_size, size};
463 binding.BindOutput(
464 network_->outputs_[i].c_str(),
465 Ort::Value::CreateTensor<DataType>(
466 inputs_outputs_->memory_info_,
467 static_cast<DataType*>(
468 inputs_outputs_->output_tensors_data_device_[i]) +
469 start * size,
470 size * batch_size, dims, 2));
471 }
472
473 int64_t dims[] = {batch_size, kInputPlanes, 8, 8};
474 binding.BindInput(
475 network_->inputs_[0].c_str(),
476 Ort::Value::CreateTensor<DataType>(
477 inputs_outputs_->memory_info_,
478 static_cast<DataType*>(inputs_outputs_->input_tensor_data_device_) +
479 start * kInputPlanes * 8 * 8,
480 batch_size * kInputPlanes * 8 * 8, dims, 4));
481 return binding;
482}

484template <typename DataType>
485void OnnxComputation<DataType>::ComputeBlocking() {
486 LCTRACE_FUNCTION_SCOPE;
487 int batch_size = network_->batch_size_;
488 if (batch_size < 0) {
489 batch_size =
490 std::max(static_cast<int>(input_size_), network_->min_batch_size_);
491 }
492 // Only the DML onnxruntime execution provider is documented as needing
493 // locking, but it seems all GPU backends need it.
494 if (network_->provider_ != OnnxProvider::CPU) {
495 network_->lock_.lock();
496 }
497 for (size_t i = 0; i < (size_t)input_size_;) {
498 int step = (input_size_ - i + batch_size - 1) / batch_size;
499 if (step > network_->steps_) step = network_->steps_;
500 int batch = batch_size * step;
501 if (network_->provider_ == OnnxProvider::TRT && network_->batch_size_ > 0) {
502 batch = std::min((int)input_size_ - (int)i, batch);
503 }
504
505 auto binding = PrepareInputs(i, batch, step);
506
507 Ort::RunOptions options = {};
508#ifdef USE_ONNX_CUDART
509 if (network_->provider_ == OnnxProvider::TRT ||
510 network_->provider_ == OnnxProvider::CUDA) {
511 if (i == 0) {
512 ReportCUDAErrors(
513 cudaStreamWaitEvent(network_->upload_stream_,
514 inputs_outputs_->inputs_processed_event_));
515 }
516 const char* src_masks =
517 static_cast<char*>(inputs_outputs_->input_tensor_data_);
518 char* dst_masks =
519 static_cast<char*>(inputs_outputs_->input_tensor_upload_device_);
520 src_masks += i * kInputPlanes * sizeof(uint64_t);
521 dst_masks += i * kInputPlanes * (sizeof(uint64_t) + sizeof(DataType));
522 ReportCUDAErrors(cudaMemcpyAsync(
523 dst_masks, src_masks, batch * kInputPlanes * sizeof(uint64_t),
524 cudaMemcpyHostToDevice, network_->upload_stream_));
525 char* src_values =
526 static_cast<char*>(inputs_outputs_->input_tensor_data_);
527 src_values += network_->max_batch_size_ * kInputPlanes * sizeof(uint64_t);
528 src_values += i * kInputPlanes * sizeof(DataType);
529 char* dst_values = dst_masks + batch * kInputPlanes * sizeof(uint64_t);
530 ReportCUDAErrors(cudaMemcpyAsync(
531 dst_values, src_values, batch * kInputPlanes * sizeof(DataType),
532 cudaMemcpyHostToDevice, network_->upload_stream_));
533 ReportCUDAErrors(cudaEventRecord(inputs_outputs_->inputs_uploaded_event_,
534 network_->upload_stream_));
535 ReportCUDAErrors(cudaStreamWaitEvent(
536 network_->compute_stream_, inputs_outputs_->inputs_uploaded_event_));
537 if (network_->fp16_) {
538 half* dst =
539 reinterpret_cast<half*>(inputs_outputs_->input_tensor_data_device_);
540 dst += i * kInputPlanes * 8 * 8;
541 expandPlanesOnnx(dst, dst_masks, batch * kInputPlanes,
542 network_->compute_stream_);
543 } else if (network_->bf16_) {
544 __nv_bfloat16* dst = reinterpret_cast<__nv_bfloat16*>(
545 inputs_outputs_->input_tensor_data_device_);
546 dst += i * kInputPlanes * 8 * 8;
547 expandPlanesOnnx(dst, dst_masks, batch * kInputPlanes,
548 network_->compute_stream_);
549 } else {
550 float* dst = reinterpret_cast<float*>(
551 inputs_outputs_->input_tensor_data_device_);
552 dst += i * kInputPlanes * 8 * 8;
553 expandPlanesOnnx(dst, dst_masks, batch * kInputPlanes,
554 network_->compute_stream_);
555 }
556
557 ReportCUDAErrors(cudaEventRecord(inputs_outputs_->inputs_processed_event_,
558 network_->upload_stream_));
559 if (i == 0) {
560 ReportCUDAErrors(
561 cudaStreamWaitEvent(network_->compute_stream_,
562 inputs_outputs_->outputs_download_event_));
563 }
564 options.AddConfigEntry("disable_synchronize_execution_providers", "1");
565 } else
566#endif
567 {
568 binding.SynchronizeInputs();
569 }
570 network_->session_[step - 1].Run(options, binding);
571#ifdef USE_ONNX_CUDART
572 if (network_->provider_ == OnnxProvider::TRT ||
573 network_->provider_ == OnnxProvider::CUDA) {
574 for (size_t j = 0; j < inputs_outputs_->output_tensors_step_.size();
575 j++) {
576 ReportCUDAErrors(
577 cudaEventRecord(inputs_outputs_->evaluation_done_event_,
578 network_->compute_stream_));
579 ReportCUDAErrors(
580 cudaStreamWaitEvent(network_->download_stream_,
581 inputs_outputs_->evaluation_done_event_));
582 size_t offset = i * inputs_outputs_->output_tensors_step_[j];
583 ReportCUDAErrors(cudaMemcpyAsync(
584 static_cast<DataType*>(inputs_outputs_->output_tensors_data_[j]) +
585 offset,
586 static_cast<DataType*>(
587 inputs_outputs_->output_tensors_data_device_[j]) +
588 offset,
589 batch * inputs_outputs_->output_tensors_step_[j] * sizeof(DataType),
590 cudaMemcpyDeviceToHost, network_->download_stream_));
591 ReportCUDAErrors(
592 cudaEventRecord(inputs_outputs_->outputs_download_event_,
593 network_->download_stream_));
594 }
595 } else
596#endif
597 {
598 binding.SynchronizeOutputs();
599 }
600 i += batch;
601 }
602 if (network_->provider_ != OnnxProvider::CPU) {
603 network_->lock_.unlock();
604 }
605#ifdef USE_ONNX_CUDART
606 if (network_->provider_ == OnnxProvider::TRT ||
607 network_->provider_ == OnnxProvider::CUDA) {
608 ReportCUDAErrors(
609 cudaEventSynchronize(inputs_outputs_->outputs_download_event_));
610 }
611#endif
612 if (network_->wdl_head_ != -1) {
613 const DataType* data = static_cast<DataType*>(
614 inputs_outputs_->output_tensors_data_[network_->wdl_head_]);
615 for (size_t i = 0; i < input_size_; i++) {
616 float w = AsFloat(data[i * 3 + 0]);
617 float d = AsFloat(data[i * 3 + 1]);
618 float l = AsFloat(data[i * 3 + 2]);
619 if (network_->cpu_wdl_) {
620 // Value softmax done cpu side.
621 float m = std::max({w, d, l});
622 w = std::exp(w - m);
623 d = std::exp(d - m);
624 l = std::exp(l - m);
625 float sum = w + d + l;
626 w /= sum;
627 l /= sum;
628 d /= sum;
629 }
630 inputs_outputs_->wdl_output_data_[3 * i + 0] = w;
631 inputs_outputs_->wdl_output_data_[3 * i + 1] = d;
632 inputs_outputs_->wdl_output_data_[3 * i + 2] = l;
633 }
634 }
635}

637Ort::SessionOptions OnnxNetwork::GetOptions(int threads, int batch_size,
638 uint64_t hash, int optimize) {
639 Ort::SessionOptions options;
640 options.SetIntraOpNumThreads(threads);
641 GraphOptimizationLevel level = GraphOptimizationLevel::ORT_DISABLE_ALL;
642 switch (optimize) {
643 case 0:
644 level = GraphOptimizationLevel::ORT_DISABLE_ALL;
645 break;
646 case 1:
647 level = GraphOptimizationLevel::ORT_ENABLE_BASIC;
648 break;
649 case 2:
650 level = GraphOptimizationLevel::ORT_ENABLE_EXTENDED;
651 break;
652 default:
653 level = GraphOptimizationLevel::ORT_ENABLE_ALL;
654 break;
655 }
656 options.SetGraphOptimizationLevel(level);
657

659 // Override the default (variable) batch size.
660 Ort::ThrowOnError(
661 OrtGetApiBase()
662 ->GetApi(ORT_API_VERSION)
663 ->AddFreeDimensionOverrideByName(options, "batch", batch_size));
664 }
665
666 switch (provider_) {
667 case OnnxProvider::DML: {
668 std::unordered_map<std::string, std::string> dml_options;
669 dml_options["device_id"] = std::to_string(gpu_);
670 dml_options["performance_preference"] = "high_performance";
671 options.AppendExecutionProvider("DML", dml_options);
672 break;
673 }
674 case OnnxProvider::TRT: {
675 options.SetExecutionMode(ExecutionMode::ORT_SEQUENTIAL);
676
677 std::string cache_dir = CommandLine::BinaryDirectory() + "/trt_cache";
678 std::map<std::string, std::string> trt_options;
679 trt_options["device_id"] = std::to_string(gpu_);
680 trt_options["trt_builder_optimization_level"] =
681 std::to_string(std::clamp(optimize, 0, 5));
682 trt_options["trt_fp16_enable"] = optimize >= 6 ? "1" : "0";
683#if ORT_API_VERSION >= 23
684 trt_options["trt_bf16_enable"] = optimize >= 7 ? "1" : "0";
685#endif
686 trt_options["trt_int8_enable"] = optimize >= 8 ? "1" : "0";
687 trt_options["trt_max_partition_iterations"] = "1000";
688 trt_options["trt_min_subgraph_size"] = "1";
689 trt_options["trt_engine_cache_enable"] = "1";
690 // We need the batch size as well as the hash, as it is set after loading.
691 std::ostringstream oss;
692 oss << std::hex << hash;
693 trt_options["trt_engine_cache_prefix"] =
694 "Lc0_ONNX_TRT_ORT_" + Ort::GetVersionString() + "_batch_" +
695 (batch_size < 0 ? std::to_string(batch_size)
696 : std::to_string(batch_size - batch_size_ + 1) + "-" +
697 std::to_string(batch_size)) +
698 "_" + std::to_string(optimize) + "_" + oss.str() + "_";
699 trt_options["trt_engine_cache_path"] = cache_dir;
700 trt_options["trt_timing_cache_enable"] = "1";
701 trt_options["trt_timing_cache_path"] = cache_dir;
702 trt_options["trt_layer_norm_fp32_fallback"] = "1";
703 trt_options["trt_force_sequential_engine_build"] = "1";
704 trt_options["trt_context_memory_sharing_enable"] = "1";
705 // Looks like we need I/O binding to enable this.
706#ifdef USE_ONNX_CUDART
707 trt_options["has_user_compute_stream"] = "1";
708#endif
709 if (batch_size < 0) {
710 trt_options["trt_profile_min_shapes"] =
711 inputs_[0] + ":" + std::to_string(min_batch_size_) + "x112x8x8";
712 trt_options["trt_profile_max_shapes"] =
713 inputs_[0] + ":" + std::to_string(max_batch_size_) + "x112x8x8";
714 trt_options["trt_profile_opt_shapes"] =
715 inputs_[0] + ":" + std::to_string(max_batch_size_ / 4) + "x112x8x8";
716 } else {
717 trt_options["trt_profile_min_shapes"] =
718 inputs_[0] + ":" + std::to_string(batch_size - batch_size_ + 1) +
719 "x112x8x8";
720 trt_options["trt_profile_max_shapes"] =
721 inputs_[0] + ":" + std::to_string(batch_size) + "x112x8x8";
722 trt_options["trt_profile_opt_shapes"] =
723 inputs_[0] + ":" + std::to_string(batch_size) + "x112x8x8";
724 }
725 std::vector<const char*> keys;
726 std::vector<const char*> values;
727 for (const auto& [key, value] : trt_options) {
728 keys.push_back(key.c_str());
729 values.push_back(value.c_str());
730 }
731
732 const auto& api = Ort::GetApi();

735 Ort::ThrowOnError(api.UpdateTensorRTProviderOptions(
736 trt_options_v2, keys.data(), values.data(), keys.size()));
737#ifdef USE_ONNX_CUDART
738 Ort::ThrowOnError(api.UpdateTensorRTProviderOptionsWithValue(
739 trt_options_v2, "user_compute_stream", compute_stream_));
740#endif
741 options.AppendExecutionProvider_TensorRT_V2(*trt_options_v2);
742 api.ReleaseTensorRTProviderOptions(trt_options_v2);
743 break;
744 }
745 case OnnxProvider::ROCM: {
746 OrtROCMProviderOptions rocm_options;
747 rocm_options.device_id = gpu_;
748 options.AppendExecutionProvider_ROCM(rocm_options);
749 break;
750 }
751 case OnnxProvider::MIGRAPHX: {
752 std::unordered_map<std::string, std::string> migraphx_options;
753 migraphx_options["device_id"] = std::to_string(gpu_);
754 migraphx_options["migraphx_exhaustive_tune"] = optimize >= 5 ? "1" : "0";
755 migraphx_options["migraphx_fp16_enable"] = optimize >= 6 ? "1" : "0";
756 migraphx_options["migraphx_bf16_enable"] = optimize >= 7 ? "1" : "0";

762 std::filesystem::create_directories(cache_dir);

764 migraphx_options["migraphx_model_cache_dir"] = cache_dir.string();

766 options.AppendExecutionProvider("MIGraphX", migraphx_options);
767 break;
768 }
769 case OnnxProvider::CUDA: {
770 OrtCUDAProviderOptions cuda_options;
771 cuda_options.device_id = gpu_;
772#ifdef USE_ONNX_CUDART
773 cuda_options.has_user_compute_stream = true;
774 cuda_options.user_compute_stream = compute_stream_;
775#endif
776 options.AppendExecutionProvider_CUDA(cuda_options);
777 break;
778 }
779 case OnnxProvider::CPU:
780 // The CPU execution provider is always available.
781 break;
782 }
783 return options;

786OnnxNetwork::OnnxNetwork(const WeightsFile& file, const OptionsDict& opts,
787 OnnxProvider provider, bool cpu_wdl)
788 : onnx_env_(ORT_LOGGING_LEVEL_WARNING, "lc0"),
789 capabilities_{file.format().network_format().input(),
790 file.format().network_format().output(),
791 file.format().network_format().moves_left()},
792 fp16_(file.onnx_model().data_type() == pblczero::OnnxModel::FLOAT16),
793 bf16_(file.onnx_model().data_type() == pblczero::OnnxModel::BFLOAT16),
794 cpu_wdl_(cpu_wdl),
795 provider_(provider) {
796 onnx_env_.DisableTelemetryEvents();
797
798 gpu_ = opts.GetOrDefault<int>("gpu", 0);
799
800#ifdef USE_ONNX_CUDART
801 if (provider_ == OnnxProvider::CUDA || provider_ == OnnxProvider::TRT) {
802 cudaDeviceProp deviceProp = {};
803 if (!cudaGetDeviceProperties(&deviceProp, gpu_)) {
804 CERR << "GPU: " << deviceProp.name;
805 CERR << "GPU memory: " << deviceProp.totalGlobalMem / std::pow(2.0f, 30)
806 << " Gb";
807 int clockRate = 0;
808 ReportCUDAErrors(
809 cudaDeviceGetAttribute(&clockRate, cudaDevAttrClockRate, gpu_));
810 CERR << "GPU clock frequency: " << clockRate / 1e3f << " MHz";
811 }
812#if CUDART_VERSION >= 12080
813 int runtime_version;
814 ReportCUDAErrors(cudaRuntimeGetVersion(&runtime_version));
815 if (runtime_version >= 12080) {
816 int attr;
817 ReportCUDAErrors(
818 cudaDeviceGetAttribute(&attr, cudaDevAttrGpuPciDeviceId, gpu_));
819 uint32_t pci_device = attr;
820 CERR << "GPU device ID: " << std::hex << (pci_device & 0xffff) << ":"
821 << (pci_device >> 16);
822 ReportCUDAErrors(
823 cudaDeviceGetAttribute(&attr, cudaDevAttrGpuPciSubsystemId, gpu_));
824 uint32_t pci_subsystem = attr;
825 CERR << "GPU subsystem ID: " << std::hex << (pci_subsystem & 0xffff)
826 << ":" << (pci_subsystem >> 16) << std::dec;
827 }
828#endif
829 }
830#endif
831
832 int threads =
833 opts.GetOrDefault<int>("threads", provider == OnnxProvider::CPU ? 1 : 0);
834 int default_batch = -1;
835 int default_steps = 1;
836 int default_min_batch = 1;
837 switch (provider) {
838 case OnnxProvider::DML:
839 case OnnxProvider::MIGRAPHX:
840 default_batch = 16;
841 default_steps = 4;
842 break;
843 case OnnxProvider::TRT:
844 default_min_batch = 4;
845 default:
846 break;
847 }
848
849 int optimize = opts.GetOrDefault<int>("optimize", 3);
850 batch_size_ = opts.GetOrDefault<int>("batch", default_batch);
851 steps_ = opts.GetOrDefault<int>("steps", default_steps);
852 min_batch_size_ = opts.GetOrDefault<int>("min_batch", default_min_batch);
853
854 // Sanity checks.
855 if (batch_size_ <= 0) {
856 batch_size_ = -1; // Variable batch size.
857 steps_ = 1;
858 }
859 if (batch_size_ * steps_ > max_batch_size_) {
860 batch_size_ = max_batch_size_ / steps_;
861 }
862
863 const auto& md = file.onnx_model();
864 if (!md.has_input_planes()) {
865 throw Exception("NN doesn't have input planes defined.");
866 }
867 inputs_.emplace_back(md.input_planes());
868 if (!md.has_output_policy()) {
869 throw Exception("NN doesn't have policy head defined.");
870 }
871 policy_head_ = outputs_.size();
872 outputs_.emplace_back(md.output_policy());
873 if (md.has_output_wdl()) {
874 wdl_head_ = outputs_.size();
875 outputs_.emplace_back(md.output_wdl());
876 } else if (md.has_output_value()) {
877 value_head_ = outputs_.size();
878 outputs_.emplace_back(md.output_value());
879 } else {
880 throw Exception("NN doesn't have value head.");
881 }
882 if (md.has_output_mlh()) {
883 mlh_head_ = outputs_.size();
884 outputs_.emplace_back(md.output_mlh());
885 }
886 uint64_t hash = 0;
887 if (provider == OnnxProvider::TRT) {
888 hash = std::hash<std::string_view>()(md.model());
889 }
890 switch (provider) {
891 case OnnxProvider::TRT:
892 case OnnxProvider::CUDA:
893#ifdef USE_ONNX_CUDART
894 ReportCUDAErrors(cudaSetDevice(gpu_));
895 ReportCUDAErrors(cudaStreamCreate(&compute_stream_));
896 ReportCUDAErrors(cudaStreamCreate(&upload_stream_));
897 ReportCUDAErrors(cudaStreamCreate(&download_stream_));
898#else
899 CERR << "WARNING: Simplified version without CUDA enhancements.";
900#endif
901 break;
902 default:
903 break;
904 }
905
906 for (int step = 1; step <= steps_; step++)
907 session_.emplace_back(
908 onnx_env_, file.onnx_model().model().data(),
909 file.onnx_model().model().size(),
910 GetOptions(threads, batch_size_ * step, hash, optimize));
911}

913template <OnnxProvider kProvider>
914std::unique_ptr<Network> MakeOnnxNetwork(const std::optional<WeightsFile>& w,
915 const OptionsDict& opts) {
916 if (!w) throw Exception("The ONNX backend requires a network file.");
917
918 if (w->has_onnx_model()) {
919 return std::make_unique<OnnxNetwork>(*w, opts, kProvider, false);
920 } else {
921 WeightsToOnnxConverterOptions converter_options;
922 converter_options.ir = opts.GetOrDefault<int>("ir", -1);
923 converter_options.alt_mish = opts.GetOrDefault<bool>(
924 "alt_mish", kProvider == OnnxProvider::CPU ? true : false);
925 converter_options.alt_layernorm = opts.GetOrDefault<bool>(
926 "alt_layernorm",
927 kProvider == OnnxProvider::DML &&
928 w->format().network_format().ffn_activation() ==
929 pblczero::NetworkFormat::ACTIVATION_RELU_2
930 ? true
931 : false);
932 converter_options.no_shape = opts.GetOrDefault<bool>("no_shape", false);
933 converter_options.policy_head =
934 opts.GetOrDefault<std::string>("policy_head", "vanilla");
935 converter_options.value_head =
936 opts.GetOrDefault<std::string>("value_head", "winner");
937 converter_options.no_wdl_softmax = true;
938 // No execution provider has a better mish version, some don't even have it.
939 converter_options.real_mish = false;
940
941 std::string datatype;
942 if (opts.Exists<std::string>("datatype")) {
943 datatype = opts.Get<std::string>("datatype");
944 } else {
945 bool fp16 = opts.GetOrDefault<bool>(
946 "fp16", kProvider == OnnxProvider::CPU ? false : true);
947 datatype = fp16 ? "f16" : "f32";
948 }
949 converter_options.data_type =
950 WeightsToOnnxConverterOptions::StringToDataType(datatype);
951 converter_options.opset = opts.GetOrDefault<int>(
952 "opset", converter_options.data_type ==
953 WeightsToOnnxConverterOptions::DataType::kBFloat16
954 ? 22
955 : 17);
956
957 auto converted = ConvertWeightsToOnnx(*w, converter_options);
958 return std::make_unique<OnnxNetwork>(converted, opts, kProvider, true);
959 }
