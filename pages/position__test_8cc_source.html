29TEST(Position, SetFenGetFen) {
30 std::vector<Position> positions;
31 ChessBoard board;
32 std::vector<std::string> source_fens = {
33 "r3k2r/p1ppqpb1/bn2pnp1/3PN3/1p2P3/2N2Q1p/PPPBBPPP/R3K2R w KQkq - 1 1",
34 // has en_passant space e3 - black to move
35 "r3k2r/p1ppqpb1/bn2pnp1/3PN3/1p2P3/2N2Q1p/PPPBBPPP/R3K2R b KQkq e3 1 1",
36 // has en_passant space c6 - white to move
37 "rnbqkbnr/pp1ppppp/8/2p5/4P3/8/PPPP1PPP/RNBQKBNR w KQkq c6 0 2",
38 "8/2p5/3p4/KP5r/1R3p1k/8/4P1P1/8 w - - 1 1",
39 "r2q1rk1/pP1p2pp/Q4n2/bbp1p3/Np6/1B3NBn/pPPP1PPP/R3K2R b KQ - 0 1",
40 "3b4/rp1r1k2/8/1RP2p1p/p1KP4/P3P2P/5P2/1R2B3 b - - 2 30",
41 "rnbq1k1r/pp1Pbppp/2p5/8/2B5/8/PPP1NnPP/RNBQK2R w KQ - 1 8",
42 "r4rk1/1pp1qppp/p1np1n2/2b1p1B1/2B1P1b1/P1NP1N2/1PP1QPPP/R4RK1 w - - 0 "
43 "10",
44 "8/8/8/4k3/8/8/2K5/8 w - - 0 1", "8/8/8/4k3/1N6/8/2K5/8 w - - 0 1"};
45 for (size_t i = 0; i < source_fens.size(); i++) {
46 board.Clear();
47 PositionHistory history;
48 int no_capture_ply;
49 int game_move;
50 board.SetFromFen(source_fens[i], &no_capture_ply, &game_move);
51 history.Reset(board, no_capture_ply,
52 2 * game_move - (board.flipped() ? 1 : 2));
53 Position pos = history.Last();
54 std::string target_fen = PositionToFen(pos);
55 EXPECT_EQ(source_fens[i], target_fen);
56 }
57}
