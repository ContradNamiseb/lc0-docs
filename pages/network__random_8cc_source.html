44class RandomNetworkComputation : public NetworkComputation {
45 public:
50 RandomNetworkComputation(int delay, int seed, bool uniform_mode)
51 : delay_ms_(delay), seed_(seed), uniform_mode_(uniform_mode) {}
52
54 void AddInput(InputPlanes&& input) override {
55 std::uint64_t hash = seed_;
56 for (const auto& plane : input) {
57 hash = HashCat({hash, plane.mask});
58 std::uint32_t tmp;
59 std::memcpy(&tmp, &plane.value, sizeof(float));
60 const std::uint64_t value_hash = tmp;
61 hash = HashCat({hash, value_hash});
62 }
63 inputs_.push_back(hash);
64 }
65
67 void ComputeBlocking() override {
68 if (delay_ms_) {
69 std::this_thread::sleep_for(std::chrono::milliseconds(delay_ms_));
70 }
71 }
72
74 int GetBatchSize() const override { return inputs_.size(); }
75
76 float GetQVal(int sample) const override {
77 if (uniform_mode_) return 0.0f;
78 return (int(inputs_[sample] % 200000) - 100000) / 100000.0;
79 }
80
81 float GetDVal(int sample) const override {
82 if (uniform_mode_) return 0.0f;
83 // Maximum D value is 1 - abs(Q) for W, D, L to be in range [0.0, 1.0].
84 float q = GetQVal(sample);
85 float max_d = 1.0f - std::fabs(q);
86 // Hash in arbitrary constant to make D return different value from Q.
87 float d = max_d * (HashCat({inputs_[sample], 1234}) % 10000) / 10000.0;
88 return d;
89 }
90
91 float GetMVal(int /* sample */) const override { return 0.0f; }
92
93 float GetPVal(int sample, int move_id) const override {
94 if (uniform_mode_) return 1.0f;
95
96 // Note that this function returns the policy value *before* softmax.
97 // We choose a uniform distribution over [0, a], implying that the
98 // proportion between the smallest and largest policy value *after* softmax
99 // exponentiation (but before normalization) is equal to S = exp(-a).
100 // Choosing a = 3.0 leads to S = 0.05.
101 const float a = 3.0f;
102 return (HashCat({inputs_[sample], static_cast<unsigned long>(move_id)}) %
103 10000) *
104 (a / 10000.0f);
105 }
106
107 private:
108 std::vector<std::uint64_t> inputs_;
109 int delay_ms_ = 0;
110 int seed_ = 0;
111 bool uniform_mode_ = false;
112};
