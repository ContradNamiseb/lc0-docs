38// Combines multiple stoppers into one.
39class ChainedSearchStopper : public SearchStopper {
40 public:
41 ChainedSearchStopper() = default;
42 // Calls stoppers one by one until one of them returns true. If one of
43 // stoppers modifies hints, next stoppers in the chain see that.
44 bool ShouldStop(const IterationStats&, StoppersHints*) override;
45 // Can be nullptr, in that canse stopper is not added.
46 void AddStopper(std::unique_ptr<SearchStopper> stopper);
47 void OnSearchDone(const IterationStats&) override;
48
49 private:
50 std::vector<std::unique_ptr<SearchStopper>> stoppers_;
51};
