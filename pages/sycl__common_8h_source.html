63struct DeviceCapabilities {
64 public:
65 struct DeviceInfo {
66 int max_workgroup_size;
67 };
68
69 static int Init(const sycl::queue& queue) {
70 const auto& device = queue.get_device();
71 std::lock_guard<std::mutex> lock(cache_mutex_);
72
73 DeviceInfo info;
74 info.max_workgroup_size =
75 device.get_info<sycl::info::device::max_work_group_size>();
76 device_cache_[device] = info;
77
78 return info.max_workgroup_size;
79 }
80
81 static int GetMaxWorkgroupSize(const sycl::queue& queue) {
82 const auto& device = queue.get_device();
83 auto it = device_cache_.find(device);
84 if (it != device_cache_.end()) {
85 // Cache hit.
86 return it->second.max_workgroup_size;
87 }
88 // Cache miss.
89 return Init(queue);
90 }
91
92 // Clear cache if needed
93 static void ClearCache() {
94 std::lock_guard<std::mutex> lock(cache_mutex_);
95 device_cache_.clear();
96 }
97
98 private:
99 static std::unordered_map<sycl::device, DeviceInfo> device_cache_;
100 static std::mutex cache_mutex_;
101};
