269class SearchWorker {
270 public:
274 SearchWorker(Search* search, const SearchParams& params)
275 : search_(search),
276 history_(search_->played_history_),
277 params_(params),
278 moves_left_support_(search_->backend_attributes_.has_mlh) {
279 task_workers_ = params.GetTaskWorkersPerSearchWorker();
280 if (task_workers_ < 0) {
281 if (search_->backend_attributes_.runs_on_cpu) {
282 task_workers_ = 0;
283 } else {
284 int working_threads = std::max(
285 search_->thread_count_.load(std::memory_order_acquire) - 1, 1);
286 task_workers_ = std::min(
287 std::thread::hardware_concurrency() / working_threads - 1, 4U);
288 }
289 }
290 for (int i = 0; i < task_workers_; i++) {
291 task_workspaces_.emplace_back();
292 task_threads_.emplace_back([this, i]() { this->RunTasks(i); });
293 }
294 target_minibatch_size_ = params_.GetMiniBatchSize();
295 if (target_minibatch_size_ == 0) {
296 target_minibatch_size_ =
297 search_->backend_attributes_.recommended_batch_size;
298 }
299 max_out_of_order_ =
300 std::max(1, static_cast<int>(params_.GetMaxOutOfOrderEvalsFactor() *
301 target_minibatch_size_));
302 }
303
305 ~SearchWorker() {
306 {
307 task_count_.store(-1, std::memory_order_release);
308 Mutex::Lock lock(picking_tasks_mutex_);
309 exiting_ = true;
310 task_added_.notify_all();
311 }
312 for (size_t i = 0; i < task_threads_.size(); i++) {
313 task_threads_[i].join();
314 }
315 }
316
319 void RunBlocking() {
320 LOGFILE << "Started search thread.";
321 try {
322 // A very early stop may arrive before this point, so the test is at the
323 // end to ensure at least one iteration runs before exiting.
324 do {
325 ExecuteOneIteration();
326 } while (search_->IsSearchActive());
327 } catch (std::exception& e) {
328 std::cerr << "Unhandled exception in worker thread: " << e.what()
329 << std::endl;
330 abort();
331 }
332 }
