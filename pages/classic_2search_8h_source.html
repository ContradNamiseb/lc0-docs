206// Single thread worker of the search engine.
207// That used to be just a function Search::Worker(), but to parallelize it
208// within one thread, have to split into stages.
209class SearchWorker {
210 public:
211 SearchWorker(Search* search, const SearchParams& params)
212 : search_(search),
213 history_(search_->played_history_),
214 params_(params),
215 moves_left_support_(search_->backend_attributes_.has_mlh) {
216 task_workers_ = params.GetTaskWorkersPerSearchWorker();
217 if (task_workers_ < 0) {
218 if (search_->backend_attributes_.runs_on_cpu) {
219 task_workers_ = 0;
220 } else {
221 int working_threads = std::max(
222 search_->thread_count_.load(std::memory_order_acquire) - 1, 1);
223 task_workers_ = std::min(
224 std::thread::hardware_concurrency() / working_threads - 1, 4U);
225 }
226 }
227 for (int i = 0; i < task_workers_; i++) {
228 task_workspaces_.emplace_back();
229 task_threads_.emplace_back([this, i]() { this->RunTasks(i); });
230 }
231 target_minibatch_size_ = params_.GetMiniBatchSize();
232 if (target_minibatch_size_ == 0) {
233 target_minibatch_size_ =
234 search_->backend_attributes_.recommended_batch_size;
235 }
236 max_out_of_order_ =
237 std::max(1, static_cast<int>(params_.GetMaxOutOfOrderEvalsFactor() *
238 target_minibatch_size_));
239 }
240
241 ~SearchWorker() {
242 {
243 task_count_.store(-1, std::memory_order_release);
244 Mutex::Lock lock(picking_tasks_mutex_);
245 exiting_ = true;
246 task_added_.notify_all();
247 }
248 for (size_t i = 0; i < task_threads_.size(); i++) {
249 task_threads_[i].join();
250 }
251 }
252
253 // Runs iterations while needed.
254 void RunBlocking() {
255 LOGFILE << "Started search thread.";
256 try {
257 // A very early stop may arrive before this point, so the test is at the
258 // end to ensure at least one iteration runs before exiting.
259 do {
260 ExecuteOneIteration();
261 } while (search_->IsSearchActive());
262 } catch (std::exception& e) {
263 std::cerr << "Unhandled exception in worker thread: " << e.what()
264 << std::endl;
265 abort();
266 }
267 }

269 // Does one full iteration of MCTS search:
270 // 1. Initialize internal structures.
271 // 2. Gather minibatch.
272 // 3. Prefetch into cache.
273 // 4. Run NN computation.
274 // 5. Retrieve NN computations (and terminal values) into nodes.
275 // 6. Propagate the new nodes' information to all their parents in the tree.
276 // 7. Update the Search's status and progress information.
277 void ExecuteOneIteration();
278
279 // The same operations one by one:
280 // 1. Initialize internal structures.
281 // @computation is the computation to use on this iteration.
282 void InitializeIteration();
283
284 // 2. Gather minibatch.
285 void GatherMinibatch();
286
287 // 2b. Copy collisions into shared_collisions_.
288 void CollectCollisions();
289
290 // 3. Prefetch into cache.
291 void MaybePrefetchIntoCache();
292
293 // 4. Run NN computation.
294 void RunNNComputation();
295
296 // 5. Retrieve NN computations (and terminal values) into nodes.
297 void FetchMinibatchResults();
298
299 // 6. Propagate the new nodes' information to all their parents in the tree.
300 void DoBackupUpdate();
301
302 // 7. Update the Search's status and progress information.
303 void UpdateCounters();
