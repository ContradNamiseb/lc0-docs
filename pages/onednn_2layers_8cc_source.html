38BaseLayer::BaseLayer(int c, int h, int w, BaseLayer* ip)
39 : input_(ip), C(c), H(h), W(w) {
40 if (ip) {
41 data_type_ = ip->data_type_;
42 convolution_type_ = ip->convolution_type_;
43 } else {
44 data_type_ = dnnl::memory::data_type::undef;
45 convolution_type_ = dnnl::algorithm::convolution_auto;
46 }
47}

49ConvLayer::ConvLayer(BaseLayer* ip, int C, int H, int W, int filter, int Cin,
50 ActivationFunction activation, bool skip)
51 : BaseLayer(C, H, W, ip),
52 c_input_(Cin),
53 filter_size_(filter),
54 activation_(activation),
55 use_skip_(skip) {}

57void ConvLayer::LoadWeights(dnnl::memory& w1, dnnl::memory& b1,
58 dnnl::engine& eng, dnnl::stream& stream) {
59 auto filter_md =
60 dnnl::memory::desc({C, c_input_, filter_size_, filter_size_}, data_type_,
61 dnnl::memory::format_tag::oihw);
62 filter_mem = dnnl::memory(filter_md, eng);
63 dnnl::reorder(w1, filter_mem).execute(stream, w1, filter_mem);
64
65 auto bias_md =
66 dnnl::memory::desc({C}, data_type_, dnnl::memory::format_tag::a);
67 bias_mem = dnnl::memory(bias_md, eng);
68 dnnl::reorder(b1, bias_mem).execute(stream, b1, bias_mem);
69}

71void ConvLayer::Eval(int N, dnnl::memory& output, dnnl::memory& input,
72 dnnl::engine& eng, dnnl::stream& stream) {
73 std::lock_guard<std::mutex> lock(lock_);
74 if (last_batch_ != N) {
75 auto t_in_md = dnnl::memory::desc({N, c_input_, H, W}, data_type_,
76 dnnl::memory::format_tag::any);
77
78 auto t_filter_md =
79 dnnl::memory::desc({C, c_input_, filter_size_, filter_size_},
80 data_type_, dnnl::memory::format_tag::any);
81
82 auto t_out_md = dnnl::memory::desc({N, C, H, W}, data_type_,
83 dnnl::memory::format_tag::any);
84
85 const int padding = filter_size_ / 2;
86 auto conv_d = dnnl::convolution_forward::desc(
87 dnnl::prop_kind::forward_inference,
88 filter_size_ == 3 ? convolution_type_
89 : dnnl::algorithm::convolution_auto,
90 t_in_md, t_filter_md, bias_mem.get_desc(), t_out_md, {1, 1},
91 {padding, padding}, {padding, padding});
92 dnnl::post_ops conv_ops;
93 if (use_skip_) {
94 conv_ops.append_sum();
95 }
96 if (activation_ == ACTIVATION_RELU) {
97 conv_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_relu, 0.0f, 0.0f);
98 } else if (activation_ == ACTIVATION_TANH) {
99 conv_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_tanh, 0.0f, 0.0f);
100 }
101 dnnl::primitive_attr conv_attr;
102 conv_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
103 conv_attr.set_post_ops(conv_ops);
104 auto conv_pd =
105 dnnl::convolution_forward::primitive_desc(conv_d, conv_attr, eng);
106 auto scratchpad_md = conv_pd.scratchpad_desc();
107 conv_ = dnnl::convolution_forward(conv_pd);
108
109 in_md = conv_pd.src_desc();
110 out_md = conv_pd.dst_desc();
111
112 // Apparently convolution doesn't go well with mish post op.
113 if (activation_ == ACTIVATION_MISH) {
114 auto mish_d = dnnl::eltwise_forward::desc(
115 dnnl::prop_kind::forward_inference, dnnl::algorithm::eltwise_mish,
116 out_md, 0.f, 0.f);
117 dnnl::primitive_attr mish_attr;
118 mish_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
119 auto mish_pd =
120 dnnl::eltwise_forward::primitive_desc(mish_d, mish_attr, eng);
121 mish_ = dnnl::eltwise_forward(mish_pd);
122 if (scratchpad_md.get_size() < mish_pd.scratchpad_desc().get_size()) {
123 scratchpad_md = mish_pd.scratchpad_desc();
124 }
125 }
126
127 if (!conv_filter_mem ||
128 conv_pd.weights_desc() != conv_filter_mem.get_desc()) {
129 // This may be a transformation for Winograd convolution, so keep the
130 // original weights.
131 conv_filter_mem = dnnl::memory(conv_pd.weights_desc(), eng);
132 dnnl::reorder(filter_mem, conv_filter_mem)
133 .execute(stream, filter_mem, conv_filter_mem);
134 }
135
136 dnnl::primitive_attr reorder_attr;
137 reorder_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
138 auto in_reorder_pd = dnnl::reorder::primitive_desc(
139 eng, input.get_desc(), eng, in_md, reorder_attr);
140 in_reorder_ = dnnl::reorder(in_reorder_pd);
141 if (scratchpad_md.get_size() < in_reorder_pd.scratchpad_desc().get_size()) {
142 scratchpad_md = in_reorder_pd.scratchpad_desc();
143 }
144
145 if (use_skip_) {
146 auto skip_reorder_pd = dnnl::reorder::primitive_desc(
147 eng, output.get_desc(), eng, out_md, reorder_attr);
148 skip_reorder_ = dnnl::reorder(skip_reorder_pd);
149 if (scratchpad_md.get_size() <
150 skip_reorder_pd.scratchpad_desc().get_size()) {
151 scratchpad_md = skip_reorder_pd.scratchpad_desc();
152 }
153 }
154
155 scratchpad_mem = dnnl::memory(scratchpad_md, eng);
156
157 last_batch_ = N;
158 }
159
160 if (in_md != input.get_desc()) {
161 auto tmp = dnnl::memory(in_md, eng);
162 in_reorder_.execute(stream, {{DNNL_ARG_SRC, input},
163 {DNNL_ARG_DST, tmp},
164 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
165 input = tmp;
166 }
167
168 if (!output || out_md != output.get_desc()) {
169 if (use_skip_) {
170 auto tmp = dnnl::memory(out_md, eng);
171 skip_reorder_.execute(stream, {{DNNL_ARG_SRC, output},
172 {DNNL_ARG_DST, tmp},
173 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
174 output = tmp;
175 } else {
176 output = dnnl::memory(out_md, eng);
177 }
178 }
179
180 conv_.execute(stream, {{DNNL_ARG_SRC, input},
181 {DNNL_ARG_WEIGHTS, conv_filter_mem},
182 {DNNL_ARG_BIAS, bias_mem},
183 {DNNL_ARG_DST, output},
184 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
185
186 if (activation_ == ACTIVATION_MISH) {
187 mish_.execute(stream, {{DNNL_ARG_SRC, output},
188 {DNNL_ARG_DST, output},
189 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
190 }
191}

198void SELayer::LoadWeights(dnnl::memory& w1, dnnl::memory& b1, dnnl::memory& w2,
199 dnnl::memory& b2, dnnl::engine& eng,
200 dnnl::stream& stream) {
201 auto filter_md = dnnl::memory::desc({numFc1Out_, C}, data_type_,
202 dnnl::memory::format_tag::ab);
203 filter_mem = dnnl::memory(filter_md, eng);
204 dnnl::reorder(w1, filter_mem).execute(stream, w1, filter_mem);
205
206 auto bias_md =
207 dnnl::memory::desc({numFc1Out_}, data_type_, dnnl::memory::format_tag::a);
208 bias_mem = dnnl::memory(bias_md, eng);
209 dnnl::reorder(b1, bias_mem).execute(stream, b1, bias_mem);
210
211 auto filter2_md = dnnl::memory::desc({2 * C, numFc1Out_}, data_type_,
212 dnnl::memory::format_tag::ab);
213
214 filter2_mem = dnnl::memory(filter2_md, eng);
215 dnnl::reorder(w2, filter2_mem).execute(stream, w2, filter2_mem);
216
217 auto bias2_md =
218 dnnl::memory::desc({2 * C}, data_type_, dnnl::memory::format_tag::a);
219
220 bias2_mem = dnnl::memory(bias2_md, eng);
221 dnnl::reorder(b2, bias2_mem).execute(stream, b2, bias2_mem);
222}

224void SELayer::Eval(int N, dnnl::memory& output, dnnl::memory& input,
225 dnnl::engine& eng, dnnl::stream& stream) {
226 std::lock_guard<std::mutex> lock(lock_);
227 if (last_batch_ != N) {
228 // Also the broadcast input memory format for the binary primitives.
229 auto t_pool_out_md = dnnl::memory::desc({N, C, 1, 1}, data_type_,
230 dnnl::memory::format_tag::any);
231
232 // Also the output memory format for the fc2 inner products.
233 auto t_fc1_in_md =
234 dnnl::memory::desc({N, C}, data_type_, dnnl::memory::format_tag::any);
235
236 auto t_fc1_out_md = dnnl::memory::desc({N, numFc1Out_}, data_type_,
237 dnnl::memory::format_tag::any);
238
239 auto t_filter_md = dnnl::memory::desc({numFc1Out_, C}, data_type_,
240 dnnl::memory::format_tag::any);
241
242 auto t_filter2_md = dnnl::memory::desc({2 * C, numFc1Out_}, data_type_,
243 dnnl::memory::format_tag::any);
244
245 auto t_fc2_out_md = dnnl::memory::desc({N, 2 * C}, data_type_,
246 dnnl::memory::format_tag::any);
247
248 auto pooling_d = dnnl::pooling_forward::desc(
249 dnnl::prop_kind::forward_inference, dnnl::algorithm::pooling_avg,
250 input.get_desc(), t_pool_out_md, {1, 1}, {H, W}, {0, 0}, {0, 0});
251 dnnl::primitive_attr pooling_attr;
252 pooling_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
253 auto pooling_pd =
254 dnnl::pooling_forward::primitive_desc(pooling_d, pooling_attr, eng);
255 pooling_ = dnnl::pooling_forward(pooling_pd);
256 auto scratchpad_md = pooling_pd.scratchpad_desc();
257
258 // This is also the optimized memory format descriptor for the binary
259 // primitives.
260 pool_out_md = pooling_pd.dst_desc();
261
262 auto fc_d = dnnl::inner_product_forward::desc(
263 dnnl::prop_kind::forward_inference, t_fc1_in_md, t_filter_md,
264 bias_mem.get_desc(), t_fc1_out_md);
265 dnnl::post_ops fc_ops;
266 if (activation_ == ACTIVATION_RELU) {
267 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_relu, 0.0f, 0.0f);
268 } else if (activation_ == ACTIVATION_MISH) {
269 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_mish, 0.0f, 0.0f);
270 } else if (activation_ == ACTIVATION_TANH) {
271 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_tanh, 0.0f, 0.0f);
272 }
273 dnnl::primitive_attr fc_attr;
274 fc_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
275 fc_attr.set_post_ops(fc_ops);
276 auto fc_pd =
277 dnnl::inner_product_forward::primitive_desc(fc_d, fc_attr, eng);
278 fc_ = dnnl::inner_product_forward(fc_pd);
279 if (scratchpad_md.get_size() < fc_pd.scratchpad_desc().get_size()) {
280 scratchpad_md = fc_pd.scratchpad_desc();
281 }
282
283 fc1_in_md = fc_pd.src_desc().reshape({N, C, 1, 1});
284 fc1_out_md = fc_pd.dst_desc();
285 if (fc_pd.weights_desc() != filter_mem.get_desc()) {
286 auto tmp = dnnl::memory(fc_pd.weights_desc(), eng);
287 dnnl::reorder(filter_mem, tmp).execute(stream, filter_mem, tmp);
288 filter_mem = tmp;
289 }
290
291 auto fc2_d = dnnl::inner_product_forward::desc(
292 dnnl::prop_kind::forward_inference, fc1_out_md, t_filter2_md,
293 bias2_mem.get_desc(), t_fc2_out_md);
294 dnnl::primitive_attr fc2_attr;
295 fc2_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
296 auto fc2_pd =
297 dnnl::inner_product_forward::primitive_desc(fc2_d, fc2_attr, eng);
298 fc2_ = dnnl::inner_product_forward(fc2_pd);
299 if (scratchpad_md.get_size() < fc2_pd.scratchpad_desc().get_size()) {
300 scratchpad_md = fc2_pd.scratchpad_desc();
301 }
302
303 if (fc2_pd.weights_desc() != filter2_mem.get_desc()) {
304 auto tmp = dnnl::memory(fc2_pd.weights_desc(), eng);
305 dnnl::reorder(filter2_mem, tmp).execute(stream, filter2_mem, tmp);
306 filter2_mem = tmp;
307 }
308
309 fc2_out_md = fc2_pd.dst_desc();
310
311 auto sigmoid_d = dnnl::eltwise_forward::desc(
312 dnnl::prop_kind::forward_inference, dnnl::algorithm::eltwise_logistic,
313 pool_out_md, 0.f, 0.f);
314 dnnl::primitive_attr sigmoid_attr;
315 sigmoid_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
316 auto sigmoid_pd =
317 dnnl::eltwise_forward::primitive_desc(sigmoid_d, sigmoid_attr, eng);
318 sigmoid_ = dnnl::eltwise_forward(sigmoid_pd);
319 if (scratchpad_md.get_size() < sigmoid_pd.scratchpad_desc().get_size()) {
320 scratchpad_md = sigmoid_pd.scratchpad_desc();
321 }
322
323 auto mul_d =
324 dnnl::binary::desc(dnnl::algorithm::binary_mul, input.get_desc(),
325 pool_out_md, output.get_desc());
326 dnnl::post_ops mul_ops;
327 mul_ops.append_sum();
328 if (eng.get_kind() == dnnl::engine::kind::gpu) {
329 // Using binary post-ops is a gain on gpu but a huge loss on cpu.
330 mul_ops.append_binary(dnnl::algorithm::binary_add, pool_out_md);
331 if (activation_ == ACTIVATION_RELU) {
332 mul_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_relu, 0.0f, 0.0f);
333 } else if (activation_ == ACTIVATION_MISH) {
334 mul_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_mish, 0.0f, 0.0f);
335 } else if (activation_ == ACTIVATION_TANH) {
336 mul_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_tanh, 0.0f, 0.0f);
337 }
338 }
339 dnnl::primitive_attr mul_attr;
340 mul_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
341 mul_attr.set_post_ops(mul_ops);
342 auto mul_pd = dnnl::binary::primitive_desc(mul_d, mul_attr, eng);
343 mul_ = dnnl::binary(mul_pd);
344 if (scratchpad_md.get_size() < mul_pd.scratchpad_desc().get_size()) {
345 scratchpad_md = mul_pd.scratchpad_desc();
346 }
347
348 if (eng.get_kind() != dnnl::engine::kind::gpu) {
349 auto add_d =
350 dnnl::binary::desc(dnnl::algorithm::binary_add, output.get_desc(),
351 pool_out_md, output.get_desc());
352 dnnl::post_ops add_ops;
353 if (activation_ == ACTIVATION_RELU) {
354 add_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_relu, 0.0f, 0.0f);
355 } else if (activation_ == ACTIVATION_MISH) {
356 add_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_mish, 0.0f, 0.0f);
357 } else if (activation_ == ACTIVATION_TANH) {
358 add_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_tanh, 0.0f, 0.0f);
359 }
360 dnnl::primitive_attr add_attr;
361 add_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
362 add_attr.set_post_ops(add_ops);
363 auto add_pd = dnnl::binary::primitive_desc(add_d, add_attr, eng);
364 add_ = dnnl::binary(add_pd);
365 if (scratchpad_md.get_size() < add_pd.scratchpad_desc().get_size()) {
366 scratchpad_md = add_pd.scratchpad_desc();
367 }
368 }
369
370 dnnl::primitive_attr reorder_attr;
371 reorder_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
372 auto fc1_reorder_pd = dnnl::reorder::primitive_desc(
373 eng, pool_out_md, eng, fc1_in_md, reorder_attr);
374 fc1_reorder_ = dnnl::reorder(fc1_reorder_pd);
375 if (scratchpad_md.get_size() <
376 fc1_reorder_pd.scratchpad_desc().get_size()) {
377 scratchpad_md = fc1_reorder_pd.scratchpad_desc();
378 }
379
380 auto mul_reorder_pd = dnnl::reorder::primitive_desc(
381 eng, fc2_out_md.submemory_desc({N, C}, {0, 0}).reshape({N, C, 1, 1}),
382 eng, pool_out_md, reorder_attr);
383 mul_reorder_ = dnnl::reorder(mul_reorder_pd);
384 if (scratchpad_md.get_size() <
385 mul_reorder_pd.scratchpad_desc().get_size()) {
386 scratchpad_md = mul_reorder_pd.scratchpad_desc();
387 }
388
389 auto add_reorder_pd = dnnl::reorder::primitive_desc(
390 eng, fc2_out_md.submemory_desc({N, C}, {0, C}).reshape({N, C, 1, 1}),
391 eng, pool_out_md, reorder_attr);
392 add_reorder_ = dnnl::reorder(add_reorder_pd);
393 if (scratchpad_md.get_size() <
394 add_reorder_pd.scratchpad_desc().get_size()) {
395 scratchpad_md = add_reorder_pd.scratchpad_desc();
396 }
397
398 scratchpad_mem = dnnl::memory(scratchpad_md, eng);
399
400 last_batch_ = N;
401 }
402
403 auto pool_out_mem = dnnl::memory(pool_out_md, eng);
404 auto fc1_out_mem = dnnl::memory(fc1_out_md, eng);
405 auto fc2_out_mem = dnnl::memory(fc2_out_md, eng);
406
407 pooling_.execute(stream, {{DNNL_ARG_SRC, input},
408 {DNNL_ARG_DST, pool_out_mem},
409 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
410
411 dnnl::memory fc1_in_mem;
412 if (fc1_in_md != pool_out_md) {
413 fc1_in_mem = dnnl::memory(fc1_in_md, eng);
414 fc1_reorder_.execute(stream, {{DNNL_ARG_SRC, pool_out_mem},
415 {DNNL_ARG_DST, fc1_in_mem},
416 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
417 } else {
418 fc1_in_mem = pool_out_mem;
419 }
420
421 fc_.execute(stream, {{DNNL_ARG_SRC, fc1_in_mem},
422 {DNNL_ARG_WEIGHTS, filter_mem},
423 {DNNL_ARG_BIAS, bias_mem},
424 {DNNL_ARG_DST, fc1_out_mem},
425 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
426
427 fc2_.execute(stream, {{DNNL_ARG_SRC, fc1_out_mem},
428 {DNNL_ARG_WEIGHTS, filter2_mem},
429 {DNNL_ARG_BIAS, bias2_mem},
430 {DNNL_ARG_DST, fc2_out_mem},
431 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
432
433 dnnl::memory mul_in_mem;
434 mul_in_mem = dnnl::memory(pool_out_md, eng);
435 mul_reorder_.execute(stream, {{DNNL_ARG_SRC, fc2_out_mem},
436 {DNNL_ARG_DST, mul_in_mem},
437 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
438
439 sigmoid_.execute(stream, {{DNNL_ARG_SRC, mul_in_mem},
440 {DNNL_ARG_DST, mul_in_mem},
441 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
442
443 dnnl::memory add_in_mem;
444 add_in_mem = dnnl::memory(pool_out_md, eng);
445 add_reorder_.execute(stream, {{DNNL_ARG_SRC, fc2_out_mem},
446 {DNNL_ARG_DST, add_in_mem},
447 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
448
449 if (eng.get_kind() == dnnl::engine::kind::gpu) {
450 mul_.execute(stream, {{DNNL_ARG_SRC_0, input},
451 {DNNL_ARG_SRC_1, mul_in_mem},
452 {DNNL_ARG_ATTR_MULTIPLE_POST_OP(1) | DNNL_ARG_SRC_1,
453 add_in_mem},
454 {DNNL_ARG_DST, output},
455 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
456 } else {
457 mul_.execute(stream, {{DNNL_ARG_SRC_0, input},
458 {DNNL_ARG_SRC_1, mul_in_mem},
459 {DNNL_ARG_DST, output},
460 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
461
462 add_.execute(stream, {{DNNL_ARG_SRC_0, output},
463 {DNNL_ARG_SRC_1, add_in_mem},
464 {DNNL_ARG_DST, output},
465 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
466 }
467}

473void FCLayer::LoadWeights(dnnl::memory& w1, dnnl::memory& b1, dnnl::engine& eng,
474 dnnl::stream& stream) {
475 const int num_outputs = C * H * W;
476
477 auto filter_md = dnnl::memory::desc(
478 {num_outputs, input_->GetC(), input_->GetH(), input_->GetW()}, data_type_,
479 dnnl::memory::format_tag::abcd);
480 filter_mem = dnnl::memory(filter_md, eng);
481 dnnl::reorder(w1, filter_mem).execute(stream, w1, filter_mem);
482
483 auto bias_md = dnnl::memory::desc({num_outputs}, data_type_,
484 dnnl::memory::format_tag::a);
485 bias_mem = dnnl::memory(bias_md, eng);
486 dnnl::reorder(b1, bias_mem).execute(stream, b1, bias_mem);
487}

489void FCLayer::Eval(int N, dnnl::memory& output, dnnl::memory& input,
490 dnnl::engine& eng, dnnl::stream& stream) {
491 std::lock_guard<std::mutex> lock(lock_);
492 if (last_batch_ != N) {
493 const int num_outputs = C * H * W;
494
495 auto t_in_md =
496 dnnl::memory::desc({N, input_->GetC(), input_->GetH(), input_->GetW()},
497 data_type_, dnnl::memory::format_tag::any);
498
499 auto t_filter_md = dnnl::memory::desc(
500 {num_outputs, input_->GetC(), input_->GetH(), input_->GetW()},
501 data_type_, dnnl::memory::format_tag::any);
502
503 auto t_out_md = dnnl::memory::desc({N, C, H, W}, data_type_,
504 dnnl::memory::format_tag::any);
505
506 auto fc_d = dnnl::inner_product_forward::desc(
507 dnnl::prop_kind::forward_inference, t_in_md, t_filter_md,
508 bias_mem.get_desc(), t_out_md.reshape({N, num_outputs}));
509 dnnl::post_ops fc_ops;
510 if (activation_ == ACTIVATION_RELU) {
511 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_relu, 0.0f, 0.0f);
512 } else if (activation_ == ACTIVATION_MISH) {
513 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_mish, 0.0f, 0.0f);
514 } else if (activation_ == ACTIVATION_TANH) {
515 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_tanh, 0.0f, 0.0f);
516 }
517 dnnl::primitive_attr fc_attr;
518 fc_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
519 fc_attr.set_post_ops(fc_ops);
520 auto fc_pd =
521 dnnl::inner_product_forward::primitive_desc(fc_d, fc_attr, eng);
522 fc_ = dnnl::inner_product_forward(fc_pd);
523 auto scratchpad_md = fc_pd.scratchpad_desc();
524
525 in_md = fc_pd.src_desc();
526 out_md = fc_pd.dst_desc().reshape({N, C, H, W});
527 if (fc_pd.weights_desc() != filter_mem.get_desc()) {
528 auto tmp = dnnl::memory(fc_pd.weights_desc(), eng);
529 dnnl::reorder(filter_mem, tmp).execute(stream, filter_mem, tmp);
530 filter_mem = tmp;
531 }
532
533 dnnl::primitive_attr reorder_attr;
534 reorder_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
535 auto in_reorder_pd = dnnl::reorder::primitive_desc(
536 eng, input.get_desc(), eng, in_md, reorder_attr);
537 in_reorder_ = dnnl::reorder(in_reorder_pd);
538 if (scratchpad_md.get_size() < in_reorder_pd.scratchpad_desc().get_size()) {
539 scratchpad_md = in_reorder_pd.scratchpad_desc();
540 }
541
542 scratchpad_mem = dnnl::memory(fc_pd.scratchpad_desc(), eng);
543
544 last_batch_ = N;
545 }
546
547 if (in_md != input.get_desc()) {
548 auto tmp = dnnl::memory(in_md, eng);
549 in_reorder_.execute(stream, {{DNNL_ARG_SRC, input},
550 {DNNL_ARG_DST, tmp},
551 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
552 input = tmp;
553 }
554
555 if (!output || out_md != output.get_desc()) {
556 output = dnnl::memory(out_md, eng);
557 }
558
559 fc_.execute(stream, {{DNNL_ARG_SRC, input},
560 {DNNL_ARG_WEIGHTS, filter_mem},
561 {DNNL_ARG_BIAS, bias_mem},
562 {DNNL_ARG_DST, output},
563 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
564}

602void AttentionPolicyHead::Eval(int N, dnnl::memory& output, dnnl::memory& input,
603 dnnl::engine& eng, dnnl::stream& stream) {
604 std::lock_guard<std::mutex> lock(lock_);
605 if (last_batch_ != N) {
606 in_md = dnnl::memory::desc({N, C, H, W}, data_type_,
607 dnnl::memory::format_tag::nhwc);
608 out_md = dnnl::memory::desc({N, 67, 8, 8}, data_type_,
609 dnnl::memory::format_tag::nchw);
610
611 auto fc_out_md = dnnl::memory::desc({N * 64, embedding_size_}, data_type_,
612 dnnl::memory::format_tag::ab);
613 fc_out_mem = dnnl::memory(fc_out_md, eng);
614 auto foo_md = dnnl::memory::desc({N, H, W, C}, data_type_,
615 dnnl::memory::format_tag::nchw);
616 auto fc_d = dnnl::inner_product_forward::desc(
617 dnnl::prop_kind::forward_inference, foo_md.reshape({N * H * W, C}),
618 fc_filter_mem.get_desc(), fc_bias_mem.get_desc(), fc_out_md);
619 dnnl::post_ops fc_ops;
620 // SELU activation.
621 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_elu, 1.67326324f,
622 0.0f);
623 fc_ops.append_eltwise(1.0f, dnnl::algorithm::eltwise_linear, 1.05070098f,
624 0.0f);
625 dnnl::primitive_attr fc_attr;
626 fc_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
627 fc_attr.set_post_ops(fc_ops);
628 auto fc_pd =
629 dnnl::inner_product_forward::primitive_desc(fc_d, fc_attr, eng);
630 fc_ = dnnl::inner_product_forward(fc_pd);
631 auto scratchpad_md = fc_pd.scratchpad_desc();
632
633 // Q
634 auto fcQK_out_md = dnnl::memory::desc({N * 64, policy_d_model_}, data_type_,
635 dnnl::memory::format_tag::ab);
636 fcQ_out_mem = dnnl::memory(fcQK_out_md, eng);
637
638 auto fcQK_d = dnnl::inner_product_forward::desc(
639 dnnl::prop_kind::forward_inference, fc_out_md,
640 fcQ_filter_mem.get_desc(), fcQ_bias_mem.get_desc(), fcQK_out_md);
641 dnnl::primitive_attr common_attr;
642 common_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
643 auto fcQK_pd =
644 dnnl::inner_product_forward::primitive_desc(fcQK_d, common_attr, eng);
645 fcQK_ = dnnl::inner_product_forward(fcQK_pd);
646 if (scratchpad_md.get_size() < fcQK_pd.scratchpad_desc().get_size()) {
647 scratchpad_md = fcQK_pd.scratchpad_desc();
648 }
649
650 // K
651 fcK_out_mem = dnnl::memory(fcQK_out_md, eng);
652 const float scaling = sqrtf(policy_d_model_);
653 auto mul_A_md = dnnl::memory::desc({N, 64, policy_d_model_}, data_type_,
654 dnnl::memory::format_tag::abc);
655 auto mul_B_md = dnnl::memory::desc({N, policy_d_model_, 64}, data_type_,
656 dnnl::memory::format_tag::acb);
657 auto mul_C_md =
658 dnnl::memory::desc({N, 64, 64}, data_type_, {64 * 67, 64, 1});
659 auto mul_d = dnnl::matmul::desc(mul_A_md, mul_B_md, mul_C_md);
660 dnnl::primitive_attr mul_attr;
661 mul_attr.set_scratchpad_mode(dnnl::scratchpad_mode::user);
662 mul_attr.set_output_scales(0, {1.0f / scaling});
663 auto mul_pd = dnnl::matmul::primitive_desc(mul_d, mul_attr, eng);
664 mul_ = dnnl::matmul(mul_pd);
665 if (scratchpad_md.get_size() < mul_pd.scratchpad_desc().get_size()) {
666 scratchpad_md = mul_pd.scratchpad_desc();
667 }
668
669 // The promotion offsets are stored in the output array for now.
670 auto promo_md =
671 out_md.submemory_desc({N, 1, 4, 8}, {0, 64, 0, 0}).reshape({N, 4, 8});
672
673 if (eng.get_kind() == dnnl::engine::kind::gpu) {
674 // The gpu matmul primitive ignores memory offsets, so a copy is needed.
675 auto reorder_pd = dnnl::reorder::primitive_desc(
676 eng, mul_B_md.submemory_desc({N, policy_d_model_, 8}, {0, 0, 56}),
677 eng, mul_B_md.submemory_desc({N, policy_d_model_, 8}, {0, 0, 0}),
678 common_attr);
679 hack_reorder_ = dnnl::reorder(reorder_pd);
680 if (scratchpad_md.get_size() < reorder_pd.scratchpad_desc().get_size()) {
681 scratchpad_md = reorder_pd.scratchpad_desc();
682 }
683
684 auto pmul_d = dnnl::matmul::desc(
685 pmul_mem.get_desc(),
686 mul_B_md.submemory_desc({N, policy_d_model_, 8}, {0, 0, 0}),
687 mul_A_md.submemory_desc({N, 4, 8}, {0, 0, 0}));
688 auto pmul_pd = dnnl::matmul::primitive_desc(pmul_d, common_attr, eng);
689 pmul_ = dnnl::matmul(pmul_pd);
690 if (scratchpad_md.get_size() < pmul_pd.scratchpad_desc().get_size()) {
691 scratchpad_md = pmul_pd.scratchpad_desc();
692 }
693
694 reorder_pd = dnnl::reorder::primitive_desc(
695 eng, mul_A_md.submemory_desc({N, 4, 8}, {0, 0, 0}), eng, promo_md,
696 common_attr);
697 hack_reorder_2_ = dnnl::reorder(reorder_pd);
698 if (scratchpad_md.get_size() < reorder_pd.scratchpad_desc().get_size()) {
699 scratchpad_md = reorder_pd.scratchpad_desc();
700 }
701
702 } else {
703 auto pmul_d = dnnl::matmul::desc(
704 pmul_mem.get_desc(),
705 mul_B_md.submemory_desc({N, policy_d_model_, 8}, {0, 0, 56}),
706 promo_md);
707 auto pmul_pd = dnnl::matmul::primitive_desc(pmul_d, common_attr, eng);
708 pmul_ = dnnl::matmul(pmul_pd);
709 if (scratchpad_md.get_size() < pmul_pd.scratchpad_desc().get_size()) {
710 scratchpad_md = pmul_pd.scratchpad_desc();
711 }
712 }
713
714 auto in_reorder_pd = dnnl::reorder::primitive_desc(eng, input.get_desc(),
715 eng, in_md, common_attr);
716 in_reorder_ = dnnl::reorder(in_reorder_pd);
717 if (scratchpad_md.get_size() < in_reorder_pd.scratchpad_desc().get_size()) {
718 scratchpad_md = in_reorder_pd.scratchpad_desc();
719 }
720
721 scratchpad_mem = dnnl::memory(scratchpad_md, eng);
722
723 last_batch_ = N;
724 }
725
726 // Convert to NHWC.
727 if (in_md != input.get_desc()) {
728 auto tmp = dnnl::memory(in_md, eng);
729 in_reorder_.execute(stream, {{DNNL_ARG_SRC, input},
730 {DNNL_ARG_DST, tmp},
731 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
732 input = tmp;
733 }
734
735 if (!output || out_md != output.get_desc()) {
736 output = dnnl::memory(out_md, eng);
737 }
738
739 fc_.execute(stream, {{DNNL_ARG_SRC, input},
740 {DNNL_ARG_WEIGHTS, fc_filter_mem},
741 {DNNL_ARG_BIAS, fc_bias_mem},
742 {DNNL_ARG_DST, fc_out_mem},
743 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
744
745 fcQK_.execute(stream, {{DNNL_ARG_SRC, fc_out_mem},
746 {DNNL_ARG_WEIGHTS, fcQ_filter_mem},
747 {DNNL_ARG_BIAS, fcQ_bias_mem},
748 {DNNL_ARG_DST, fcQ_out_mem},
749 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
750
751 fcQK_.execute(stream, {{DNNL_ARG_SRC, fc_out_mem},
752 {DNNL_ARG_WEIGHTS, fcK_filter_mem},
753 {DNNL_ARG_BIAS, fcK_bias_mem},
754 {DNNL_ARG_DST, fcK_out_mem},
755 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
756
757 mul_.execute(stream, {{DNNL_ARG_SRC, fcQ_out_mem},
758 {DNNL_ARG_WEIGHTS, fcK_out_mem},
759 {DNNL_ARG_DST, output},
760 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
761
762 if (eng.get_kind() == dnnl::engine::kind::gpu) {
763 hack_reorder_.execute(stream, {{DNNL_ARG_SRC, fcK_out_mem},
764 {DNNL_ARG_DST, fcQ_out_mem},
765 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
766 pmul_.execute(stream, {{DNNL_ARG_SRC, pmul_mem},
767 {DNNL_ARG_WEIGHTS, fcQ_out_mem},
768 {DNNL_ARG_DST, fcK_out_mem},
769 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
770 hack_reorder_2_.execute(stream, {{DNNL_ARG_SRC, fcK_out_mem},
771 {DNNL_ARG_DST, output},
772 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
773
774 } else {
775 pmul_.execute(stream, {{DNNL_ARG_SRC, pmul_mem},
776 {DNNL_ARG_WEIGHTS, fcK_out_mem},
777 {DNNL_ARG_DST, output},
778 {DNNL_ARG_SCRATCHPAD, scratchpad_mem}});
779 }
780}
