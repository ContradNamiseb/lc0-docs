85std::pair<std::string, std::unordered_map<std::string, std::string>>
86ParseCommand(const std::string& line) {
87 std::unordered_map<std::string, std::string> params;
88 std::string* value = nullptr;
89
90 std::istringstream iss(line);
91 std::string token;
92 iss >> token >> std::ws;
93
94 // If empty line, return empty command.
95 if (token.empty()) return {};
96
97 const auto command = kKnownCommands.find(token);
98 if (command == kKnownCommands.end()) {
99 throw Exception("Unknown command: " + line);
100 }
101
102 // Special parsing for setoption to keep strings unmodified.
103 if (command->first == "setoption") {
104 iss >> token;
105 if (token != "name") {
106 throw Exception("setoption must be followed by name");
107 }
108 int name_pos = iss.eof() ? line.length() : static_cast<int>(iss.tellg());
109 std::optional<int> value_pos;
110 while (iss >> token) {
111 if (token == "value") {
112 value_pos = iss.eof() ? line.length() : static_cast<int>(iss.tellg());
113 params["value"] = Trim(line.substr(*value_pos));
114 break;
115 }
116 }
117 params["name"] = Trim(line.substr(
118 name_pos, value_pos ? *value_pos - name_pos - 5 : std::string::npos));
119 return {"setoption", params};
120 }
121
122 std::string whitespace;
123 while (iss >> token) {
124 auto iter = command->second.find(token);
125 if (iter == command->second.end()) {
126 if (!value) throw Exception("Unexpected token: " + token);
127 *value += whitespace + token;
128 whitespace = " ";
129 } else {
130 value = &params[token];
131 iss >> std::ws;
132 whitespace = "";
133 }
134 }
135 return {command->first, params};
136}

138std::string GetOrEmpty(
139 const std::unordered_map<std::string, std::string>& params,
140 const std::string& key) {
141 const auto iter = params.find(key);
142 if (iter == params.end()) return {};
143 return iter->second;

146int GetNumeric(const std::unordered_map<std::string, std::string>& params,
147 const std::string& key) {
148 const auto iter = params.find(key);
149 if (iter == params.end()) {
150 throw Exception("Unexpected error");
151 }
152 const std::string& str = iter->second;
153 try {
154 if (str.empty()) {
155 throw Exception("expected value after " + key);
156 }
157 return std::stoi(str);
158 } catch (std::invalid_argument&) {
159 throw Exception("invalid value " + str);
160 } catch (const std::out_of_range&) {
161 throw Exception("out of range value " + str);
162 }
163}

179bool UciLoop::DispatchCommand(
180 const std::string& command,
181 const std::unordered_map<std::string, std::string>& params) {
182 if (command == "uci") {
183 uci_responder_->SendId();
184 for (const auto& option : options_->ListOptionsUci()) {
185 uci_responder_->SendRawResponse(option);
186 }
187 uci_responder_->SendRawResponse("uciok");
188 } else if (command == "isready") {
189 engine_->EnsureReady();
190 uci_responder_->SendRawResponse("readyok");
191 } else if (command == "setoption") {
192 if (GetOrEmpty(params, "name").empty()) {
193 throw Exception("setoption requires name");
194 } else {
195 options_->SetUciOption(GetOrEmpty(params, "name"),
196 GetOrEmpty(params, "value"));
197 }
198 } else if (command == "ucinewgame") {
199 engine_->NewGame();
200 } else if (command == "position") {
201 if (ContainsKey(params, "fen") == ContainsKey(params, "startpos")) {
202 throw Exception("Position requires either fen or startpos");
203 }
204 const std::vector<std::string> moves =
205 StrSplitAtWhitespace(GetOrEmpty(params, "moves"));
206 const std::string fen = GetOrEmpty(params, "fen");
207 engine_->SetPosition(fen.empty() ? ChessBoard::kStartposFen : fen, moves);
208 } else if (command == "go") {
209 GoParams go_params;
210 if (ContainsKey(params, "infinite")) {
211 if (!GetOrEmpty(params, "infinite").empty()) {
212 throw Exception("Unexpected token " + GetOrEmpty(params, "infinite"));
213 }
214 go_params.infinite = true;
215 }
216 if (ContainsKey(params, "searchmoves")) {
217 go_params.searchmoves =
218 StrSplitAtWhitespace(GetOrEmpty(params, "searchmoves"));
219 }
220 if (ContainsKey(params, "ponder")) {
221 if (!GetOrEmpty(params, "ponder").empty()) {
222 throw Exception("Unexpected token " + GetOrEmpty(params, "ponder"));
223 }
224 go_params.ponder = true;
225 }
226#define UCIGOOPTION(x) \
227 if (ContainsKey(params, #x)) { \
228 go_params.x = GetNumeric(params, #x); \
229 }
230 UCIGOOPTION(wtime);
231 UCIGOOPTION(btime);
232 UCIGOOPTION(winc);
233 UCIGOOPTION(binc);
234 UCIGOOPTION(movestogo);
235 UCIGOOPTION(depth);
236 UCIGOOPTION(mate);
237 UCIGOOPTION(nodes);
238 UCIGOOPTION(movetime);
239#undef UCIGOOPTION
240 engine_->Go(go_params);
241 } else if (command == "wait") {
242 engine_->Wait();
243 } else if (command == "stop") {
244 engine_->Stop();
245 } else if (command == "ponderhit") {
246 engine_->PonderHit();
247 } else if (command == "xyzzy") {
248 uci_responder_->SendRawResponse("Nothing happens.");
249 } else if (command == "quit") {
250 return false;
251 } else {
252 throw Exception("Unknown command: " + command);
253 }
254 return true;
255}

257bool UciLoop::ProcessLine(const std::string& line) {
258 auto command = ParseCommand(line);
259 // Ignore empty line.
260 if (command.first.empty()) return true;
261 return DispatchCommand(command.first, command.second);
262}

285void StringUciResponder::OutputBestMove(BestMoveInfo* info) {
286 const bool c960 = IsChess960();
287 std::string res = "bestmove " + info->bestmove.ToString(c960);
288 if (!info->ponder.is_null()) res += " ponder " + info->ponder.ToString(c960);
289 if (info->player != -1) res += " player " + std::to_string(info->player);
290 if (info->game_id != -1) res += " gameid " + std::to_string(info->game_id);
291 if (info->is_black)
292 res += " side " + std::string(*info->is_black ? "black" : "white");
293 SendRawResponse(res);

296void StringUciResponder::OutputThinkingInfo(std::vector<ThinkingInfo>* infos) {
297 std::vector<std::string> reses;
298 const bool c960 = IsChess960();
299 for (const auto& info : *infos) {
300 std::string res = "info";
301 if (info.player != -1) res += " player " + std::to_string(info.player);
302 if (info.game_id != -1) res += " gameid " + std::to_string(info.game_id);
303 if (info.is_black)
304 res += " side " + std::string(*info.is_black ? "black" : "white");
305 if (info.depth >= 0)
306 res += " depth " + std::to_string(std::max(info.depth, 1));
307 if (info.seldepth >= 0) res += " seldepth " + std::to_string(info.seldepth);
308 if (info.time >= 0) res += " time " + std::to_string(info.time);
309 if (info.nodes >= 0) res += " nodes " + std::to_string(info.nodes);
310 if (info.mate) res += " score mate " + std::to_string(*info.mate);
311 if (info.score) res += " score cp " + std::to_string(*info.score);
312 if (info.wdl && options_ && options_->Get<bool>(kShowWDL)) {
313 res += " wdl " + std::to_string(info.wdl->w) + " " +
314 std::to_string(info.wdl->d) + " " + std::to_string(info.wdl->l);
315 }
316 if (info.moves_left && options_ && options_->Get<bool>(kShowMovesleft)) {
317 res += " movesleft " + std::to_string(*info.moves_left);
318 }
319 if (info.hashfull >= 0) res += " hashfull " + std::to_string(info.hashfull);
320 if (info.nps >= 0) res += " nps " + std::to_string(info.nps);
321 if (info.eps >= 0 && options_ && options_->Get<bool>(kShowEPS)) {
322 res += " eps " + std::to_string(info.eps);
323 }
324 if (info.tb_hits >= 0) res += " tbhits " + std::to_string(info.tb_hits);
325 if (info.multipv >= 0) res += " multipv " + std::to_string(info.multipv);
326
327 if (!info.pv.empty()) {
328 res += " pv";
329 for (const auto& move : info.pv) res += " " + move.ToString(c960);
330 }
331 if (!info.comment.empty()) res += " string " + info.comment;
332 reses.push_back(std::move(res));
333 }
334 SendRawResponses(reses);
335}
