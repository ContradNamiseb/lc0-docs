56// This is needed to get the config file from the parameters without calling
57// ProcessAllFlags() that should be called only once, and needs the config file.
58std::string ConfigFile::ProcessConfigFlag(
59 const std::vector<std::string>& args) {
60 std::string filename = kDefaultConfigFileParam;
61 for (auto iter = args.begin(), end = args.end(); iter != end; ++iter) {
62 std::string param = *iter;
63
64 if (param.substr(0, 2) == "--") {
65 param = param.substr(2);
66 const auto pos = param.find('=');
67 if (pos != std::string::npos) {
68 if (param.substr(0, pos) == kConfigFileId.long_flag()) {
69 filename = param.substr(pos + 1);
70 }
71 }
72 }
73 if (param.size() == 2 && param[0] == '-') {
74 if (param[1] == kConfigFileId.short_flag() && iter + 1 != end) {
75 filename = *(iter + 1);
76 ++iter;
77 }
78 }
79 }
80 return filename;
81}

83bool ConfigFile::Init() {
84 arguments_.clear();
85
86 // Get the path from the config file parameter.
87 std::string filename = ProcessConfigFlag(CommandLine::Arguments());
88
89 // If filename is an empty string then return true. This is to override
90 // loading the default configuration file.
91 if (filename == "") return true;
92
93 // Parses the file into the arguments_ vector.
94 if (!ParseFile(filename)) return false;
95
96 return true;
97}

99bool ConfigFile::ParseFile(std::string& filename) {
100 // Check to see if we are using the default config file or not.
101 const bool using_default_config =
102 filename == std::string(kDefaultConfigFileParam);
103
104 std::ifstream input;
105
106 // If no logfile was set on the command line, then the default is
107 // to check in the binary directory.
108 if (using_default_config) {
109 std::vector<std::string> config_dirs = {CommandLine::BinaryDirectory()};
110 const std::string user_config_path = GetUserConfigDirectory();
111 if (!user_config_path.empty()) {
112 config_dirs.emplace_back(user_config_path + "lc0");
113 }
114 for (const auto& dir : GetSystemConfigDirectoryList()) {
115 config_dirs.emplace_back(dir + (dir.back() == '/' ? "" : "/") + "lc0");
116 }
117
118 for (const auto& dir : config_dirs) {
119 filename = dir + '/' + kDefaultConfigFile;
120 input.open(filename);
121 if (input.is_open()) break;
122 }
123 } else {
124 input.open(filename);
125 }
126
127 if (!input.is_open()) {
128 // It is okay if we cannot open the default file since it is normal
129 // for it to not exist.
130 if (using_default_config) return true;
131
132 CERR << "Could not open configuration file: " << filename;
133 return false;
134 }
135
136 CERR << "Found configuration file: " << filename;
137
138 for (std::string line; getline(input, line);) {
139 // Remove all leading and trailing whitespace.
140 line = Trim(line);
141 // Ignore comments.
142 if (line.substr(0, 1) == "#") continue;
143 // Skip blank lines.
144 if (line.length() == 0) continue;
145 // Allow long form arguments that omit '--'. If omitted, add here.
146 if (line.substr(0, 1) != "-" && line.substr(0, 2) != "--") {
147 line = "--" + line;
148 }
149 // Fail now if the argument does not begin with '--'.
150 if (line.substr(0, 2) != "--") {
151 CERR << "Only '--' arguments are supported in the "
152 << "configuration file: '" << line << "'.";
153 return false;
154 }
155 // Add the line to the arguments list.
156 arguments_.emplace_back(line);
157 }
158
159 return true;
160}
