140class DemuxingBackend {
141 public:
142 ~DemuxingBackend() {
143 while (!threads_.empty()) {
144 threads_.back().join();
145 threads_.pop_back();
146 }
147 while (!queue_.empty()) {
148 queue_.front()->source_->NotifyComplete();
149 queue_.pop();
150 }
151 }
152
153 void Assign(std::unique_ptr<Network>&& network, const OptionsDict& opts,
154 std::atomic<bool>& abort) {
155 network_ = std::move(network);
156 int nn_threads = opts.GetOrDefault<int>("threads", 0);
157 if (nn_threads == 0) {
158 nn_threads = network_->GetThreads();
159 }
160 for (int i = 0; i < nn_threads; i++) {
161 threads_.emplace_back([&] { Worker(abort); });
162 }
163 }
164
165 void Enqueue(DemuxingWork* work) {
166 {
167 std::unique_lock lock(mutex_);
168 queue_.push(work);
169 }
170 dataready_cv_.notify_one();
171 }
172
173 void Abort() {
174 {
175 std::unique_lock lock(mutex_);
176 }
177 dataready_cv_.notify_all();
178 }
179
180 void Worker(std::atomic<bool>& abort) {
181 while (!abort.load(std::memory_order_relaxed)) {
182 DemuxingWork* work = nullptr;
183 {
184 std::unique_lock lock(mutex_);
185 dataready_cv_.wait(lock, [&] {
186 return abort.load(std::memory_order_relaxed) || !queue_.empty();
187 });
188 if (abort.load(std::memory_order_relaxed)) return;
189 if (!queue_.empty()) {
190 work = queue_.front();
191 queue_.pop();
192 }
193 }
194 if (work) {
195 work->computation_ = network_->NewComputation();
196 auto& planes = work->source_->planes_;
197 for (int i = work->start_; i < work->end_; i++) {
198 work->computation_->AddInput(std::move(planes[i]));
199 }
200 work->computation_->ComputeBlocking();
201 work->source_->NotifyComplete();
202 }
203 }
204 }
205
206 private:
207 std::mutex mutex_;
208 std::condition_variable dataready_cv_;
209 std::vector<std::thread> threads_;
210 std::unique_ptr<Network> network_;
211 std::queue<DemuxingWork*> queue_;
212};

218 public:
219 DemuxingNetwork(const std::optional<WeightsFile>& weights,
220 const OptionsDict& options)
221 : backends_(std::max(size_t(1), options.ListSubdicts().size())) {
222 const auto parents = options.ListSubdicts();
223 if (parents.empty()) {
224 // If options are empty, or multiplexer configured in root object,
225 // initialize on root object and default backend.
226 auto backends = NetworkFactory::Get()->GetBackendsList();
227 AddBackend(0, backends[0], weights, options);
228 }
229
230 int i = 0;
231 for (const auto& name : parents) {
232 AddBackend(i++, name, weights, options.GetSubdict(name));
233 }
234 }

236 void AddBackend(int index, const std::string& name,
237 const std::optional<WeightsFile>& weights,
238 const OptionsDict& opts) {
239 const std::string backend = opts.GetOrDefault<std::string>("backend", name);
240
241 auto network = NetworkFactory::Get()->Create(backend, weights, opts);
242
243 min_batch_size_ = std::min(min_batch_size_, network->GetMiniBatchSize());
244 batch_step_ = std::max(batch_step_, network->GetPreferredBatchStep());
245 is_cpu_ &= network->IsCpu();
246 if (index == 0) {
247 capabilities_ = network->GetCapabilities();
248 } else {
249 capabilities_.Merge(network->GetCapabilities());
250 }
251 backends_[index].Assign(std::move(network), opts, abort_);
252 }

255 return std::make_unique<DemuxingComputation>(this);
256 }
257

259 return capabilities_;

273 abort_.store(true, std::memory_order_relaxed);
274 for (auto& b : backends_) {
275 b.Abort();
276 }
277 }

288void DemuxingComputation::ComputeBlocking() {
289 if (GetBatchSize() == 0) return;
290 // Calculate batch_step_ size split count.
291 int splits = 1 + (GetBatchSize() - 1) / network_->batch_step_;
292 // Calculate the minimum number of splits per backend.
293 int split_size_per_backend = splits / network_->backends_.size();
294 // Calculate how many backends get extra work.
295 int extra_split_backends =
296 splits - split_size_per_backend * network_->backends_.size();
297
298 // Find the first backend which got less work from the previous batch.
299 int start_index =
300 network_->start_index_.fetch_add(std::max(1, extra_split_backends),
301 std::memory_order_relaxed) %
302 network_->backends_.size();
303
304 int end_index =
305 (start_index + extra_split_backends) % network_->backends_.size();
306 int work_start = 0;
307 int work_items = split_size_per_backend > 0 ? network_->backends_.size()
308 : extra_split_backends;
309 // First store the work item count and reserve memory from them.
310 dataready_.store(work_items, std::memory_order_relaxed);
311 parents_.reserve(work_items);
312 int i = start_index;
313 // First send work to backends which get extra work.
314 int split_size = split_size_per_backend + 1;
315 for (; i != end_index; i = (i + 1) % network_->backends_.size()) {
316 assert(work_start != GetBatchSize());
317 int work_end = work_start + split_size * network_->batch_step_;
318 work_end = std::min(work_end, GetBatchSize());
319 parents_.emplace_back(this, work_start, work_end);
320 network_->backends_[i].Enqueue(&parents_.back());
321 work_start = work_end;
322 }
323 // Queue remaining work items which don't get extra work.
324 split_size--;
325 if (split_size > 0) {
326 do {
327 assert(work_start != GetBatchSize());
328 int work_end = work_start + split_size * network_->batch_step_;
329 work_end = std::min(work_end, GetBatchSize());
330 parents_.emplace_back(this, work_start, work_end);
331 network_->backends_[i].Enqueue(&parents_.back());
332 work_start = work_end;
333 i = (i + 1) % network_->backends_.size();
334 } while (i != start_index);
335 }
336 assert(work_start == GetBatchSize());
337 assert(work_items == (int)parents_.size());
338 // Wait until all backends complete their work.
339 std::unique_lock<std::mutex> lock(mutex_);
340 dataready_cv_.wait(lock, [this]() {
341 return dataready_.load(std::memory_order_acquire) <= 0;
342 });
343}
