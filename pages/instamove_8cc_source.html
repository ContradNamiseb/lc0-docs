45class InstamoveSearch : public SearchBase {
46 public:
47 using SearchBase::SearchBase;
48
49 private:
50 virtual Move GetBestMove(const GameState& game_state) = 0;
51
52 void SetPosition(const GameState& game_state) final {
53 game_state_ = game_state;
54 }
55
56 void StartSearch(const GoParams& go_params) final {
57 responded_bestmove_.store(false, std::memory_order_relaxed);
58 bestmove_ = GetBestMove(game_state_);
59 if (!go_params.infinite && !go_params.ponder) RespondBestMove();
60 }
61 void WaitSearch() final {
62 while (!responded_bestmove_.load()) {
63 std::this_thread::sleep_for(std::chrono::milliseconds(10));
64 }
65 }
66 void StopSearch() final { RespondBestMove(); }
67 void AbortSearch() final { responded_bestmove_.store(true); }
68 void RespondBestMove() {
69 if (responded_bestmove_.exchange(true)) return;
70 BestMoveInfo info{bestmove_};
71 // TODO Remove this when move will be encoded from white perspective.
72 if (game_state_.CurrentPosition().IsBlackToMove()) {
73 info.bestmove.Flip();
74 } else if (!info.ponder.is_null()) {
75 info.ponder.Flip();
76 }
77 uci_responder_->OutputBestMove(&info);
78 }
79
80 void SetBackend(Backend* backend) override {
81 batchsplit_backend_ = CreateBatchSplitingBackend(backend);
82 backend_ = batchsplit_backend_.get();
83 }
84 void StartClock() final {}
85
86 Move bestmove_;
87 std::atomic<bool> responded_bestmove_{false};
88 std::unique_ptr<Backend> batchsplit_backend_;
89 GameState game_state_;
90};

93class PolicyHeadSearch : public InstamoveSearch {
94 public:
95 using InstamoveSearch::InstamoveSearch;
96
97 Move GetBestMove(const GameState& game_state) final {
98 const std::vector<Position> positions = game_state.GetPositions();
99 MoveList legal_moves = positions.back().GetBoard().GenerateLegalMoves();
100 std::vector<EvalResult> res = backend_->EvaluateBatch(
101 std::vector<EvalPosition>{EvalPosition{positions, legal_moves}});
102 const size_t best_move_idx =
103 std::max_element(res[0].p.begin(), res[0].p.end()) - res[0].p.begin();
104
105 std::vector<ThinkingInfo> infos = {{
106 .depth = 1,
107 .seldepth = 1,
108 .nodes = 1,
109 .score = 90 * std::tan(1.5637541897 * res[0].q),
110 .wdl =
111 ThinkingInfo::WDL{
112 static_cast<int>(std::round(500 * (1 + res[0].q - res[0].d))),
113 static_cast<int>(std::round(1000 * res[0].d)),
114 static_cast<int>(std::round(500 * (1 - res[0].q - res[0].d)))},
115 }};
116 uci_responder_->OutputThinkingInfo(&infos);
117
118 Move best_move = legal_moves[best_move_idx];
119 return best_move;
120 }
121};

124class ValueHeadSearch : public InstamoveSearch {
125 public:
126 using InstamoveSearch::InstamoveSearch;
127 Move GetBestMove(const GameState& game_state) final {
128 std::unique_ptr<BackendComputation> computation =
129 backend_->CreateComputation();
130
131 PositionHistory history(game_state.GetPositions());
132 const ChessBoard& board = history.Last().GetBoard();
133 const std::vector<Move> legal_moves = board.GenerateLegalMoves();
134
135 struct Score {
136 float negative_q; // Negative because NN evaluates from opponent's
137 // perspective.
138 float d;
139 std::optional<int> mate;
140
141 bool operator<(const Score& other) const {
142 // Mate always beats non-mate
143 if (mate && !other.mate) return true;
144 if (!mate && other.mate) return false;
145 // Both mates: shorter is better
146 if (mate && other.mate) return *mate < *other.mate;
147 // Neither mate: lower negative_q is better
148 return negative_q < other.negative_q;
149 }
150 };
151
152 std::vector<Score> results(legal_moves.size());
153
154 for (size_t i = 0; i < legal_moves.size(); i++) {
155 Move move = legal_moves[i];
156 history.Append(move);
157 switch (history.ComputeGameResult()) {
158 case GameResult::UNDECIDED:
159 computation->AddInput(
160 EvalPosition{history.GetPositions(), {}},
161 EvalResultPtr{.q = &results[i].negative_q, .d = &results[i].d});
162 break;
163 case GameResult::DRAW:
164 results[i] = {.negative_q = 0, .d = 1, .mate = std::nullopt};
165 break;
166 default:
167 // A legal move to a non-drawn terminal without tablebases must be a
168 // win.
169 results[i] = {.negative_q = -1, .d = 0, .mate = 1};
170 }
171 history.Pop();
172 }
173
174 computation->ComputeBlocking();
175
176 const size_t best_idx =
177 std::min_element(results.begin(), results.end()) - results.begin();
178
179 const Score& r = results[best_idx];
180 auto to_int = [](double x) { return static_cast<int>(std::round(x)); };
181 std::vector<ThinkingInfo> infos{
182 {.depth = 1,
183 .seldepth = 1,
184 .nodes = static_cast<int64_t>(legal_moves.size()),
185 .mate = r.mate,
186 .score = r.mate ? std::nullopt
187 : std::make_optional<int>(
188 -90 * std::tan(1.5637541897 * r.negative_q)),
189 .wdl = r.mate
190 ? std::nullopt
191 : std::make_optional<ThinkingInfo::WDL>(ThinkingInfo::WDL{
192 .w = to_int(500 * (1 - r.negative_q - r.d)),
193 .d = to_int(1000 * r.d),
194 .l = to_int(500 * (1 + r.negative_q - r.d)),
195 })}};
196 uci_responder_->OutputThinkingInfo(&infos);
197 Move best_move = legal_moves[best_idx];
198 return best_move;
199 }
200};
