30uint64_t ReadFixed(const std::uint8_t** iter, size_t size,

68void ProtoMessage::MergeFromString(std::string_view str) {
69 const std::uint8_t* iter = reinterpret_cast<const std::uint8_t*>(str.data());
70 const std::uint8_t* const end = iter + str.size();
71 while (iter < end) {
72 uint64_t wire_field_id = ReadVarInt(&iter, end);
73 uint64_t field_id = wire_field_id >> 3;
74 switch (wire_field_id & 0x7) {
75 case 0:
76 // Varint field, so read one more varint.
77 SetVarInt(field_id, ReadVarInt(&iter, end));
78 break;
79 case 1:
80 // Fixed64, read 8 bytes.
81 SetInt64(field_id, ReadFixed(&iter, 8, end));
82 break;
83 case 2: {
84 // String/submessage. Varint length and then buffer of that length.
85 size_t size = ReadVarInt(&iter, end);
86 CheckOutOfBounds(iter, size, end);
87 SetString(field_id,
88 std::string_view(reinterpret_cast<const char*>(iter), size));
89 iter += size;
90 break;
91 }
92 case 5:
93 // Fixed32, read 4 bytes.
94 SetInt32(field_id, ReadFixed(&iter, 4, end));
95 break;
96 default:
97 throw Exception("The file seems to be unparseable.");
98 }
99 }
