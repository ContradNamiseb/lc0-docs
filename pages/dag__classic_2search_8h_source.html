226class SearchWorker {
227 public:
228 static constexpr int kTaskCountDigits = std::numeric_limits<int>::digits + 1;
229 static constexpr int kTasksTakenShift = kTaskCountDigits / 2;
230 static constexpr int kTasksTakenOne = 1 << kTasksTakenShift;
231 // Suspend is -1 for the low half.
232 static constexpr int kTaskCountSuspend = kTasksTakenOne - 1;
233
234 SearchWorker(Search* search, const SearchParams& params)
235 : search_(search),
236 history_(search_->played_history_),
237 params_(params),
238 moves_left_support_(search_->backend_attributes_.has_mlh) {
239 task_workers_ = params.GetTaskWorkersPerSearchWorker();
240 if (task_workers_ < 0) {
241 if (search_->backend_attributes_.runs_on_cpu) {
242 task_workers_ = 0;
243 } else {
244 int working_threads = std::max(
245 search_->thread_count_.load(std::memory_order_acquire) - 1, 1);
246 task_workers_ = std::min(
247 std::thread::hardware_concurrency() / working_threads - 1, 4U);
248 }
249 }
250 for (int i = 0; i < task_workers_; i++) {
251 task_workspaces_.emplace_back();
252 task_threads_.emplace_back([this, i]() {
253 LOGFILE << "Task worker " << i << " starting.";
254 this->RunTasks(i);
255 LOGFILE << "Task worker " << i << " exiting.";
256 });
257 }
258 target_minibatch_size_ = params_.GetMiniBatchSize();
259 if (target_minibatch_size_ == 0) {
260 target_minibatch_size_ =
261 search_->backend_attributes_.recommended_batch_size;
262 }
263 max_out_of_order_ =
264 std::max(1, static_cast<int>(params_.GetMaxOutOfOrderEvalsFactor() *
265 target_minibatch_size_));
266 }
267
268 ~SearchWorker();
269
271 void RunBlocking() {
272 LOGFILE << "Started search thread.";
273 try {
274 // A very early stop may arrive before this point, so the test is at the
275 // end to ensure at least one iteration runs before exiting.
276 do {
277 ExecuteOneIteration();
278 } while (search_->IsSearchActive());
279 } catch (std::exception& e) {
280 std::cerr << "Unhandled exception in worker thread: " << e.what()
281 << std::endl;
282 abort();
283 }
284 }

368 std::string DebugString() const {

372 << " Maxvisit:" << maxvisit << " NNQueried:" << nn_queried

374 << " Collision:" << is_collision << " OOO:" << ooo_completed

376 for (auto it = path.cbegin(); it != path.cend(); ++it) {
377 if (it != path.cbegin()) oss << "->";
378 auto n = std::get<0>(*it);
379 const auto& nl = n->GetLowNode();
380 oss << n << ":" << n->GetNInFlight();
381 if (nl) {
382 oss << "(" << nl << ")";
