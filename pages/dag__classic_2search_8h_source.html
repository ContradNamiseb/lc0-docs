262class SearchWorker {
263 public:
264 static constexpr int kTaskCountDigits = std::numeric_limits<int>::digits + 1;
265 static constexpr int kTasksTakenShift = kTaskCountDigits / 2;
266 static constexpr int kTasksTakenOne = 1 << kTasksTakenShift;
267 // Suspend is -1 for the low half.
268 static constexpr int kTaskCountSuspend = kTasksTakenOne - 1;
269
270 SearchWorker(Search* search, const SearchParams& params)
271 : search_(search),
272 history_(search_->played_history_),
273 params_(params),
274 moves_left_support_(search_->backend_attributes_.has_mlh) {
275 task_workers_ = params.GetTaskWorkersPerSearchWorker();
276 if (task_workers_ < 0) {
277 if (search_->backend_attributes_.runs_on_cpu) {
278 task_workers_ = 0;
279 } else {
280 int working_threads = std::max(
281 search_->thread_count_.load(std::memory_order_acquire) - 1, 1);
282 task_workers_ = std::min(
283 std::thread::hardware_concurrency() / working_threads - 1, 4U);
284 }
285 }
286 for (int i = 0; i < task_workers_; i++) {
287 task_workspaces_.emplace_back();
288 task_threads_.emplace_back([this, i]() {
289 LOGFILE << "Task worker " << i << " starting.";
290 this->RunTasks(i);
291 LOGFILE << "Task worker " << i << " exiting.";
292 });
293 }
294 target_minibatch_size_ = params_.GetMiniBatchSize();
295 if (target_minibatch_size_ == 0) {
296 target_minibatch_size_ =
297 search_->backend_attributes_.recommended_batch_size;
298 }
299 max_out_of_order_ =
300 std::max(1, static_cast<int>(params_.GetMaxOutOfOrderEvalsFactor() *
301 target_minibatch_size_));
302 }
303
304 ~SearchWorker();
305
307 void RunBlocking() {
308 LOGFILE << "Started search thread.";
309 try {
310 // A very early stop may arrive before this point, so the test is at the
311 // end to ensure at least one iteration runs before exiting.
312 do {
313 ExecuteOneIteration();
314 } while (search_->IsSearchActive());
315 } catch (std::exception& e) {
316 std::cerr << "Unhandled exception in worker thread: " << e.what()
317 << std::endl;
318 abort();
319 }
320 }

408 std::string DebugString() const {
409 std::ostringstream oss;
410 oss << "<NodeToProcess> This:" << this << " Depth:" << path.size()
411 << " Node:" << node << " Multivisit:" << multivisit
412 << " Maxvisit:" << maxvisit << " NNQueried:" << nn_queried
413 << " TTHit:" << is_tt_hit << " CacheHit:" << is_cache_hit
414 << " Collision:" << is_collision << " OOO:" << ooo_completed
415 << " Repetitions:" << repetitions << " Path:";
416 for (auto it = path.cbegin(); it != path.cend(); ++it) {
417 if (it != path.cbegin()) oss << "->";
418 auto n = std::get<0>(*it);
419 const auto& nl = n->GetLowNode();
420 oss << n << ":" << n->GetNInFlight();
421 if (nl) {
422 oss << "(" << nl << ")";
423 }
424 }
425 oss << " --- " << std::get<0>(path.back())->DebugString();
426 if (node->GetLowNode())
427 oss << " --- " << node->GetLowNode()->DebugString();
428
429 return oss.str();
430 }
