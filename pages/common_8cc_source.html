88void PopulateCommonStopperOptions(RunType for_what, OptionsParser* options) {
89 options->Add<IntOption>(kKLDGainAverageIntervalId, 1, 10000000) = 100;
90 options->Add<FloatOption>(kMinimumKLDGainPerNodeId, 0.0f, 1.0f) = 0.0f;
91 options->Add<FloatOption>(kSmartPruningFactorId, 0.0f, 10.0f) =

94 options->Add<BoolOption>(kNodesAsPlayoutsId) = false;
95 options->Add<IntOption>(kRamLimitMbId, 0, 100000000) = 0;

98// Parameters needed for selfplay and uci, but not benchmark nor infinite mode.
99void PopulateIntrinsicStoppers(ChainedSearchStopper* stopper,
100 const OptionsDict& options) {
101 // KLD gain.
102 const auto min_kld_gain = options.Get<float>(kMinimumKLDGainPerNodeId);
103 if (min_kld_gain > 0.0f) {
104 stopper->AddStopper(std::make_unique<KldGainStopper>(
105 min_kld_gain, options.Get<int>(kKLDGainAverageIntervalId)));
106 }
107
108 // Should be last in the chain.
109 const auto smart_pruning_factor = options.Get<float>(kSmartPruningFactorId);
110 if (smart_pruning_factor > 0.0f) {
111 stopper->AddStopper(std::make_unique<SmartPruningStopper>(
112 smart_pruning_factor, options.Get<int>(kMinimumSmartPruningBatchesId)));
113 }
114}

117// Stoppers for uci mode only.
118// Adds standard UCI search controls like depth, nodes, move time, mate search,
119// and memory limits to the stopper chain.
120void PopulateCommonUciStoppers(ChainedSearchStopper* stopper,
121 const OptionsDict& options,
122 const GoParams& params, size_t total_memory,
123 size_t avg_node_size, uint32_t nodes,
124 int64_t move_overhead) {
125 const bool infinite = params.infinite || params.ponder || params.mate;
126
127 // RAM limit watching stopper.
128 const int ram_limit_mb = options.Get<int>(kRamLimitMbId);
129 if (ram_limit_mb) {
130 stopper->AddStopper(std::make_unique<MemoryWatchingStopper>(
131 ram_limit_mb, total_memory, avg_node_size, nodes,
132 options.Get<float>(kSmartPruningFactorId) > 0.0f));
133 }
134
135 // "go nodes" stopper.
136 int64_t node_limit = 4000000000;
137 if (params.nodes.has_value()) {
138 if (options.Get<bool>(kNodesAsPlayoutsId)) {
139 stopper->AddStopper(std::make_unique<PlayoutsStopper>(
140 *params.nodes, options.Get<float>(kSmartPruningFactorId) > 0.0f));
141 } else {
142 node_limit = *params.nodes;
143 }
144 }
145 // Always limit nodes to avoid exceeding the limit 4000000000.
146 stopper->AddStopper(std::make_unique<VisitsStopper>(
147 node_limit, options.Get<float>(kSmartPruningFactorId) > 0.0f));
148
149 // "go movetime" stopper.
150 if (params.movetime && !infinite) {
151 stopper->AddStopper(
152 std::make_unique<TimeLimitStopper>(*params.movetime - move_overhead));
153 }
154
155 // "go depth" stopper.
156 if (params.depth) {
157 stopper->AddStopper(std::make_unique<DepthStopper>(*params.depth));
158 }
159
160 // "go mate" stopper.
161 if (params.mate) {
162 stopper->AddStopper(std::make_unique<MateStopper>(*params.mate));
163 }
164
165 // Add internal search tree stoppers when we want to automatically stop.
166 if (!infinite) PopulateIntrinsicStoppers(stopper, options);
167}

172class CommonTimeManager : public TimeManager {
173 public:
174 CommonTimeManager(std::unique_ptr<TimeManager> child_mgr,
175 const OptionsDict& options, int64_t move_overhead)
176 : child_mgr_(std::move(child_mgr)),
177 options_(options),
178 move_overhead_(move_overhead) {}
179
180 private:
181 std::unique_ptr<SearchStopper> GetStopper(const GoParams& params,
182 const Position& position,
183 size_t avg_node_size,
184 size_t total_memory,
185 uint32_t nodes) override {
186 auto result = std::make_unique<ChainedSearchStopper>();
187 if (child_mgr_)
188 result->AddStopper(child_mgr_->GetStopper(params, position, avg_node_size,
189 total_memory, nodes));
190 PopulateCommonUciStoppers(result.get(), options_, params, avg_node_size,
191 total_memory, nodes, move_overhead_);
192 return result;
193 }
194
195 const std::unique_ptr<TimeManager> child_mgr_;
196 const OptionsDict& options_;
197 const int64_t move_overhead_;
198};
