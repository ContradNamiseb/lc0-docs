54// Terminology:
55// * Edge - a potential edge with a move and policy information.
56// * Node - an existing edge with number of visits and evaluation.
57// * LowNode - a node with number of visits, evaluation and edges.
58//
59// Storage:
60// * Potential edges are stored in a simple array inside the LowNode as edges_.
61// * Existing edges are stored in a linked list starting with a child_ pointer
62// in the LowNode and continuing with a sibling_ pointer in each Node.
63// * Existing edges have a copy of their potential edge counterpart, index_
64// among potential edges and are linked to the target LowNode via the
65// low_node_ pointer.
66//
67// Example:
68// LowNode
69// |
70// +-------------+-------------+----------------+--------------+
71// | | | | |
72// Edge 0(Nf3) Edge 1(Bc5) Edge 2(a4) Edge 3(Qxf7) Edge 4(a3)
73// (dangling) | (dangling) | (dangling)
74// Node, Q=0.5 Node, Q=-0.2
75//
76// Is represented as:
77// +-----------------+
78// | LowNode |
79// +-----------------+ +--------+
80// | edges_ | -------------------------------------> | Edge[] |
81// | | +------------+ +--------+
82// | child_ | -> | Node | | Nf3 |
83// | | +------------+ | Bc5 |
84// | ... | | edge_ | | a4 |
85// | | | index_ = 1 | | Qxf7 |
86// | | | q_ = 0.5 | +------------+ | a3 |
87// | | | sibling_ | -> | Node | +--------+
88// | | +------------+ +------------+
89// | | | edge_ |
90// +-----------------+ | index_ = 3 |
91// | q_ = -0.2 |
92// | sibling_ | -> nullptr
93// +------------+

103// Atomic unique_ptr based on the public domain code from
104// https://stackoverflow.com/a/42811152 .
107template <class T>
108class atomic_unique_ptr {
109 using pointer = T*;
110 using unique_pointer = std::unique_ptr<T>;
111
112 public:
113 // Manage no pointer.
114 constexpr atomic_unique_ptr() noexcept : ptr() {}
115
116 // Make pointer @p managed.
117 explicit atomic_unique_ptr(pointer p) noexcept : ptr(p) {}
118
119 // Move the managed pointer ownership from another atomic_unique_ptr.
120 atomic_unique_ptr(atomic_unique_ptr&& p) noexcept : ptr(p.release()) {}
121 // Move the managed pointer ownership from another atomic_unique_ptr.
122 atomic_unique_ptr& operator=(atomic_unique_ptr&& p) noexcept {
123 reset(p.release());
124 return *this;
125 }
126
127 // Move the managed object ownership from a unique_ptr.
128 atomic_unique_ptr(unique_pointer&& p) noexcept : ptr(p.release()) {}
129 // Move the managed object ownership from a unique_ptr.
130 atomic_unique_ptr& operator=(unique_pointer&& p) noexcept {
131 reset(p.release());
132 return *this;
133 }
134
135 // Replace the managed pointer, deleting the old one.
136 void reset(pointer p = pointer()) noexcept {
137 auto old = ptr.exchange(p, std::memory_order_acq_rel);
138 if (old) delete old;
139 }
140 // Release ownership of and delete the owned pointer.
141 ~atomic_unique_ptr() { reset(); }
142
143 // Returns the managed pointer.
144 operator pointer() const noexcept { return get(); }
145 // Returns the managed pointer.
146 pointer operator->() const noexcept { return get(); }
147 // Returns the managed pointer.
148 pointer get() const noexcept { return ptr.load(std::memory_order_acquire); }
149
150 // Checks whether there is a managed pointer.
151 explicit operator bool() const noexcept { return get() != pointer(); }
152
153 // Replace the managed pointer, only releasing returning the old one.
154 pointer set(pointer p = pointer()) noexcept {
155 return ptr.exchange(p, std::memory_order_acq_rel);
156 }
157 // Return the managed pointer and release its ownership.
158 pointer release() noexcept { return set(pointer()); }
159
160 // Move managed pointer from @source, iff the managed pointer equals
161 // @expected.
162 bool compare_exchange(pointer& expected,
163 atomic_unique_ptr<T>& source) noexcept {
164 if (ptr.compare_exchange_strong(expected, source.get(),
165 std::memory_order_acq_rel)) {
166 source.release();
167 return true;
168 } else {
169 return false;
170 }
171 }
172
173 private:
174 std::atomic<pointer> ptr;
175};

882 void operator++() {
883 // If it was the last edge in array, become end(), otherwise advance.
884 if (++current_idx_ == total_count_) {
885 edge_ = nullptr;
886 } else {
887 ++edge_;
888 Actualize();
889 }
890 }
891 Edge_Iterator& operator*() { return *this; }
892
893 // If there is node, return it. Otherwise spawn a new one and return it.
894 Node* GetOrSpawnNode(Node* parent) {
895 if (node_) return node_; // If there is already a node, return it.
896
897 // We likely need to add a new node, prepare it now.
898 auto low_parent = parent->GetLowNode()->GetEdgeAt(current_idx_);
899 atomic_unique_ptr<Node> new_node =
900 std::make_unique<Node>(low_parent, current_idx_);
901 while (true) {
902 auto node = Actualize(); // But maybe other thread already did that.
903 if (node_) return node_; // If it did, return.
904
905 // New node needs to be added, but we might be in a race with another
906 // thread doing what we do or adding a different index to the same
907 // sibling.
908
909 // Suppose there are nodes with idx 3 and 7, and we want to insert one
910 // with idx 5. Here is how it looks like:
911 // node_ptr_ -> &Node(idx_.3).sibling_ -> Node(idx_.7)
912 // Here is how we do that:
913 // 1. Store pointer to a node idx_.7:
914 // node_ptr_ -> &Node(idx_.3).sibling_ -> nullptr
915 // tmp -> Node(idx_.7)
916 // 2. Create fresh Node(idx_.5):
917 // node_ptr_ -> &Node(idx_.3).sibling_ -> Node(idx_.5)
918 // tmp -> Node(idx_.7)
919 // 3. Attach stored pointer back to a list:
920 // node_ptr_ ->
921 // &Node(idx_.3).sibling_ -> Node(idx_.5).sibling_ -> Node(idx_.7)
922
923 // Atomically add the new node into the right place.
924 // Set new node's sibling to the expected sibling seen by Actualize in
925 // node_ptr_.
926 auto new_sibling = new_node->GetSibling();
927 new_sibling->set(node);
928 // Try to atomically insert the new node and stop if it works.
929 if (node_ptr_->compare_exchange(node, new_node)) break;
930 // Recover from failure and try again.
931 // Release expected sibling to avoid double free.
932 new_sibling->release();
933 }
934 // 4. Actualize:
935 // node_ -> &Node(idx_.5)
936 // node_ptr_ -> &Node(idx_.5).sibling_ -> Node(idx_.7)
937 Actualize();
938 return node_;
939 }
940
941 private:
942 // Moves node_ptr_ as close as possible to the target index and returns the
943 // contents of node_ptr_ for use by atomic insert in GetOrSpawnNode.
944 Node* Actualize() {
945 // If node_ptr_ is behind, advance it.
946 // This is needed (and has to be 'while' rather than 'if') as other threads
947 // could spawn new nodes between &node_ptr_ and *node_ptr_ while we didn't
948 // see.
949 // Read the direct pointer just once as other threads may change it between
950 // uses.
951 auto node = node_ptr_->get();
952 while (node != nullptr && node->Index() < current_idx_) {
953 node_ptr_ = node->GetSibling();
954 node = node_ptr_->get();
955 }
956 // If in the end node_ptr_ points to the node that we need, populate node_
957 // and advance node_ptr_.
958 if (node != nullptr && node->Index() == current_idx_) {
959 node_ = node;
960 node_ptr_ = node->GetSibling();
961 } else {
962 node_ = nullptr;
963 }
964
965 return node;
966 }
967
968 // Pointer to a pointer to the next node. Has to be a pointer to pointer
969 // as we'd like to update it when spawning a new node.
970 Ptr node_ptr_;
971 uint16_t current_idx_ = 0;
972 uint16_t total_count_ = 0;
973};
