39// A hash-keyed cache. Thread-safe. Takes ownership of all values, which are
40// deleted upon eviction; thus, using values stored requires pinning them, which
41// in turn requires Unpin()ing them after use. The use of HashKeyedCacheLock is
42// recommend to automate this element-memory management.
43// Unlike LRUCache, doesn't even consider trying to support LRU order.
44// Does not support delete.
45// Does not support replace! Inserts to existing elements are silently ignored.
46// FIFO eviction.
47// Assumes that eviction while pinned is rare enough to not need to optimize
48// unpin for that case.
49template <class V>
50class HashKeyedCache {
51 static const double constexpr kLoadFactor = 1.9;
52
53 public:
54 HashKeyedCache(int capacity = 128)
55 : capacity_(capacity),
56 hash_(static_cast<size_t>(capacity * kLoadFactor + 1)) {}
57
58 ~HashKeyedCache() {
59 EvictToCapacity(0);
60 assert(size_ == 0);
61 assert(allocated_ == 0);
62 }
63
64 // Inserts the element under key @key with value @val. Unless the key is
65 // already in the cache.
66 void Insert(uint64_t key, std::unique_ptr<V> val) {
67 if (capacity_.load(std::memory_order_relaxed) == 0) return;
68
69 SpinMutex::Lock lock(mutex_);
70
71 size_t idx = key % hash_.size();
72 while (true) {
73 if (!hash_[idx].in_use) break;
74 if (hash_[idx].key == key) {
75 // Already exists.
76 return;
77 }
78 ++idx;
79 if (idx >= hash_.size()) idx -= hash_.size();
80 }
81 hash_[idx].key = key;
82 hash_[idx].value = std::move(val);
83 hash_[idx].pins = 0;
84 hash_[idx].in_use = true;
85 insertion_order_.push_back(key);
86 ++size_;
87 ++allocated_;
88
89 EvictToCapacity(capacity_);
90 }
91
92 // Checks whether a key exists. Doesn't pin. Of course the next moment the
93 // key may be evicted.
94 bool ContainsKey(uint64_t key) {
95 if (capacity_.load(std::memory_order_relaxed) == 0) return false;
96
97 SpinMutex::Lock lock(mutex_);
98 size_t idx = key % hash_.size();
99 while (true) {
100 if (!hash_[idx].in_use) break;
101 if (hash_[idx].key == key) {
102 return true;
103 }
104 ++idx;
105 if (idx >= hash_.size()) idx -= hash_.size();
106 }
107 return false;
108 }
109
110 // Looks up and pins the element by key. Returns nullptr if not found.
111 // If found, a call to Unpin must be made for each such element.
112 // Use of HashedKeyCacheLock is recommended to automate this pin management.
113 V* LookupAndPin(uint64_t key) {
114 if (capacity_.load(std::memory_order_relaxed) == 0) return nullptr;
115
116 SpinMutex::Lock lock(mutex_);
117
118 size_t idx = key % hash_.size();
119 while (true) {
120 if (!hash_[idx].in_use) break;
121 if (hash_[idx].key == key) {
122 ++hash_[idx].pins;
123 return hash_[idx].value.get();
124 }
125 ++idx;
126 if (idx >= hash_.size()) idx -= hash_.size();
127 }
128 return nullptr;
129 }
130
131 // Unpins the element given key and value. Use of HashedKeyCacheLock is
132 // recommended to automate this pin management.
133 void Unpin(uint64_t key, V* value) {
134 SpinMutex::Lock lock(mutex_);
135
136 // Checking evicted list first.
137 for (auto it = evicted_.begin(); it != evicted_.end(); ++it) {
138 auto& entry = *it;
139 if (key == entry.key && value == entry.value.get()) {
140 if (--entry.pins == 0) {
141 --allocated_;
142 evicted_.erase(it);
143 return;
144 } else {
145 return;
146 }
147 }
148 }
149 // Now the main list.
150 size_t idx = key % hash_.size();
151 while (true) {
152 if (!hash_[idx].in_use) break;
153 if (hash_[idx].key == key && hash_[idx].value.get() == value) {
154 --hash_[idx].pins;
155 return;
156 }
157 ++idx;
158 if (idx >= hash_.size()) idx -= hash_.size();
159 }
160 assert(false);
161 }
162
163 // Sets the capacity of the cache. If new capacity is less than current size
164 // of the cache, oldest entries are evicted. In any case the hashtable is
165 // rehashed.
166 void SetCapacity(int capacity) {
167 // This is the one operation that can be expected to take a long time, which
168 // usually means a SpinMutex is not a great idea. However we should only
169 // very rarely have any contention on the lock while this function is
170 // running, since its called very rarely and almost always before things
171 // start happening.
172 SpinMutex::Lock lock(mutex_);
173
174 if (capacity_.load(std::memory_order_relaxed) == capacity) return;
175 EvictToCapacity(capacity);
176 capacity_.store(capacity);
177
178 std::vector<Entry> new_hash(
179 static_cast<size_t>(capacity * kLoadFactor + 1));
180
181 if (size_ != 0) {
182 for (Entry& item : hash_) {
183 if (!item.in_use) continue;
184 size_t idx = item.key % new_hash.size();
185 while (true) {
186 if (!new_hash[idx].in_use) break;
187 ++idx;
188 if (idx >= new_hash.size()) idx -= new_hash.size();
189 }
190 new_hash[idx].key = item.key;
191 new_hash[idx].value = std::move(item.value);
192 new_hash[idx].pins = item.pins;
193 new_hash[idx].in_use = true;
194 }
195 }
196 hash_.swap(new_hash);
197 }
198
199 // Clears the cache;
200 void Clear() {
201 SpinMutex::Lock lock(mutex_);
202 EvictToCapacity(0);
203 }
204
205 int GetSize() const {
206 SpinMutex::Lock lock(mutex_);
207 return size_;
208 }
209 int GetCapacity() const { return capacity_.load(std::memory_order_relaxed); }
210 static constexpr size_t GetItemStructSize() { return sizeof(Entry); }
211
212 private:
213 struct Entry {
214 Entry() {}
215 Entry(uint64_t key, std::unique_ptr<V> value)
216 : key(key), value(std::move(value)) {}
217 uint64_t key;
218 std::unique_ptr<V> value;
219 int pins = 0;
220 bool in_use = false;
221 };
222
223 void EvictItem() REQUIRES(mutex_) {
224 --size_;
225 uint64_t key = insertion_order_.front();
226 insertion_order_.pop_front();
227 size_t idx = key % hash_.size();
228 while (true) {
229 if (hash_[idx].in_use && hash_[idx].key == key) {
230 break;
231 }
232 ++idx;
233 if (idx >= hash_.size()) idx -= hash_.size();
234 }
235 if (hash_[idx].pins == 0) {
236 --allocated_;
237 hash_[idx].value.reset();
238 hash_[idx].in_use = false;
239 } else {
240 evicted_.emplace_back(hash_[idx].key, std::move(hash_[idx].value));
241 evicted_.back().pins = hash_[idx].pins;
242 hash_[idx].pins = 0;
243 hash_[idx].in_use = false;
244 }
245 size_t next = idx + 1;
246 if (next >= hash_.size()) next -= hash_.size();
247 while (true) {
248 if (!hash_[next].in_use) {
249 break;
250 }
251 size_t target = hash_[next].key % hash_.size();
252 if (!InRange(target, idx + 1, next)) {
253 std::swap(hash_[next], hash_[idx]);
254 idx = next;
255 }
256 ++next;
257 if (next >= hash_.size()) next -= hash_.size();
258 }
259 }
260
261 bool InRange(size_t target, size_t start, size_t end) {
262 if (start <= end) {
263 return target >= start && target <= end;
264 } else {
265 return target >= start || target <= end;
266 }
267 }
268
269 void EvictToCapacity(int capacity) REQUIRES(mutex_) {
270 if (capacity < 0) capacity = 0;
271 while (size_ > capacity) {
272 EvictItem();
273 }
274 }
275
276 std::atomic<int> capacity_;
277 int size_ GUARDED_BY(mutex_) = 0;
278 int allocated_ GUARDED_BY(mutex_) = 0;
279 // Fresh in back, stale at front.
280 std::deque<uint64_t> GUARDED_BY(mutex_) insertion_order_;
281 std::vector<Entry> GUARDED_BY(mutex_) evicted_;
282 std::vector<Entry> GUARDED_BY(mutex_) hash_;
283
284 mutable SpinMutex mutex_;
285};

287// Convenience class for pinning cache items.
288template <class V>
289class HashKeyedCacheLock {
290 public:
291 // Looks up the value in @cache by @key and pins it if found.
292 HashKeyedCacheLock(HashKeyedCache<V>* cache, uint64_t key)
293 : cache_(cache), key_(key), value_(cache->LookupAndPin(key_)) {}
294
295 // Unpins the cache entry (if holds).
296 ~HashKeyedCacheLock() {
297 if (value_) cache_->Unpin(key_, value_);
298 }
299
300 HashKeyedCacheLock(const HashKeyedCacheLock&) = delete;
301
302 // Returns whether lock holds any value.
303 bool holds_value() const { return value_; }
304 operator bool() const { return value_; }
305
306 // Gets the value.
307 V* operator->() const { return value_; }
308 V* operator*() const { return value_; }
309
310 HashKeyedCacheLock() {}
311 HashKeyedCacheLock(HashKeyedCacheLock&& other)
312 : cache_(other.cache_), key_(other.key_), value_(other.value_) {
313 other.value_ = nullptr;
314 }
315 void operator=(HashKeyedCacheLock&& other) {
316 if (value_) cache_->Unpin(key_, value_);
317 cache_ = other.cache_;
318 key_ = other.key_;
319 value_ = other.value_;
320 other.value_ = nullptr;
321 }
322
323 private:
324 HashKeyedCache<V>* cache_ = nullptr;
325 uint64_t key_;
326 V* value_ = nullptr;
327};
