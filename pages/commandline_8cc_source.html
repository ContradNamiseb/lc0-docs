37void CommandLine::Init(int argc, const char** argv) {
38#ifdef _WIN32
39 // Under windows argv[0] may not have the extension. Also _get_pgmptr() had
40 // issues in some windows 10 versions, so check returned values carefully.
41 char* pgmptr = nullptr;
42 if (!_get_pgmptr(&pgmptr) && pgmptr != nullptr && *pgmptr) {
43 binary_ = pgmptr;
44 } else {
45 binary_ = argv[0];
46 }
47#else
48 binary_ = argv[0];
49#endif
50 arguments_.clear();
51 std::ostringstream params;
52 for (int i = 1; i < argc; ++i) {
53 params << ' ' << argv[i];
54 arguments_.push_back(argv[i]);
55 }
56 LOGFILE << "Command line: " << binary_ << params.str();
57}

59bool CommandLine::ConsumeCommand(std::string_view command) {
60 if (arguments_.empty()) return false;
61 if (arguments_[0] != command) return false;
62 arguments_.erase(arguments_.begin());
63 return true;
64}

71std::string CommandLine::BinaryDirectory() {
72 std::string path = binary_;
73 const auto pos = path.find_last_of("\\/");
74 if (pos == std::string::npos) {
75 return ".";
76 }
77 path.resize(pos);
78 return path;
79}
