53void Benchmark::Run(bool run_shorter_benchmark) {
54 OptionsParser options;
55 SharedBackendParams::Populate(&options);
56 options.Add<IntOption>(kThreadsOptionId, 1, 128) = kDefaultThreads;
57 options.GetMutableDefaultsOptions()->Set(SharedBackendParams::kNNCacheSizeId,
58 200000);
59 classic::SearchParams::Populate(&options);
60
61 options.Add<IntOption>(kNodesId, -1, 999999999) = -1;
62 options.Add<StringOption>(kFenId) = "";
63 if (run_shorter_benchmark) {
64 options.Add<IntOption>(kMovetimeId, -1, 999999999) = 500;
65 options.Add<IntOption>(kNumPositionsId, 1, 34) = 10;
66 } else {
67 options.Add<IntOption>(kMovetimeId, -1, 999999999) = 10000;
68 options.Add<IntOption>(kNumPositionsId, 1, 34) = 34;
69 }
70
71 if (!options.ProcessAllFlags()) return;
72
73 try {
74 auto option_dict = options.GetOptionsDict();
75
76 auto backend = CreateMemCache(
77 BackendManager::Get()->CreateFromParams(option_dict), option_dict);
78
79 const int visits = option_dict.Get<int>(kNodesId);
80 const int movetime = option_dict.Get<int>(kMovetimeId);
81 const std::string fen = option_dict.Get<std::string>(kFenId);
82 int num_positions = option_dict.Get<int>(kNumPositionsId);
83
84 std::vector<std::double_t> times;
85 std::vector<std::int64_t> playouts;
86 std::uint64_t cnt = 1;
87
88 if (fen.length() > 0) {
89 positions = {fen};
90 num_positions = 1;
91 }
92 std::vector<std::string> testing_positions(
93 positions.cbegin(), positions.cbegin() + num_positions);
94
95 for (std::string position : testing_positions) {
96 std::cout << "\nPosition: " << cnt++ << "/" << testing_positions.size()
97 << " " << position << std::endl;
98
99 auto stopper = std::make_unique<classic::ChainedSearchStopper>();
100 if (movetime > -1) {
101 stopper->AddStopper(
102 std::make_unique<classic::TimeLimitStopper>(movetime));
103 }
104 if (visits > -1) {
105 stopper->AddStopper(
106 std::make_unique<classic::VisitsStopper>(visits, false));
107 }
108
109 classic::NodeTree tree;
110 std::vector<std::string> moves;
111 if (auto iter = position.find("moves "); iter != std::string::npos) {
112 moves = StrSplitAtWhitespace(position.substr(iter + 6));
113 position = position.substr(0, iter);
114 }
115 tree.ResetToPosition(position, moves);
116
117 const auto start = std::chrono::steady_clock::now();
118 auto search = std::make_unique<classic::Search>(
119 tree, backend.get(),
120 std::make_unique<CallbackUciResponder>(
121 std::bind(&Benchmark::OnBestMove, this, std::placeholders::_1),
122 std::bind(&Benchmark::OnInfo, this, std::placeholders::_1)),
123 MoveList(), start, std::move(stopper), false, false, option_dict,
124 nullptr);
125 search->StartThreads(option_dict.Get<int>(kThreadsOptionId));
126 search->Wait();
127 const auto end = std::chrono::steady_clock::now();
128
129 const auto time =
130 std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
131 times.push_back(time.count());
132 playouts.push_back(search->GetTotalPlayouts());
133 }
134
135 const auto total_playouts =
136 std::accumulate(playouts.begin(), playouts.end(), 0);
137 const auto total_time = std::accumulate(times.begin(), times.end(), 0);
138 std::cout << "\n==========================="
139 << "\nTotal time (ms) : " << total_time
140 << "\nNodes searched : " << total_playouts
141 << "\nNodes/second : "
142 << std::lround(1000.0 * total_playouts / (total_time + 1))
143 << std::endl;
144 } catch (Exception& ex) {
145 std::cerr << ex.what() << std::endl;
146 }
147}

153void Benchmark::OnInfo(const std::vector<ThinkingInfo>& infos) {
154 std::string line = "Benchmark time " + std::to_string(infos[0].time);
155 line += " ms, " + std::to_string(infos[0].nodes) + " nodes, ";
156 line += std::to_string(infos[0].nps) + " nps";
157 if (!infos[0].pv.empty()) line += ", move " + infos[0].pv[0].ToString(true);
158 std::cout << line << std::endl;
159}
