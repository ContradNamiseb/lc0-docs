37// The Layer objects only hold memory for weights, biases, etc
38// memory for input and output tensors is provided by caller of Eval.

40class BaseLayer {
41 public:
42 int GetC() const { return C; }
43 int GetH() const { return H; }
44 int GetW() const { return W; }
45
46 BaseLayer(int c, int h, int w, BaseLayer* ip);
47 virtual ~BaseLayer() = default;
48 size_t GetOutputSize(int N) const { return sizeof(float) * N * C * H * W; }
49 void SetDataType(dnnl::memory::data_type type) { data_type_ = type; }
50 void SetConvolutionType(dnnl::algorithm type) { convolution_type_ = type; }
51 virtual void Eval(int N, dnnl::memory& output, dnnl::memory& input,
52 dnnl::engine& eng, dnnl::stream& stream) = 0;
53
54 protected:
55 BaseLayer* input_;
56
57 int C; // Output tensor dimensions.
58 int H;
59 int W;
60 dnnl::memory::data_type data_type_;
61 dnnl::algorithm convolution_type_;
62 std::mutex lock_;
63};
