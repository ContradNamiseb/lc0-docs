94// GenericOnnxConst takes inline constant (usually short and known at compile
95// time) and converts to initializer for OnnxBuilder.
96template <typename T>
97class GenericOnnxConst : public OnnxConst {
98 public:
99 GenericOnnxConst(const std::vector<T> data, std::initializer_list<int> dims)
100 : data_(data), dims_(dims) {}
101
102 private:
103 std::vector<int> GetDimensions() const override { return dims_; }
104 std::string GetRawData() const override {
105 return {reinterpret_cast<const char*>(data_.data()),
106 reinterpret_cast<const char*>(data_.data() + data_.size())};
107 }
108
109 std::vector<T> data_;
110 std::vector<int> dims_;
111};
