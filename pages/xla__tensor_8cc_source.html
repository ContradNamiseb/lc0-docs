37std::string AsHexString(std::string_view buf) {
38 std::string result;
39 result.reserve(buf.size() * 2);
40 constexpr char hex[] = "0123456789abcdef";
41 for (unsigned char c : buf) {
42 result.push_back(hex[c >> 4]);
43 result.push_back(hex[c & 0xf]);
44 }
45 return result;
46}

50std::string XlaTensor::DebugString() {
51 constexpr size_t kMaxSize = 1000;
52 constexpr size_t kSuffixSize = 200;
53 std::string result = "XlaTensor(";
54 result += "shape=[";
55 for (size_t i = 0; i < shape().size(); ++i) {
56 if (i > 0) result += ", ";
57 result += std::to_string(shape()[i]);
58 }
59 result += "], type=";
60 result += pblczero::XlaShapeProto::Type_Name(type());
61 result += ") size=" + std::to_string(size());
62 result += " data=";
63 if (size() <= kMaxSize) {
64 result += AsHexString({static_cast<const char*>(data()), size()});
65 } else {
66 result += AsHexString(
67 {static_cast<const char*>(data()), kMaxSize - kSuffixSize - 2});
68 result += "....";
69 result += AsHexString(
70 {static_cast<const char*>(data()) + size() - kSuffixSize, kSuffixSize});
71 }
72 return result;
73}

95void XlaMutableTensor::Cast(pblczero::XlaShapeProto::Type new_type) {
96 if (new_type == type_) return;
97 const size_t new_size = GetBufferSize(new_type, shape_);
98 std::unique_ptr<char[]> new_data;
99 const void* src = data_.get();
100 if (new_size > capacity_ ||
101 GetXlaTypeSize(new_type) > GetXlaTypeSize(type_)) {
102 capacity_ = std::max(new_size, capacity_);
103 new_data.reset(new char[capacity_]);
104 std::swap(data_, new_data);
105 }
106 void* dst = data_.get();
107 if (new_type != pblczero::XlaShapeProto::F32 &&
108 type_ != pblczero::XlaShapeProto::F32) {
109 throw Exception(
110 "Only float32 casts are supported, attempting to cast from " +
111 pblczero::XlaShapeProto::Type_Name(type_) + " to " +
112 pblczero::XlaShapeProto::Type_Name(new_type));
113 }
114 auto convert = [&](auto&& func) {
115 using src_t = typename DeduceType<decltype(func)>::arg;
116 using dst_t = typename DeduceType<decltype(func)>::result;
117 const size_t count = std::accumulate(shape_.begin(), shape_.end(), 1,
118 std::multiplies<int64_t>());
119 const src_t* src_ptr = static_cast<const src_t*>(src);
120 dst_t* dst_ptr = static_cast<dst_t*>(dst);
121 for (size_t i = 0; i < count; ++i) dst_ptr[i] = func(src_ptr[i]);
122 };
123 if (type_ == pblczero::XlaShapeProto::F32) {
124 switch (new_type) {
125 case pblczero::XlaShapeProto::F16:
126 convert(FP32toFP16);
127 break;
128 case pblczero::XlaShapeProto::BF16:
129 convert(FP32toBF16);
130 break;
131 case pblczero::XlaShapeProto::F8E5M2:
132 convert(FP32toFP8E5M2_Saturate);
133 break;
134 default:
135 throw Exception("Unsupported cast F32 -> " +
136 pblczero::XlaShapeProto::Type_Name(new_type));
137 }
138 } else {
139 switch (type_) {
140 case pblczero::XlaShapeProto::F16:
141 convert(FP16toFP32);
142 break;
143 case pblczero::XlaShapeProto::BF16:
144 convert(BF16toFP32);
145 break;
146 case pblczero::XlaShapeProto::F8E5M2:
147 convert(FP8E5M2toFP32);
148 break;
149 default:
150 throw Exception("Unsupported cast " +
151 pblczero::XlaShapeProto::Type_Name(type_) + " -> F32");
152 }
153 }
154 size_ = new_size;
155 type_ = new_type;
156}

158pblczero::XlaShapeProto::Type StringToXlaType(const std::string& type) {
159 for (const auto& types : {
160 pblczero::XlaShapeProto::S4,
161 pblczero::XlaShapeProto::S8,
162 pblczero::XlaShapeProto::S16,
163 pblczero::XlaShapeProto::S32,
164 pblczero::XlaShapeProto::S64,
165 pblczero::XlaShapeProto::U4,
166 pblczero::XlaShapeProto::U8,
167 pblczero::XlaShapeProto::U16,
168 pblczero::XlaShapeProto::U32,
169 pblczero::XlaShapeProto::U64,
170 pblczero::XlaShapeProto::F16,
171 pblczero::XlaShapeProto::F32,
172 pblczero::XlaShapeProto::BF16,
173 pblczero::XlaShapeProto::F64,
174 pblczero::XlaShapeProto::F8E5M2,
175 pblczero::XlaShapeProto::F8E4M3FN,
176 pblczero::XlaShapeProto::F8E4M3B11FNUZ,
177 pblczero::XlaShapeProto::F8E5M2FNUZ,
178 pblczero::XlaShapeProto::F8E4M3FNUZ,
179 pblczero::XlaShapeProto::C64,
180 pblczero::XlaShapeProto::C128,
181 }) {
182 if (StringsEqualIgnoreCase(type,
183 pblczero::XlaShapeProto::Type_Name(types))) {
184 return types;
185 }
186 }
187 throw Exception("Cannot convert to XLA type: " + type);
188}
