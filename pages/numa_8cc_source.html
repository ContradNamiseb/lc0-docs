41void Numa::Init() {
42#if defined(_WIN64) && _WIN32_WINNT >= 0x0601
43 SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX* buffer;
44 DWORD len = 0;
45 GetLogicalProcessorInformationEx(RelationProcessorCore, NULL, &len);
46 buffer = static_cast<SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX*>(malloc(len));
47 GetLogicalProcessorInformationEx(RelationProcessorCore, buffer, &len);
48 if (buffer->Processor.Flags & LTP_PC_SMT) {
49 threads_per_core_ = BitBoard(buffer->Processor.GroupMask[0].Mask).count();
50 }
51 free(buffer);
52
53 int group_count = GetActiveProcessorGroupCount();
54 int thread_count = GetActiveProcessorCount(ALL_PROCESSOR_GROUPS);
55 int core_count = thread_count / threads_per_core_;
56 CERR << "Detected " << core_count << " core(s) and " << thread_count
57 << " thread(s) in " << group_count << " group(s).";
58 for (int group_id = 0; group_id < group_count; group_id++) {
59 int group_threads = GetActiveProcessorCount(group_id);
60 int group_cores = group_threads / threads_per_core_;
61 CERR << "Group " << group_id << " has " << group_cores
62 << " core(s) and " << group_threads << " thread(s).";
63 }
64#endif
65}

67void Numa::BindThread(int id) {
68#if defined(_WIN64) && _WIN32_WINNT >= 0x0601
69 int group_count = GetActiveProcessorGroupCount();
70 int thread_count = GetActiveProcessorCount(ALL_PROCESSOR_GROUPS);
71 int core_count = thread_count / threads_per_core_;
72 int core_id = id;
73 GROUP_AFFINITY affinity = {};
74 for (int group_id = 0; group_id < group_count; group_id++) {
75 int group_threads = GetActiveProcessorCount(group_id);
76 int group_cores = group_threads / threads_per_core_;
77 // Allocate cores of each group in order, and distribute remaining threads
78 // to all groups.
79 if ((id < core_count && core_id < group_cores) ||
80 (id >= core_count && (id - core_count) % group_count == group_id)) {
81 affinity.Group = group_id;
82 affinity.Mask = ~0ULL >> (64 - group_threads);
83 SetThreadGroupAffinity(GetCurrentThread(), &affinity, NULL);
84 break;
85 }
86 core_id -= group_cores;
87 }
88#else
89 // Silence warning.
90 (void)id;
91#endif
92}
