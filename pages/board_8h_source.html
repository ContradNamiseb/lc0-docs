166 class Castlings {
167 public:
168 Castlings()
169 : our_queenside_rook(kFileA),
170 their_queenside_rook(kFileA),
171 our_kingside_rook(kFileH),
172 their_kingside_rook(kFileH),
173 data_(0) {}
174
175 void set_we_can_00() { data_ |= 1; }
176 void set_we_can_000() { data_ |= 2; }
177 void set_they_can_00() { data_ |= 4; }
178 void set_they_can_000() { data_ |= 8; }
179
180 void reset_we_can_00() { data_ &= ~1; }
181 void reset_we_can_000() { data_ &= ~2; }
182 void reset_they_can_00() { data_ &= ~4; }
183 void reset_they_can_000() { data_ &= ~8; }
184
185 bool we_can_00() const { return data_ & 1; }
186 bool we_can_000() const { return data_ & 2; }
187 bool they_can_00() const { return data_ & 4; }
188 bool they_can_000() const { return data_ & 8; }
189 bool no_legal_castle() const { return data_ == 0; }
190
191 void Mirror() {
192 std::swap(our_queenside_rook, their_queenside_rook);
193 std::swap(our_kingside_rook, their_kingside_rook);
194 data_ = ((data_ & 0b11) << 2) + ((data_ & 0b1100) >> 2);
195 }
196
197 // Note: this is not a strict xfen compatible output. Without access to the
198 // board its not possible to know whether there is ambiguity so all cases
199 // with any non-standard rook positions are encoded in the x-fen format
200 std::string as_string() const {
201 if (data_ == 0) return "-";
202 std::string result;
203 if (our_queenside_rook == kFileA && our_kingside_rook == kFileH &&
204 their_queenside_rook == kFileA && their_kingside_rook == kFileH) {
205 if (we_can_00()) result += 'K';
206 if (we_can_000()) result += 'Q';
207 if (they_can_00()) result += 'k';
208 if (they_can_000()) result += 'q';
209 } else {
210 if (we_can_00()) result += our_kingside_rook.ToString(true);
211 if (we_can_000()) result += our_queenside_rook.ToString(true);
212 if (they_can_00()) result += their_kingside_rook.ToString(false);
213 if (they_can_000()) result += their_queenside_rook.ToString(false);
214 }
215 return result;
216 }
217
218 std::string DebugString() const {
219 std::string result;
220 if (data_ == 0) result = "-";
221 if (we_can_00()) result += 'K';
222 if (we_can_000()) result += 'Q';
223 if (they_can_00()) result += 'k';
224 if (they_can_000()) result += 'q';
225 result += '[';
226 result += our_queenside_rook.ToString(true);
227 result += our_kingside_rook.ToString(true);
228 result += their_queenside_rook.ToString(false);
229 result += their_kingside_rook.ToString(false);
230 result += ']';
231 return result;
232 }
233
234 uint8_t as_int() const { return data_; }
235 bool operator==(const Castlings& other) const = default;
236
237 // Position of "left" (queenside) rook in starting game position.
238 File our_queenside_rook;
239 File their_queenside_rook;
240 // Position of "right" (kingside) rook in starting position.
241 File our_kingside_rook;
242 File their_kingside_rook;
243
244 private:
245 // - Bit 0 -- "our" side's kingside castle.
246 // - Bit 1 -- "our" side's queenside castle.
247 // - Bit 2 -- opponent's side's kingside castle.
248 // - Bit 3 -- opponent's side's queenside castle.
249 uint8_t data_;
250 };
