106 // Parses a move from move_str.
107 // The input string should be in the "normal" notation rather than from the
108 // player to move, i.e. "e7e5" for the black pawn move.

114 uint64_t Hash() const {
115 return HashCat({our_pieces_.as_int(), their_pieces_.as_int(),
116 rooks_.as_int(), bishops_.as_int(), pawns_.as_int(),
117 (static_cast<uint32_t>(our_king_.as_idx()) << 24) |
118 (static_cast<uint32_t>(their_king_.as_idx()) << 16) |
119 (static_cast<uint32_t>(castlings_.as_int()) << 8) |
120 static_cast<uint32_t>(flipped_)});
121 }

123 class Castlings {
124 public:
125 Castlings()
126 : our_queenside_rook(kFileA),
127 their_queenside_rook(kFileA),
128 our_kingside_rook(kFileH),
129 their_kingside_rook(kFileH),
130 data_(0) {}
131
132 void set_we_can_00() { data_ |= 1; }
133 void set_we_can_000() { data_ |= 2; }
134 void set_they_can_00() { data_ |= 4; }
135 void set_they_can_000() { data_ |= 8; }
136
137 void reset_we_can_00() { data_ &= ~1; }
138 void reset_we_can_000() { data_ &= ~2; }
139 void reset_they_can_00() { data_ &= ~4; }
140 void reset_they_can_000() { data_ &= ~8; }
141
142 bool we_can_00() const { return data_ & 1; }
143 bool we_can_000() const { return data_ & 2; }
144 bool they_can_00() const { return data_ & 4; }
145 bool they_can_000() const { return data_ & 8; }
146 bool no_legal_castle() const { return data_ == 0; }
147
148 void Mirror() {
149 std::swap(our_queenside_rook, their_queenside_rook);
150 std::swap(our_kingside_rook, their_kingside_rook);
151 data_ = ((data_ & 0b11) << 2) + ((data_ & 0b1100) >> 2);
152 }
153
154 // Note: this is not a strict xfen compatible output. Without access to the
155 // board its not possible to know whether there is ambiguity so all cases
156 // with any non-standard rook positions are encoded in the x-fen format
157 std::string as_string() const {
158 if (data_ == 0) return "-";
159 std::string result;
160 if (our_queenside_rook == kFileA && our_kingside_rook == kFileH &&
161 their_queenside_rook == kFileA && their_kingside_rook == kFileH) {
162 if (we_can_00()) result += 'K';
163 if (we_can_000()) result += 'Q';
164 if (they_can_00()) result += 'k';
165 if (they_can_000()) result += 'q';
166 } else {
167 if (we_can_00()) result += our_kingside_rook.ToString(true);
168 if (we_can_000()) result += our_queenside_rook.ToString(true);
169 if (they_can_00()) result += their_kingside_rook.ToString(false);
170 if (they_can_000()) result += their_queenside_rook.ToString(false);
171 }
172 return result;
173 }
174
175 std::string DebugString() const {
176 std::string result;
177 if (data_ == 0) result = "-";
178 if (we_can_00()) result += 'K';
179 if (we_can_000()) result += 'Q';
180 if (they_can_00()) result += 'k';
181 if (they_can_000()) result += 'q';
182 result += '[';
183 result += our_queenside_rook.ToString(true);
184 result += our_kingside_rook.ToString(true);
185 result += their_queenside_rook.ToString(false);
186 result += their_kingside_rook.ToString(false);
187 result += ']';
188 return result;
189 }
190
191 uint8_t as_int() const { return data_; }
192 bool operator==(const Castlings& other) const = default;
193
194 // Position of "left" (queenside) rook in starting game position.
195 File our_queenside_rook;
196 File their_queenside_rook;
197 // Position of "right" (kingside) rook in starting position.
198 File our_kingside_rook;
199 File their_kingside_rook;
200
201 private:
202 // - Bit 0 -- "our" side's kingside castle.
203 // - Bit 1 -- "our" side's queenside castle.
204 // - Bit 2 -- opponent's side's kingside castle.
205 // - Bit 3 -- opponent's side's queenside castle.
206 uint8_t data_;
207 };

212 BitBoard theirs() const { return their_pieces_; }
213 BitBoard pawns() const { return pawns_ & kPawnMask; }
214 BitBoard en_passant() const { return pawns_ - kPawnMask; }
215 BitBoard bishops() const { return bishops_ - rooks_; }

217 BitBoard queens() const { return rooks_ & bishops_; }
218 BitBoard knights() const {

220 rooks_ - bishops_;
221 }
222 BitBoard kings() const {
223 return BitBoard::FromSquare(our_king_) | BitBoard::FromSquare(their_king_);
224 }
225 const Castlings& castlings() const { return castlings_; }
226 bool flipped() const { return flipped_; }

228 bool operator==(const ChessBoard& other) const = default;
229 bool operator!=(const ChessBoard& other) const = default;

231 private:
232 // Sets the piece on the square.
233 void PutPiece(Square square, PieceType piece, bool is_theirs);
234 // Check internal state is consistent after state transformations.
235 bool IsValid() const;
236
237 // All white pieces.
238 BitBoard our_pieces_;
239 // All black pieces.
240 BitBoard their_pieces_;
241 // Rooks and queens.
242 BitBoard rooks_;
243 // Bishops and queens;
244 BitBoard bishops_;
245 // Pawns.
246 // Ranks 1 and 8 have special meaning. Pawn at rank 1 means that
247 // corresponding white pawn on rank 4 can be taken en passant. Rank 8 is the
248 // same for black pawns. Those "fake" pawns are not present in our_pieces_ and
249 // their_pieces_ bitboards.
250 BitBoard pawns_;
251 Square our_king_;
252 Square their_king_;
253 Castlings castlings_;
254 bool flipped_ = false; // aka "Black to move".
255};
