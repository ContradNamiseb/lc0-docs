104 MuxingNetwork(const std::optional<WeightsFile>& weights,
105 const OptionsDict& options) {
106 // int threads, int max_batch)
107 //: network_(std::move(network)), max_batch_(max_batch) {
108
109 const auto parents = options.ListSubdicts();
110 if (parents.empty()) {
111 // If options are empty, or multiplexer configured in root object,
112 // initialize on root object and default backend.
113 auto backends = NetworkFactory::Get()->GetBackendsList();
114 AddBackend(backends[0], weights, options);
115 }
116
117 for (const auto& name : parents) {
118 AddBackend(name, weights, options.GetSubdict(name));
119 }

126 void AddBackend(const std::string& name,
127 const std::optional<WeightsFile>& weights,
128 const OptionsDict& opts) {
129 const int max_batch = opts.GetOrDefault<int>("max_batch", 256);
130 const std::string backend = opts.GetOrDefault<std::string>("backend", name);
131
132 networks_.emplace_back(
133 NetworkFactory::Get()->Create(backend, weights, opts));
134 Network* net = networks_.back().get();
135
136 int nn_threads = opts.GetOrDefault<int>("threads", 0);
137 if (nn_threads == 0) {
138 nn_threads = net->GetThreads();
139 }
140
141 min_batch_size_ = std::min(min_batch_size_, net->GetMiniBatchSize());
142 is_cpu_ &= net->IsCpu();
143
144 if (networks_.size() == 1) {
145 capabilities_ = net->GetCapabilities();
146 } else {
147 capabilities_.Merge(net->GetCapabilities());
148 }
149
150 for (int i = 0; i < nn_threads; ++i) {
151 threads_.emplace_back(
152 [this, net, max_batch]() { Worker(net, max_batch); });
153 }
154 }

170 void Enqueue(MuxingComputation* computation) {
171 std::lock_guard<std::mutex> lock(mutex_);
172 queue_.push(computation);

176 ~MuxingNetwork() {
177 Abort();
178 Wait();

180 while (!queue_.empty()) {
181 queue_.front()->NotifyReady();
182 queue_.pop();
183 }

186 void Worker(Network* network, const int max_batch) {
187 // While Abort() is not called (and it can only be called from destructor).
188 while (!abort_) {
189 std::vector<MuxingComputation*> children;
190 // Create new computation in "upstream" network, to gather batch into
191 // there.
192 std::shared_ptr<NetworkComputation> parent(network->NewComputation());
193 {
194 std::unique_lock<std::mutex> lock(mutex_);
195 // Wait until there's come work to compute.
196 cv_.wait(lock, [&] { return abort_ || !queue_.empty(); });
197 if (abort_) break;
198
199 // While there is a work in queue, add it.
200 while (!queue_.empty()) {
201 // If we are reaching batch size limit, stop adding.
202 // However, if a single input batch is larger than output batch limit,
203 // we still have to add it.
204 if (parent->GetBatchSize() != 0 &&
205 parent->GetBatchSize() + queue_.front()->GetBatchSize() >
206 max_batch) {
207 break;
208 }
209 // Remember which of "input" computations we serve.
210 children.push_back(queue_.front());
211 queue_.pop();
212 // Make "input" computation populate data into output batch.
213 children.back()->PopulateToParent(parent);
214 }
215 }
216
217 // Compute.
218 parent->ComputeBlocking();
219 // Notify children that data is ready!
220 for (auto child : children) child->NotifyReady();
221 }
222 }
223
224 void Abort() {
225 {
226 std::lock_guard<std::mutex> lock(mutex_);
227 abort_ = true;
228 }
229 cv_.notify_all();
230 }
231
232 void Wait() {
233 while (!threads_.empty()) {
234 threads_.back().join();
235 threads_.pop_back();
236 }
237 }
238
239 private:
240 std::vector<std::unique_ptr<Network>> networks_;
241 std::queue<MuxingComputation*> queue_;
242 bool abort_ = false;
243 NetworkCapabilities capabilities_;
244 int min_batch_size_ = std::numeric_limits<int>::max();
245 bool is_cpu_ = true;
246
247 std::mutex mutex_;
248 std::condition_variable cv_;
249
250 std::vector<std::thread> threads_;
251};
