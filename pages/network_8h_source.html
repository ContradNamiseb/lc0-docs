40// All input planes are 64 value vectors, every element of which is either
41// 0 or some value, unique for the plane. Therefore, input is defined as
42// a bitmask showing where to set the value, and the value itself.
47struct InputPlane {
48 InputPlane() = default;
50 void SetAll() { mask = ~0ull; }
53 void Fill(float val) {
54 SetAll();
55 value = val;
56 }
58 std::uint64_t mask = 0ull;
60 float value = 1.0f;
61};
62using InputPlanes = std::vector<InputPlane>;

105// The plan:
106// 1. Search must not look directly into any fields of NetworkFormat anymore.
107// 2. Backends populate NetworkCapabilities that show search how to use NN, both
108// for input and output.
109// 3. Input part of NetworkCapabilities is just copy of InputFormat for now, and
110// is likely to stay so (because search not knowing how to use NN is not very
111// useful), but it's fine if it will change.
112// 4. On the other hand, output part of NetworkCapabilities is set of
113// independent parameters (like WDL, moves left head etc), because search can
114// look what's set and act accordingly. Backends may derive it from
115// output head format fields or other places.
116
122struct NetworkCapabilities {
123 pblczero::NetworkFormat::InputFormat input_format;
124 pblczero::NetworkFormat::OutputFormat output_format;
125 pblczero::NetworkFormat::MovesLeftFormat moves_left;
126 // TODO expose information of whether GetDVal() is usable or always zero.
127
128 // Combines capabilities by setting the most restrictive ones. May throw
129 // exception.
136 void Merge(const NetworkCapabilities& other) {
137 if (input_format != other.input_format) {
138 throw Exception("Incompatible input formats, " +
139 std::to_string(input_format) + " vs " +
140 std::to_string(other.input_format));
141 }
142 if (output_format != other.output_format) {
143 throw Exception("Incompatible output formats, " +
144 std::to_string(output_format) + " vs " +
145 std::to_string(other.output_format));
146 }
147 if (!other.has_mlh()) moves_left = pblczero::NetworkFormat::MOVES_LEFT_NONE;
148 }
149
152 bool has_mlh() const {
153 return moves_left != pblczero::NetworkFormat::MOVES_LEFT_NONE;
154 }
155
158 bool has_wdl() const {
159 return output_format == pblczero::NetworkFormat::OUTPUT_WDL;
160 }
161};
