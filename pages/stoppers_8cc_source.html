94MemoryWatchingStopper::MemoryWatchingStopper(int ram_limit_mb,
95 size_t total_memory,
96 size_t avg_node_size,
97 uint32_t nodes,
98 bool populate_remaining_playouts)
99 : VisitsStopper(
100 [&]() -> size_t {
101 const auto ram_limit = ram_limit_mb * 1000000LL;
102 const auto nodes_memory = avg_node_size * nodes;
103 if (ram_limit + nodes_memory < total_memory) return 0;
104 return (ram_limit + nodes_memory - total_memory) / avg_node_size;
105 }(),
106 populate_remaining_playouts) {
107 LOGFILE << "RAM limit " << ram_limit_mb << "MB. Memory allocated is "
108 << (total_memory - avg_node_size * nodes) / 1000000
109 << "MB. Remaining memory is enough for " << GetVisitsLimit()
110 << " nodes.";
111}

159KldGainStopper::KldGainStopper(float min_gain, int average_interval)
160 : min_gain_(min_gain), average_interval_(average_interval) {}
161
162bool KldGainStopper::ShouldStop(const IterationStats& stats, StoppersHints*) {
163 Mutex::Lock lock(mutex_);
164 const auto new_child_nodes = stats.total_nodes - 1.0;
165 if (new_child_nodes < prev_child_nodes_ + average_interval_) return false;
166
167 const auto new_visits = stats.edge_n;
168 if (!prev_visits_.empty()) {
169 double kldgain = 0.0;
170 for (decltype(new_visits)::size_type i = 0; i < new_visits.size(); i++) {
171 double o_p = prev_visits_[i] / prev_child_nodes_;
172 double n_p = new_visits[i] / new_child_nodes;
173 if (prev_visits_[i] != 0) kldgain += o_p * log(o_p / n_p);
174 }
175 if (kldgain / (new_child_nodes - prev_child_nodes_) < min_gain_) {
176 LOGFILE << "Stopping search: KLDGain per node too small.";
177 return true;
178 }
179 }
180 prev_visits_ = new_visits;
181 prev_child_nodes_ = new_child_nodes;
182 return false;
183}

194SmartPruningStopper::SmartPruningStopper(float smart_pruning_factor,
195 int64_t minimum_batches)
196 : smart_pruning_factor_(smart_pruning_factor),
197 minimum_batches_(minimum_batches) {}
198
199bool SmartPruningStopper::ShouldStop(const IterationStats& stats,
200 StoppersHints* hints) {
201 if (smart_pruning_factor_ <= 0.0) return false;
202 Mutex::Lock lock(mutex_);
203 if (stats.edge_n.size() == 1) {
204 LOGFILE << "Only one possible move. Moving immediately.";
205 return true;
206 }
207 if (stats.edge_n.size() <= static_cast<size_t>(stats.num_losing_edges) +
208 (stats.may_resign ? 0 : 1)) {
209 LOGFILE << "At most one non losing move, stopping search.";
210 return true;
211 }
212 if (stats.win_found) {
213 LOGFILE << "Terminal win found, stopping search.";
214 return true;
215 }
216 if (stats.nodes_since_movestart > 0 && !first_eval_time_) {
217 first_eval_time_ = stats.time_since_movestart;
218 return false;
219 }
220 if (!first_eval_time_) return false;
221 if (stats.edge_n.size() == 0) return false;
222 if (stats.time_since_movestart <
223 *first_eval_time_ + kSmartPruningToleranceMs) {
224 return false;
225 }
226
227 const auto nodes = stats.nodes_since_movestart + kSmartPruningToleranceNodes;
228 const auto time = stats.time_since_movestart - *first_eval_time_;
229 // If nps is populated by someone who knows better, use it. Otherwise use the
230 // value calculated here.
231 const auto nps = hints->GetEstimatedNps().value_or(1000LL * nodes / time + 1);
232
233 const double remaining_time_s = hints->GetEstimatedRemainingTimeMs() / 1000.0;
234 const auto remaining_playouts =
235 std::min(remaining_time_s * nps / smart_pruning_factor_,
236 hints->GetEstimatedRemainingPlayouts() / smart_pruning_factor_);
237
238 // May overflow if (nps/smart_pruning_factor) > 180 000 000, but that's not
239 // very realistic.
240 hints->UpdateEstimatedRemainingPlayouts(remaining_playouts);
241 if (stats.batches_since_movestart < minimum_batches_) return false;
242
243 uint32_t largest_n = 0;
244 uint32_t second_largest_n = 0;
245 for (auto n : stats.edge_n) {
246 if (n > largest_n) {
247 second_largest_n = largest_n;
248 largest_n = n;
249 } else if (n > second_largest_n) {
250 second_largest_n = n;
251 }
252 }
253
254 if (remaining_playouts < (largest_n - second_largest_n)) {
255 LOGFILE << std::fixed << remaining_playouts
256 << " playouts remaining. Best move has " << largest_n
257 << " visits, second best -- " << second_largest_n
258 << ". Difference is " << (largest_n - second_largest_n)
259 << ", so stopping the search after "
260 << stats.batches_since_movestart << " batches.";
261
262 return true;
263 }
264
265 return false;
266}
