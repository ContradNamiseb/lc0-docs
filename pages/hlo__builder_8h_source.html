169// A context class for annotating parts of the HLO computation with metadata,
170// like original ONNX op, its name, and source file name and line.
171// The class saves the current metadata in constructor and restores it in
172// destructor, making it possible to use it in a scoped way.
173class HloContext {
174 public:
175 HloContext(HloBuilder* builder)
176 : builder_(builder), saved_metadata_(builder->metadata_) {}
177 ~HloContext() { builder_->metadata_ = saved_metadata_; }
178 void SetOpType(std::string_view op_type) const {
179 builder_->metadata_.set_op_type(op_type);
180 }
181 void SetOpName(std::string_view op_name) const {
182 builder_->metadata_.set_op_name(op_name);
183 }
184
185 private:
186 HloBuilder* builder_;
187 pblczero::XlaOpMetadata saved_metadata_;
188};
