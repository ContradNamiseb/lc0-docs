95bool ProcessParameters(OptionsParser* options) {
96 options->Add<StringOption>(kInputFilenameId);
97 options->Add<StringOption>(kOutputFilenameId);
98 options->Add<StringOption>(kHloTextOutputFilenameId);
99 options->Add<StringOption>(kHloProtoOutputFilenameId);
100 options->Add<IntOption>(kOnnxBatchSizeId, -1, 2048) = -1;
101 options->Add<IntOption>(kOnnxOpsetId, 7, 18) = 17;
102 options->Add<IntOption>(kOnnxIrId, -1, 10) = -1;
103 options->Add<IntOption>(kHloBatchSizeId, 1, 2048) = 333;
104 options->Add<ChoiceOption>(
105 kOnnxDataTypeId, std::vector<std::string>{"f32", "f16", "bf16"}) = "f32";
106 options->Add<BoolOption>(kHloAllowPartialResultId);
107
108 options->Add<StringOption>(kInputPlanesName) = "/input/planes";
109 options->Add<StringOption>(kOutputPolicyHead) = "/output/policy";
110 options->Add<StringOption>(kOutputWdl) = "/output/wdl";
111 options->Add<StringOption>(kOutputValue) = "/output/value";
112 options->Add<StringOption>(kOutputMlh) = "/output/mlh";
113 options->Add<BoolOption>(kOnnxToPytorch) = false;
114 options->Add<StringOption>(kValueHead) = "winner";
115 options->Add<StringOption>(kPolicyHead) = "vanilla";
116 if (!options->ProcessAllFlags()) return false;
117
118 const OptionsDict& dict = options->GetOptionsDict();
119 dict.EnsureExists<std::string>(kInputFilenameId);
120 if (!dict.OwnExists<std::string>(kOutputFilenameId) &&
121 !dict.OwnExists<std::string>(kHloTextOutputFilenameId) &&
122 !dict.OwnExists<std::string>(kHloProtoOutputFilenameId)) {
123 throw Exception(
124 "At least one of --output, --hlo-text-output or --hlo-proto-output "
125 "must be specified.");
126 }
127 return true;
128}

132void ConvertLeelaToOnnx() {
133 OptionsParser options_parser;
134 if (!ProcessParameters(&options_parser)) return;
135
136 const OptionsDict& dict = options_parser.GetOptionsDict();
137 auto weights_file =
138 LoadWeightsFromFile(dict.Get<std::string>(kInputFilenameId));
139
140 ShowNetworkFormatInfo(weights_file);
141 if (weights_file.has_onnx_model()) {
142 COUT << "The leela network already has ONNX network embedded, extracting.";
143 } else {
144 ShowNetworkWeightsInfo(weights_file);
145 COUT << "Converting Leela network to the ONNX.";
146 WeightsToOnnxConverterOptions onnx_options;
147 onnx_options.input_planes_name = dict.Get<std::string>(kInputPlanesName);
148 onnx_options.output_policy_head = dict.Get<std::string>(kOutputPolicyHead);
149 onnx_options.output_wdl = dict.Get<std::string>(kOutputWdl);
150 onnx_options.output_value = dict.Get<std::string>(kOutputValue);
151 onnx_options.opset = dict.Get<int>(kOnnxOpsetId);
152 onnx_options.ir = dict.Get<int>(kOnnxIrId);
153 onnx_options.batch_size = dict.Get<int>(kOnnxBatchSizeId);
154 onnx_options.data_type = WeightsToOnnxConverterOptions::StringToDataType(
155 dict.Get<std::string>(kOnnxDataTypeId));
156 // onnx2pytorch only needs an alternate layernorm-implementation, so it's
157 // currently only enables that. Might need to be extended in the future.
158 onnx_options.alt_layernorm = dict.Get<bool>(kOnnxToPytorch);
159 onnx_options.value_head = dict.Get<std::string>(kValueHead);
160 onnx_options.policy_head = dict.Get<std::string>(kPolicyHead);
161 weights_file = ConvertWeightsToOnnx(weights_file, onnx_options);
162 }
163
164 const auto& onnx = weights_file.onnx_model();
165 if (dict.OwnExists<std::string>(kOutputFilenameId)) {
166 WriteStringToFile(dict.Get<std::string>(kOutputFilenameId), onnx.model());
167 }
168 if (dict.OwnExists<std::string>(kHloTextOutputFilenameId) ||
169 dict.OwnExists<std::string>(kHloProtoOutputFilenameId)) {
170 Onnx2HloOptions hlo_options;
171 hlo_options.debugging_allow_partial_result =
172 dict.Get<bool>(kHloAllowPartialResultId);
173 pblczero::ModelProto onnx_model;
174 onnx_model.ParseFromString(onnx.model());
175 auto hlo_result = ConvertOnnxToHlo(
176 onnx_model, dict.Get<int>(kHloBatchSizeId), hlo_options);
177 if (dict.OwnExists<std::string>(kHloTextOutputFilenameId)) {
178 std::string filename = dict.Get<std::string>(kHloTextOutputFilenameId);
179 if (filename == "-") {
180 PrettyPrintHlo(hlo_result.hlo_module, {}, std::cout);
181 } else {
182 std::ofstream file(filename.c_str());
183 PrettyPrintHlo(hlo_result.hlo_module, {}, file);
184 }
185 }
186 if (dict.OwnExists<std::string>(kHloProtoOutputFilenameId)) {
187 WriteStringToFile(dict.Get<std::string>(kHloProtoOutputFilenameId),
188 hlo_result.hlo_module.OutputAsString());
189 }
190 }
191 ShowNetworkOnnxInfo(weights_file, false);
192 COUT << "Done.";
193}
