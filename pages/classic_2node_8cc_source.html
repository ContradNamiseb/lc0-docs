47// Node garbage collector

50namespace {
51// Periodicity of garbage collection, milliseconds.
52const int kGCIntervalMs = 100;
53
54// Every kGCIntervalMs milliseconds release nodes in a separate GC thread.
55class NodeGarbageCollector {
56 public:
57 NodeGarbageCollector() : gc_thread_([this]() { Worker(); }) {}
58
60 void AddToGcQueue(std::unique_ptr<Node> node, size_t solid_size = 0) {
61 if (!node) return;
62 Mutex::Lock lock(gc_mutex_);
63 subtrees_to_gc_.emplace_back(std::move(node));
64 subtrees_to_gc_solid_size_.push_back(solid_size);
65 }
66
67 ~NodeGarbageCollector() {
68 // Flips stop flag and waits for a worker thread to stop.
69 stop_.store(true);
70 gc_thread_.join();
71 }
72
73 private:
74 void GarbageCollect() {
75 while (!stop_.load()) {
76 // Node will be released in destructor when mutex is not locked.
77 std::unique_ptr<Node> node_to_gc;
78 size_t solid_size = 0;
79 {
80 // Lock the mutex and move last subtree from subtrees_to_gc_ into
81 // node_to_gc.
82 Mutex::Lock lock(gc_mutex_);
83 if (subtrees_to_gc_.empty()) return;
84 node_to_gc = std::move(subtrees_to_gc_.back());
85 subtrees_to_gc_.pop_back();
86 solid_size = subtrees_to_gc_solid_size_.back();
87 subtrees_to_gc_solid_size_.pop_back();
88 }
89 // Solid is a hack...
90 if (solid_size != 0) {
91 for (size_t i = 0; i < solid_size; i++) {
92 node_to_gc.get()[i].~Node();
93 }
94 std::allocator<Node> alloc;
95 alloc.deallocate(node_to_gc.release(), solid_size);
96 }
97 }
98 }
99
100 void Worker() {
101 while (!stop_.load()) {
102 std::this_thread::sleep_for(std::chrono::milliseconds(kGCIntervalMs));
103 GarbageCollect();
104 };
105 }
106
107 mutable Mutex gc_mutex_;
108 std::vector<std::unique_ptr<Node>> subtrees_to_gc_ GUARDED_BY(gc_mutex_);
109 std::vector<size_t> subtrees_to_gc_solid_size_ GUARDED_BY(gc_mutex_);
110
111 // When true, Worker() should stop and exit.
112 std::atomic<bool> stop_{false};
113 std::thread gc_thread_;
114};

134// Policy priors (P) are stored in a compressed 16-bit format.
135//
136// Source values are 32-bit floats:
137// * bit 31 is sign (zero means positive)
138// * bit 30 is sign of exponent (zero means nonpositive)
139// * bits 29..23 are value bits of exponent
140// * bits 22..0 are significand bits (plus a "virtual" always-on bit: s âˆˆ [1,2))
141// The number is then sign * 2^exponent * significand, usually.
142// See https://www.h-schmidt.net/FloatConverter/IEEE754.html for details.
143//
144// In compressed 16-bit value we store bits 27..12:
145// * bit 31 is always off as values are always >= 0
146// * bit 30 is always off as values are always < 2
147// * bits 29..28 are only off for values < 4.6566e-10, assume they are always on
148// * bits 11..0 are for higher precision, they are dropped leaving only 11 bits
149// of precision
150//
151// When converting to compressed format, bit 11 is added to in order to make it
152// a rounding rather than truncation.
153//
154// Out of 65556 possible values, 2047 are outside of [0,1] interval (they are in
155// interval (1,2)). This is fine because the values in [0,1] are skewed towards
156// 0, which is also exactly how the components of policy tend to behave (since
157// they add up to 1).
158
159// If the two assumed-on exponent bits (3<<28) are in fact off, the input is
160// rounded up to the smallest value with them on. We accomplish this by
161// subtracting the two bits from the input and checking for a negative result
162// (the subtraction works despite crossing from exponent to significand). This
163// is combined with the round-to-nearest addition (1<<11) into one op.
167void Edge::SetP(float p) {
168 assert(0.0f <= p && p <= 1.0f);
169 constexpr int32_t roundings = (1 << 11) - (3 << 28);
170 int32_t tmp;
171 std::memcpy(&tmp, &p, sizeof(float));
172 tmp += roundings;
173 p_ = (tmp < 0) ? 0 : static_cast<uint16_t>(tmp >> 12);
174}
175
178float Edge::GetP() const {
179 // Reshift into place and set the assumed-set exponent bits.
180 uint32_t tmp = (static_cast<uint32_t>(p_) << 12) | (3 << 28);
181 float ret;
182 std::memcpy(&ret, &tmp, sizeof(uint32_t));
183 return ret;
184}

269bool Node::MakeSolid() {
270 if (solid_children_ || num_edges_ == 0 || IsTerminal()) return false;
271 // Can only make solid if no immediate leaf children are in flight since we
272 // allow the search code to hold references to leaf nodes across locks.
273 Node* old_child_to_check = child_.get();
274 uint32_t total_in_flight = 0;
275 while (old_child_to_check != nullptr) {
276 if (old_child_to_check->GetN() <= 1 &&
277 old_child_to_check->GetNInFlight() > 0) {
278 return false;
279 }
280 if (old_child_to_check->IsTerminal() &&
281 old_child_to_check->GetNInFlight() > 0) {
282 return false;
283 }
284 total_in_flight += old_child_to_check->GetNInFlight();
285 old_child_to_check = old_child_to_check->sibling_.get();
286 }
287 // If the total of children in flight is not the same as self, then there are
288 // collisions against immediate children (which don't update the GetNInFlight
289 // of the leaf) and its not safe.
290 if (total_in_flight != GetNInFlight()) {
291 return false;
292 }
293 std::allocator<Node> alloc;
294 auto* new_children = alloc.allocate(num_edges_);
295 for (int i = 0; i < num_edges_; i++) {
296 new (&(new_children[i])) Node(this, i);
297 }
298 std::unique_ptr<Node> old_child = std::move(child_);
299 while (old_child) {
300 int index = old_child->index_;
301 new_children[index] = std::move(*old_child.get());
302 // This isn't needed, but it helps crash things faster if something has gone
303 // wrong.
304 old_child->parent_ = nullptr;
305 gNodeGc.AddToGcQueue(std::move(old_child));
306 new_children[index].UpdateChildrenParents();
307 old_child = std::move(new_children[index].sibling_);
308 }
309 // This is a hack.
310 child_ = std::unique_ptr<Node>(new_children);
311 solid_children_ = true;
312 return true;
313}

316void Node::SortEdges() {

326void Node::MakeTerminal(GameResult result, float plies_left, Terminal type) {
327 if (type != Terminal::TwoFold) SetBounds(result, result);
328 terminal_type_ = type;
329 m_ = plies_left;
330 if (result == GameResult::DRAW) {
331 wl_ = 0.0f;
332 d_ = 1.0f;
333 } else if (result == GameResult::WHITE_WON) {
334 wl_ = 1.0f;
335 d_ = 0.0f;
336 } else if (result == GameResult::BLACK_WON) {
337 wl_ = -1.0f;
338 d_ = 0.0f;
339 // Terminal losses have no uncertainty and no reason for their U value to be
340 // comparable to another non-loss choice. Force this by clearing the policy.
341 if (GetParent() != nullptr) GetOwnEdge()->SetP(0.0f);
342 }
343}
344
346void Node::MakeNotTerminal() {
347 terminal_type_ = Terminal::NonTerminal;
348 n_ = 0;
349
350 // If we have edges, we've been extended (1 visit), so include children too.
351 if (edges_) {
352 n_++;
353 for (const auto& child : Edges()) {
354 const auto n = child.GetN();
355 if (n > 0) {
356 n_ += n;
357 // Flip Q for opponent.
358 // Default values don't matter as n is > 0.
359 wl_ += -child.GetWL(0.0f) * n;
360 d_ += child.GetD(0.0f) * n;
361 }
362 }
363
364 // Recompute with current eval (instead of network's) and children's eval.
365 wl_ /= n_;
366 d_ /= n_;
367 }
368}

394void Node::FinalizeScoreUpdate(float v, float d, float m, int multivisit) {
395 // Recompute Q.
396 wl_ += multivisit * (v - wl_) / (n_ + multivisit);
397 d_ += multivisit * (d - d_) / (n_ + multivisit);
398 m_ += multivisit * (m - m_) / (n_ + multivisit);
399
400 // Increment N.
401 n_ += multivisit;
402 // Decrement virtual loss.
403 n_in_flight_ -= multivisit;
404}

407void Node::AdjustForTerminal(float v, float d, float m, int multivisit) {
408 // Recompute Q.
409 wl_ += multivisit * v / n_;
410 d_ += multivisit * d / n_;
411 m_ += multivisit * m / n_;
412}

415void Node::RevertTerminalVisits(float v, float d, float m, int multivisit) {
416 // Compute new n_ first, as reducing a node to 0 visits is a special case.
417 const int n_new = n_ - multivisit;
418 if (n_new <= 0) {
419 // If n_new == 0, reset all relevant values to 0.
420 wl_ = 0.0;
421 d_ = 1.0;
422 m_ = 0.0;
423 n_ = 0;
424 } else {
425 // Recompute Q and M.
426 wl_ -= multivisit * (v - wl_) / n_new;
427 d_ -= multivisit * (d - d_) / n_new;
428 m_ -= multivisit * (m - m_) / n_new;
429 // Decrement N.
430 n_ -= multivisit;
431 }
432}

435void Node::UpdateChildrenParents() {
436 if (!solid_children_) {
437 Node* cur_child = child_.get();
438 while (cur_child != nullptr) {
439 cur_child->parent_ = this;
440 cur_child = cur_child->sibling_.get();
441 }
442 } else {
443 Node* child_array = child_.get();
444 for (int i = 0; i < num_edges_; i++) {
445 child_array[i].parent_ = this;

458void Node::ReleaseChildrenExceptOne(Node* node_to_save) {
459 if (solid_children_) {
460 std::unique_ptr<Node> saved_node;
461 if (node_to_save != nullptr) {
462 saved_node = std::make_unique<Node>(this, node_to_save->index_);
463 *saved_node = std::move(*node_to_save);
464 }
465 gNodeGc.AddToGcQueue(std::move(child_), num_edges_);
466 child_ = std::move(saved_node);
467 if (child_) {
468 child_->UpdateChildrenParents();
469 }
470 solid_children_ = false;
471 } else {
472 // Stores node which will have to survive (or nullptr if it's not found).
473 std::unique_ptr<Node> saved_node;
474 // Pointer to unique_ptr, so that we could move from it.
475 for (std::unique_ptr<Node>* node = &child_; *node;
476 node = &(*node)->sibling_) {
477 // If current node is the one that we have to save.
478 if (node->get() == node_to_save) {
479 // Kill all remaining siblings.
480 gNodeGc.AddToGcQueue(std::move((*node)->sibling_));
481 // Save the node, and take the ownership from the unique_ptr.
482 saved_node = std::move(*node);
483 break;
484 }
485 }
486 // Make saved node the only child. (kills previous siblings).
487 gNodeGc.AddToGcQueue(std::move(child_));
488 child_ = std::move(saved_node);
489 }
490 if (!child_) {
491 num_edges_ = 0;
492 edges_.reset(); // Clear edges list.
493 }
494}

513void NodeTree::MakeMove(Move move) {
514 Node* new_head = nullptr;
515 for (auto& n : current_head_->Edges()) {
516 if (n.GetMove() == move) {
517 new_head = n.GetOrSpawnNode(current_head_);
518 // Ensure head is not terminal, so search can extend or visit children of
519 // "terminal" positions, e.g., WDL hits, converted terminals, 3-fold draw.
520 if (new_head->IsTerminal()) new_head->MakeNotTerminal();
521 break;
522 }
523 }
524 current_head_->ReleaseChildrenExceptOne(new_head);
525 new_head = current_head_->child_.get();
526 current_head_ =
527 new_head ? new_head : current_head_->CreateSingleChildNode(move);
528 history_.Append(move);
529}

532void NodeTree::TrimTreeAtHead() {
533 // If solid, this will be empty before move and will be moved back empty
534 // afterwards which is fine.
535 auto tmp = std::move(current_head_->sibling_);
536 // Send dependent nodes for GC instead of destroying them immediately.
537 current_head_->ReleaseChildren();
538 *current_head_ = Node(current_head_->GetParent(), current_head_->index_);
539 current_head_->sibling_ = std::move(tmp);
540}

544bool NodeTree::ResetToPosition(const GameState& pos) {
545 if (gamebegin_node_ && (history_.Starting() != pos.startpos)) {
546 // Completely different position.
547 DeallocateTree();
548 }
549
550 if (!gamebegin_node_) {
551 gamebegin_node_ = std::make_unique<Node>(nullptr, 0);
552 }
553
554 history_.Reset(pos.startpos);
555
556 Node* old_head = current_head_;
557 current_head_ = gamebegin_node_.get();
558 bool seen_old_head = (gamebegin_node_.get() == old_head);
559 for (const Move m : pos.moves) {
560 MakeMove(m);
561 if (old_head == current_head_) seen_old_head = true;
562 }
563
564 // MakeMove guarantees that no siblings exist; but, if we didn't see the old
565 // head, it means we might have a position that was an ancestor to a
566 // previously searched position, which means that the current_head_ might
567 // retain old n_ and q_ (etc) data, even though its old children were
568 // previously trimmed; we need to reset current_head_ in that case.
569 if (!seen_old_head) TrimTreeAtHead();
570 return seen_old_head;
571}

574bool NodeTree::ResetToPosition(const std::string& starting_fen,
575 const std::vector<std::string>& moves) {
576 GameState state;
577 state.startpos = Position::FromFen(starting_fen);
578 ChessBoard cur_board = state.startpos.GetBoard();
579 state.moves.reserve(moves.size());
580 for (const auto& move : moves) {
581 Move m = cur_board.ParseMove(move);
582 state.moves.push_back(m);
583 cur_board.ApplyMove(m);
584 cur_board.Mirror();
585 }
586 return ResetToPosition(state);
587}
