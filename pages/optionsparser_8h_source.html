83 // Add an option to the list of available options (from command line flags
84 // or UCI params)
85 // Usage:
86 // options->Add<StringOption>(name, func, long_flag, short_flag) = def_val;
87 template <typename Option, typename... Args>
88 typename Option::ValueType& Add(Args&&... args) {
89 options_.emplace_back(
90 std::make_unique<Option>(std::forward<Args>(args)...));
91 return defaults_.GetRef<typename Option::ValueType>(
92 options_.back()->GetId());
93 }
94
95 // Returns list of options in UCI format.
96 std::vector<std::string> ListOptionsUci() const;
97
98 // Set the UCI option from string value.
99 void SetUciOption(const std::string& name, const std::string& value,
100 const std::string& context = "");
101 // Processes all flags from the command line and an optional
102 // configuration file. Returns false if there is an invalid flag.
103 bool ProcessAllFlags();
104 // Processes either the command line or configuration file flags.
105 bool ProcessFlags(const std::vector<std::string>& args);
106
107 // Get the options dict for given context.
108 const OptionsDict& GetOptionsDict(const std::string& context = {});
109 // Gets the dictionary for given context which caller can modify.
110 OptionsDict* GetMutableOptions(const std::string& context = {});
111 // Gets the mutable list of default options.
112 OptionsDict* GetMutableDefaultsOptions() { return &defaults_; }
113 // Adds a subdictionary for a given context.
114 void AddContext(const std::string&);
115 // Prints help to std::cout.
116 void ShowHelp() const;
117
118 private:
119 // Returns an option based on the long flag.
120 Option* FindOptionByLongFlag(const std::string& flag) const;
121 // Returns an option based by its uci name.
122 Option* FindOptionByUciName(const std::string& name) const;
123 // Returns an option based by its id.
124 Option* FindOptionById(const OptionId& id) const;
125
126 std::vector<std::unique_ptr<Option>> options_;
127 OptionsDict defaults_;
128 OptionsDict& values_;
129 OptionId::VisibilityMode visibility_mode_ = OptionId::kNormalMode;
130};
