42std::vector<std::string> BackendManager::GetBackendNames() const {
43 std::vector<std::pair<int, std::string>> priority_and_names;
44 std::transform(algorithms_.begin(), algorithms_.end(),
45 std::back_inserter(priority_and_names),
46 [](const std::unique_ptr<BackendFactory>& factory) {
47 return std::make_pair(factory->GetPriority(),
48 std::string(factory->GetName()));
49 });
50 std::sort(priority_and_names.begin(), priority_and_names.end(),
51 std::greater<>());
52 std::vector<std::string> result;
53 std::transform(priority_and_names.begin(), priority_and_names.end(),
54 std::back_inserter(result),
55 [](const std::pair<int, std::string>& p) { return p.second; });
56#ifdef DEFAULT_BACKEND
57 std::string name = DEFAULT_BACKEND;
58 auto pos = std::find(result.begin(), result.end(), name);
59 if (pos == result.end()) throw Exception("Unknown backend: " + name);
60 std::rotate(result.begin(), pos, pos + 1);
61#endif
62 return result;
63}

65BackendFactory* BackendManager::GetFactoryByName(std::string_view name) const {
66 auto iter =
67 std::find_if(algorithms_.begin(), algorithms_.end(),
68 [name](const std::unique_ptr<BackendFactory>& factory) {
69 return factory->GetName() == name;
70 });
71 return iter == algorithms_.end() ? nullptr : iter->get();
72}

81std::unique_ptr<Backend> BackendManager::CreateFromName(
82 std::string_view name, const OptionsDict& options) const {
83 BackendFactory* factory = GetFactoryByName(name);
84 if (!factory) throw Exception("Unknown backend: " + std::string(name));
85 return factory->Create(options);

88void BackendManager::RemoveBackend(const BackendFactory* factory) {
89 auto iter = std::find_if(algorithms_.begin(), algorithms_.end(),
90 [factory](const std::unique_ptr<BackendFactory>& f) {
91 return f.get() == factory;
92 });
93 if (iter == algorithms_.end()) {
94 throw Exception("Attempt to remove unregistered backend");
95 }
96 algorithms_.erase(iter);
97}
