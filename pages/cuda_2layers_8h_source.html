47// The Layer objects only hold memory for weights, biases, etc
48// memory for input and output tensors is provided by caller of Eval.
49
50template <typename DataType>
51class BaseLayer {
52 public:
53 int GetC() const { return C; }
54 int GetH() const { return H; }
55 int GetW() const { return W; }
56 bool isNHWC() const { return nhwc_; }
57
58 BaseLayer(int c, int h, int w, BaseLayer* ip);
59 BaseLayer(int c, int h, int w, BaseLayer* ip, bool nhwc);
60 BaseLayer(int c, int h, int w, BaseLayer* ip, bool nhwc, bool use_gemm_ex);
61 virtual ~BaseLayer() = default;
62 size_t GetOutputSize(int N) const { return sizeof(DataType) * N * C * H * W; }
63
64 // Input2 is optional (skip connection).
65 virtual void Eval(int N, DataType* output, const DataType* input,
66 const DataType* input2, void* scratch, size_t scratch_size,
67 cudnnHandle_t cudnn, cublasHandle_t cublas,
68 cudaStream_t stream, DataType*** = nullptr) = 0;
69
70 protected:
71 BaseLayer* input_;
72
73 int C; // Output tensor dimensions.
74 int H;
75 int W;
76
77 bool nhwc_; // tensor layout
78 const bool use_gemm_ex_;
79
80 void cublasRowMajorMatrixMul(const DataType* A, const DataType* B,
81 DataType* Out, int M, int N, int K,
82 int batchSize, cublasHandle_t cublas);
83};
