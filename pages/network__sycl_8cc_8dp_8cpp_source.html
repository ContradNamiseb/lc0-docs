55static size_t getMaxAttentionHeadSize(
56 const MultiHeadWeights::PolicyHead& weights, int N) {
57 const size_t embedding_op_size = weights.ip_pol_b.size();
58 const size_t policy_d_model = weights.ip2_pol_b.size();
59 assert(policy_d_model == weights.ip3_pol_b.size());
60
61 size_t encoder_d_model = 0;
62 size_t encoder_dff = 0;
63
64 if (weights.pol_encoder.size() > 0) {
65 encoder_d_model = weights.pol_encoder[0].mha.q_b.size();
66 encoder_dff = weights.pol_encoder[0].ffn.dense1_b.size();
67
68 assert(encoder_d_model == weights.pol_encoder[0].mha.k_b.size());
69 assert(encoder_d_model == weights.pol_encoder[0].mha.v_b.size());
70 assert(embedding_op_size == weights.pol_encoder[0].ffn.dense2_b.size());
71 }
72
73 const size_t encoder_heads = weights.pol_encoder_head_count;
74
75 size_t size =
76 N * 64 *
77 std::max(std::max(embedding_op_size, encoder_dff), policy_d_model);
78
79 // size of matmul_qk matrix = encoder_heads_ * Batch * 64 * 64
80 const size_t matmul_qk_size = encoder_heads * N * 64 * 64;
81 const size_t output_size = N * (64 * 64 + 8 * 24);
82 size = std::max(size, std::max(matmul_qk_size, output_size));
83
84 size_t qkv_size = N * 64 * encoder_d_model;
85 // We store qkv in single allocation, and other intermediate tensors are
86 // sometimes stored by splitting an allocation into two halves.
87 size = std::max(2 * size, 3 * qkv_size);
88 return size;
89}

92static size_t getMaxAttentionBodySize(const MultiHeadWeights& weights, int N) {
93 const size_t embedding_op_size = weights.ip_emb_b.size();
94
95 size_t encoder_d_model = 0;
96 size_t encoder_dff = 0;
97
98 if (weights.encoder.size() > 0) {
99 encoder_d_model = weights.encoder[0].mha.q_b.size();
100 encoder_dff = weights.encoder[0].ffn.dense1_b.size();
101
102 assert(encoder_d_model == weights.encoder[0].mha.k_b.size());
103 assert(encoder_d_model == weights.encoder[0].mha.v_b.size());
104 assert(embedding_op_size == weights.encoder[0].ffn.dense2_b.size());
105 }
106
107 const size_t encoder_heads = weights.encoder_head_count;
108
109 size_t size =
110 N * 64 *
111 std::max(std::max(embedding_op_size, encoder_dff), encoder_d_model);
112
113 // size of matmul_qk matrix = encoder_heads_ * Batch * 64 * 64
114 const size_t matmul_qk_size = encoder_heads * N * 64 * 64;
115 const size_t output_size = N * (64 * 64 + 8 * 24);
116 size = std::max(size, std::max(matmul_qk_size, output_size));
117
118 size_t qkv_size = N * 64 * encoder_d_model;
119 // We store qkv in single allocation, and other intermediate tensors are
120 // sometimes stored by splitting an allocation into two halves.
121 size = std::max(2 * size, 3 * qkv_size);
122 return size;
123}

128template <typename DataType>
129class SyclNetworkComputation : public NetworkComputation {
130 public:
131 SyclNetworkComputation(SyclNetwork<DataType>* network, bool wdl,
132 bool moves_left);
133 ~SyclNetworkComputation();
134
135 void AddInput(InputPlanes&& input) override {
136 const auto iter_mask =
137 &inputs_outputs_->input_masks_mem_shared_[batch_size_ * kInputPlanes];
138 const auto iter_val =
139 &inputs_outputs_->input_val_mem_shared_[batch_size_ * kInputPlanes];
140
141 int i = 0;
142 for (const auto& plane : input) {
143 iter_mask[i] = plane.mask;
144 iter_val[i] = plane.value;
145 i++;
146 }
147
148 batch_size_++;
149 }
150
151 void ComputeBlocking() override;
152
153 int GetBatchSize() const override { return batch_size_; }
154
155 float GetQVal(int sample) const override {
156 if (wdl_) {
157 auto w = inputs_outputs_->op_value_mem_shared_[3 * sample + 0];
158 auto l = inputs_outputs_->op_value_mem_shared_[3 * sample + 2];
159 return w - l;
160 }
161 return inputs_outputs_->op_value_mem_shared_[sample];
162 }
163
164 float GetDVal(int sample) const override {
165 if (wdl_) {
166 auto d = inputs_outputs_->op_value_mem_shared_[3 * sample + 1];
167 return d;
168 }
169 return 0.0f;
170 }
171
172 float GetPVal(int sample, int move_id) const override {
173 return inputs_outputs_->op_policy_mem_[sample * kNumOutputPolicy + move_id];
174 }
175
176 float GetMVal(int sample) const override {
177 if (moves_left_) {
178 return inputs_outputs_->op_moves_left_mem_shared_[sample];
179 }
180 return 0.0f;
181 }
182
183 private:
184 // Memory holding inputs, outputs.
185 std::unique_ptr<InputsOutputs> inputs_outputs_;
186 int batch_size_;
187 bool wdl_;
188 bool moves_left_;
189
190 SyclNetwork<DataType>* network_;
191};

632 void forwardEval(InputsOutputs* io, int batchSize) {

635#ifdef DEBUG_RAW_NPS
636 auto t_start = std::chrono::high_resolution_clock::now();
637#endif

639 // Expand packed planes to full planes.
640 uint64_t* ipDataMasks = io->input_masks_mem_shared_;
641 float* ipDataValues = io->input_val_mem_shared_;
642 sycl::queue io_sycl_queue_ = io->q_ct1;

644 DataType* tensor_mem[3];
645 void* scratch_mem;
646 DataType*** offset_pointers;
647 DataType*** head_offset_pointers;

649 if (multi_stream_) {
650 // We use tensor and scratch memory from InputOutputs (so that multiple
651 // requests can run in parallel)
652 for (int i = 0; i < 3; i++) tensor_mem[i] = (DataType*)io->tensor_mem_[i];
653 scratch_mem = io->scratch_mem_;
654 offset_pointers = (DataType***)&io->offset_pointers_;
655 head_offset_pointers = (DataType***)&io->head_offset_pointers_;
656 // stream = io->stream_;
657 // cublas = io->cublas_;
658 } else {
659 for (int i = 0; i < 3; i++) tensor_mem[i] = tensor_mem_[i];
660 scratch_mem = scratch_mem_;
661 offset_pointers = (DataType***)&offset_pointers_;
662 head_offset_pointers = (DataType***)&head_offset_pointers_;
663 // stream = &dpct::get_default_queue(); // default stream
664 // cublas = cublas_;
665 }
666
667 bool fp16 = std::is_same<sycl::half, DataType>::value;
668 if (fp16) {
669 expandPlanes_Fp16_NCHW((sycl::half*)(tensor_mem[0]), ipDataMasks,
670 ipDataValues, batchSize * kInputPlanes,
671 io_sycl_queue_);
672 } else {
673 expandPlanes_Fp32_NCHW((float*)(tensor_mem[0]), ipDataMasks, ipDataValues,
674 batchSize * kInputPlanes, io_sycl_queue_);
675 }
676
677 float* opPol = io->op_policy_mem_gpu_;
678 float* opVal = io->op_value_mem_shared_;
679 float* opMov = io->op_moves_left_mem_shared_;
680
681 // Figure out if the memory requirment for running the res block would fit
682 // in the L2 cache.
683 bool enableCacheOpt = false;
684 DataType* skip_connection =
685 use_res_block_winograd_fuse_opt_ ? tensor_mem[1] : tensor_mem[2];
686
687 // #if DPCT_COMPAT_RT_VERSION >= 11000
688 const int pre_transform_tensor_size =
689 batchSize * numFilters_ * 8 * 8 * sizeof(DataType);
690 const int transformed_tensor_size = pre_transform_tensor_size * 36 / 16;
691 const int res_block_mem =
692 transformed_tensor_size * 2 + pre_transform_tensor_size;
693
694 // cudaStreamAttrValue stream_attribute = {};
695 // stream_attribute.accessPolicyWindow.base_ptr = tensor_mem[2];
696 // stream_attribute.accessPolicyWindow.num_bytes = res_block_mem;
697 // stream_attribute.accessPolicyWindow.hitRatio = 1.0f;
698 // stream_attribute.accessPolicyWindow.hitProp =
699 // cudaAccessPropertyPersisting;
700 // stream_attribute.accessPolicyWindow.missProp =
701 // cudaAccessPropertyStreaming;
702
703 // if (allow_cache_opt_ && use_res_block_winograd_fuse_opt_ &&
704 // (res_block_mem <= scratch_size_) && (res_block_mem <=
705 // l2_cache_size_)) {
706 // we can use a single alloc to hold all the required tensors, and enable
707 // persistent L2 caching on it
708 /*
709 DPCT1007:87: Migration of cudaStreamSetAttribute is not supported.
710 */
711 // cudaStreamSetAttribute(stream, cudaStreamAttributeAccessPolicyWindow,
712 // &stream_attribute);
713
714 // enableCacheOpt = true;
715 // skip_connection =
716 // tensor_mem[2] + 2 * transformed_tensor_size / sizeof(DataType);
717 // }
718 // #endif
719
720 int l = 0;
721
722 DataType* flow = tensor_mem[0];
723 DataType* spare1 = tensor_mem[1];
724 DataType* spare2 = tensor_mem[2];
725
726 if (numBlocks_ > 0) {
727 // Input.
728 network_[l++]->Eval(batchSize, skip_connection, tensor_mem[0], nullptr,
729 scratch_mem, scratch_size_, io_sycl_queue_,
730 nullptr); // input conv
731
732 // Residual block.
733 for (int block = 0; block < numBlocks_; block++) {
734 if (use_res_block_winograd_fuse_opt_) {
735 network_[l++]->Eval(batchSize, tensor_mem[2], skip_connection,
736 nullptr, enableCacheOpt ? nullptr : scratch_mem,
737 scratch_size_, io_sycl_queue_, nullptr); // block
738 } else {
739 network_[l++]->Eval(batchSize, tensor_mem[0], tensor_mem[2], nullptr,
740 scratch_mem, scratch_size_, io_sycl_queue_,
741 nullptr); // conv1
742
743 network_[l++]->Eval(batchSize, tensor_mem[2], tensor_mem[0],
744 tensor_mem[2], scratch_mem, scratch_size_,
745 io_sycl_queue_, nullptr); // conv2
746 }
747 }
748
749 flow = tensor_mem[2];
750 spare1 = tensor_mem[0];
751 spare2 = tensor_mem[1];
752 }
753
754 if (attn_body_) {
755 network_[l++]->Eval(
756 batchSize, tensor_mem[1],
757 (numBlocks_ > 0) ? tensor_mem[2] : tensor_mem[0],
758 (numBlocks_ > 0) ? tensor_mem[0] : tensor_mem[2], scratch_mem,
759 scratch_size_, io_sycl_queue_,
760 offset_pointers); // Entire attention body of the network
761
762 flow = tensor_mem[1];
763 spare1 = tensor_mem[0];
764 spare2 = tensor_mem[2];
765 }
766
767 // Policy head.
768
769 if (attn_policy_) {
770 network_[l++]->Eval(
771 batchSize, spare1, flow, spare2, scratch_mem, scratch_size_,
772 io_sycl_queue_,
773 head_offset_pointers); // Entire Attention policy head except for the
774 // policy map
775 if (fp16) {
776 network_[l++]->Eval(batchSize, spare2, spare1, nullptr, scratch_mem,
777 scratch_size_, io_sycl_queue_,
778 nullptr); // policy map layer
779
780 copyTypeConverted(opPol, (sycl::half*)spare2,
781 batchSize * kNumOutputPolicy,
782 io_sycl_queue_); // POLICY output
783 } else {
784 network_[l++]->Eval(batchSize, (DataType*)opPol, spare1, nullptr,
785 scratch_mem, scratch_size_, io_sycl_queue_,
786 nullptr); // policy map layer // POLICY output
787 }
788
789 } else if (conv_policy_) {
790 network_[l++]->Eval(batchSize, spare1, flow, nullptr, scratch_mem,
791 scratch_size_, io_sycl_queue_,
792 nullptr); // policy conv1
793
794 network_[l++]->Eval(batchSize, spare2, spare1, nullptr, scratch_mem,
795 scratch_size_, io_sycl_queue_,
796 nullptr); // policy conv2
797
798 if (fp16) {
799 network_[l++]->Eval(batchSize, spare1, spare2, nullptr, scratch_mem,
800 scratch_size_, io_sycl_queue_,
801 nullptr); // policy map layer
802
803 copyTypeConverted(opPol, (sycl::half*)(spare1),
804 batchSize * kNumOutputPolicy,
805 io_sycl_queue_); // POLICY output
806
807 } else {
808 network_[l++]->Eval(batchSize, (DataType*)opPol, spare2, nullptr,
809 scratch_mem, scratch_size_, io_sycl_queue_,
810 nullptr);
811 // policy map layer // POLICY output
812 }
813
814 } else {
815 network_[l++]->Eval(batchSize, spare1, flow, nullptr, scratch_mem,
816 scratch_size_, io_sycl_queue_, nullptr); // pol conv
817
818 if (fp16) {
819 network_[l++]->Eval(batchSize, spare2, spare1, nullptr, scratch_mem,
820 scratch_size_, io_sycl_queue_, nullptr); // pol FC
821
822 copyTypeConverted(opPol, (sycl::half*)(spare2),
823 batchSize * kNumOutputPolicy,
824 io_sycl_queue_); // POLICY
825 } else {
826 network_[l++]->Eval(batchSize, (DataType*)opPol, spare1, nullptr,
827 scratch_mem, scratch_size_, io_sycl_queue_,
828 nullptr); // pol FC // POLICY
829 }
830 }
831
832 // value head
833 if (fp16) {
834 network_[l++]->Eval(batchSize, spare1, flow, spare2, scratch_mem,
835 scratch_size_, io_sycl_queue_,
836 nullptr); // value head
837
838 copyTypeConverted(opVal, (sycl::half*)spare1,
839 wdl_ ? 3 * batchSize : batchSize, io_sycl_queue_);
840 } else {
841 network_[l++]->Eval(batchSize, (DataType*)opVal, flow, spare2,
842 scratch_mem, scratch_size_, io_sycl_queue_,
843 nullptr); // value head
844 }
845
846 if (moves_left_) {
847 // Moves left head
848 network_[l++]->Eval(batchSize, spare1, flow, nullptr, scratch_mem,
849 scratch_size_, io_sycl_queue_,
850 nullptr); // moves conv or embedding
851
852 network_[l++]->Eval(batchSize, spare2, spare1, nullptr, scratch_mem,
853 scratch_size_, io_sycl_queue_, nullptr); // moves FC1
854
855 // Moves left FC2
856 if (fp16) {
857 // TODO: consider fusing the bias-add of FC2 with format conversion.
858
859 network_[l++]->Eval(batchSize, spare1, spare2, nullptr, scratch_mem,
860 scratch_size_, io_sycl_queue_, nullptr);
861
862 copyTypeConverted(opMov, (sycl::half*)(spare1), batchSize,
863 io_sycl_queue_);
864
865 } else {
866 network_[l++]->Eval(batchSize, (DataType*)opMov, spare2, nullptr,
867 scratch_mem, scratch_size_, io_sycl_queue_,
868 nullptr);
869 }
870 }
871
872 // Copy policy output from device memory to host memory.
873 auto event =
874 io_sycl_queue_.memcpy(io->op_policy_mem_, io->op_policy_mem_gpu_,
875 sizeof(float) * kNumOutputPolicy * batchSize);
876
877 if (!multi_stream_) {
878 // ReportCUDAErrors(
879 // DPCT_CHECK_ERROR(dpct::get_current_device().queues_wait_and_throw()));
880 // The next thread can start using the GPU now.
881 lock_.unlock();
882 }
883
884 event.wait();
885
886 if (wdl_) {
887 // Value softmax done cpu side.
888 for (int i = 0; i < batchSize; i++) {
889 float w = io->op_value_mem_shared_[3 * i + 0];
890 float d = io->op_value_mem_shared_[3 * i + 1];
891 float l = io->op_value_mem_shared_[3 * i + 2];
892 float m = std::max({w, d, l});
893 w = std::exp(w - m);
894 d = std::exp(d - m);
895 l = std::exp(l - m);
896 float sum = w + d + l;
897 w /= sum;
898 l /= sum;
899 d /= sum;
900 io->op_value_mem_shared_[3 * i + 0] = w;
901 io->op_value_mem_shared_[3 * i + 1] = d;
902 io->op_value_mem_shared_[3 * i + 2] = l;
903 }
904 }
