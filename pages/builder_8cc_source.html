38OnnxBuilder::OnnxBuilder(int opset, int ir) : opset_(opset) {
39 if (opset < 7 || opset > 22) {
40 throw Exception("Only ONNX opsets between 7 and 22 are supported.");
41 }
42 // Map of latest opset corresponding to IR version.
43 std::map<int, int> opset_to_ir = {{8, 3}, {9, 4}, {10, 5},
44 {11, 6}, {14, 7}, {18, 8},
45 {20, 9}, {22, 10}, {99, 11}};
46 if (ir < 0) ir = opset_to_ir.upper_bound(opset - 1)->second;
47 if (ir < 3 || ir > 10) {
48 throw Exception("Only ONNX IR between 3 and 10 is supported.");
49 }
50 model_.set_ir_version(ir);
51 model_.set_domain("org.lczero.models.*");
52 model_.set_producer_name("Lc0");
53 model_.set_producer_version(GetVersionStr());
54 model_.add_opset_import()->set_version(opset);
55 // TODO change to real network name when it becomes available.
56 model_.mutable_graph()->set_name("org.lczero/converted");
57}

60void FillValueInfo(pblczero::ValueInfoProto* vip, const std::string& name,
61 std::initializer_list<int> dims,
62 pblczero::TensorProto::DataType datatype) {
63 vip->set_name(name);
64 auto* type = vip->mutable_type()->mutable_tensor_type();
65 type->set_elem_type(datatype);
66 auto* shape = type->mutable_shape();
67 for (const auto d : dims) {
68 auto* dim = shape->add_dim();
69 if (d < 0) {
70 dim->set_dim_param("batch");
71 } else {
72 dim->set_dim_value(d);
73 }
74 }
75}

115std::string OnnxBuilder::AddInitializer(const std::string& name,
116 const OnnxConst& weights) {
117 auto* init = model_.mutable_graph()->add_initializer();
118 init->set_name(name);
119 init->set_data_type(weights.GetDataType());
120 for (const int dim : weights.GetDimensions()) init->add_dims(dim);
121 init->set_raw_data(weights.GetRawData());
122
123 return name;

141std::string OnnxBuilder::Conv(const std::string& name,
142 const std::string& input_name,
143 const OnnxConst& kernel_weights,
144 const OnnxConst& bias_weights, int pads) {
145 auto* node = model_.mutable_graph()->add_node();
146 auto shape = kernel_weights.GetDimensions().back();
147 auto out = PopulateStdNodeFields(node, name, input_name, "Conv");
148 node->add_input(AddInitializer(name + "/w/kernel", kernel_weights));
149 node->add_input(AddInitializer(name + "/w/bias", bias_weights));
150 AddIntsAttribute(node, "pads", {pads, pads, pads, pads});
151 AddIntsAttribute(node, "kernel_shape", {shape, shape});
152 return out;
153}

301std::vector<std::string> OnnxBuilder::Split(const std::string& name,
302 const std::string& input, int axis,
303 std::initializer_list<int> split) {
304 auto* node = model_.mutable_graph()->add_node();
305 node->set_name(name);
306 node->set_op_type("Split");
307 node->add_input(input);
308 AddIntAttribute(node, "axis", axis);
309 if (split.size() > 0) {
310 if (opset_ < 13) {
311 AddIntsAttribute(node, "split", split);
312 } else {
313 node->add_input(AddInitializer(
314 name + "/split",
315 Int64OnnxConst(std::vector<int64_t>(begin(split), end(split)),
316 {static_cast<int>(split.size())})));
317 }
318 std::vector<std::string> out;
319 for (size_t i = 1; i <= split.size(); i++) {
320 out.push_back(name + "/out" + std::to_string(i));
321 node->add_output(out.back());
322 }
323 return out;
324 }
325 if (opset_ >= 18) AddIntAttribute(node, "num_outputs", 2);
326 node->add_output(name + "/out1");
327 node->add_output(name + "/out2");
328 return {name + "/out1", name + "/out2"};
329}

331std::string OnnxBuilder::Slice(const std::string& name,
332 const std::string& input,
333 std::initializer_list<int> starts,
334 std::initializer_list<int> ends) {
335 auto* node = model_.mutable_graph()->add_node();
336 auto out = PopulateStdNodeFields(node, name, input, "Slice");
337 if (opset_ < 10) {
338 AddIntsAttribute(node, "starts", starts);
339 AddIntsAttribute(node, "ends", ends);
340 } else {
341 node->add_input(AddInitializer(
342 name + "/starts", Int32OnnxConst(std::vector<int>(starts),
343 {static_cast<int>(starts.size())})));
344 node->add_input(AddInitializer(
345 name + "/ends", Int32OnnxConst(std::vector<int>(ends),
346 {static_cast<int>(ends.size())})));
347 }
348 return out;
349}
