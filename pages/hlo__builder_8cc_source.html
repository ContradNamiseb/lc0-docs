38HloTensorType::HloTensorType(const pblczero::XlaShapeProto& proto)
39 : type_(proto.element_type()), dimensions_(proto.dimensions()) {
40 switch (type_) {
41 case pblczero::XlaShapeProto::PRIMITIVE_TYPE_INVALID:
42 case pblczero::XlaShapeProto::TUPLE:
43 case pblczero::XlaShapeProto::OPAQUE_TYPE:
44 case pblczero::XlaShapeProto::TOKEN:
45 throw Exception("Invalid element type for tensor type");
46 default:
47 break;
48 }
49}
50
51pblczero::XlaShapeProto HloTensorType::ToProto() const {
52 pblczero::XlaShapeProto ret;
53 ret.set_element_type(type_);
54 *ret.mutable_dimensions() = dimensions_;
55 ret.mutable_layout();
56 for (size_t i = 0; i < dimensions_.size(); ++i) {
57 ret.add_is_dynamic_dimension(false);
58 ret.mutable_layout()->add_minor_to_major(dimensions_.size() - i - 1);
59 }
60 return ret;
61}

63std::string HloTensorType::ToString() const {
64 std::string ret = pblczero::XlaShapeProto::Type_Name(type_);
65 ret += "[";
66 for (size_t i = 0; i < dimensions_.size(); ++i) {
67 ret += (i == 0 ? "" : ", ") + std::to_string(dimensions_[i]);
68 }
69 return ret + "]";
70}

72size_t HloTensorType::NumElements() const {
73 return std::accumulate(dimensions_.begin(), dimensions_.end(), 1,
74 std::multiplies<int64_t>());

77// Creates an instruction and populates required fields of the
78// HloInstructionProto: result shape, opcode and operands.
79// Appends the instruction to the entry computation.
80pblczero::HloInstructionProto* HloBuilder::MakeInstruction(
81 std::string_view opcode, const pblczero::XlaShapeProto& shape,
82 const std::vector<const pblczero::HloInstructionProto*> operands) {
83 auto instr = std::make_unique<pblczero::HloInstructionProto>();
84 auto ret = instr.get();
85 ret->set_opcode(opcode);
86 *ret->mutable_shape() = shape;
87 if (!metadata_.OutputAsString().empty()) {
88 *ret->mutable_metadata() = metadata_;
89 }
90 ret->set_id(next_instruction_id_++);
91 for (const auto& operand : operands) {
92 ret->add_operand_ids(operand->id());
93 }
94 entry_computation_.push_back(std::move(instr));
95 return ret;
96}

98// Creates an elementwise instruction, which always have two operands of the
99// same shape.
100pblczero::HloInstructionProto* HloBuilder::MakeElementwiseInstruction(
101 std::string_view opcode, HloFlow lhs, HloFlow rhs) {
102 if (lhs->shape().dimensions() != rhs->shape().dimensions()) {
103 throw Exception("Elementwise operands must have the same shape");
104 }
105 return MakeInstruction(opcode, lhs->shape(), {lhs, rhs});

131HloFlow HloBuilder::Convolution(
132 HloFlow input, HloFlow filter, const pblczero::XlaWindow& window,
133 const pblczero::XlaConvolutionDimensionNumbers& dn) {
134 if (input->shape().dimensions_size() != filter->shape().dimensions_size()) {
135 throw Exception(
136 "Convolution input and filter shapes must have the "
137 "same number of dimensions");
138 }
139 pblczero::XlaShapeProto shape = input->shape();
140 auto* out_dims = shape.mutable_dimensions();
141 const auto& in_dims = input->shape().dimensions();
142 const auto& filter_dims = filter->shape().dimensions();
143 (*out_dims)[dn.output_batch_dimension()] =
144 in_dims[dn.input_batch_dimension()];
145 (*out_dims)[dn.output_feature_dimension()] =
146 filter_dims[dn.kernel_output_feature_dimension()];
147 for (size_t i = 0; i < dn.input_spatial_dimensions_size(); ++i) {
148 (*out_dims)[dn.output_spatial_dimensions(i)] =
149 in_dims[dn.input_spatial_dimensions(i)];
150 }
151 auto* flow = MakeInstruction("convolution", shape, {input, filter});
152 *flow->mutable_window() = window;
153 *flow->mutable_convolution_dimension_numbers() = dn;
154 return flow;
155}

157HloFlow HloBuilder::Broadcast(
158 HloFlow input, const HloTensorType& target_shape,
159 const std::vector<int64_t>& broadcast_dimensions) {
160 auto flow = MakeInstruction("broadcast", target_shape.ToProto(), {input});
161 if (broadcast_dimensions.size() != input->shape().dimensions_size()) {
162 throw Exception(
163 "Broadcast must have the same size as the input shape: "
164 "broadcast_dimensions=" +
165 std::to_string(broadcast_dimensions.size()) +
166 ", input_shape=" + HloTensorType(input->shape()).ToString());
167 }
168 const auto& input_shape = input->shape();
169 for (size_t i = 0; i < broadcast_dimensions.size(); ++i) {
170 auto dim = broadcast_dimensions[i];
171 const auto& input_dim = input_shape.dimensions(i);
172 if (input_dim != target_shape.GetDimension(dim)) {
173 throw Exception(
174 "Broadcast dimension must be equal to the target shape dimension");
175 }
176 flow->add_dimensions(dim);
177 }
178 return flow;
179}

181HloFlow HloBuilder::Reduce(HloFlow input, HloFlow initial,
182 HloComputation function,
183 const std::vector<int64_t>& reduction_dimensions) {
184 HloTensorType target_shape(input->shape().element_type());
185 for (size_t i = 0; i < input->shape().dimensions_size(); ++i) {
186 if (std::find(reduction_dimensions.begin(), reduction_dimensions.end(),
187 i) == reduction_dimensions.end()) {
188 target_shape.AddDimension(input->shape().dimensions(i));
189 }
190 }
191 auto flow =
192 MakeInstruction("reduce", target_shape.ToProto(), {input, initial});
193 *flow->mutable_dimensions() = {reduction_dimensions.begin(),
194 reduction_dimensions.end()};
195 flow->add_called_computation_ids(function.idx());
196 return flow;
197}

242HloFlow HloBuilder::Gather(HloFlow input, HloFlow indices,
243 size_t index_vector_dim,
244 const std::vector<int64_t>& offset_dims,
245 const std::vector<int64_t>& slice_sizes,
246 const std::vector<int64_t>& collapsed_slice_dims,
247 const std::vector<int64_t>& start_index_map,
248 bool indices_are_sorted, bool unique_indicies) {
249 HloTensorType input_shape(input->shape());
250 HloTensorType indices_shape(indices->shape());
251
252 if (indices_shape.Rank() == index_vector_dim) {
253 indices_shape.AddDimension(1);
254 }
255
256 HloTensorType output_shape(input_shape.GetElementType());
257 size_t output_rank = offset_dims.size() + indices_shape.Rank() - 1;
258 size_t offset_dims_idx = 0;
259 size_t gather_dims_idx = 0;
260
261 for (size_t i = 0; i < output_rank; ++i) {
262 const bool is_in_offset = std::find(offset_dims.begin(), offset_dims.end(),
263 i) != offset_dims.end();
264 if (is_in_offset) {
265 while (std::find(collapsed_slice_dims.begin(), collapsed_slice_dims.end(),
266 offset_dims_idx) != collapsed_slice_dims.end()) {
267 offset_dims_idx++;
268 }
269 output_shape.AddDimension(slice_sizes[offset_dims_idx++]);
270 } else {
271 if (gather_dims_idx == index_vector_dim) ++gather_dims_idx;
272 output_shape.AddDimension(indices_shape.GetDimension(gather_dims_idx++));
273 }
274 }
275
276 auto flow =
277 MakeInstruction("gather", output_shape.ToProto(), {input, indices});
278 *flow->mutable_gather_slice_sizes() = slice_sizes;
279 *flow->mutable_gather_dimension_numbers()->mutable_offset_dims() =
280 offset_dims;
281 *flow->mutable_gather_dimension_numbers()->mutable_collapsed_slice_dims() =
282 collapsed_slice_dims;
283 *flow->mutable_gather_dimension_numbers()->mutable_start_index_map() =
284 start_index_map;
285 flow->mutable_gather_dimension_numbers()->set_index_vector_dim(
286 index_vector_dim);
287 flow->set_indices_are_sorted(indices_are_sorted);
288 flow->set_unique_indices(unique_indicies);
289 return flow;
290}

292HloFlow HloBuilder::Dot(HloFlow lhs, HloFlow rhs,
293 const pblczero::XlaDotDimensionNumbers& dn) {
294 HloTensorType lhs_shape(lhs->shape());
295 HloTensorType rhs_shape(rhs->shape());
296 HloTensorType new_shape(lhs_shape.GetElementType());
297 if (lhs_shape.GetElementType() != rhs_shape.GetElementType()) {
298 throw Exception("Dot operands must have the same element type");
299 }
300 if (dn.lhs_batch_dimensions_size() != dn.rhs_batch_dimensions_size()) {
301 throw Exception("Dot batch dimensions must have the same size");
302 }
303 for (size_t i = 0; i < dn.lhs_batch_dimensions_size(); ++i) {
304 auto lhs_dim = lhs_shape.GetDimension(dn.lhs_batch_dimensions(i));
305 auto rhs_dim = rhs_shape.GetDimension(dn.rhs_batch_dimensions(i));
306 if (lhs_dim != rhs_dim) {
307 throw Exception("Dot batch dimensions must have the same size");
308 }
309 new_shape.AddDimension(lhs_dim);
310 }
311 if (dn.lhs_contracting_dimensions_size() !=
312 dn.rhs_contracting_dimensions_size()) {
313 throw Exception("Dot contracting dimensions must have the same size");
314 }
315 for (size_t i = 0; i < dn.lhs_contracting_dimensions_size(); ++i) {
316 auto lhs_dim = lhs_shape.GetDimension(dn.lhs_contracting_dimensions(i));
317 auto rhs_dim = rhs_shape.GetDimension(dn.rhs_contracting_dimensions(i));
318 if (lhs_dim != rhs_dim) {
319 throw Exception("Dot contracting dimensions must have the same size");
320 }
321 }
322 // Sorry, github copilot generated the code below (well, above too). Enjoy!
323 for (size_t i = 0; i < lhs->shape().dimensions_size(); ++i) {
324 if (std::find(dn.lhs_batch_dimensions().begin(),
325 dn.lhs_batch_dimensions().end(),
326 i) == dn.lhs_batch_dimensions().end() &&
327 std::find(dn.lhs_contracting_dimensions().begin(),
328 dn.lhs_contracting_dimensions().end(),
329 i) == dn.lhs_contracting_dimensions().end()) {
330 new_shape.AddDimension(lhs_shape.GetDimension(i));
331 }
332 }
333 for (size_t i = 0; i < rhs->shape().dimensions_size(); ++i) {
334 if (std::find(dn.rhs_batch_dimensions().begin(),
335 dn.rhs_batch_dimensions().end(),
336 i) == dn.rhs_batch_dimensions().end() &&
337 std::find(dn.rhs_contracting_dimensions().begin(),
338 dn.rhs_contracting_dimensions().end(),
339 i) == dn.rhs_contracting_dimensions().end()) {
340 new_shape.AddDimension(rhs_shape.GetDimension(i));
341 }
342 }
343 auto flow = MakeInstruction("dot", new_shape.ToProto(), {lhs, rhs});
344 *flow->mutable_dot_dimension_numbers() = dn;
345 return flow;
346}

414HloFlow HloBuilder::Concatenate(const std::vector<HloFlow>& inputs,
415 int64_t dimension) {
416 if (inputs.empty()) {
417 throw Exception("Concatenate must have at least one input");
418 }
419 HloTensorType shape(inputs[0]->shape());
420 for (size_t i = 1; i < inputs.size(); ++i) {
421 if (inputs[i]->shape().element_type() != shape.GetElementType()) {
422 throw Exception("Concatenate operands must have the same element type");
423 }
424 if (inputs[i]->shape().dimensions_size() != shape.Rank()) {
425 throw Exception("Concatenate operands must have the same rank");
426 }
427 for (size_t j = 0; j < shape.Rank(); ++j) {
428 if (j == static_cast<size_t>(dimension)) {
429 shape.SetDimension(
430 j, shape.GetDimension(j) + inputs[i]->shape().dimensions(j));
431 } else if (inputs[i]->shape().dimensions(j) != shape.GetDimension(j)) {
432 std::string shapes;
433 for (const auto& input : inputs) {
434 shapes += HloTensorType(input->shape()).ToString() + ", ";
435 }
436 throw Exception("Concatenate operands must have the same shape, got " +
437 shapes + "axis=" + std::to_string(dimension));
438 }
439 }
440 }
441 auto flow = MakeInstruction("concatenate", shape.ToProto(), inputs);
442 flow->add_dimensions(dimension);
443 return flow;
444}

446HloFlow HloBuilder::Slice(
447 HloFlow input,
448 const std::vector<pblczero::HloInstructionProto::SliceDimensions>& slice) {
449 HloTensorType current_shape(input->shape());
450 if (slice.size() != current_shape.Rank()) {
451 throw Exception(
452 "Slice dimensions must have the same size as the input shape");
453 }
454 HloTensorType new_shape(current_shape.GetElementType());
455 for (size_t i = 0; i < slice.size(); ++i) {
456 const auto& dim = slice[i];
457 if (dim.start() < 0 || dim.start() >= current_shape.GetDimension(i) ||
458 dim.limit() < 0 || dim.limit() > current_shape.GetDimension(i) ||
459 dim.start() >= dim.limit() || dim.stride() != 1) {
460 throw Exception("Invalid slice dimensions, input shape=" +
461 current_shape.ToString() + ", dim=" + std::to_string(i) +
462 ", slice_start=" + std::to_string(dim.start()) +
463 ", slice_limit=" + std::to_string(dim.limit()));
464 }
465 // This / dim.stride() is pretty approximate, therefore there's a check
466 // above.
467 new_shape.AddDimension((dim.limit() - dim.start()) / dim.stride());
468 }
469 auto flow = MakeInstruction("slice", new_shape.ToProto(), {input});
470 *flow->mutable_slice_dimensions() = slice;
471 return flow;
472}

475// Go over all "parameter" instructions of the computation and assign
476// "parameter_number" field with increasing numbers.
477// Normally it's not requiredm but in our case it's simpler.
478// Outputs shapes and instruction names of parameters.
479std::pair<std::vector<pblczero::XlaShapeProto>, std::vector<std::string>>
480AssignParameterIndices(const InstructionList& comp) {
481 std::vector<pblczero::XlaShapeProto> parameter_shapes;
482 std::vector<std::string> parameter_names;
483 size_t idx = 0;
484 for (const auto& instr : comp) {
485 if (instr->opcode() == "parameter") {
486 instr->set_parameter_number(idx++);
487 parameter_shapes.push_back(instr->shape());
488 parameter_names.push_back(std::string(instr->name()));
489 }
490 }
491 return {parameter_shapes, parameter_names};
492}

494// Finalizes HloComputationProto (sets name, renumbers parameters, adds
495// computation shape and root instruction).
496pblczero::HloComputationProto MakeComputation(const InstructionList& comp,
497 std::string_view name,
498 size_t id) {
499 pblczero::HloComputationProto ret;
500 ret.set_id(id);
501 ret.set_name(name);
502 auto [shapes, names] = AssignParameterIndices(comp);
503 for (auto& instr : comp) *ret.add_instructions() = *instr;
504 *ret.mutable_program_shape()->mutable_parameters() = shapes;
505 *ret.mutable_program_shape()->mutable_parameter_names() = names;
506 *ret.mutable_program_shape()->mutable_result() = comp.back()->shape();
507 ret.set_root_id(comp.back()->id());
508 return ret;
509}

512// Assigns unique names to all instructions in the module.
513// In StableHLO instructions are allowed to have numeric names, but in XLA HLO
514// they are not, so we use "i"+number.
515void HloBuilder::AssignInstructionNames() {
516 // Every instruction in the module should have an unique name, numeric names
517 // are allowed.
518 size_t idx = 0;
519 for (auto& instr : entry_computation_) {
520 instr->set_name("i" + std::to_string(idx++));
521 }
522 for (auto& comp : dependent_computations_) {
523 for (auto& instr : *comp.mutable_instructions()) {
524 instr.set_name("i" + std::to_string(idx++));
525 }
526 }
527}

536HloComputation HloBuilder::AddComputation(std::string_view name,
537 const HloBuilder& builder) {
538 std::unordered_map<size_t, size_t> id_map;
539 if (computation_names_.contains(std::string(name))) {
540 throw Exception("Computation with name " + std::string(name) +
541 " already exists");
542 }
543
544 const size_t computation_add_idx = dependent_computations_.size();
545
546 // Insert all dependent computations of the passed builder.
547 for (const auto& [name, id] : builder.computation_names_) {
548 auto iter = computation_names_.find(name);
549 if (iter != computation_names_.end()) {
550 // TODO check that the computation is the same.
551 id_map[id] = iter->second;
552 continue;
553 }
554 const size_t new_id = dependent_computations_.size();
555 id_map[id] = new_id;
556 computation_names_[name] = new_id;
557 dependent_computations_.push_back(
558 ReassignInstructionIds(builder.dependent_computations_[id]));
559 dependent_computations_.back().set_id(new_id);
560 }
561
562 // Insert passed builder's entry computation as current builder's dependent
563 // computation.
564 {
565 const size_t new_id = dependent_computations_.size();
566 computation_names_[std::string(name)] = new_id;
567 dependent_computations_.push_back(ReassignInstructionIds(
568 MakeComputation(builder.entry_computation_, name, new_id)));
569 }
570
571 // Remap operand ids in the dependent computations.
572 for (size_t i = computation_add_idx; i < dependent_computations_.size();
573 ++i) {
574 auto* comp = &dependent_computations_[i];
575 for (auto& instr : *comp->mutable_instructions()) {
576 for (int64_t& called_id : *instr.mutable_called_computation_ids()) {
577 called_id = id_map.at(called_id);
578 }
579 }
580 }
581
582 return HloComputation(dependent_computations_.back().id());
583}

585pblczero::HloComputationProto HloBuilder::ReassignInstructionIds(
586 pblczero::HloComputationProto computation) {
587 std::unordered_map<size_t, size_t> id_map;
588 for (auto& instr : *computation.mutable_instructions()) {
589 id_map[instr.id()] = next_instruction_id_++;
590 }
591 for (auto& instr : *computation.mutable_instructions()) {
592 instr.set_id(id_map.at(instr.id()));
