24OpenCLBuffers::OpenCLBuffers(const OpenCL_Network& opencl_net)
25 : m_opencl_net(opencl_net), m_opencl(opencl_net.getOpenCL()) {
26 auto& program = m_opencl.m_program;
27 auto& context = m_opencl.m_context;
28 auto& device = m_opencl.m_device;
29
30 m_convolve1_kernel = cl::Kernel(program, "convolve1");
31 m_merge_kernel = cl::Kernel(program, "merge_bn");
32 m_in_transform_kernel = cl::Kernel(program, "in_transform");
33 m_sgemm_kernel = cl::Kernel(program, "XgemmBatched");
34 m_out_transform_bn_kernel = cl::Kernel(program, "out_transform_fused_bn");
35 m_out_transform_bn_in_kernel =
36 cl::Kernel(program, "out_transform_fused_bn_in");
37 m_global_avg_pooling_kernel = cl::Kernel(program, "global_avg_pooling");
38 m_apply_se_kernel = cl::Kernel(program, "apply_se");
39 m_policymap_kernel = cl::Kernel(program, "policymap");
40 m_sgemv_kernel = cl::Kernel(program, "Xgemv");
41 m_commandqueue = cl::CommandQueue(context, device);
42
43 auto& layers = m_opencl_net.m_layers;
44
45 constexpr auto tiles = WINOGRAD_P;
46 constexpr auto width = 8;
47 constexpr auto height = 8;
48
49 m_finalSize_pol = 0;
50 m_finalSize_val = 0;
51 m_finalSize_mov = 0;
52
53 auto max_channels = unsigned{0};
54 for (const auto& layer : layers) {
55 max_channels =
56 std::max(max_channels, std::max(layer.channels, layer.outputs));
57 if (layer.is_policy || layer.is_conv_policy) {
58 m_finalSize_pol = layer.ip_out_size * sizeof(net_t);
59 }
60 if (layer.is_value) {
61 m_finalSize_val = layer.ip_out_size * sizeof(net_t);
62 }
63 if (layer.is_moves_left) {
64 m_finalSize_mov = layer.ip_out_size * sizeof(net_t);
65 }
66 }
67
68 const auto mwg = m_opencl.m_sgemm_tuners.mwg;
69 const auto nwg = m_opencl.m_sgemm_tuners.nwg;
70 const auto vwm = m_opencl.m_sgemm_tuners.vwm;
71 const auto vwn = m_opencl.m_sgemm_tuners.vwn;
72
73 const auto m_ceil = ceilMultiple(ceilMultiple(max_channels, mwg), vwm);
74 const auto n_ceil = ceilMultiple(ceilMultiple(tiles, nwg), vwn);
75
76 const auto max_batch_size = m_opencl_net.getMaxMatchSize();
77 const auto alloc_inSize =
78 max_batch_size * width * height * max_channels * sizeof(net_t);
79 const auto alloc_vm_size =
80 max_batch_size * WINOGRAD_TILE * m_ceil * n_ceil * sizeof(net_t);
81 const auto alloc_pool_size =
82 max_batch_size * 2 * max_channels * sizeof(net_t);
83
84 auto v_zeros = std::vector<float>(alloc_vm_size);
85
86 m_inBuffer = cl::Buffer(m_opencl.m_context, CL_MEM_READ_WRITE, alloc_inSize);
87 m_inBuffer2 = cl::Buffer(m_opencl.m_context, CL_MEM_READ_WRITE, alloc_inSize);
88 m_VBuffer = cl::Buffer(
89 m_opencl.m_context,
90 CL_MEM_READ_WRITE | CL_MEM_HOST_NO_ACCESS | CL_MEM_COPY_HOST_PTR,
91 alloc_vm_size, v_zeros.data(), nullptr);
92 m_MBuffer =
93 cl::Buffer(m_opencl.m_context, CL_MEM_READ_WRITE | CL_MEM_HOST_NO_ACCESS,
94 alloc_vm_size);
95
96 try {
97 m_pinnedOutBuffer_pol = cl::Buffer(
98 m_opencl.m_context, CL_MEM_WRITE_ONLY | CL_MEM_ALLOC_HOST_PTR,
99 max_batch_size * m_finalSize_pol);
100 } catch (const cl::Error& e) {
101 CERR << "Error in m_pinnedOutBuffer_pol: " << e.what() << ": " << e.err()
102 << std::endl;
103 throw;
104 }
105
106 try {
107 m_pinnedOutBuffer_val = cl::Buffer(
108 m_opencl.m_context, CL_MEM_WRITE_ONLY | CL_MEM_ALLOC_HOST_PTR,
109 max_batch_size * m_finalSize_val);
110 } catch (const cl::Error& e) {
111 CERR << "Error in m_pinnedOutBuffer_val: " << e.what() << ": " << e.err()
112 << std::endl;
113 throw;
114 }
115
116 if (m_finalSize_mov > 0) {
117 try {
118 m_pinnedOutBuffer_mov = cl::Buffer(
119 m_opencl.m_context, CL_MEM_WRITE_ONLY | CL_MEM_ALLOC_HOST_PTR,
120 max_batch_size * m_finalSize_mov);
121 } catch (const cl::Error& e) {
122 CERR << "Error in m_pinnedOutBuffer_mov: " << e.what() << ": " << e.err()
123 << std::endl;
124 throw;
125 }
126 }
127
128 m_pool_buffer =
129 cl::Buffer(m_opencl.m_context, CL_MEM_READ_WRITE | CL_MEM_HOST_NO_ACCESS,
130 alloc_pool_size);
131}

133void OpenCLBuffers::forward(const std::vector<net_t>& input,
134 std::vector<net_t>& output_pol,
135 std::vector<net_t>& output_val,
136 std::vector<net_t>& output_mov,
137 const int batch_size) {
138 auto& layers = m_opencl_net.m_layers;
139
140 const auto inSize = sizeof(net_t) * input.size();
141 m_commandqueue.enqueueWriteBuffer(m_inBuffer, CL_FALSE, 0, inSize,
142 input.data());
143
144 auto skip_in_trans = false;
145 for (auto iter = cbegin(layers); iter != cend(layers); iter++) {
146 const auto& layer = *iter;
147 const auto niter = std::next(iter);
148
149 if (layer.is_input_convolution) {
150 assert(niter != cend(layers));
151 auto conv_weights = begin(layer.weights);
152 auto conv_biases = begin(layer.weights) + 1;
153 auto skip_next_in_trans = false;
154 if (niter->is_residual_block) {
155 skip_next_in_trans = true;
156 }
157 convolve3(layer.channels, layer.outputs, m_inBuffer, m_inBuffer,
158 m_VBuffer, m_MBuffer, conv_weights, nullptr, conv_biases,
159 skip_in_trans, skip_next_in_trans, true, true, batch_size);
160 skip_in_trans = skip_next_in_trans;
161 } else if (layer.is_residual_block) {
162 assert(layer.channels == layer.outputs);
163 assert(niter != cend(layers));
164 auto conv1_weights = begin(layer.weights);
165 auto conv1_biases = begin(layer.weights) + 1;
166 auto conv2_weights = begin(layer.weights) + 2;
167 auto conv2_biases = begin(layer.weights) + 3;
168
169 convolve3(layer.channels, // channels
170 layer.outputs, // outputs
171 m_inBuffer, // bufferIn
172 m_inBuffer2, // bufferOut
173 m_VBuffer, // bufferV
174 m_MBuffer, // bufferM
175 conv1_weights, // weights
176 nullptr, // bufferResidual
177 conv1_biases, // biases
178 skip_in_trans, // skip_in_transform
179 true, // fuse_in_transform
180 false, // store_inout
181 true, // relu
182 batch_size); // batch_size
183
184 auto skip_next_in_trans = false;
185 if (niter->is_residual_block) {
186 skip_next_in_trans = true;
187 }
188 auto relu = true;
189 auto residual = &m_inBuffer;
190 auto out_buffer = m_inBuffer;
191 auto store_inout = true;
192 if (niter->is_se_unit) {
193 // SE unit does relu
194 relu = false;
195 residual = nullptr;
196 out_buffer = m_inBuffer2;
197 store_inout = false;
198 }
199 convolve3(layer.channels, // channels
200 layer.outputs, // outputs
201 m_inBuffer2, // bufferIn
202 out_buffer, // bufferOut
203 m_VBuffer, // bufferV
204 m_MBuffer, // bufferM
205 conv2_weights, // weights
206 residual, // bufferResidual
207 conv2_biases, // biases
208 true, // skip_in_transform
209 skip_next_in_trans, // fuse_in_transform
210 store_inout, // store_inout
211 relu, // relu
212 batch_size); // batch_size
213 skip_in_trans = skip_next_in_trans;
214 } else if (layer.is_se_unit) {
215 // inBuffer: residual connection from start of the residual block
216 // inBuffer2: Last block output
217 // Output will be written in inBuffer
218 assert(niter != cend(layers));
219 auto se_weights = begin(layer.weights);
220 squeeze_excitation(layer.outputs, // channels
221 layer.se_fc_outputs, // fc_outputs
222 m_inBuffer2, // bufferIn
223 m_pool_buffer, // bufferTemp1
224 m_MBuffer, // bufferTemp2
225 se_weights, // weights
226 m_inBuffer, // residual
227 batch_size); // batch_size
228 } else if (layer.is_conv_policy) {
229 assert(niter != cend(layers));
230 auto conv1_weights = begin(layer.weights);
231 auto conv1_biases = begin(layer.weights) + 1;
232 auto conv2_weights = begin(layer.weights) + 2;
233 auto conv2_biases = begin(layer.weights) + 3;
234 auto indices = begin(layer.weights) + 4;
235
236 convolve3(layer.channels, // channels
237 layer.channels, // outputs
238 m_inBuffer, // bufferIn
239 m_inBuffer2, // bufferOut
240 m_VBuffer, // bufferV
241 m_MBuffer, // bufferM
242 conv1_weights, // weights
243 nullptr, // bufferResidual
244 conv1_biases, // biases
245 skip_in_trans, // skip_in_transform
246 true, // fuse_in_transform
247 false, // store_inout
248 true, // relu
249 batch_size); // batch_size
250
251 // m_inBuffer needs to be preserved for value head
252 convolve3(layer.channels, // channels
253 layer.outputs, // outputs
254 m_inBuffer2, // bufferIn
255 m_inBuffer2, // bufferOut
256 m_VBuffer, // bufferV
257 m_MBuffer, // bufferM
258 conv2_weights, // weights
259 nullptr, // bufferResidual
260 conv2_biases, // biases
261 true, // skip_in_transform
262 false, // fuse_in_transform
263 false, // store_inout
264 false, // relu
265 batch_size); // batch_size
266
267 policymap(batch_size, m_inBuffer2, m_pinnedOutBuffer_pol, indices[0],
268 layer.outputs * 8 * 8, layer.ip_in_size, layer.ip_out_size);
269
270 } else {
271 assert(layer.is_value || layer.is_policy || layer.is_moves_left);
272
273 cl::Buffer out_buffer;
274 if (layer.is_policy) {
275 out_buffer = m_pinnedOutBuffer_pol;
276 } else if (layer.is_value) {
277 out_buffer = m_pinnedOutBuffer_val;
278 } else {
279 out_buffer = m_pinnedOutBuffer_mov;
280 }
281
282 auto conv_weights = begin(layer.weights);
283 auto conv_biases = begin(layer.weights) + 1;
284 auto ip_w = begin(layer.weights) + 2;
285 auto ip_b = begin(layer.weights) + 3;
286
287 convolve1(layer.channels, layer.outputs, m_inBuffer, m_inBuffer2,
288 m_VBuffer, conv_weights, conv_biases, batch_size);
289
290 innerproduct(m_inBuffer2, ip_w, ip_b, out_buffer, layer.ip_in_size,
291 layer.ip_out_size, layer.is_value, batch_size);
292 }
293 }
294
295 auto pinnedOutBufferHost_pol = m_commandqueue.enqueueMapBuffer(
296 m_pinnedOutBuffer_pol, CL_FALSE, CL_MAP_READ, 0,
297 batch_size * m_finalSize_pol);
298 auto pinnedOutBufferHost_val = m_commandqueue.enqueueMapBuffer(
299 m_pinnedOutBuffer_val, CL_FALSE, CL_MAP_READ, 0,
300 batch_size * m_finalSize_val);
301 void* pinnedOutBufferHost_mov = nullptr;
302 if (m_finalSize_mov > 0) {
303 pinnedOutBufferHost_mov = m_commandqueue.enqueueMapBuffer(
304 m_pinnedOutBuffer_mov, CL_FALSE, CL_MAP_READ, 0,
305 batch_size * m_finalSize_mov);
306 }
307
308 m_commandqueue.finish();
309
310 std::memcpy(output_pol.data(), pinnedOutBufferHost_pol,
311 batch_size * m_finalSize_pol);
312 std::memcpy(output_val.data(), pinnedOutBufferHost_val,
313 batch_size * m_finalSize_val);
314 if (m_finalSize_mov > 0) {
315 std::memcpy(output_mov.data(), pinnedOutBufferHost_mov,
316 batch_size * m_finalSize_mov);
317 }
318
319 m_commandqueue.enqueueUnmapMemObject(m_pinnedOutBuffer_pol,
320 pinnedOutBufferHost_pol);
321 m_commandqueue.enqueueUnmapMemObject(m_pinnedOutBuffer_val,
322 pinnedOutBufferHost_val);
323 if (m_finalSize_mov > 0) {
324 m_commandqueue.enqueueUnmapMemObject(m_pinnedOutBuffer_mov,
325 pinnedOutBufferHost_mov);
326 }
327}

329void OpenCLBuffers::convolve3(int channels, int outputs, cl::Buffer& bufferIn,
330 cl::Buffer& bufferOut, cl::Buffer& bufferV,
331 cl::Buffer& bufferM, weight_slice_t weights,
332 cl::Buffer* bufferResidual, weight_slice_t biases,
333 bool skip_in_transform, bool fuse_in_transform,
334 bool store_inout, bool relu, int batch_size) {
335 auto mwg = m_opencl.m_sgemm_tuners.mwg;
336 auto nwg = m_opencl.m_sgemm_tuners.nwg;
337 auto kwg = m_opencl.m_sgemm_tuners.kwg;
338 auto vwm = m_opencl.m_sgemm_tuners.vwm;
339 auto vwn = m_opencl.m_sgemm_tuners.vwn;
340 auto mdimc = m_opencl.m_sgemm_tuners.mdimc;
341 auto ndimc = m_opencl.m_sgemm_tuners.ndimc;
342 auto wavefront_size = m_opencl.m_wavefront_size;
343
344 assert(mwg != 0);
345 assert(nwg != 0);
346 assert(kwg != 0);
347 assert(mdimc != 0);
348 assert(ndimc != 0);
349 assert(vwm != 0);
350 assert(vwn != 0);
351 assert(wavefront_size != 0);
352
353 constexpr auto tiles = WINOGRAD_P;
354 constexpr auto width = 8;
355 constexpr auto height = 8;
356
357 auto wgs = ceilMultiple(tiles, wavefront_size);
358 auto m_ceil = int(ceilMultiple(ceilMultiple(outputs, mwg), vwm));
359 auto n_ceil = int(ceilMultiple(ceilMultiple(batch_size * tiles, nwg), vwn));
360 auto k_ceil = int(ceilMultiple(ceilMultiple(channels, kwg), vwm));
361
362 if (!skip_in_transform) {
363 try {
364 m_in_transform_kernel.setArg(0, bufferIn);
365 m_in_transform_kernel.setArg(1, bufferV);
366 m_in_transform_kernel.setArg(2, channels);
367 m_in_transform_kernel.setArg(3, k_ceil);
368 m_in_transform_kernel.setArg(4, n_ceil);
369
370 m_commandqueue.enqueueNDRangeKernel(
371 m_in_transform_kernel, cl::NullRange,
372 cl::NDRange(wgs, channels, batch_size));
373 } catch (const cl::Error& e) {
374 CERR << "Error in convolve3/in: " << e.what() << ": " << e.err()
375 << std::endl;
376 throw;
377 }
378 }
379
380 try {
381 m_sgemm_kernel.setArg(0, m_ceil);
382 m_sgemm_kernel.setArg(1, n_ceil);
383 m_sgemm_kernel.setArg(2, k_ceil);
384 m_sgemm_kernel.setArg(3, weights[0]);
385 m_sgemm_kernel.setArg(4, bufferV);
386 m_sgemm_kernel.setArg(5, bufferM);
387
388 cl::NDRange local_sgemm = {mdimc, ndimc, 1};
389
390 cl::NDRange size_sgemm = {(m_ceil * mdimc) / mwg, (n_ceil * ndimc) / nwg,
391 (cl::size_type)WINOGRAD_TILE};
392
393 m_commandqueue.enqueueNDRangeKernel(m_sgemm_kernel, cl::NullRange,
394 size_sgemm, local_sgemm);
395 } catch (const cl::Error& e) {
396 CERR << "Error in convolve3/sgemm: " << e.what() << ": " << e.err()
397 << std::endl;
398 throw;
399 }
400
401 try {
402 if (fuse_in_transform) {
403 assert(relu); // No relu not supported
404
405 // TODO : Eventually this might also be something tuneable?
406 constexpr auto dim_size = 2;
407 m_out_transform_bn_in_kernel.setArg(0, bufferM);
408 if (store_inout) {
409 m_out_transform_bn_in_kernel.setArg(1, bufferOut);
410 } else {
411 m_out_transform_bn_in_kernel.setArg(1, nullptr);
412 }
413 m_out_transform_bn_in_kernel.setArg(2, bufferV);
414 m_out_transform_bn_in_kernel.setArg(3, outputs);
415 m_out_transform_bn_in_kernel.setArg(4, m_ceil);
416 m_out_transform_bn_in_kernel.setArg(5, n_ceil);
417 // k_ceil of the next convolution
418 auto k_ceil2 = int(ceilMultiple(ceilMultiple(outputs, kwg), vwm));
419 m_out_transform_bn_in_kernel.setArg(6, k_ceil2);
420 if (bufferResidual) {
421 m_out_transform_bn_in_kernel.setArg(7, *bufferResidual);
422 } else {
423 m_out_transform_bn_in_kernel.setArg(7, nullptr);
424 }
425 m_out_transform_bn_in_kernel.setArg(8, biases[0]);
426 m_out_transform_bn_in_kernel.setArg(
427 9, cl::Local(dim_size * width * height * sizeof(float)));
428
429 m_commandqueue.enqueueNDRangeKernel(
430 m_out_transform_bn_in_kernel, cl::NullRange,
431 cl::NDRange(outputs, wgs, batch_size), cl::NDRange(dim_size, wgs, 1));
432 } else {
433 m_out_transform_bn_kernel.setArg(0, bufferM);
434 m_out_transform_bn_kernel.setArg(1, bufferOut);
435 m_out_transform_bn_kernel.setArg(2, outputs);
436 m_out_transform_bn_kernel.setArg(3, m_ceil);
437 m_out_transform_bn_kernel.setArg(4, n_ceil);
438 m_out_transform_bn_kernel.setArg(5, static_cast<int>(relu));
439 if (bufferResidual) {
440 m_out_transform_bn_kernel.setArg(6, *bufferResidual);
441 } else {
442 m_out_transform_bn_kernel.setArg(6, nullptr);
443 }
444 m_out_transform_bn_kernel.setArg(7, biases[0]);
445
446 m_commandqueue.enqueueNDRangeKernel(
447 m_out_transform_bn_kernel, cl::NullRange,
448 cl::NDRange(outputs, wgs, batch_size));
449 }
450 } catch (const cl::Error& e) {
451 CERR << "Error in convolve3/out: " << e.what() << ": " << e.err()
452 << std::endl;
453 throw;
454 }
455}

457void OpenCLBuffers::squeeze_excitation(
458 int channels, int fc_outputs, cl::Buffer& bufferIn, cl::Buffer& bufferTemp1,
459 cl::Buffer& bufferTemp2, weight_slice_t weights, cl::Buffer& bufferResidual,
460 int batch_size) {
461 constexpr int width = 8;
462
463 try {
464 m_global_avg_pooling_kernel.setArg(0, batch_size * channels);
465 m_global_avg_pooling_kernel.setArg(1, bufferIn);
466 m_global_avg_pooling_kernel.setArg(2, bufferTemp1);
467
468 m_commandqueue.enqueueNDRangeKernel(
469 m_global_avg_pooling_kernel, cl::NullRange,
470 cl::NDRange(width, batch_size * channels), cl::NDRange(width, 1));
471 } catch (const cl::Error& e) {
472 CERR << "Error in squeeze_excitation/pooling: " << e.what() << ": "
473 << e.err() << std::endl;
474 throw;
475 }
476
477 innerproduct(bufferTemp1, weights, weights + 1, bufferTemp2, channels,
478 fc_outputs, true, batch_size);
479
480 innerproduct(bufferTemp2, weights + 2, weights + 3, bufferTemp1, fc_outputs,
481 2 * channels, false, batch_size);
482
483 try {
484 m_apply_se_kernel.setArg(0, channels);
485 m_apply_se_kernel.setArg(1, batch_size);
486 m_apply_se_kernel.setArg(2, bufferIn);
487 m_apply_se_kernel.setArg(3, bufferResidual);
488 m_apply_se_kernel.setArg(4, bufferTemp1);
489
490 m_commandqueue.enqueueNDRangeKernel(
491 m_apply_se_kernel, cl::NullRange,
492 cl::NDRange(width, batch_size * channels));
493 } catch (const cl::Error& e) {
494 CERR << "Error in squeeze_excitation/apply_se: " << e.what() << ": "
495 << e.err() << std::endl;
496 throw;
497 }
498}

500void OpenCLBuffers::convolve1(int channels, int outputs,
501 cl::Buffer& bufferInput, cl::Buffer& bufferOutput,
502 cl::Buffer& bufferMerge,
503 weight_slice_t conv_weights,
504 weight_slice_t conv_biases, int batch_size) {
505 // fixed for 8x8.
506 constexpr int width = 8;
507 constexpr int height = 8;
508 constexpr int boardsize = width * height;
509 constexpr int rowTiles = 8;
510
511 // Input channel grouping in multiples of 8.
512 constexpr int channelGroup = 8;
513 constexpr int channelShift = 3;
514 constexpr int rowGroup = 1;
515 // Assumes that if outputs > 16, then outputs is divisible by 16.
516 size_t outputGroup = std::min(outputs, 16);
517
518#ifndef NDEBUG
519 // Total output size after reducing.
520 size_t outSize = width * height * outputs * sizeof(net_t);
521
522 // Produce channel * output planes and merge them at the end.
523 size_t mergeSize = (channels >> channelShift) * outSize;
524 assert(mergeSize <= bufferMerge.getInfo<CL_MEM_SIZE>());
525#endif
526
527 // Copy the rows locally.
528 size_t stripSize = width * sizeof(float);
529
530 int rowBuffer = std::min<int>(channelGroup, 7);
531 size_t rowSize = channelGroup * outputGroup * rowBuffer * sizeof(float);
532
533 try {
534 m_convolve1_kernel.setArg(0, bufferInput);
535 m_convolve1_kernel.setArg(1, bufferMerge);
536 m_convolve1_kernel.setArg(2, conv_weights[0]);
537 m_convolve1_kernel.setArg(3,
538 cl::Local(stripSize * channelGroup * rowGroup));
539 m_convolve1_kernel.setArg(4, cl::Local(rowSize));
540
541 m_commandqueue.enqueueNDRangeKernel(
542 m_convolve1_kernel, cl::NullRange,
543 cl::NDRange(channels, outputs, batch_size * rowTiles),
544 cl::NDRange(channelGroup, outputGroup, rowGroup));
545 } catch (const cl::Error& e) {
546 CERR << "Error in convolve1: " << e.what() << ": " << e.err() << std::endl;
547 throw;
548 }
549
550 assert(channels % (1 << channelShift) == 0);
551
552 try {
553 m_merge_kernel.setArg(0, bufferMerge);
554 m_merge_kernel.setArg(1, bufferOutput);
555 m_merge_kernel.setArg(2, channels >> channelShift);
556 m_merge_kernel.setArg(3, conv_biases[0]);
557
558 m_commandqueue.enqueueNDRangeKernel(
559 m_merge_kernel, cl::NullRange,
560 cl::NDRange(outputs, boardsize, batch_size),
561 cl::NDRange(std::min(8, outputs), 8, 1));
562 } catch (const cl::Error& e) {
563 CERR << "Error in merge: " << e.what() << ": " << e.err() << std::endl;
564 throw;
565 }
566}

568void OpenCLBuffers::innerproduct(cl::Buffer& input, weight_slice_t weights,
569 weight_slice_t biases, cl::Buffer& output,
570 const int inputs, const int outputs,
571 const int relu, int batch_size) {
572 // TODO: Tune these.
573 size_t wgs1 = 64;
574 size_t wpt1 = 1;
575
576 auto m_ceil = int(ceilMultiple(outputs, wgs1 * wpt1));
577 auto global_size = m_ceil / wpt1;
578 auto local_size = wgs1;
579
580 try {
581 // Sets the kernel arguments.
582 m_sgemv_kernel.setArg(0, static_cast<int>(outputs));
583 m_sgemv_kernel.setArg(1, static_cast<int>(inputs));
584 m_sgemv_kernel.setArg(2, weights[0]);
585 m_sgemv_kernel.setArg(3, static_cast<int>(0));
586 m_sgemv_kernel.setArg(4, static_cast<int>(inputs));
587 m_sgemv_kernel.setArg(5, input);
