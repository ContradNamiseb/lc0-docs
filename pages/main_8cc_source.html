45void ChooseAndRunEngine() {
46 // First try the engine which is explicitly specified on the command line.
47 for (const std::string_view search_name :
48 SearchManager::Get()->GetSearchNames()) {
49 if (CommandLine::ConsumeCommand(search_name)) {
50 RunEngine(SearchManager::Get()->GetFactoryByName(search_name));
51 return;
52 }
53 }
54
55 // Then if DEFAULT_SEARCH is defined, run the engine specified by it.
56#ifdef DEFAULT_SEARCH
57 SearchFactory* factory =
58 SearchManager::Get()->GetFactoryByName(DEFAULT_SEARCH);
59 if (!factory) throw Exception("Unknown search algorithm: " DEFAULT_SEARCH);
60 RunEngine(factory);
61 return;
62#endif
63
64 // Then try to run the engine which is specified by the name of the binary.
65 const std::string& binary_name = CommandLine::BinaryName();
66 for (const std::string_view search_name :
67 SearchManager::Get()->GetSearchNames()) {
68 if (binary_name.find(search_name) != std::string::npos) {
69 RunEngine(SearchManager::Get()->GetFactoryByName(search_name));
70 return;
71 }
72 }
73
74 // Finally, run "classic" search through the new API.
75 RunEngine(SearchManager::Get()->GetFactoryByName("classic"));
76}

92 CommandLine::Init(argc, argv);
93 if (CommandLine::BinaryName().find("simple") == std::string::npos) {
94 CommandLine::RegisterMode("selfplay", "Play games with itself");
95 CommandLine::RegisterMode("benchmark", "Quick benchmark");
96 CommandLine::RegisterMode("bench", "Very quick benchmark");
97 CommandLine::RegisterMode("backendbench",
98 "Quick benchmark of backend only");
99 CommandLine::RegisterMode("leela2onnx", "Convert Leela network to ONNX.");
100 CommandLine::RegisterMode("onnx2leela",
101 "Convert ONNX network to Leela net.");
102 CommandLine::RegisterMode("describenet",
103 "Shows details about the Leela network.");
104 }
105 for (const std::string_view search_name :
106 SearchManager::Get()->GetSearchNames()) {
107 CommandLine::RegisterMode(
108 std::string(search_name),
109 "Use \"" + std::string(search_name) + "\" search");
110 }
111
112 if (CommandLine::ConsumeCommand("selfplay")) {
113 // Selfplay mode.
114 StdoutUciResponder uci_responder;
115 SelfPlayLoop loop(&uci_responder);
116 loop.Run();
117 } else if (CommandLine::ConsumeCommand("benchmark")) {
118 // Benchmark mode, longer version.
119 Benchmark benchmark;
120 benchmark.Run();
121 } else if (CommandLine::ConsumeCommand("bench")) {
122 // Benchmark mode, shorter version.
123 Benchmark benchmark;
124 benchmark.Run(/*run_shorter_benchmark=*/true);
125 } else if (CommandLine::ConsumeCommand("backendbench")) {
126 // Backend Benchmark mode.
127 BackendBenchmark benchmark;
128 benchmark.Run();
129 } else if (CommandLine::ConsumeCommand("leela2onnx")) {
130 lczero::ConvertLeelaToOnnx();
131 } else if (CommandLine::ConsumeCommand("onnx2leela")) {
132 lczero::ConvertOnnxToLeela();
133 } else if (CommandLine::ConsumeCommand("describenet")) {
134 lczero::DescribeNetworkCmd();
135 } else {
136 lczero::ChooseAndRunEngine();
137 }
138 } catch (std::exception& e) {
139 std::cerr << "Unhandled exception: " << e.what() << std::endl;
140 abort();
141 }
