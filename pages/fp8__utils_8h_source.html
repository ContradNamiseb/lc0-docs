44inline uint8_t FP32toFP8E5M2(float f32, bool saturate = true) {
45 unsigned int x;
46 unsigned int sign = 0;
47 memcpy(&x, &f32, sizeof(float));
48 if (x & 0x80000000) sign = 0x80;
49 x &= 0x7fffffff;
50 if (x >= 0x47700000) {
51 if (x > 0x7f800000) {
52 x = 0x7f;
53 } else if (saturate) {
54 x = 0x7b;
55 } else {
56 x = 0x7c;
57 }
58 } else if (x < 0x37000000) {
59 x = 0;
60 } else if (x < 0x38700000) {
61 float t = 128 + std::abs(f32);
62 memcpy(&x, &t, sizeof(float));
63 } else {
64 // Adjust exponent and round to nearest even.
65 if (x & 0x2fffff) {
66 x -= 0x37f00000;
67 } else {
68 x -= 0x38000000;
69 }
70 x >>= 21;
71 }
72 return x | sign;
73}

82inline float FP8E5M2toFP32(uint8_t f8) {
83 unsigned int x;
84 float f;
85 x = f8 & 0x7f;
86 if ((x & 0x7c) == 0) {
87 f = 1.5258789e-5f * x;
88 memcpy(&x, &f, sizeof(float));
89 } else if (x >= 0x7c) {
90 if (x & 0x1) x |= 0x2;
91 x = (x + 0x380) << 21;
92 } else {
93 x = (x + 0x1c0) << 21;
94 }
95 if (f8 & 0x80) x |= 0x80000000;
96 memcpy(&f, &x, sizeof(float));
97 return f;
98}

104 H = _mm_insert_epi8(H, f8, 1);

111inline uint8_t FP32toFP8E4M3FN(float f32, bool saturate = true) {
112 unsigned int x;
113 unsigned int sign = 0;
114 memcpy(&x, &f32, sizeof(float));
115 if (x & 0x80000000) sign = 0x80;
116 x &= 0x7fffffff;
117 if (x >= 0x43e80000) {
118 if (saturate && x <= 0x7f800000) {
119 x = 0x7e;
120 } else {
121 x = 0x7f;
122 }
123 } else if (x < 0x3a800000) {
124 x = 0;
125 } else if (x < 0x3c800000) {
126 float t = 16384 + std::abs(f32);
127 memcpy(&x, &t, sizeof(float));
128 } else {
129 // Adjust exponent and round to nearest even.
130 if (x & 0x17ffff) {
131 x -= 0x3bf80000;
132 } else {
133 x -= 0x3c000000;
134 }
135 x >>= 20;
136 }
137 return x | sign;
138}

140inline float FP8E4M3FNtoFP32(uint8_t f8) {
141 unsigned int x;
142 float f;
143 x = f8 & 0x7f;
144 if ((x & 0x78) == 0) {
145 f = .001953125f * x;
146 memcpy(&x, &f, sizeof(float));
147 } else if (x == 0x7f) {
148 x = 0x7ff00000;
149 } else {
150 x = (x + 0x3c0) << 20;
151 }
152 if (f8 & 0x80) x |= 0x80000000;
153 memcpy(&f, &x, sizeof(float));
154 return f;
155}
