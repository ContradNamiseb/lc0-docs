100class NetworkAsBackendComputation : public BackendComputation {
101 public:
102 NetworkAsBackendComputation(NetworkAsBackend* backend)
103 : backend_(backend),
104 computation_(backend_->network_->NewComputation()),
105 entries_(backend_->attrs_.maximum_batch_size) {}
106
107 size_t UsedBatchSize() const override { return entries_.size(); }
108
109 AddInputResult AddInput(const EvalPosition& pos,
110 EvalResultPtr result) override {
111 int transform;
112 const size_t idx = entries_.emplace_back(Entry{
113 .input = EncodePositionForNN(backend_->input_format_, pos.pos, 8,
114 backend_->fill_empty_history_, &transform),
115 .legal_moves = MoveList(pos.legal_moves.begin(), pos.legal_moves.end()),
116 .result = result,
117 .transform = 0});
118 entries_[idx].transform = transform;
119 return ENQUEUED_FOR_EVAL;
120 }
121
122 void ComputeBlocking() override {
123 for (auto& entry : entries_) computation_->AddInput(std::move(entry.input));
124 computation_->ComputeBlocking();
125 LCTRACE_FUNCTION_SCOPE;
126 for (size_t i = 0; i < entries_.size(); ++i) {
127 const EvalResultPtr& result = entries_[i].result;
128 if (result.q) *result.q = computation_->GetQVal(i);
129 if (result.d) *result.d = computation_->GetDVal(i);
130 if (result.m) *result.m = computation_->GetMVal(i);
131 if (!result.p.empty()) SoftmaxPolicy(result.p, computation_.get(), i);
132 }
133 }
134
135 void SoftmaxPolicy(std::span<float> dst,
136 const NetworkComputation* computation, int idx) {
137 LCTRACE_FUNCTION_SCOPE;
138 const std::vector<Move>& moves = entries_[idx].legal_moves;
139 const int transform = entries_[idx].transform;
140 // Copy the values to the destination array and compute the maximum.
141 const float max_p = std::accumulate(
142 moves.begin(), moves.end(), -std::numeric_limits<float>::infinity(),
143 [&, counter = 0](float max_p, const Move& move) mutable {
144 return std::max(max_p, dst[counter++] = computation->GetPVal(
145 idx, MoveToNNIndex(move, transform)));
146 });
147 // Compute the softmax and compute the total.
148 const float temperature = backend_->softmax_policy_temperature_;
149 float total = std::accumulate(
150 dst.begin(), dst.end(), 0.0f, [&](float total, float& val) {
151 return total + (val = FastExp((val - max_p) * temperature));
152 });
153 const float scale = total > 0.0f ? 1.0f / total : 1.0f;
154 // Scale the values to sum to 1.0.
155 std::for_each(dst.begin(), dst.end(), [&](float& val) { val *= scale; });
156 }
157
158 private:
159 struct Entry {
160 InputPlanes input;
161 MoveList legal_moves;
162 EvalResultPtr result;
163 int transform;
164 };
165
166 NetworkAsBackend* backend_;
167 std::unique_ptr<NetworkComputation> computation_;
168 AtomicVector<Entry> entries_;
169};

182std::unique_ptr<Backend> NetworkAsBackendFactory::Create(
183 const OptionsDict& options) {
184 const std::string backend_options =
185 options.Get<std::string>(SharedBackendParams::kBackendOptionsId);
186 OptionsDict network_options;
187 network_options.AddSubdictFromString(backend_options);
188
189 std::string net_path =
190 options.Get<std::string>(SharedBackendParams::kWeightsId);
191 std::optional<WeightsFile> weights = LoadWeights(net_path);
192 std::unique_ptr<Network> network =
193 factory_(std::move(weights), network_options);
194 network_options.CheckAllOptionsRead(name_);
195 return std::make_unique<NetworkAsBackend>(std::move(network), options);
196}
