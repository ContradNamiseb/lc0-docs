60std::unique_ptr<SearchStopper> AlphazeroTimeManager::GetStopper(
61 const GoParams& params, const Position& position, size_t /*total_memory*/,
62 size_t /*avg_node_size*/, uint32_t /*nodes*/) {
63 const bool is_black = position.IsBlackToMove();
64 const std::optional<int64_t>& time = (is_black ? params.btime : params.wtime);
65 // If no time limit is given, don't stop on this condition.
66 if (params.infinite || params.ponder || !time) return nullptr;
67
68 auto total_moves_time = *time - move_overhead_;
69
70 float this_move_time = total_moves_time * (alphazerotimepct_ / 100.0f);
71
72 LOGFILE << "Budgeted time for the move: " << this_move_time << "ms"
73 << "Remaining time " << *time << "ms(-" << move_overhead_
74 << "ms overhead)";
75
76 return std::make_unique<TimeLimitStopper>(this_move_time);
77}
