68void SharedBackendParams::Populate(OptionsParser* options) {
69 options->Add<FloatOption>(kPolicySoftmaxTemp, 0.1f, 10.0f) = 1.359f;
70 std::vector<std::string> history_fill_opt{"no", "fen_only", "always"};
71 options->Add<ChoiceOption>(kHistoryFill, history_fill_opt) = "fen_only";
72
73#if defined(EMBED)
74 options->Add<StringOption>(SharedBackendParams::kWeightsId) = kEmbed;
75#else
76 options->Add<StringOption>(SharedBackendParams::kWeightsId) = kAutoDiscover;
77#endif
78 const auto backends = BackendManager::Get()->GetBackendNames();
79 options->Add<ChoiceOption>(SharedBackendParams::kBackendId, backends) =
80 backends.empty() ? "<none>" : backends[0];
81 options->Add<StringOption>(SharedBackendParams::kBackendOptionsId);
82 options->Add<IntOption>(SharedBackendParams::kNNCacheSizeId, 0, 999999999) =
83 2000000;
84}
