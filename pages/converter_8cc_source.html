184pblczero::TensorProto::DataType Converter::GetDataType() const {
185 switch (options_.data_type) {
186 case WeightsToOnnxConverterOptions::DataType::kFloat32:
187 return pblczero::TensorProto::FLOAT;
188 case WeightsToOnnxConverterOptions::DataType::kFloat16:
189 return pblczero::TensorProto::FLOAT16;
190 case WeightsToOnnxConverterOptions::DataType::kBFloat16:
191 return pblczero::TensorProto::BFLOAT16;
192 default:
193 return pblczero::TensorProto::UNDEFINED;
194 }
195}

197std::unique_ptr<OnnxConst> Converter::GetWeghtsConverter(
198 const std::vector<float>& weights, std::initializer_list<int> dims,
199 std::initializer_list<int> order) {
200 switch (options_.data_type) {
201 case WeightsToOnnxConverterOptions::DataType::kFloat32:
202 return std::make_unique<FloatOnnxWeightsAdapter>(weights, dims, order);
203 case WeightsToOnnxConverterOptions::DataType::kFloat16:
204 return std::make_unique<Float16OnnxWeightsAdapter>(weights, dims, order);
205 case WeightsToOnnxConverterOptions::DataType::kBFloat16:
206 return std::make_unique<BFloat16OnnxWeightsAdapter>(weights, dims, order);
207 }
208 throw Exception("Data type " +
209 std::to_string(static_cast<int>(options_.data_type)) +
210 " is not supported in weights converter");
211}

213std::unique_ptr<OnnxConst> Converter::GetScalarConverter(float in) {
214 switch (options_.data_type) {
215 case WeightsToOnnxConverterOptions::DataType::kFloat32:
216 return std::make_unique<FloatOnnxConst>(FloatOnnxConst({in}, {1}));
217 case WeightsToOnnxConverterOptions::DataType::kFloat16:
218 return std::make_unique<Float16OnnxConst>(
219 Float16OnnxConst({FP32toFP16(in)}, {1}));
220 case WeightsToOnnxConverterOptions::DataType::kBFloat16:
221 return std::make_unique<BFloat16OnnxConst>(
222 BFloat16OnnxConst({FP32toBF16(in)}, {1}));
223 }
224 throw Exception("Data type " +
225 std::to_string(static_cast<int>(options_.data_type)) +
226 " is not supported in scalar converter");
227}

230 std::string flow,
231 std::string name) {
232 if (options_.opset >= 22 ||

241 std::string flow, std::string name) {
242 if (options_.opset >= 22 ||
243 options_.data_type !=
244 WeightsToOnnxConverterOptions::DataType::kBFloat16) {
245 return flow;
246 }
247 return builder->Cast(name + "/to_bf16", flow,
248 pblczero::TensorProto::BFLOAT16);

251std::string Converter::MakeMish(OnnxBuilder* builder, const std::string& input,
252 const std::string& name) {
253 if (!options_.alt_mish) {
254 std::string flow = input;
255 flow = StartOptionalBf16Fix(builder, flow, name);
256 if (options_.opset >= 18 && options_.real_mish) {
257 flow = builder->Mish(name, flow);
258 return EndOptionalBf16Fix(builder, flow, name);
259 }
260 flow = builder->Softplus(name + "/softplus", flow);
261 flow = EndOptionalBf16Fix(builder, flow, name);
262 flow = builder->Tanh(name + "/tanh", flow);
263 return builder->Mul(name, flow, input);
264 } else {
265 auto in = input;
266 auto one = builder->AddInitializer(name + "/one", *GetScalarConverter(1));
267 auto two = builder->AddInitializer(name + "/two", *GetScalarConverter(2));
268 auto e = builder->Exp(name + "/e", in);
269 auto flow = builder->Add(name + "/e+2", e, two);
270 flow = builder->Mul(name + "/e*e+2e", e, flow);
271 flow = builder->Div(name + "/2/(e*e+2e)", two, flow);
272 flow = builder->Add(name + "/1+2/(e*e+2e)", flow, one);
273 flow = builder->Div(name + "/in/(1+2/(e*e+2e))", in, flow);
274 return flow;
275 }
276}

278std::string Converter::MakeSwish(OnnxBuilder* builder, const std::string& input,
279 const std::string& name) {
280 auto flow = builder->Sigmoid(name + "/sigmoid", input);
281 return builder->Mul(name, flow, input);

284std::string Converter::MakeActivation(OnnxBuilder* builder,
285 const std::string& input,
286 const std::string& name,
287 ActivationFunction activation) {
288 switch (activation) {
289 case ACTIVATION_RELU:
290 return builder->Relu(name + "/relu", input);
291 case ACTIVATION_MISH:
292 return MakeMish(builder, input, name + "/mish");
293 case ACTIVATION_SELU: {
294 auto flow = input;
295 flow = StartOptionalBf16Fix(builder, flow, name);
296 flow = builder->Selu(name + "/selu", flow);
297 return EndOptionalBf16Fix(builder, flow, name);
298 }
299 case ACTIVATION_SWISH:
300 return MakeSwish(builder, input, name + "/swish");
301 case ACTIVATION_RELU_2: {
302 auto flow = builder->Relu(name + "/sqrrelu/relu", input);
303 return builder->Mul(name + "/sqrrelu/sqr", flow, flow);
304 }
305 case ACTIVATION_NONE:
306 return input;
307 default:
308 throw Exception("Unsupported activation in " + name);
309 }
310}

312std::string Converter::MakeSqueezeAndExcite(
313 OnnxBuilder* builder, const MultiHeadWeights::SEunit& se_unit,
314 const std::string& input, const std::string& name) {
315 const int se_filters = se_unit.b1.size();
316
317 if (!se_reshape_init_) {
318 builder->AddInitializer("/const/se_reshape",
319 Int64OnnxConst({-1, NumFilters() * 2, 1, 1}, {4}));
320 se_reshape_init_ = true;
321 }
322 auto flow = input;
323 flow = builder->ReduceMean(name + "/reduce_mean", flow, {2, 3}, false);
324 flow = builder->MatMul(
325 name + "/matmul1", flow,
326 *GetWeghtsConverter(se_unit.w1, {NumFilters(), se_filters}, {1, 0}));
327 flow = builder->Add(name + "/add1", flow,
328 *GetWeghtsConverter(se_unit.b1, {se_filters}));
329 flow = MakeActivation(builder, flow, name, default_activation_);
330 flow = builder->MatMul(
331 name + "/matmul2", flow,
332 *GetWeghtsConverter(se_unit.w2, {se_filters, 2 * NumFilters()}, {1, 0}));
333 flow = builder->Add(name + "/add2", flow,
334 *GetWeghtsConverter(se_unit.b2, {2 * NumFilters()}));
335 flow = builder->Reshape(name + "/reshape", flow, "/const/se_reshape");
336
337 auto splits = builder->Split(name + "/split", flow, 1);
338
339 flow = builder->Sigmoid(name + "/sigmoid", splits[0]);
340 flow = builder->Mul(name + "/mul", flow, input);
341 return builder->Add(name + "/add3", flow, splits[1]);
342}

344std::string Converter::MakeConvBlock(
345 OnnxBuilder* builder, const MultiHeadWeights::ConvBlock& weights,
346 int input_channels, int output_channels, const std::string& input,
347 const std::string& name, const MultiHeadWeights::SEunit* seunit,
348 const std::string& mixin, bool activation, int filters) {
349 auto flow = input;
350 if (options_.opset < 22 &&
351 options_.data_type ==
352 WeightsToOnnxConverterOptions::DataType::kBFloat16) {
353 flow =
354 builder->Cast(name + "/to_float", flow, pblczero::TensorProto::FLOAT);
355 flow = builder->Conv(
356 name, flow, *std::make_unique<FloatOnnxWeightsAdapter>(
357 weights.weights, std::initializer_list<int>(
358 {output_channels, input_channels,
359 filters, filters})),
360 *std::make_unique<FloatOnnxWeightsAdapter>(
361 weights.biases, std::initializer_list<int>({output_channels})),
362 (filters - 1) / 2);
363 flow =
364 builder->Cast(name + "/to_bf16", flow, pblczero::TensorProto::BFLOAT16);
365 } else {
366 flow = builder->Conv(
367 name, flow,
368 *GetWeghtsConverter(weights.weights, {output_channels, input_channels,
369 filters, filters}),
370 *GetWeghtsConverter(weights.biases, {output_channels}),
371 (filters - 1) / 2);
372 }
373 if (seunit) flow = MakeSqueezeAndExcite(builder, *seunit, flow, name + "/se");
374 if (!mixin.empty()) flow = builder->Add(name + "/mixin", flow, mixin);
375 if (activation) {
376 flow = MakeActivation(builder, flow, name, default_activation_);
377 }
378 return flow;
379}

381std::string Converter::MakeResidualBlock(OnnxBuilder* builder,
382 const MultiHeadWeights::Residual& res,
383 const std::string& input,
384 const std::string& name) {

388 name + "/conv2", res.has_se ? &res.se : nullptr, input);

391std::string Converter::MakeSmolgen(OnnxBuilder* builder,
392 const MultiHeadWeights::EncoderLayer& layer,
393 int embedding_size, int heads,
394 const std::string& encoder_in,
395 const std::string& name) {
396 const auto smolgen_activation = static_cast<ActivationFunction>(
397 src_.format().network_format().smolgen_activation());
398 const auto activation = smolgen_activation == ACTIVATION_DEFAULT
399 ? default_activation_
400 : smolgen_activation;
401 const int smolgen_hidden_channels =
402 layer.mha.smolgen.compress.size() / embedding_size;
403 const int smolgen_hidden_sz = layer.mha.smolgen.dense1_b.size();
404 const int smolgen_gen_sz = layer.mha.smolgen.dense2_b.size() / heads;
405 auto flow = builder->MatMul(
406 name + "/smolgen/compress", encoder_in,
407 *GetWeghtsConverter(layer.mha.smolgen.compress,
408 {embedding_size, smolgen_hidden_channels}, {1, 0}));
409 flow = builder->Reshape(
410 name + "/smolgen/compress/reshape", flow,
411 builder->AddInitializer(
412 "/const" + name + "/smolgen/compress/shape",
413 Int64OnnxConst({-1, 64 * smolgen_hidden_channels}, {2})));
414 flow = builder->MatMul(
415 name + "/smolgen/dense1/w", flow,
416 *GetWeghtsConverter(layer.mha.smolgen.dense1_w,
417 {64 * smolgen_hidden_channels, smolgen_hidden_sz},
418 {1, 0}));
419 flow = builder->Add(
420 name + "/smolgen/dense1/b", flow,
421 *GetWeghtsConverter(layer.mha.smolgen.dense1_b, {smolgen_hidden_sz}));
422 flow = MakeActivation(builder, flow, name + "/smolgen/dense1", activation);
423 flow = MakeLayerNorm(
424 builder, flow, name + "/smolgen/ln1",
425 *GetWeghtsConverter(layer.mha.smolgen.ln1_gammas, {smolgen_hidden_sz}),
426 *GetWeghtsConverter(layer.mha.smolgen.ln1_betas, {smolgen_hidden_sz}),
427 1e-3);
428 flow = builder->MatMul(
429 name + "/smolgen/dense2/w", flow,
430 *GetWeghtsConverter(layer.mha.smolgen.dense2_w,
431 {smolgen_hidden_sz, smolgen_gen_sz * heads}, {1, 0}));
432 flow = builder->Add(name + "/smolgen/dense2/b", flow,
433 *GetWeghtsConverter(layer.mha.smolgen.dense2_b,
434 {smolgen_gen_sz * heads}));
435 flow = MakeActivation(builder, flow, name + "/smolgen/dense2", activation);
436 flow = MakeLayerNorm(builder, flow, name + "/smolgen/ln2",
437 *GetWeghtsConverter(layer.mha.smolgen.ln2_gammas,
438 {smolgen_gen_sz * heads}),
439 *GetWeghtsConverter(layer.mha.smolgen.ln2_betas,
440 {smolgen_gen_sz * heads}),
441 1e-3);
442 flow =
443 builder->Reshape(name + "/smolgen/gen_from/reshape", flow,
444 builder->AddInitializer(
445 "/const" + name + "/smolgen/gen_from/shape",
446 Int64OnnxConst({-1, heads, smolgen_gen_sz}, {3})));
447 flow = builder->MatMul(name + "/smolgen/smol_weight_gen", flow,
448 "/const/smolgen_w");
449 flow = builder->Reshape(
450 name + "/smolgen/out/reshape", flow,
451 builder->AddInitializer("/const" + name + "/smolgen/out/shape",
452 Int64OnnxConst({-1, heads, 64, 64}, {4})));
453 return flow;
454}

456std::string Converter::MakeLayerNorm(OnnxBuilder* builder,
457 const std::string& input,
458 const std::string& name,
459 const lczero::OnnxConst& gammas,
460 const lczero::OnnxConst& betas,
461 float eps) {
462 if (!options_.alt_layernorm) {
463 return builder->LayerNormalization(name, input, gammas, betas, 1, eps);
464 }
465 auto in = input;
466 if (GetDataType() != pblczero::TensorProto::FLOAT) {
467 in = builder->Cast(name + "/to_float", in, pblczero::TensorProto::FLOAT);
468 }
469 auto flow = builder->ReduceMean(name + "/mean", in, {1});
470 in = builder->Sub(name + "/centered", in, flow);
471 flow = builder->Mul(name + "/squared", in, in);
472 flow = builder->ReduceMean(name + "/var", flow, {1});
473 flow =
474 builder->Add(name + "/var_eps", flow,
475 static_cast<const OnnxConst&>(FloatOnnxConst({eps}, {1})));
476 flow = builder->Sqrt(name + "/std", flow);
477 flow = builder->Reciprocal(name + "/inv_std", flow);
478 flow = builder->Mul(name + "/normalized", in, flow);
479 if (GetDataType() != pblczero::TensorProto::FLOAT) {
480 flow = builder->Cast(name + "/to_data_type", flow, GetDataType());
481 }
482 flow = builder->Mul(name + "/gammas", flow, gammas);
483 flow = builder->Add(name + "/betas", flow, betas);
484 return flow;
485}

487std::string Converter::MakeFFN(OnnxBuilder* builder,
488 const MultiHeadWeights::FFN& ffn,
489 int embedding_size, const std::string& ffn_in,
490 const std::string& name,
491 ActivationFunction activation, float alpha) {
492 const int dff_size = ffn.dense1_b.size();
493 auto flow = builder->MatMul(
494 name + "/ffn/dense1/w", ffn_in,
495 *GetWeghtsConverter(ffn.dense1_w, {embedding_size, dff_size}, {1, 0}));
496 flow = builder->Add(name + "/ffn/dense1/b", flow,
497 *GetWeghtsConverter(ffn.dense1_b, {dff_size}));
498 flow = MakeActivation(builder, flow, name + "/ffn/dense1", activation);
499 flow = builder->MatMul(
500 name + "/ffn/dense2/w", flow,
501 *GetWeghtsConverter(ffn.dense2_w, {dff_size, embedding_size}, {1, 0}));
502 flow = builder->Add(name + "/ffn/dense2/b", flow,
503 *GetWeghtsConverter(ffn.dense2_b, {embedding_size}));
504 if (alpha != 1.0) {
505 flow = builder->Mul(name + "/ffn/alpha", flow, *GetScalarConverter(alpha));
506 }
507 flow = builder->Add(name + "/ffn/skip", flow, ffn_in);
508 return flow;
509}

511std::string Converter::MakeEncoderLayer(
512 OnnxBuilder* builder, const MultiHeadWeights::EncoderLayer& layer,
513 int embedding_size, int heads, const std::string& encoder_in,
514 const std::string& name, ActivationFunction activation, float alpha) {
515 const int d_model = layer.mha.q_b.size();
516 const int depth = d_model / heads;
517
518 auto mha_shape =
519 builder->AddInitializer("/const" + name + "/mha/shape",
520 Int64OnnxConst({-1, 64, heads, depth}, {4}));
521 auto flow = builder->MatMul(
522 name + "/mha/Q/w", encoder_in,
523 *GetWeghtsConverter(layer.mha.q_w, {embedding_size, d_model}, {1, 0}));
524 flow = builder->Add(name + "/mha/Q/b", flow,
525 *GetWeghtsConverter(layer.mha.q_b, {d_model}));
526 flow = builder->Reshape(name + "/mha/Q/reshape", flow, mha_shape);
527 auto Q = builder->Transpose(name + "/mha/Q/transpose", flow, {0, 2, 1, 3});
528 flow = builder->MatMul(
529 name + "/mha/K/w", encoder_in,
530 *GetWeghtsConverter(layer.mha.k_w, {embedding_size, d_model}, {1, 0}));
531 flow = builder->Add(name + "/mha/K/b", flow,
532 *GetWeghtsConverter(layer.mha.k_b, {d_model}));
533 flow = builder->Reshape(name + "/mha/K/reshape", flow, mha_shape);
534 auto K = builder->Transpose(name + "/mha/K/transpose", flow, {0, 2, 3, 1});
535 flow = builder->MatMul(
536 name + "/mha/V/w", encoder_in,
537 *GetWeghtsConverter(layer.mha.v_w, {embedding_size, d_model}, {1, 0}));
538 flow = builder->Add(name + "/mha/V/b", flow,
539 *GetWeghtsConverter(layer.mha.v_b, {d_model}));
540 flow = builder->Reshape(name + "/mha/V/reshape", flow, mha_shape);
541 auto V = builder->Transpose(name + "/mha/V/transpose", flow, {0, 2, 1, 3});
542 flow = builder->MatMul(name + "/mha/QK/matmul", Q, K);
543 flow = builder->Mul(name + "/mha/QK/scale", flow,
544 *GetScalarConverter(1.0f / sqrtf(depth)));
545 if (layer.mha.has_smolgen) {
546 auto smolgen_weights =
547 MakeSmolgen(builder, layer, embedding_size, heads, encoder_in, name);
548 flow = builder->Add(name + "/smolgen_weights", flow, smolgen_weights);
549 }
550 flow = builder->Softmax(name + "/mha/QK/softmax", flow, 3);
551 flow = builder->MatMul(name + "/mha/QKV/matmul", flow, V);
552 if (heads > 1) {
553 flow = builder->Transpose(name + "/mha/out/transpose", flow, {0, 2, 1, 3});
554 }
555 flow = builder->Reshape(
556 name + "/mha/out/reshape", flow,
557 builder->AddInitializer("/const" + name + "/mha/out/shape",
558 Int64OnnxConst({-1, d_model}, {2})));
559 flow =
560 builder->MatMul(name + "/mha/out/dense/w", flow,
561 *GetWeghtsConverter(layer.mha.dense_w,
562 {d_model, embedding_size}, {1, 0}));
563 flow = builder->Add(name + "/mha/out/dense/b", flow,
564 *GetWeghtsConverter(layer.mha.dense_b, {embedding_size}));
565 if (alpha != 1.0) {
566 flow =
567 builder->Mul(name + "/alpha*input", flow, *GetScalarConverter(alpha));
568 }
569 flow = builder->Add(name + "/mha/out/skip", flow, encoder_in);
570 flow = MakeLayerNorm(builder, flow, name + "/ln1",
571 *GetWeghtsConverter(layer.ln1_gammas, {embedding_size}),
572 *GetWeghtsConverter(layer.ln1_betas, {embedding_size}),
573 default_eps_);
574 const auto ffn_activation = static_cast<ActivationFunction>(
575 src_.format().network_format().ffn_activation());
576 flow = MakeFFN(
577 builder, layer.ffn, embedding_size, flow, name,
578 ffn_activation == ACTIVATION_DEFAULT ? activation : ffn_activation,
579 alpha);
580 flow = MakeLayerNorm(builder, flow, name + "/ln2",
581 *GetWeghtsConverter(layer.ln2_gammas, {embedding_size}),
582 *GetWeghtsConverter(layer.ln2_betas, {embedding_size}),
583 default_eps_);
584 return flow;
585}

639 builder->AddInitializer(
640 "/const/pos_encoding",
641 *GetWeghtsConverter(
642 std::vector<float>(kPosEncoding[0], kPosEncoding[0] + 64 * 64),

644 pad);
645 }
646 flow = builder->Concat("/attn_body/padded_input", {flow, pad}, 2);

695std::string Converter::MakeAttentionBody(OnnxBuilder* builder,
696 const std::string& input,
697 const MultiHeadWeights& weights) {
698 if (weights.has_smolgen) {
699 builder->AddInitializer(
700 "/const/smolgen_w",
701 *GetWeghtsConverter(
702 weights.smolgen_w,
703 {static_cast<int>(weights.smolgen_w.size() / 4096), 4096}, {1, 0}));
704 }
705 auto input_embedding = src_.format().network_format().input_embedding();
706 using network_format = pblczero::NetworkFormat;
707 auto flow = builder->Transpose("/attn_body/transpose", input, {0, 2, 3, 1});
708 int fist_stage_out_C = 0;
709
710 if (NumResBlocks() > 0) {
711 flow = builder->Reshape(
712 "/attn_body/reshape", flow,
713 builder->AddInitializer("/const/att_body_shape",
714 Int64OnnxConst({-1, NumFilters()}, {2})));
715 fist_stage_out_C = NumFilters();
716 } else if (input_embedding == network_format::INPUT_EMBEDDING_PE_MAP) {
717 flow = AttentionBodyMapEmbedding(builder, flow);
718 fist_stage_out_C = 176;
719 } else if (input_embedding == network_format::INPUT_EMBEDDING_PE_DENSE) {
720 int embedding_dense_size = weights.ip_emb_preproc_b.size() / 64;
721 flow = AttentionBodyDenseEmbedding(builder, flow, weights,
722 embedding_dense_size);
723 fist_stage_out_C = 112 + embedding_dense_size;
724 } else {
725 throw Exception("Attention body missing input embedding.");
726 }
727
728 int embedding_size = weights.ip_emb_b.size();
729 flow = builder->MatMul(
730 "/attn_body/matmul", flow,
731 *GetWeghtsConverter(weights.ip_emb_w, {fist_stage_out_C, embedding_size},
732 {1, 0}));
733 flow = builder->Add("/attn_body/add", flow,
734 *GetWeghtsConverter(weights.ip_emb_b, {embedding_size}));
735 flow = MakeActivation(builder, flow, "/attn_body", default_activation_);
736
737 if (input_embedding == network_format::INPUT_EMBEDDING_PE_DENSE) {
738 flow = MakeLayerNorm(
739 builder, flow, "/attn_body/ln",
740 *GetWeghtsConverter(weights.ip_emb_ln_gammas, {embedding_size}),
741 *GetWeghtsConverter(weights.ip_emb_ln_betas, {embedding_size}), 1e-3);
742 }
743
744 if (weights.ip_mult_gate.size() > 0 || weights.ip_add_gate.size() > 0) {
745 flow = builder->Reshape(
746 "/attn_body/ma_gating/rehape", flow,
747 builder->AddInitializer("/const/ma_gating/shape1",
748 Int64OnnxConst({-1, 64, embedding_size}, {3})));
749 if (weights.ip_mult_gate.size() > 0) {
750 flow = builder->Mul("/ip_mul_gate", flow,
751 *GetWeghtsConverter(weights.ip_mult_gate,
752 {64, embedding_size}, {1, 0}));
753 }
754 if (weights.ip_add_gate.size() > 0) {
755 flow = builder->Add("/ip_add_gate", flow,
756 *GetWeghtsConverter(weights.ip_add_gate,
757 {64, embedding_size}, {1, 0}));
758 }
759 }
760
761 flow = builder->Reshape(
762 "/attn_body/rehape", flow,
763 builder->AddInitializer("/const/ma_gating/shape2",
764 Int64OnnxConst({-1, embedding_size}, {2})));
765
766 float alpha = std::pow(2.0f * NumEncBlocks(), -0.25f);
767
768 if (input_embedding == network_format::INPUT_EMBEDDING_PE_DENSE) {
769 const auto ffn_activation = static_cast<ActivationFunction>(
770 src_.format().network_format().ffn_activation());
771 flow =
772 MakeFFN(builder, weights.ip_emb_ffn, embedding_size, flow, "/attn_body",
773 ffn_activation == ACTIVATION_DEFAULT ? default_activation_
774 : ffn_activation,
775 alpha);
776 flow = MakeLayerNorm(
777 builder, flow, "/attn_body/ln2",
778 *GetWeghtsConverter(weights.ip_emb_ffn_ln_gammas, {embedding_size}),
779 *GetWeghtsConverter(weights.ip_emb_ffn_ln_betas, {embedding_size}),
780 1e-3);
781 }
782
783 for (size_t i = 0; i < NumEncBlocks(); i++) {
784 flow = MakeEncoderLayer(
785 builder, weights.encoder[i], embedding_size, weights.encoder_head_count,
786 flow, "/encoder" + std::to_string(i), default_activation_, alpha);
787 }
788 return flow;
789}

803std::string Converter::MakeAttentionPolicy(
804 OnnxBuilder* builder, const std::string& input,
805 const MultiHeadWeights& weights, const MultiHeadWeights::PolicyHead& head) {
806 if (head.ip2_pol_b.empty()) {
807 throw Exception("The policy head selected '" + options_.policy_head + "'" +
808 " is empty.");
809 }
810 const int embedding_size = weights.ip_emb_b.size();
811 const int policy_embedding_size = head.ip_pol_b.size();
812 const int policy_d_model = head.ip2_pol_b.size();
813 auto flow = input;
814 auto activation = src_.format().network_format().network() >=
815 pblczero::NetworkFormat::
816 NETWORK_ATTENTIONBODY_WITH_HEADFORMAT
817 ? default_activation_
818 : ACTIVATION_SELU;
819 if (NumEncBlocks() == 0) {
820 flow = builder->Transpose("/policy/dense1/transpose", flow, {0, 2, 3, 1});
821
822 flow = builder->Reshape(
823 "/policy/dense1/reshape", flow,
824 builder->AddInitializer("/const/policy_shape",
825 Int64OnnxConst({-1, NumFilters()}, {2})));
826 }
827 flow = builder->MatMul(
828 "/policy/dense1/matmul", flow,
829 *GetWeghtsConverter(head.ip_pol_w,
830 {NumEncBlocks() > 0 ? embedding_size : NumFilters(),
831 policy_embedding_size},
832 {1, 0}));
833 flow = builder->Add("/policy/dense1/add", flow,
834 *GetWeghtsConverter(head.ip_pol_b,
835 {policy_embedding_size}));
836 flow = MakeActivation(builder, flow, "/policy/dense1", activation);
837
838 for (size_t i = 0; i < head.pol_encoder.size(); i++) {
839 std::string name = "/policy/enc_layer_" + std::to_string(i);
840
841 flow =
842 MakeEncoderLayer(builder, head.pol_encoder[i], policy_embedding_size,
843 head.pol_encoder_head_count, flow, name, activation);
844 }
845 auto encoder_out = flow;
846 flow = builder->MatMul(
847 "/policy/Q/matmul", encoder_out,
848 *GetWeghtsConverter(head.ip2_pol_w,
849 {policy_embedding_size, policy_d_model}, {1, 0}));
850 flow = builder->Add("/policy/Q/add", flow,
851 *GetWeghtsConverter(head.ip2_pol_b, {policy_d_model}));
852 auto Q = builder->Reshape(
853 "/policy/Q/reshape", flow,
854 builder->AddInitializer("/const/QK_shape",
855 Int64OnnxConst({-1, 64, policy_d_model}, {3})));
856 flow = builder->MatMul(
857 "/policy/K/matmul", encoder_out,
858 *GetWeghtsConverter(head.ip3_pol_w,
859 {policy_embedding_size, policy_d_model}, {1, 0}));
860 flow = builder->Add("/policy/K/add", flow,
861 *GetWeghtsConverter(head.ip3_pol_b, {policy_d_model}));
862 auto K = builder->Reshape("/policy/K/reshape", flow, "/const/QK_shape");
863 flow = builder->Transpose("/policy/K/transpose", K, {0, 2, 1});
864 flow = builder->MatMul("/policy/matmul", Q, flow);
865 flow = builder->Mul("/policy/scale", flow,
866 *GetScalarConverter(1.0f / sqrtf(policy_d_model)));
867 auto prom = builder->Slice("policy/promotion/slice", K, {0, 56, 0},
868 {INT_MAX, 64, policy_d_model});
869 prom = builder->MatMul(
870 "/policy/promotion/matmul", prom,
871 *GetWeghtsConverter(head.ip4_pol_w, {policy_d_model, 4}, {1, 0}));
872 prom = builder->Transpose("/policy/promotion/transpose", prom, {0, 2, 1});
873 auto prom2 = builder->Split("/policy/promotion/split", prom, 1, {3, 1});
874 prom = builder->Add("/policy/promotion/add", prom2[0], prom2[1]);
875 prom = builder->Transpose("/policy/promotion/transpose2", prom, {0, 2, 1});
876 prom = builder->Reshape(
877 "/policy/promotion/reshape", prom,
878 builder->AddInitializer("/const/policy_promotion_shape",
879 Int64OnnxConst({-1, 1, 24}, {3})));
880 auto sl = builder->Slice("policy/promotion/slice2", flow, {0, 48, 56},
881 {INT_MAX, 56, 64});
882 sl = builder->Reshape(
883 "/policy/promotion/reshape2", sl,
884 builder->AddInitializer("/const/policy_promotion_shape2",
885 Int64OnnxConst({-1, 64, 1}, {3})));
886 sl = builder->Concat("/policy/promotion/concat", {sl, sl, sl}, 2);
887 sl = builder->Reshape(
888 "/policy/promotion/reshape3", sl,
889 builder->AddInitializer("/const/policy_promotion_shape3",
890 Int64OnnxConst({-1, 8, 24}, {3})));
891 prom = builder->Add("/policy/promotion/add2", sl, prom);
892 prom = builder->Reshape(
893 "/policy/promotion/reshape4", prom,
894 builder->AddInitializer("/const/policy_promotion_shape4",
895 Int64OnnxConst({-1, 3, 64}, {3})));
896 flow = builder->Concat("/policy/concat", {flow, prom}, 1);
897 flow = builder->Reshape(
898 "/policy/reshape", flow,
899 builder->AddInitializer("/const/policy_out_shape",
900 Int64OnnxConst({-1, 67 * 64}, {2})));
901 return builder->Gather(
902 options_.output_policy_head, flow,
903 builder->AddInitializer(
904 "/const/mapping_table",
905 Int32OnnxConst(
906 MakePolicyMap(kAttnPolicyMap, std::size(kAttnPolicyMap)),
907 {1858})),
908 1);
909}

911void Converter::MakePolicyHead(pblczero::OnnxModel* onnx, OnnxBuilder* builder,
912 const std::string& input,
913 const MultiHeadWeights& weights) {
914 // Check that selected policy head exists.
915 if (!weights.policy_heads.contains(options_.policy_head)) {
916 throw Exception("The policy head you specified '" + options_.policy_head +
917 "'" + " does not exist in this net.");
918 }
919 const MultiHeadWeights::PolicyHead& head =
920 weights.policy_heads.at(options_.policy_head);
921 if (src_.format().network_format().policy() ==
922 pblczero::NetworkFormat::POLICY_ATTENTION) {
923 auto output = MakeAttentionPolicy(builder, input, weights, head);
924 builder->AddOutput(output, {-1, 1858}, GetDataType());
925 onnx->set_output_policy(output);
926 return;
927 } else if (head.policy.weights.empty()) {
928 throw Exception("The policy head selected '" + options_.policy_head + "'" +
929 " is empty.");
930 } else if (!head.policy1.weights.empty()) {
931 // Conv policy head.
932 if (NumEncBlocks() > 0) {
933 throw Exception(
934 "Convolutional policy not supported with attention body.");
935 }
936 auto flow = MakeConvBlock(builder, head.policy1, NumFilters(), NumFilters(),
937 input, "/policy/conv1");
938 flow = MakeConvBlock(builder, head.policy, NumFilters(), 80, flow,
939 "/policy/conv2", nullptr, "", false);
940 flow = builder->Reshape(
941 "/policy/flatten", flow,
942 builder->AddInitializer("/const/policy_shape",
943 Int64OnnxConst({-1, 80 * 8 * 8}, {2})));
944 auto output = builder->Gather(
945 options_.output_policy_head, flow,
946 builder->AddInitializer(
947 "/const/mapping_table",
948 Int32OnnxConst(
949 MakePolicyMap(kConvPolicyMap, std::size(kConvPolicyMap)),
950 {1858})),
951 1);
952 builder->AddOutput(output, {options_.batch_size, 1858}, GetDataType());
953 onnx->set_output_policy(output);
954 } else {
955 // Dense policy head.
956 if (NumEncBlocks() > 0) {
957 throw Exception("Classical policy not supported with attention body.");
958 }
959 const int pol_channels = head.policy.biases.size();
960 auto flow = MakeConvBlock(builder, head.policy, NumFilters(), pol_channels,
961 input, "/policy/conv", nullptr, "", true, 1);
962 flow =
963 builder->Reshape("/policy/reshape", flow,
964 builder->AddInitializer(
965 "/const/policy_shape",
966 Int64OnnxConst({-1, pol_channels * 8 * 8}, {2})));
967 flow = builder->MatMul(
968 "/policy/dense/matmul", flow,
969 *GetWeghtsConverter(head.ip_pol_w,
970 {pol_channels * 8 * 8, 1858}, {1, 0}));
971 auto output = builder->Add(
972 options_.output_policy_head, flow,
973 *GetWeghtsConverter(head.ip_pol_b, {1858}));
974 builder->AddOutput(output, {options_.batch_size, 1858}, GetDataType());
975 onnx->set_output_policy(output);
976 }
977}

979void Converter::MakeValueHead(pblczero::OnnxModel* onnx, OnnxBuilder* builder,
980 const std::string& input,
981 const MultiHeadWeights& weights) {
982 // Check that selected value head exists.
983 if (!weights.value_heads.contains(options_.value_head)) {
984 throw Exception("The value head you specified '" + options_.value_head +
985 "'" + " does not exist in this net.");
986 }
987 const MultiHeadWeights::ValueHead& head =
988 weights.value_heads.at(options_.value_head);
989 if (head.ip1_val_b.empty()) {
990 throw Exception("The value head selected '" + options_.value_head + "'" +
991 " is empty.");
992 }
993 std::string flow;
994 const int val_channels = NumEncBlocks() > 0 ? head.ip_val_b.size() : 32;
995 if (NumEncBlocks() > 0) {
996 int embedding_size = weights.ip_emb_b.size();
997 flow = builder->MatMul(
998 "/value/embed/matmul", input,
999 *GetWeghtsConverter(head.ip_val_w, {embedding_size, val_channels},
1000 {1, 0}));
1001 flow = builder->Add("/value/embed/add", flow,
1002 *GetWeghtsConverter(head.ip_val_b, {val_channels}));
1003 flow = MakeActivation(builder, flow, "/value/embed", default_activation_);
1004 } else {
1005 flow = MakeConvBlock(builder, head.value, NumFilters(), val_channels, input,
1006 "/value/conv", nullptr, "", true, 1);
1007 }
1008 flow = builder->Reshape(
1009 "/value/reshape", flow,
1010 builder->AddInitializer("/const/value_shape",
1011 Int64OnnxConst({-1, val_channels * 8 * 8}, {2})));
1012 flow = builder->MatMul(
1013 "/value/dense1/matmul", flow,
1014 *GetWeghtsConverter(head.ip1_val_w, {val_channels * 8 * 8, 128}, {1, 0}));
1015 flow = builder->Add("/value/dense1/add", flow,
1016 *GetWeghtsConverter(head.ip1_val_b, {128}));
1017 flow = MakeActivation(builder, flow, "/value/dense1", default_activation_);
1018
1019 const bool wdl = src_.format().network_format().value() ==
1020 pblczero::NetworkFormat::VALUE_WDL;
1021 if (wdl) {
1022 flow =
1023 builder->MatMul("/value/dense2/matmul", flow,
1024 *GetWeghtsConverter(head.ip2_val_w, {128, 3}, {1, 0}));
1025 flow = builder->Add("/value/dense2/add", flow,
1026 *GetWeghtsConverter(head.ip2_val_b, {3}));
1027 if (!options_.no_wdl_softmax) {
1028 flow = builder->Softmax(options_.output_wdl, flow);
1029 }
1030 builder->AddOutput(flow, {options_.batch_size, 3}, GetDataType());
1031 onnx->set_output_wdl(flow);
1032 } else {
1033 flow =
1034 builder->MatMul("/value/dense2/matmul", flow,
1035 *GetWeghtsConverter(head.ip2_val_w, {128, 1}, {1, 0}));
1036 flow = builder->Add("/value/dense2/add", flow,
1037 *GetWeghtsConverter(head.ip2_val_b, {1}));
1038 auto output = builder->Tanh(options_.output_value, flow);
1039 builder->AddOutput(output, {options_.batch_size, 1}, GetDataType());
1040 onnx->set_output_value(output);
1041 }

1044void Converter::MakeMovesLeftHead(pblczero::OnnxModel* onnx,
1045 OnnxBuilder* builder,
1046 const std::string& input,
1047 const MultiHeadWeights& weights) {
1048 if (src_.format().network_format().moves_left() !=
1049 pblczero::NetworkFormat::MOVES_LEFT_V1) {
1050 return;
1051 }
1052 const int mlh_channels = NumEncBlocks() > 0
1053 ? weights.ip_mov_b.size()
1054 : weights.moves_left.biases.size();
1055 const int mlh_fc1_outputs = weights.ip1_mov_b.size();
1056 std::string flow;
1057 if (NumEncBlocks() > 0) {
1058 int embedding_size = weights.ip_emb_b.size();
1059 flow = builder->MatMul(
1060 "/mlh/embed/matmul", input,
1061 *GetWeghtsConverter(weights.ip_mov_w, {embedding_size, mlh_channels},
1062 {1, 0}));
1063 flow = builder->Add("/mlh/embed/add", flow,
1064 *GetWeghtsConverter(weights.ip_mov_b, {mlh_channels}));
1065 flow = MakeActivation(builder, flow, "/mlh/embed", default_activation_);
1066 } else {
1067 flow =
1068 MakeConvBlock(builder, weights.moves_left, NumFilters(), mlh_channels,
1069 input, "/mlh/conv", nullptr, "", true, 1);
1070 }
1071 flow = builder->Reshape(
1072 "/mlh/reshape", flow,
1073 builder->AddInitializer("/const/mlh_shape",
1074 Int64OnnxConst({-1, mlh_channels * 8 * 8}, {2})));
1075 flow = builder->MatMul(
1076 "/mlh/dense1/matmul", flow,
1077 *GetWeghtsConverter(weights.ip1_mov_w,
1078 {mlh_channels * 8 * 8, mlh_fc1_outputs}, {1, 0}));
1079 flow =
1080 builder->Add("/mlh/dense1/add", flow,
1081 *GetWeghtsConverter(weights.ip1_mov_b, {mlh_fc1_outputs}));
1082 flow = MakeActivation(builder, flow, "/mlh/dense1", default_activation_);
1083 flow = builder->MatMul(
1084 "/mlh/dense2/matmul", flow,
1085 *GetWeghtsConverter(weights.ip2_mov_w, {mlh_fc1_outputs, 1}, {1, 0}));
1086 flow = builder->Add("/mlh/dense2/add", flow,
1087 *GetWeghtsConverter(weights.ip2_mov_b, {1}));
1088 // Explicity ReLU activation.
1089 auto output = builder->Relu(options_.output_mlh, flow);
1090 builder->AddOutput(output, {options_.batch_size, 1}, GetDataType());
1091 onnx->set_output_mlh(output);
1092}

1094void Converter::GenerateOnnx(pblczero::OnnxModel* onnx) {
1095 MultiHeadWeights weights(src_.weights());
1096 OnnxBuilder builder(options_.opset, options_.ir);
1097
1098 if (GetDataType() == pblczero::TensorProto::FLOAT16) {
1099 onnx->set_data_type(pblczero::OnnxModel::FLOAT16);
1100 } else if (GetDataType() == pblczero::TensorProto::BFLOAT16) {
1101 onnx->set_data_type(pblczero::OnnxModel::BFLOAT16);
1102 } else {
1103 onnx->set_data_type(pblczero::OnnxModel::FLOAT);
1104 }
1105 onnx->set_input_planes(options_.input_planes_name);
1106 builder.AddInput(options_.input_planes_name, {options_.batch_size, 112, 8, 8},
1107 GetDataType());
1108
1109 auto flow = options_.input_planes_name;
1110
1111 // Input convolution.
1112 if (NumResBlocks() > 0) {
1113 flow = MakeConvBlock(&builder, weights.input, kInputPlanes, NumFilters(),
1114 flow, "/inputconv");
1115 }
1116
1117 // Residual tower.
1118 for (size_t i = 0; i < NumResBlocks(); ++i) {
1119 flow = MakeResidualBlock(&builder, weights.residual[i], flow,
1120 "/block" + std::to_string(i));
1121 }
1122
1123 if (NumEncBlocks() > 0) {
1124 flow = MakeAttentionBody(&builder, flow, weights);
1125 }
1126
1127 // Policy head.
1128 MakePolicyHead(onnx, &builder, flow, weights);
1129 // Value head.
1130 MakeValueHead(onnx, &builder, flow, weights);
1131 // Moves left head.
1132 MakeMovesLeftHead(onnx, &builder, flow, weights);
1133
1134 onnx->set_model(builder.OutputAsString());
1135}

1210void Converter::Convert(pblczero::Net* dst) {
1211 if (src_.has_onnx_model() && src_.format().network_format().network() ==
1212 pblczero::NetworkFormat::NETWORK_ONNX) {
1213 *dst = src_;
1214 return;
1215 }

1223
