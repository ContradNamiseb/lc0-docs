43// Implementation of reader-preferenced shared mutex. Based on fair shared
44// mutex.
45class CAPABILITY("mutex") RpSharedMutex {
46 public:
47 RpSharedMutex() : waiting_readers_(0) {}
48
49 void lock() ACQUIRE() {
50 while (true) {
51 mutex_.lock();
52 if (waiting_readers_ == 0) return;
53 mutex_.unlock();
54 }
55 }
56 void unlock() RELEASE() { mutex_.unlock(); }
57 void lock_shared() ACQUIRE_SHARED() {
58 ++waiting_readers_;
59 mutex_.lock_shared();
60 }
61 void unlock_shared() RELEASE_SHARED() {
62 --waiting_readers_;
63 mutex_.unlock_shared();
64 }
65
66 private:
67 std::shared_timed_mutex mutex_;
68 std::atomic<int> waiting_readers_;
69};

137// A very simple spin lock.
138class CAPABILITY("mutex") SpinMutex {
139 public:
140 // std::unique_lock<SpinMutex> wrapper.
141 class SCOPED_CAPABILITY Lock {
142 public:
143 Lock(SpinMutex& m) ACQUIRE(m) : lock_(m) {}
144 ~Lock() RELEASE() {}
145
146 private:
147 std::unique_lock<SpinMutex> lock_;
148 };
149
150 void lock() ACQUIRE() {
151 int spins = 0;
152 while (true) {
153 int val = 0;
154 if (mutex_.compare_exchange_weak(val, 1, std::memory_order_acq_rel)) {
155 break;
156 }
157 ++spins;
158 // Help avoid complete resource starvation by yielding occasionally if
159 // needed.
160 if (spins % 512 == 0) {
161 std::this_thread::yield();
162 } else {
163 SpinloopPause();
164 }
165 }
166 }
167 void unlock() RELEASE() { mutex_.store(0, std::memory_order_release); }
168
169 private:
170 std::atomic<int> mutex_{0};
171};
