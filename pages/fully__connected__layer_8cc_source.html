35void ApplyBias(size_t batch_size, const size_t output_size, const float* biases,
36 const ActivationFunction activation, float* outputs) {

57template <>
58void FullyConnectedLayer<false>::Forward1D(
59 size_t batch_size, const size_t input_size, const size_t output_size,
60 const float* inputs, const float* weights, const float* biases,
61 const ActivationFunction activation, float* outputs) {
62 if (batch_size == 1) {
63 // Just a matrix-vector multiplication
64 //
65 // C A B
66 //
67 // outputs := weights x inputs
68 //
69 // cols: 1 input_size 1
70 //
71 // rows output_size output_size input_size
72 //
73 cblas_sgemv(CblasRowMajor, CblasNoTrans,
74 // M K
75 (int)output_size, (int)input_size, 1.0f, weights,
76 (int)input_size, inputs, 1, 0.0f, outputs, 1);
77 } else {
78 // more columns, matrix-matrix multiplication
79 //
80 // C A B
81 //
82 // outputs := weights x inputs
83 //
84 // cols: batch_size (N) input_size (K) batch_size (N)
85 //
86 // rows output_size (M) output_size (M) input_size (K)
87 //
88
89 // C←αAB + βC
90 // M Number of rows in matrices A and C.
91 // N Number of columns in matrices B and C.
92 // K Number of columns in matrix A; number of rows in matrix B.
93 // lda The size of the first dimension of matrix A; if you are
94 // passing a matrix A[m][n], the value should be m.
95 // cblas_sgemm(CblasRowMajor, TransA, TransB, M, N, K, alpha, A, lda, B,
96 // ldb, beta, C, N);
97 cblas_sgemm(CblasColMajor, CblasTrans, CblasNoTrans,
98 (int)output_size, // M
99 (int)batch_size, // N
100 (int)input_size, // K
101 1.0f, // alpha
102 weights, // A
103 (int)input_size, // lda, leading rank of A
104 inputs, // B
105 (int)input_size, // ldb, leading rank of B
106 0.0f, // beta
107 outputs, // C
108 (int)output_size); // ldc, leading rank of C
109 }
110 if (biases != nullptr) {
111 ApplyBias(batch_size, output_size, biases, activation, outputs);
112 }
113}

126template <>
127void FullyConnectedLayer<true>::Forward1D(
128 size_t batch_size, const size_t input_size, const size_t output_size,
129 const float* inputs, const float* weights, const float* biases,
130 const ActivationFunction activation, float* outputs) {
131 if (batch_size == 1) {
132 EigenVectorMap<float> y(outputs, output_size);
133 y.noalias() = ConstEigenMatrixMap<float>(weights, input_size, output_size)
134 .transpose() *
135 ConstEigenVectorMap<float>(inputs, input_size);
136 } else {
137 auto C_mat = EigenMatrixMap<float>(outputs, output_size, batch_size);
138 C_mat.noalias() =
139 ConstEigenMatrixMap<float>(weights, input_size, output_size)
140 .transpose() *
141 ConstEigenMatrixMap<float>(inputs, input_size, batch_size);
142 }
143 if (biases != nullptr) {
144 ApplyBias(batch_size, output_size, biases, activation, outputs);
145 }
146}
