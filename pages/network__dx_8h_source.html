95class DxNetworkComputation : public NetworkComputation {
96 public:
97 DxNetworkComputation(DxNetwork* network, bool wdl, bool moves_left);
98 ~DxNetworkComputation();
99
100 void AddInput(InputPlanes&& input) override;
101
102 void ComputeBlocking() override;
103
104 int GetBatchSize() const override { return batch_size_; }
105
106 float GetQVal(int sample) const override {
107 if (wdl_) {
108 auto w = inputs_outputs_->op_value_mem_final_[3 * sample + 0];
109 auto l = inputs_outputs_->op_value_mem_final_[3 * sample + 2];
110 return w - l;
111 } else {
112 return inputs_outputs_->op_value_mem_final_[sample];
113 }
114 }
115
116 float GetDVal(int sample) const override {
117 if (wdl_) {
118 auto d = inputs_outputs_->op_value_mem_final_[3 * sample + 1];
119 return d;
120 } else {
121 return 0.0f;
122 }
123 }
124
125 float GetPVal(int sample, int move_id) const override {
126 return inputs_outputs_
127 ->op_policy_mem_final_[sample * kNumOutputPolicy + move_id];
128 }
129
130 float GetMVal(int sample) const override {
131 if (moves_left_) {
132 return inputs_outputs_->op_moves_left_mem_final_[sample];
133 }
134 return 0.0f;
135 }
136
137 private:
138 // Memory holding inputs, outputs.
139 std::unique_ptr<InputsOutputsDx> inputs_outputs_;
140 int batch_size_;
141 bool wdl_;
142 bool moves_left_;
143
144 DxNetwork* network_;
145};
