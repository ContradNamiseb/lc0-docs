55MoveList MakeRootMoveFilter(const MoveList& searchmoves,
56 SyzygyTablebase* syzygy_tb,
57 const PositionHistory& history, bool fast_play,
58 std::atomic<int>* tb_hits, bool* dtz_success) {
59 assert(tb_hits);
60 assert(dtz_success);
61 // Search moves overrides tablebase.
62 if (!searchmoves.empty()) return searchmoves;
63 const auto& board = history.Last().GetBoard();
64 MoveList root_moves;
65 if (!syzygy_tb || !board.castlings().no_legal_castle() ||
66 (board.ours() | board.theirs()).count() > syzygy_tb->max_cardinality()) {
67 return root_moves;
68 }
69 if (syzygy_tb->root_probe(
70 history.Last(), fast_play || history.DidRepeatSinceLastZeroingMove(),
71 false, &root_moves)) {
72 *dtz_success = true;
73 tb_hits->fetch_add(1, std::memory_order_acq_rel);
74 } else if (syzygy_tb->root_probe_wdl(history.Last(), &root_moves)) {
75 tb_hits->fetch_add(1, std::memory_order_acq_rel);
76 }
77 return root_moves;
78}

82class MEvaluator {
83 public:
84 MEvaluator()
85 : enabled_{false},
86 m_slope_{0.0f},
87 m_cap_{0.0f},
88 a_constant_{0.0f},
89 a_linear_{0.0f},
90 a_square_{0.0f},
91 q_threshold_{0.0f},
92 parent_m_{0.0f} {}
93
94 MEvaluator(const SearchParams& params, const Node* parent = nullptr)
95 : enabled_{true},
96 m_slope_{params.GetMovesLeftSlope()},
97 m_cap_{params.GetMovesLeftMaxEffect()},
98 a_constant_{params.GetMovesLeftConstantFactor()},
99 a_linear_{params.GetMovesLeftScaledFactor()},
100 a_square_{params.GetMovesLeftQuadraticFactor()},
101 q_threshold_{params.GetMovesLeftThreshold()},
102 parent_m_{parent ? parent->GetM() : 0.0f},
103 parent_within_threshold_{parent ? WithinThreshold(parent, q_threshold_)
104 : false} {}
105
106 void SetParent(const Node* parent) {
107 assert(parent);
108 if (enabled_) {
109 parent_m_ = parent->GetM();
110 parent_within_threshold_ = WithinThreshold(parent, q_threshold_);
111 }
112 }
113
114 // Calculates the utility for favoring shorter wins and longer losses.
116 float GetMUtility(Node* child, float q) const {
117 if (!enabled_ || !parent_within_threshold_) return 0.0f;
118 const float child_m = child->GetM();
119 float m = std::clamp(m_slope_ * (child_m - parent_m_), -m_cap_, m_cap_);
120 m *= FastSign(-q);
121 if (q_threshold_ > 0.0f && q_threshold_ < 1.0f) {
122 // This allows a smooth M effect with higher q thresholds, which is
123 // necessary for using MLH together with contempt.
124 q = std::max(0.0f, (std::abs(q) - q_threshold_)) / (1.0f - q_threshold_);
125 }
126 m *= a_constant_ + a_linear_ * std::abs(q) + a_square_ * q * q;
127 return m;
128 }
129
130 float GetMUtility(const EdgeAndNode& child, float q) const {
131 if (!enabled_ || !parent_within_threshold_) return 0.0f;
132 if (child.GetN() == 0) return GetDefaultMUtility();
133 return GetMUtility(child.node(), q);
134 }
135
136 // The M utility to use for unvisited nodes.
138 float GetDefaultMUtility() const { return 0.0f; }
139
140 private:
141 static bool WithinThreshold(const Node* parent, float q_threshold) {
142 return std::abs(parent->GetQ(0.0f)) > q_threshold;
143 }
144
145 const bool enabled_;
146 const float m_slope_;
147 const float m_cap_;
148 const float a_constant_;
149 const float a_linear_;
150 const float a_square_;
151 const float q_threshold_;
152 float parent_m_ = 0.0f;
153 bool parent_within_threshold_ = false;
154};

209void ApplyDirichletNoise(Node* node, float eps, double alpha) {
210 float total = 0;
211 std::vector<float> noise;
212
213 for (int i = 0; i < node->GetNumEdges(); ++i) {
214 float eta = Random::Get().GetGamma(alpha, 1.0);
215 noise.emplace_back(eta);
216 total += eta;
217 }
218
219 if (total < std::numeric_limits<float>::min()) return;
220
221 int noise_idx = 0;
222 for (const auto& child : node->Edges()) {
223 auto* edge = child.edge();
224 edge->SetP(edge->GetP() * (1 - eps) + eps * noise[noise_idx++] / total);
225 }
226}

230// WDL conversion formula based on random walk model.
232inline double WDLRescale(float& v, float& d, float wdl_rescale_ratio,
233 float wdl_rescale_diff, float sign, bool invert,
234 float max_reasonable_s) {
235 if (invert) {
236 wdl_rescale_diff = -wdl_rescale_diff;
237 wdl_rescale_ratio = 1.0f / wdl_rescale_ratio;
238 }
239 auto w = (1 + v - d) / 2;
240 auto l = (1 - v - d) / 2;
241 // Safeguard against numerical issues; skip WDL transformation if WDL is too
242 // extreme.
243 const float eps = 0.0001f;
244 if (w > eps && d > eps && l > eps && w < (1.0f - eps) && d < (1.0f - eps) &&
245 l < (1.0f - eps)) {
246 auto a = FastLog(1 / l - 1);
247 auto b = FastLog(1 / w - 1);
248 auto s = 2 / (a + b);
249 // Safeguard against unrealistically broad WDL distributions coming from
250 // the NN. Originally hardcoded, made into a parameter for piece odds.
251 if (!invert) s = std::min(max_reasonable_s, s);
252 auto mu = (a - b) / (a + b);
253 auto s_new = s * wdl_rescale_ratio;
254 if (invert) {
255 std::swap(s, s_new);
256 s = std::min(max_reasonable_s, s);
257 }
258 auto mu_new = mu + sign * s * s * wdl_rescale_diff;
259 auto w_new = FastLogistic((-1.0f + mu_new) / s_new);
260 auto l_new = FastLogistic((-1.0f - mu_new) / s_new);
261 v = w_new - l_new;
262 d = std::max(0.0f, 1.0f - w_new - l_new);
263 return mu_new;
264 }
265 return 0;
266}

271void Search::SendUciInfo() REQUIRES(nodes_mutex_) REQUIRES(counters_mutex_) {
272 const auto max_pv = params_.GetMultiPv();
273 const auto edges = GetBestChildrenNoTemperature(root_node_, max_pv, 0);
274 const auto score_type = params_.GetScoreType();
275 const auto per_pv_counters = params_.GetPerPvCounters();
276 const auto draw_score = GetDrawScore(false);
277
278 std::vector<ThinkingInfo> uci_infos;
279
280 // Info common for all multipv variants.
281 ThinkingInfo common_info;
282 common_info.depth = cum_depth_ / (total_playouts_ ? total_playouts_ : 1);
283 common_info.seldepth = max_depth_;
284 common_info.time = GetTimeSinceStart();
285 if (!per_pv_counters) {
286 common_info.nodes = total_playouts_ + initial_visits_;
287 }
288 if (nps_start_time_) {
289 const auto time_since_first_batch_ms =
290 std::chrono::duration_cast<std::chrono::milliseconds>(
291 std::chrono::steady_clock::now() - *nps_start_time_)
292 .count();
293 if (time_since_first_batch_ms > 0) {
294 common_info.nps = total_playouts_ * 1000 / time_since_first_batch_ms;
295 common_info.eps = network_evaluations_ * 1000 / time_since_first_batch_ms;
296 }
297 }
298 common_info.tb_hits = tb_hits_.load(std::memory_order_acquire);
299
300 int multipv = 0;
301 const auto default_q = -root_node_->GetQ(-draw_score);
302 const auto default_wl = -root_node_->GetWL();
303 const auto default_d = root_node_->GetD();
304 for (const auto& edge : edges) {
305 ++multipv;
306 uci_infos.emplace_back(common_info);
307 auto& uci_info = uci_infos.back();
308 auto wl = edge.GetWL(default_wl);
309 auto d = edge.GetD(default_d);
310 float mu_uci = 0.0f;
311 if (score_type == "WDL_mu" || (params_.GetWDLRescaleDiff() != 0.0f &&
312 contempt_mode_ != ContemptMode::NONE)) {
313 auto sign = ((contempt_mode_ == ContemptMode::BLACK) ==
314 played_history_.IsBlackToMove())
315 ? 1.0f
316 : -1.0f;
317 mu_uci = WDLRescale(
318 wl, d, params_.GetWDLRescaleRatio(),
319 contempt_mode_ == ContemptMode::NONE
320 ? 0
321 : params_.GetWDLRescaleDiff() * params_.GetWDLEvalObjectivity(),
322 sign, true, params_.GetWDLMaxS());
323 }
324 const auto q = edge.GetQ(default_q, draw_score);
325 if (edge.IsTerminal() && wl != 0.0f) {
326 uci_info.mate = std::copysign(
327 std::round(edge.GetM(0.0f)) / 2 + (edge.IsTbTerminal() ? 101 : 1),
328 wl);
329 } else if (score_type == "centipawn_with_drawscore") {
330 uci_info.score = 90 * tan(1.5637541897 * q);
331 } else if (score_type == "centipawn") {
332 uci_info.score = 90 * tan(1.5637541897 * wl);
333 } else if (score_type == "centipawn_2019") {
334 uci_info.score = 295 * wl / (1 - 0.976953126 * std::pow(wl, 14));
335 } else if (score_type == "centipawn_2018") {
336 uci_info.score = 290.680623072 * tan(1.548090806 * wl);
337 } else if (score_type == "win_percentage") {
338 uci_info.score = wl * 5000 + 5000;
339 } else if (score_type == "Q") {
340 uci_info.score = q * 10000;
341 } else if (score_type == "W-L") {
342 uci_info.score = wl * 10000;
343 } else if (score_type == "WDL_mu") {
344 // Reports the WDL mu value whenever it is reasonable, and defaults to
345 // centipawn otherwise.
346 const float centipawn_fallback_threshold = 0.996f;
347 float centipawn_score = 45 * tan(1.56728071628 * wl);
348 uci_info.score =
349 backend_attributes_.has_wdl && mu_uci != 0.0f &&
350 std::abs(wl) + d < centipawn_fallback_threshold &&
351 (std::abs(mu_uci) < 1.0f ||
352 std::abs(centipawn_score) < std::abs(100 * mu_uci))
353 ? 100 * mu_uci
354 : centipawn_score;
355 }
356
357 auto wdl_w =
358 std::max(0, static_cast<int>(std::round(500.0 * (1.0 + wl - d))));
359 auto wdl_l =
360 std::max(0, static_cast<int>(std::round(500.0 * (1.0 - wl - d))));
361 // Using 1000-w-l so that W+D+L add up to 1000.0.
362 auto wdl_d = 1000 - wdl_w - wdl_l;
363 if (wdl_d < 0) {
364 wdl_w = std::min(1000, std::max(0, wdl_w + wdl_d / 2));
365 wdl_l = 1000 - wdl_w;
366 wdl_d = 0;
367 }
368 uci_info.wdl = ThinkingInfo::WDL{wdl_w, wdl_d, wdl_l};
369 if (backend_attributes_.has_mlh) {
370 uci_info.moves_left = static_cast<int>(
371 (1.0f + edge.GetM(1.0f + root_node_->GetM())) / 2.0f);
372 }
373 if (max_pv > 1) uci_info.multipv = multipv;
374 if (per_pv_counters) uci_info.nodes = edge.GetN();
375 bool flip = played_history_.IsBlackToMove();
376 int depth = 0;
377 for (auto iter = edge; iter;
378 iter = GetBestChildNoTemperature(iter.node(), depth), flip = !flip) {
379 uci_info.pv.push_back(iter.GetMove(flip));
380 if (!iter.node()) break; // Last edge was dangling, cannot continue.
381 depth += 1;
382 }
383 }
384
385 if (!uci_infos.empty()) last_outputted_uci_info_ = uci_infos.front();
386 if (current_best_edge_ && !edges.empty()) {
387 last_outputted_info_edge_ = current_best_edge_.edge();
388 }
389
390 uci_responder_->OutputThinkingInfo(&uci_infos);
391}

396void Search::MaybeOutputInfo() {
397 SharedMutex::Lock lock(nodes_mutex_);
398 Mutex::Lock counters_lock(counters_mutex_);
399 if (!bestmove_is_sent_ && current_best_edge_ &&
400 (current_best_edge_.edge() != last_outputted_info_edge_ ||

402 static_cast<int>(cum_depth_ /

412 std::vector<ThinkingInfo> info(1);
413 info.back().comment =
414 "WARNING: Search has reached limit and does not make any progress.";
415 uci_responder_->OutputThinkingInfo(&info);
416 }
417 }

463inline float ComputeCpuct(const SearchParams& params, uint32_t N,
464 bool is_root_node) {
465 const float init = params.GetCpuct(is_root_node);
466 const float k = params.GetCpuctFactor(is_root_node);

476std::vector<std::string> Search::GetVerboseStats(const Node* node) const {
477 assert(node == root_node_ || node->GetParent() == root_node_);
478 const bool is_root = (node == root_node_);
479 const bool is_odd_depth = !is_root;
480 const bool is_black_to_move = (played_history_.IsBlackToMove() == is_root);
481 const float draw_score = GetDrawScore(is_odd_depth);
482 const float fpu = GetFpu(params_, node, is_root, draw_score);
483 const float cpuct = ComputeCpuct(params_, node->GetN(), is_root);
484 const float U_coeff =
485 cpuct * std::sqrt(std::max(node->GetChildrenVisits(), 1u));
486 std::vector<std::tuple<uint32_t, float, EdgeAndNode>> edges;
487 edges.reserve(node->GetNumEdges());
488 for (const auto& edge : node->Edges()) {
489 edges.emplace_back(edge.GetN(),
490 edge.GetQ(fpu, draw_score) + edge.GetU(U_coeff), edge);
491 }
492 std::sort(edges.begin(), edges.end());
493
494 auto print = [](auto* oss, auto pre, auto v, auto post, auto w, int p = 0) {
495 *oss << pre << std::setw(w) << std::setprecision(p) << v << post;
496 };
497 auto print_head = [&](auto* oss, auto label, int i, auto n, auto f, auto p) {
498 *oss << std::fixed;
499 print(oss, "", label, " ", 5);
500 print(oss, "(", i, ") ", 4);
501 *oss << std::right;
502 print(oss, "N: ", n, " ", 7);
503 print(oss, "(+", f, ") ", 2);
504 print(oss, "(P: ", p * 100, "%) ", 5, p >= 0.99995f ? 1 : 2);
505 };
506 auto print_stats = [&](auto* oss, const auto* n) {
507 const auto sign = n == node ? -1 : 1;
508 if (n) {
509 auto wl = sign * n->GetWL();
510 auto d = n->GetD();
511 auto is_perspective = ((contempt_mode_ == ContemptMode::BLACK) ==
512 played_history_.IsBlackToMove())
513 ? 1.0f
514 : -1.0f;
515 WDLRescale(
516 wl, d, params_.GetWDLRescaleRatio(),
517 contempt_mode_ == ContemptMode::NONE
518 ? 0
519 : params_.GetWDLRescaleDiff() * params_.GetWDLEvalObjectivity(),
520 is_perspective, true, params_.GetWDLMaxS());
521 print(oss, "(WL: ", wl, ") ", 8, 5);
522 print(oss, "(D: ", d, ") ", 5, 3);
523 print(oss, "(M: ", n->GetM(), ") ", 4, 1);
524 print(oss, "(Q: ", wl + draw_score * d, ") ", 8, 5);
525 } else {
526 *oss << "(WL: -.-----) (D: -.---) (M: -.-) ";
527 print(oss, "(Q: ", fpu, ") ", 8, 5);
528 }
529 };
530 auto print_tail = [&](auto* oss, const auto* n) {
531 const auto sign = n == node ? -1 : 1;
532 std::optional<float> v;
533 if (n && n->IsTerminal()) {
534 v = n->GetQ(sign * draw_score);
535 } else if (n) {
536 auto history = GetPositionHistoryAtNode(n);
537 std::optional<EvalResult> nneval = backend_->GetCachedEvaluation(
538 EvalPosition{history.GetPositions(), {}});
539 if (nneval) v = -nneval->q;
540 }
541 if (v) {
542 print(oss, "(V: ", sign * *v, ") ", 7, 4);
543 } else {
544 *oss << "(V: -.----) ";
545 }
546
547 if (n) {
548 auto [lo, up] = n->GetBounds();
549 if (sign == -1) {
550 lo = -lo;
551 up = -up;
552 std::swap(lo, up);
553 }
554 *oss << (lo == up ? "(T) "
555 : lo == GameResult::DRAW && up == GameResult::WHITE_WON ? "(W) "
556 : lo == GameResult::BLACK_WON && up == GameResult::DRAW ? "(L) "
557 : "");
558 }
559 };
560
561 std::vector<std::string> infos;
562 const auto m_evaluator =
563 backend_attributes_.has_mlh ? MEvaluator(params_, node) : MEvaluator();
564 for (const auto& edge_tuple : edges) {
565 const auto& edge = std::get<2>(edge_tuple);
566 float Q = edge.GetQ(fpu, draw_score);
567 float M = m_evaluator.GetMUtility(edge, Q);
568 std::ostringstream oss;
569 oss << std::left;
570 // TODO: should this be displaying transformed index?
571 print_head(&oss, edge.GetMove(is_black_to_move).ToString(true),
572 MoveToNNIndex(edge.GetMove(), 0), edge.GetN(),
573 edge.GetNInFlight(), edge.GetP());
574 print_stats(&oss, edge.node());
575 print(&oss, "(U: ", edge.GetU(U_coeff), ") ", 6, 5);
576 print(&oss, "(S: ", Q + edge.GetU(U_coeff) + M, ") ", 8, 5);
577 print_tail(&oss, edge.node());
578 infos.emplace_back(oss.str());
579 }
580
581 // Include stats about the node in similar format to its children above.
582 std::ostringstream oss;
583 print_head(&oss, "node ", node->GetNumEdges(), node->GetN(),
584 node->GetNInFlight(), node->GetVisitedPolicy());
585 print_stats(&oss, node);
586 print_tail(&oss, node);
587 infos.emplace_back(oss.str());
588 return infos;
589}

592void Search::SendMovesStats() const REQUIRES(counters_mutex_) {
593 auto move_stats = GetVerboseStats(root_node_);
594
595 if (params_.GetVerboseStats()) {

708// Computes the best move, maybe with temperature (according to the settings).
711void Search::EnsureBestMoveKnown() REQUIRES(nodes_mutex_)
712 REQUIRES(counters_mutex_) {
713 if (bestmove_is_sent_) return;
714 if (root_node_->GetN() == 0) return;
715 if (!root_node_->HasChildren()) return;
716
717 float temperature = params_.GetTemperature();
718 const int cutoff_move = params_.GetTemperatureCutoffMove();
719 const int decay_delay_moves = params_.GetTempDecayDelayMoves();
720 const int decay_moves = params_.GetTempDecayMoves();
721 const int moves = played_history_.Last().GetGamePly() / 2;
722
723 if (cutoff_move && (moves + 1) >= cutoff_move) {
724 temperature = params_.GetTemperatureEndgame();
725 } else if (temperature && decay_moves) {
726 if (moves >= decay_delay_moves + decay_moves) {
727 temperature = 0.0;
728 } else if (moves >= decay_delay_moves) {
729 temperature *=
730 static_cast<float>(decay_delay_moves + decay_moves - moves) /
731 decay_moves;
732 }
733 // don't allow temperature to decay below endgame temperature
734 if (temperature < params_.GetTemperatureEndgame()) {
735 temperature = params_.GetTemperatureEndgame();
736 }
737 }
738
739 auto bestmove_edge = temperature
740 ? GetBestRootChildWithTemperature(temperature)
741 : GetBestChildNoTemperature(root_node_, 0);
742 final_bestmove_ = bestmove_edge.GetMove(played_history_.IsBlackToMove());
743
744 if (bestmove_edge.GetN() > 0 && bestmove_edge.node()->HasChildren()) {
745 final_pondermove_ = GetBestChildNoTemperature(bestmove_edge.node(), 1)
746 .GetMove(!played_history_.IsBlackToMove());
747 }
748}

750// Returns @count children with most visits.
752std::vector<EdgeAndNode> Search::GetBestChildrenNoTemperature(Node* parent,
753 int count,
754 int depth) const {
755 // Even if Edges is populated at this point, its a race condition to access
756 // the node, so exit quickly.
757 if (parent->GetN() == 0) return {};
758 const bool is_odd_depth = (depth % 2) == 1;
759 const float draw_score = GetDrawScore(is_odd_depth);
760 // Best child is selected using the following criteria:
761 // * Prefer shorter terminal wins / avoid shorter terminal losses.
762 // * Largest number of playouts.
763 // * If two nodes have equal number:
764 // * If that number is 0, the one with larger prior wins.
765 // * If that number is larger than 0, the one with larger eval wins.
766 std::vector<EdgeAndNode> edges;
767 for (auto& edge : parent->Edges()) {
768 if (parent == root_node_ && !root_move_filter_.empty() &&
769 std::find(root_move_filter_.begin(), root_move_filter_.end(),
770 edge.GetMove()) == root_move_filter_.end()) {
771 continue;
772 }
773 edges.push_back(edge);
774 }
775 const auto middle = (static_cast<int>(edges.size()) > count)
776 ? edges.begin() + count
777 : edges.end();
778 std::partial_sort(
779 edges.begin(), middle, edges.end(),
780 [draw_score](const auto& a, const auto& b) {
781 // The function returns "true" when a is preferred to b.
782
783 // Lists edge types from less desirable to more desirable.
784 enum EdgeRank {
785 kTerminalLoss,
786 kTablebaseLoss,
787 kNonTerminal, // Non terminal or terminal draw.
788 kTablebaseWin,
789 kTerminalWin,
790 };
791
792 auto GetEdgeRank = [](const EdgeAndNode& edge) {
793 // This default isn't used as wl only checked for case edge is
794 // terminal.
795 const auto wl = edge.GetWL(0.0f);
796 // Not safe to access IsTerminal if GetN is 0.
797 if (edge.GetN() == 0 || !edge.IsTerminal() || !wl) {
798 return kNonTerminal;
799 }
800 if (edge.IsTbTerminal()) {
801 return wl < 0.0 ? kTablebaseLoss : kTablebaseWin;
802 }
803 return wl < 0.0 ? kTerminalLoss : kTerminalWin;
804 };
805
806 // If moves have different outcomes, prefer better outcome.
807 const auto a_rank = GetEdgeRank(a);
808 const auto b_rank = GetEdgeRank(b);
809 if (a_rank != b_rank) return a_rank > b_rank;
810
811 // If both are terminal draws, try to make it shorter.
812 // Not safe to access IsTerminal if GetN is 0.
813 if (a_rank == kNonTerminal && a.GetN() != 0 && b.GetN() != 0 &&
814 a.IsTerminal() && b.IsTerminal()) {
815 if (a.IsTbTerminal() != b.IsTbTerminal()) {
816 // Prefer non-tablebase draws.
817 return a.IsTbTerminal() < b.IsTbTerminal();
818 }
819 // Prefer shorter draws.
820 return a.GetM(0.0f) < b.GetM(0.0f);
821 }
822
823 // Neither is terminal, use standard rule.
824 if (a_rank == kNonTerminal) {
825 // Prefer largest playouts then eval then prior.
826 if (a.GetN() != b.GetN()) return a.GetN() > b.GetN();
827 // Default doesn't matter here so long as they are the same as either
828 // both are N==0 (thus we're comparing equal defaults) or N!=0 and
829 // default isn't used.
830 if (a.GetQ(0.0f, draw_score) != b.GetQ(0.0f, draw_score)) {
831 return a.GetQ(0.0f, draw_score) > b.GetQ(0.0f, draw_score);
832 }
833 return a.GetP() > b.GetP();
834 }
835
836 // Both variants are winning, prefer shortest win.
837 if (a_rank > kNonTerminal) {
838 return a.GetM(0.0f) < b.GetM(0.0f);
839 }
840
841 // Both variants are losing, prefer longest losses.
842 return a.GetM(0.0f) > b.GetM(0.0f);
843 });
844
845 if (count < static_cast<int>(edges.size())) {
846 edges.resize(count);
847 }
848 return edges;
849}

858// Returns a child of a root chosen according to weighted-by-temperature visit
859// count.
861EdgeAndNode Search::GetBestRootChildWithTemperature(float temperature) const {
862 // Root is at even depth.
863 const float draw_score = GetDrawScore(/* is_odd_depth= */ false);
864
865 std::vector<float> cumulative_sums;
866 float sum = 0.0;
867 float max_n = 0.0;
868 const float offset = params_.GetTemperatureVisitOffset();
869 float max_eval = -1.0f;
870 const float fpu =
871 GetFpu(params_, root_node_, /* is_root= */ true, draw_score);
872
873 for (auto& edge : root_node_->Edges()) {
874 if (!root_move_filter_.empty() &&
875 std::find(root_move_filter_.begin(), root_move_filter_.end(),
876 edge.GetMove()) == root_move_filter_.end()) {
877 continue;
878 }
879 if (edge.GetN() + offset > max_n) {
880 max_n = edge.GetN() + offset;
881 max_eval = edge.GetQ(fpu, draw_score);
882 }
883 }
884
885 // TODO(crem) Simplify this code when samplers.h is merged.
886 const float min_eval =
887 max_eval - params_.GetTemperatureWinpctCutoff() / 50.0f;
888 for (auto& edge : root_node_->Edges()) {
889 if (!root_move_filter_.empty() &&
890 std::find(root_move_filter_.begin(), root_move_filter_.end(),
891 edge.GetMove()) == root_move_filter_.end()) {
892 continue;
893 }
894 if (edge.GetQ(fpu, draw_score) < min_eval) continue;
895 sum += std::pow(
896 std::max(0.0f,
897 (max_n <= 0.0f
898 ? edge.GetP()
899 : ((static_cast<float>(edge.GetN()) + offset) / max_n))),
900 1 / temperature);
901 cumulative_sums.push_back(sum);
902 }
903 assert(sum);
904
905 const float toss = Random::Get().GetFloat(cumulative_sums.back());
906 int idx =
907 std::lower_bound(cumulative_sums.begin(), cumulative_sums.end(), toss) -
908 cumulative_sums.begin();
909
910 for (auto& edge : root_node_->Edges()) {
911 if (!root_move_filter_.empty() &&
912 std::find(root_move_filter_.begin(), root_move_filter_.end(),
913 edge.GetMove()) == root_move_filter_.end()) {
914 continue;
915 }
916 if (edge.GetQ(fpu, draw_score) < min_eval) continue;
917 if (idx-- == 0) return edge;
918 }
919 assert(false);
920 return {};
921}

961void Search::PopulateCommonIterationStats(IterationStats* stats) {
962 stats->time_since_movestart = GetTimeSinceStart();
963
964 SharedMutex::SharedLock nodes_lock(nodes_mutex_);
965 {
966 Mutex::Lock counters_lock(counters_mutex_);
967 stats->time_since_first_batch = GetTimeSinceFirstBatch();
968 if (!nps_start_time_ && total_playouts_ > 0) {
969 nps_start_time_ = std::chrono::steady_clock::now();
970 }
971 }
972 stats->total_nodes = total_playouts_ + initial_visits_;
973 stats->nodes_since_movestart = total_playouts_;
974 stats->batches_since_movestart = total_batches_;
975 stats->average_depth = cum_depth_ / (total_playouts_ ? total_playouts_ : 1);
976 stats->edge_n.clear();
977 stats->win_found = false;
978 stats->may_resign = true;
979 stats->num_losing_edges = 0;
980 stats->time_usage_hint_ = IterationStats::TimeUsageHint::kNormal;
981 stats->mate_depth = std::numeric_limits<int>::max();
982
983 // If root node hasn't finished first visit, none of this code is safe.
984 if (root_node_->GetN() > 0) {
985 const auto draw_score = GetDrawScore(true);
986 const float fpu =
987 GetFpu(params_, root_node_, /* is_root_node */ true, draw_score);
988 float max_q_plus_m = -1000;
989 uint64_t max_n = 0;
990 bool max_n_has_max_q_plus_m = true;
991 const auto m_evaluator = backend_attributes_.has_mlh
992 ? MEvaluator(params_, root_node_)
993 : MEvaluator();
994 for (const auto& edge : root_node_->Edges()) {
995 const auto n = edge.GetN();
996 const auto q = edge.GetQ(fpu, draw_score);
997 const auto m = m_evaluator.GetMUtility(edge, q);
998 const auto q_plus_m = q + m;
999 stats->edge_n.push_back(n);
1000 if (n > 0 && edge.IsTerminal() && edge.GetWL(0.0f) > 0.0f) {
1001 stats->win_found = true;
1002 }
1003 if (n > 0 && edge.IsTerminal() && edge.GetWL(0.0f) < 0.0f) {
1004 stats->num_losing_edges += 1;
1005 }
1006 if (n > 0 && edge.IsTerminal() && edge.GetWL(0.0f) == 1.0f &&
1007 !edge.IsTbTerminal()) {
1008 stats->mate_depth =
1009 std::min(stats->mate_depth,
1010 static_cast<int>(std::round(edge.GetM(0.0f))) / 2 + 1);
1011 }
1012
1013 // If game is resignable, no need for moving quicker. This allows
1014 // proving mate when losing anyway for better score output.
1015 // Hardcoded resign threshold, because there is no available parameter.
1016 if (n > 0 && q > -0.98f) {
1017 stats->may_resign = false;
1018 }
1019 if (max_n < n) {
1020 max_n = n;
1021 max_n_has_max_q_plus_m = false;
1022 }
1023 if (max_q_plus_m <= q_plus_m) {
1024 max_n_has_max_q_plus_m = (max_n == n);
1025 max_q_plus_m = q_plus_m;
1026 }
1027 }
1028 if (!max_n_has_max_q_plus_m) {
1029 stats->time_usage_hint_ = IterationStats::TimeUsageHint::kNeedMoreTime;
1030 }
1031 }

1036void Search::WatchdogThread() {
1037 LOGFILE << "Start a watchdog thread.";
1038 StoppersHints hints;
1039 IterationStats stats;
1040 while (true) {
1041 PopulateCommonIterationStats(&stats);
1042 MaybeTriggerStop(stats, &hints);
1043 MaybeOutputInfo();
1044
1045 constexpr auto kMaxWaitTimeMs = 100;
1046 constexpr auto kMinWaitTimeMs = 1;
1047
1048 Mutex::Lock lock(counters_mutex_);
1049 // Only exit when bestmove is responded. It may happen that search threads
1050 // already all exited, and we need at least one thread that can do that.
1051 if (bestmove_is_sent_) break;
1052
1053 auto remaining_time = hints.GetEstimatedRemainingTimeMs();
1054 if (remaining_time > kMaxWaitTimeMs) remaining_time = kMaxWaitTimeMs;
1055 if (remaining_time < kMinWaitTimeMs) remaining_time = kMinWaitTimeMs;
1056 // There is no real need to have max wait time, and sometimes it's fine
1057 // to wait without timeout at all (e.g. in `go nodes` mode), but we
1058 // still limit wait time for exotic cases like when pc goes to sleep
1059 // mode during thinking.
1060 // Minimum wait time is there to prevent busy wait and other threads
1061 // starvation.
1062 watchdog_cv_.wait_for(
1063 lock.get_raw(), std::chrono::milliseconds(remaining_time),
1064 [this]() { return stop_.load(std::memory_order_acquire); });
1065 }
1066 LOGFILE << "End a watchdog thread.";
1067}

1130void SearchWorker::RunTasks(int tid) {
1131 while (true) {
1132 PickTask* task = nullptr;
1133 int id = 0;
1134 {
1135 int spins = 0;
1136 while (true) {
1137 int nta = tasks_taken_.load(std::memory_order_acquire);
1138 int tc = task_count_.load(std::memory_order_acquire);
1139 if (nta < tc) {
1140 int val = 0;
1141 if (task_taking_started_.compare_exchange_weak(
1142 val, 1, std::memory_order_acq_rel,
1143 std::memory_order_relaxed)) {
1144 nta = tasks_taken_.load(std::memory_order_acquire);
1145 tc = task_count_.load(std::memory_order_acquire);
1146 // We got the spin lock, double check we're still in the clear.
1147 if (nta < tc) {
1148 id = tasks_taken_.fetch_add(1, std::memory_order_acq_rel);
1149 task = picking_tasks_.data() + id;
1150 task_taking_started_.store(0, std::memory_order_release);
1151 break;
1152 }
1153 task_taking_started_.store(0, std::memory_order_release);
1154 }
1155 SpinloopPause();
1156 spins = 0;
1157 continue;
1158 } else if (tc != -1) {
1159 spins++;
1160 if (spins >= 512) {
1161 std::this_thread::yield();
1162 spins = 0;
1163 } else {
1164 SpinloopPause();
1165 }
1166 continue;
1167 }
1168 spins = 0;
1169 // Looks like sleep time.
1170 Mutex::Lock lock(picking_tasks_mutex_);
1171 // Refresh them now we have the lock.
1172 nta = tasks_taken_.load(std::memory_order_acquire);
1173 tc = task_count_.load(std::memory_order_acquire);
1174 if (tc != -1) continue;
1175 if (nta >= tc && exiting_) return;
1176 task_added_.wait(lock.get_raw());
1177 // And refresh again now we're awake.
1178 nta = tasks_taken_.load(std::memory_order_acquire);
1179 tc = task_count_.load(std::memory_order_acquire);
1180 if (nta >= tc && exiting_) return;
1181 }
1182 }
1183 if (task != nullptr) {
1184 switch (task->task_type) {
1185 case PickTask::kGathering: {
1186 PickNodesToExtendTask(task->start, task->base_depth,
1187 task->collision_limit, task->moves_to_base,
1188 &(task->results), &(task_workspaces_[tid]));
1189 break;
1190 }
1191 case PickTask::kProcessing: {
1192 ProcessPickedTask(task->start_idx, task->end_idx,
1193 &(task_workspaces_[tid]));
1194 break;
1195 }
1196 }
1197 picking_tasks_.data()[id].complete = true;
1198 completed_tasks_.fetch_add(1, std::memory_order_acq_rel);
1199 }
1200 }

1205void SearchWorker::ExecuteOneIteration() {
1206 // 1. Initialize internal structures.
1207 InitializeIteration();
1208
1209 if (params_.GetMaxConcurrentSearchers() != 0) {
1210 std::unique_ptr<SpinHelper> spin_helper;
1211 if (params_.GetSearchSpinBackoff()) {
1212 spin_helper = std::make_unique<ExponentialBackoffSpinHelper>();
1213 } else {
1214 // This is a hard spin lock to reduce latency but at the expense of busy
1215 // wait cpu usage. If search worker count is large, this is probably a
1216 // bad idea.
1217 spin_helper = std::make_unique<SpinHelper>();
1218 }
1219
1220 while (true) {
1221 // If search is stop, we've not gathered or done anything and we don't
1222 // want to, so we can safely skip all below. But make sure we have done
1223 // at least one iteration.
1224 if (search_->stop_.load(std::memory_order_acquire) &&
1225 search_->GetTotalPlayouts() + search_->initial_visits_ > 0) {
1226 return;
1227 }
1228
1229 int available =
1230 search_->pending_searchers_.load(std::memory_order_acquire);
1231 if (available == 0) {
1232 spin_helper->Wait();
1233 continue;
1234 }
1235
1236 if (search_->pending_searchers_.compare_exchange_weak(
1237 available, available - 1, std::memory_order_acq_rel)) {
1238 break;
1239 } else {
1240 spin_helper->Backoff();
1241 }
1242 }
1243 }
1244
1245 // 2. Gather minibatch.
1246 GatherMinibatch();
1247 task_count_.store(-1, std::memory_order_release);
1248 search_->backend_waiting_counter_.fetch_add(1, std::memory_order_relaxed);
1249
1250 // 2b. Collect collisions.
1251 CollectCollisions();
1252
1253 // 3. Prefetch into cache.
1254 MaybePrefetchIntoCache();
1255
1256 if (params_.GetMaxConcurrentSearchers() != 0) {
1257 search_->pending_searchers_.fetch_add(1, std::memory_order_acq_rel);
1258 }
1259
1260 // 4. Run NN computation.
1261 RunNNComputation();
1262 search_->backend_waiting_counter_.fetch_add(-1, std::memory_order_relaxed);
1263
1264 // 5. Retrieve NN computations (and terminal values) into nodes.
1265 FetchMinibatchResults();
1266
1267 // 6. Propagate the new nodes' information to all their parents in the tree.
1268 DoBackupUpdate();
1269
1270 // 7. Update the Search's status and progress information.
1271 UpdateCounters();
1272
1273 // If required, waste time to limit nps.
1274 if (params_.GetNpsLimit() > 0) {
1275 while (search_->IsSearchActive()) {
1276 int64_t time_since_first_batch_ms = 0;
1277 {
1278 Mutex::Lock lock(search_->counters_mutex_);
1279 time_since_first_batch_ms = search_->GetTimeSinceFirstBatch();
1280 }
1281 if (time_since_first_batch_ms <= 0) {
1282 time_since_first_batch_ms = search_->GetTimeSinceStart();
1283 }
1284 auto nps = search_->GetTotalPlayouts() * 1e3f / time_since_first_batch_ms;
1285 if (nps > params_.GetNpsLimit()) {
1286 std::this_thread::sleep_for(std::chrono::milliseconds(1));
1287 } else {
1288 break;
1289 }
1290 }
1291 }
1292}

1335void SearchWorker::GatherMinibatch() {
1336 LCTRACE_FUNCTION_SCOPE;
1337 // Total number of nodes to process.
1338 int minibatch_size = 0;
1339 int cur_n = 0;
1340 {
1341 SharedMutex::Lock lock(search_->nodes_mutex_);
1342 cur_n = search_->root_node_->GetN();
1343 }
1344 // TODO: GetEstimatedRemainingPlayouts has already had smart pruning factor
1345 // applied, which doesn't clearly make sense to include here...
1346 int64_t remaining_n =
1347 latest_time_manager_hints_.GetEstimatedRemainingPlayouts();
1348 int collisions_left = CalculateCollisionsLeft(
1349 std::min(static_cast<int64_t>(cur_n), remaining_n), params_);
1350
1351 // Number of nodes processed out of order.
1352 number_out_of_order_ = 0;
1353
1354 int thread_count = search_->thread_count_.load(std::memory_order_acquire);
1355
1356 // Gather nodes to process in the current batch.
1357 // If we had too many nodes out of order, also interrupt the iteration so
1358 // that search can exit.
1359 while (minibatch_size < target_minibatch_size_ &&
1360 number_out_of_order_ < max_out_of_order_) {
1361 // If there's something to process without touching slow neural net, do it.
1362 if (minibatch_size > 0 && computation_->UsedBatchSize() == 0) return;
1363
1364 // If there is backend work to be done, and the backend is idle - exit
1365 // immediately.
1366 // Only do this fancy work if there are multiple threads as otherwise we
1367 // early exit from every batch since there is never another search thread to
1368 // be keeping the backend busy. Which would mean that threads=1 has a
1369 // massive nps drop.
1370 if (thread_count > 1 && minibatch_size > 0 &&
1371 static_cast<int>(computation_->UsedBatchSize()) >
1372 params_.GetIdlingMinimumWork() &&
1373 thread_count - search_->backend_waiting_counter_.load(
1374 std::memory_order_relaxed) >
1375 params_.GetThreadIdlingThreshold()) {
1376 return;
1377 }
1378
1379 int new_start = static_cast<int>(minibatch_.size());
1380
1381 PickNodesToExtend(
1382 std::min({collisions_left, target_minibatch_size_ - minibatch_size,
1383 max_out_of_order_ - number_out_of_order_}));
1384
1385 // Count the non-collisions.
1386 int non_collisions = 0;
1387 for (int i = new_start; i < static_cast<int>(minibatch_.size()); i++) {
1388 auto& picked_node = minibatch_[i];
1389 if (picked_node.IsCollision()) {
1390 continue;
1391 }
1392 ++non_collisions;
1393 ++minibatch_size;
1394 }
1395
1396 bool needs_wait = false;
1397 int ppt_start = new_start;
1398 if (task_workers_ > 0 &&
1399 non_collisions >= params_.GetMinimumWorkSizeForProcessing()) {
1400 const int num_tasks = std::clamp(
1401 non_collisions / params_.GetMinimumWorkPerTaskForProcessing(), 2,
1402 task_workers_ + 1);
1403 // Round down, left overs can go to main thread so it waits less.
1404 int per_worker = non_collisions / num_tasks;
1405 needs_wait = true;
1406 ResetTasks();
1407 int found = 0;
1408 for (int i = new_start; i < static_cast<int>(minibatch_.size()); i++) {
1409 auto& picked_node = minibatch_[i];
1410 if (picked_node.IsCollision()) {
1411 continue;
1412 }
1413 ++found;
1414 if (found == per_worker) {
1415 picking_tasks_.emplace_back(ppt_start, i + 1);
1416 task_count_.fetch_add(1, std::memory_order_acq_rel);
1417 ppt_start = i + 1;
1418 found = 0;
1419 if (picking_tasks_.size() == static_cast<size_t>(num_tasks - 1)) {
1420 break;
1421 }
1422 }
1423 }
1424 }
1425 ProcessPickedTask(ppt_start, static_cast<int>(minibatch_.size()),
1426 &main_workspace_);
1427 if (needs_wait) {
1428 WaitForTasks();
1429 }
1430 bool some_ooo = false;
1431 for (int i = static_cast<int>(minibatch_.size()) - 1; i >= new_start; i--) {
1432 if (minibatch_[i].ooo_completed) {
1433 some_ooo = true;
1434 break;
1435 }
1436 }
1437 if (some_ooo) {
1438 LCTRACE_FUNCTION_SCOPE;
1439 SharedMutex::Lock lock(search_->nodes_mutex_);
1440 for (int i = static_cast<int>(minibatch_.size()) - 1; i >= new_start;
1441 i--) {
1442 // If there was any OOO, revert 'all' new collisions - it isn't possible
1443 // to identify exactly which ones are afterwards and only prune those.
1444 // This may remove too many items, but hopefully most of the time they
1445 // will just be added back in the same in the next gather.
1446 if (minibatch_[i].IsCollision()) {
1447 Node* node = minibatch_[i].node;
1448 for (node = node->GetParent();
1449 node != search_->root_node_->GetParent();
1450 node = node->GetParent()) {
1451 node->CancelScoreUpdate(minibatch_[i].multivisit);
1452 }
1453 minibatch_.erase(minibatch_.begin() + i);
1454 } else if (minibatch_[i].ooo_completed) {
1455 DoBackupUpdateSingleNode(minibatch_[i]);
1456 minibatch_.erase(minibatch_.begin() + i);
1457 --minibatch_size;
1458 ++number_out_of_order_;
1459 }
1460 }
1461 }
1462
1463 LCTRACE_FUNCTION_SCOPE;
1464 // Check for stop at the end so we have at least one node.
1465 for (size_t i = new_start; i < minibatch_.size(); i++) {
1466 auto& picked_node = minibatch_[i];
1467
1468 if (picked_node.IsCollision()) {
1469 // Check to see if we can upsize the collision to exit sooner.
1470 if (picked_node.maxvisit > 0 &&
1471 collisions_left > picked_node.multivisit) {
1472 SharedMutex::Lock lock(search_->nodes_mutex_);
1473 int extra = std::min(picked_node.maxvisit, collisions_left) -
1474 picked_node.multivisit;
1475 picked_node.multivisit += extra;
1476 Node* node = picked_node.node;
1477 for (node = node->GetParent();
1478 node != search_->root_node_->GetParent();
1479 node = node->GetParent()) {
1480 node->IncrementNInFlight(extra);
1481 }
1482 }
1483 if ((collisions_left -= picked_node.multivisit) <= 0) return;
1484 if (search_->stop_.load(std::memory_order_acquire)) return;
1485 }
1486 }
1487 }
1488}

1492void SearchWorker::ProcessPickedTask(int start_idx, int end_idx,
1493 TaskWorkspace* workspace) {
1494 LCTRACE_FUNCTION_SCOPE;
1495 auto& history = workspace->history;
1496 history = search_->played_history_;
1497
1498 for (int i = start_idx; i < end_idx; i++) {
1499 auto& picked_node = minibatch_[i];
1500 if (picked_node.IsCollision()) continue;
1501 auto* node = picked_node.node;
1502
1503 // If node is already known as terminal (win/loss/draw according to rules
1504 // of the game), it means that we already visited this node before.
1505 if (picked_node.IsExtendable()) {
1506 // Node was never visited, extend it.
1507 ExtendNode(node, picked_node.depth, picked_node.moves_to_visit, &history);
1508 if (!node->IsTerminal()) {
1509 picked_node.nn_queried = true;
1510 MoveList legal_moves;
1511 legal_moves.reserve(node->GetNumEdges());
1512 std::transform(node->Edges().begin(), node->Edges().end(),
1513 std::back_inserter(legal_moves),
1514 [](const auto& edge) { return edge.GetMove(); });
1515 picked_node.eval->p.resize(legal_moves.size());
1516 picked_node.is_cache_hit = computation_->AddInput(
1517 EvalPosition{
1518 .pos = history.GetPositions(),
1519 .legal_moves = legal_moves,
1520 },
1521 picked_node.eval->AsPtr()) ==
1522 BackendComputation::FETCHED_IMMEDIATELY;
1523 }
1524 }
1525 if (params_.GetOutOfOrderEval() && picked_node.CanEvalOutOfOrder()) {
1526 // Perform out of order eval for the last entry in minibatch_.
1527 FetchSingleNodeResult(&picked_node);
1528 picked_node.ooo_completed = true;
1529 }
1530 }

1585void SearchWorker::EnsureNodeTwoFoldCorrectForDepth(Node* child_node,
1586 int depth) {
1587 // Check whether first repetition was before root. If yes, remove
1588 // terminal status of node and revert all visits in the tree.
1589 // Length of repetition was stored in m_. This code will only do
1590 // something when tree is reused and twofold visits need to be
1591 // reverted.
1592 if (child_node->IsTwoFoldTerminal() && depth < child_node->GetM()) {
1593 // Take a mutex - any SearchWorker specific mutex... since this is
1594 // not safe to do concurrently between multiple tasks.
1595 Mutex::Lock lock(picking_tasks_mutex_);
1596 int depth_counter = 0;
1597 // Cache node's values as we reset them in the process. We could
1598 // manually set wl and d, but if we want to reuse this for reverting
1599 // other terminal nodes this is the way to go.
1600 const auto wl = child_node->GetWL();
1601 const auto d = child_node->GetD();
1602 const auto m = child_node->GetM();
1603 const auto terminal_visits = child_node->GetN();
1604 for (Node* node_to_revert = child_node; node_to_revert != nullptr;
1605 node_to_revert = node_to_revert->GetParent()) {
1606 // Revert all visits on twofold draw when making it non terminal.
1607 node_to_revert->RevertTerminalVisits(wl, d, m + (float)depth_counter,
1608 terminal_visits);
1609 depth_counter++;
1610 // Even if original tree still exists, we don't want to revert
1611 // more than until new root.
1612 if (depth_counter > depth) break;
1613 // If wl != 0, we would have to switch signs at each depth.
1614 }
1615 // Mark the prior twofold draw as non terminal to extend it again.
1616 child_node->MakeNotTerminal();
1617 // When reverting the visits, we also need to revert the initial
1618 // visits, as we reused fewer nodes than anticipated.
1619 search_->initial_visits_ -= terminal_visits;
1620 // Max depth doesn't change when reverting the visits, and
1621 // cum_depth_ only counts the average depth of new nodes, not reused
1622 // ones.
1623 }

1625
1627void SearchWorker::PickNodesToExtendTask(
1628 Node* node, int base_depth, int collision_limit,
1629 const std::vector<Move>& moves_to_base,
1630 std::vector<NodeToProcess>* receiver,
1631 TaskWorkspace* workspace) NO_THREAD_SAFETY_ANALYSIS {
1632 LCTRACE_FUNCTION_SCOPE;
1633 // TODO: Bring back pre-cached nodes created outside locks in a way that works
1634 // with tasks.
1635 // TODO: pre-reserve visits_to_perform for expected depth and likely maximum
1636 // width. Maybe even do so outside of lock scope.
1637 auto& vtp_buffer = workspace->vtp_buffer;
1638 auto& visits_to_perform = workspace->visits_to_perform;
1639 visits_to_perform.clear();
1640 auto& vtp_last_filled = workspace->vtp_last_filled;
1641 vtp_last_filled.clear();
1642 auto& current_path = workspace->current_path;
1643 current_path.clear();
1644 auto& moves_to_path = workspace->moves_to_path;
1645 moves_to_path = moves_to_base;
1646 // Sometimes receiver is reused, othertimes not, so only jump start if small.
1647 if (receiver->capacity() < 30) {
1648 receiver->reserve(receiver->size() + 30);
1649 }
1650
1651 // These 2 are 'filled pre-emptively'.
1652 std::array<float, 256> current_pol;
1653 std::array<float, 256> current_util;
1654
1655 // These 3 are 'filled on demand'.
1656 std::array<float, 256> current_score;
1657 std::array<int, 256> current_nstarted;
1658 auto& cur_iters = workspace->cur_iters;
1659
1660 Node::Iterator best_edge;
1661 Node::Iterator second_best_edge;
1662 // Fetch the current best root node visits for possible smart pruning.
1663 const int64_t best_node_n = search_->current_best_edge_.GetN();
1664
1665 int passed_off = 0;
1666 int completed_visits = 0;
1667
1668 bool is_root_node = node == search_->root_node_;
1669 const float even_draw_score = search_->GetDrawScore(false);
1670 const float odd_draw_score = search_->GetDrawScore(true);
1671 const auto& root_move_filter = search_->root_move_filter_;
1672 auto m_evaluator = moves_left_support_ ? MEvaluator(params_) : MEvaluator();
1673
1674 int max_limit = std::numeric_limits<int>::max();
1675
1676 current_path.push_back(-1);
1677 while (current_path.size() > 0) {
1678 // First prepare visits_to_perform.
1679 if (current_path.back() == -1) {
1680 // Need to do n visits, where n is either collision_limit, or comes from
1681 // visits_to_perform for the current path.
1682 int cur_limit = collision_limit;
1683 if (current_path.size() > 1) {
1684 cur_limit =
1685 (*visits_to_perform.back())[current_path[current_path.size() - 2]];
1686 }
1687 // First check if node is terminal or not-expanded. If either than create
1688 // a collision of appropriate size and pop current_path.
1689 if (node->GetN() == 0 || node->IsTerminal()) {
1690 if (is_root_node) {
1691 // Root node is special - since its not reached from anywhere else, so
1692 // it needs its own logic. Still need to create the collision to
1693 // ensure the outer gather loop gives up.
1694 if (node->TryStartScoreUpdate()) {
1695 cur_limit -= 1;
1696 minibatch_.push_back(NodeToProcess::Visit(
1697 node, static_cast<uint16_t>(current_path.size() + base_depth)));
1698 completed_visits++;
1699 }
1700 }
1701 // Visits are created elsewhere, just need the collisions here.
1702 if (cur_limit > 0) {
1703 int max_count = 0;
1704 if (cur_limit == collision_limit && base_depth == 0 &&
1705 max_limit > cur_limit) {
1706 max_count = max_limit;
1707 }
1708 receiver->push_back(NodeToProcess::Collision(
1709 node, static_cast<uint16_t>(current_path.size() + base_depth),
1710 cur_limit, max_count));
1711 completed_visits += cur_limit;
1712 }
1713 node = node->GetParent();
1714 current_path.pop_back();
1715 continue;
1716 }
1717 if (is_root_node) {
1718 // Root node is again special - needs its n in flight updated separately
1719 // as its not handled on the path to it, since there isn't one.
1720 node->IncrementNInFlight(cur_limit);
1721 }
1722
1723 // Create visits_to_perform new back entry for this level.
1724 if (vtp_buffer.size() > 0) {
1725 visits_to_perform.push_back(std::move(vtp_buffer.back()));
1726 vtp_buffer.pop_back();
1727 } else {
1728 visits_to_perform.push_back(std::make_unique<std::array<int, 256>>());
1729 }
1730 vtp_last_filled.push_back(-1);
1731
1732 // Cache all constant UCT parameters.
1733 // When we're near the leaves we can copy less of the policy, since there
1734 // is no way iteration will ever reach it.
1735 // TODO: This is a very conservative formula. It assumes every visit we're
1736 // aiming to add is going to trigger a new child, and that any visits
1737 // we've already had have also done so and then a couple extra since we go
1738 // to 2 unvisited to get second best in worst case.
1739 // Unclear we can do better without having already walked the children.
1740 // Which we are putting off until after policy is copied so we can create
1741 // visited policy without having to cache it in the node (allowing the
1742 // node to stay at 64 bytes).
1743 int max_needed = node->GetNumEdges();
1744 if (!is_root_node || root_move_filter.empty()) {
1745 max_needed = std::min(max_needed, node->GetNStarted() + cur_limit + 2);
1746 }
1747 node->CopyPolicy(max_needed, current_pol.data());
1748 for (int i = 0; i < max_needed; i++) {
1749 current_util[i] = std::numeric_limits<float>::lowest();
1750 }
1751 // Root depth is 1 here, while for GetDrawScore() it's 0-based, that's why
1752 // the weirdness.
1753 const float draw_score = ((current_path.size() + base_depth) % 2 == 0)
1754 ? odd_draw_score
1755 : even_draw_score;
1756 m_evaluator.SetParent(node);
1757 float visited_pol = 0.0f;
1758 for (Node* child : node->VisitedNodes()) {
1759 int index = child->Index();
1760 visited_pol += current_pol[index];
1761 float q = child->GetQ(draw_score);
1762 current_util[index] = q + m_evaluator.GetMUtility(child, q);
1763 }
1764 const float fpu =
1765 GetFpu(params_, node, is_root_node, draw_score, visited_pol);
1766 for (int i = 0; i < max_needed; i++) {
1767 if (current_util[i] == std::numeric_limits<float>::lowest()) {
1768 current_util[i] = fpu + m_evaluator.GetDefaultMUtility();
1769 }
1770 }
1771
1772 const float cpuct = ComputeCpuct(params_, node->GetN(), is_root_node);
1773 const float puct_mult =
1774 cpuct * std::sqrt(std::max(node->GetChildrenVisits(), 1u));
1775 int cache_filled_idx = -1;
1776 while (cur_limit > 0) {
1777 // Perform UCT for current node.
1778 float best = std::numeric_limits<float>::lowest();
1779 int best_idx = -1;
1780 float best_without_u = std::numeric_limits<float>::lowest();
1781 float second_best = std::numeric_limits<float>::lowest();
1782 bool can_exit = false;
1783 best_edge.Reset();
1784 for (int idx = 0; idx < max_needed; ++idx) {
1785 if (idx > cache_filled_idx) {
1786 if (idx == 0) {
1787 cur_iters[idx] = node->Edges();
1788 } else {
1789 cur_iters[idx] = cur_iters[idx - 1];
1790 ++cur_iters[idx];
1791 }
1792 current_nstarted[idx] = cur_iters[idx].GetNStarted();
1793 }
1794 int nstarted = current_nstarted[idx];
1795 const float util = current_util[idx];
1796 if (idx > cache_filled_idx) {
1797 current_score[idx] =
1798 current_pol[idx] * puct_mult / (1 + nstarted) + util;
1799 cache_filled_idx++;
1800 }
1801 if (is_root_node) {
1802 // If there's no chance to catch up to the current best node with
1803 // remaining playouts, don't consider it.
1804 // best_move_node_ could have changed since best_node_n was
1805 // retrieved. To ensure we have at least one node to expand, always
1806 // include current best node.
1807 if (cur_iters[idx] != search_->current_best_edge_ &&
1808 latest_time_manager_hints_.GetEstimatedRemainingPlayouts() <
1809 best_node_n - cur_iters[idx].GetN()) {
1810 continue;
1811 }
1812 // If root move filter exists, make sure move is in the list.
1813 if (!root_move_filter.empty() &&
1814 std::find(root_move_filter.begin(), root_move_filter.end(),
1815 cur_iters[idx].GetMove()) == root_move_filter.end()) {
1816 continue;
1817 }
1818 }
1819
1820 float score = current_score[idx];
1821 if (score > best) {
1822 second_best = best;
1823 second_best_edge = best_edge;
1824 best = score;
1825 best_idx = idx;
1826 best_without_u = util;
1827 best_edge = cur_iters[idx];
1828 } else if (score > second_best) {
1829 second_best = score;
1830 second_best_edge = cur_iters[idx];
1831 }
1832 if (can_exit) break;
1833 if (nstarted == 0) {
1834 // One more loop will get 2 unvisited nodes, which is sufficient to
1835 // ensure second best is correct. This relies upon the fact that
1836 // edges are sorted in policy decreasing order.
1837 can_exit = true;
1838 }
1839 }
1840 int new_visits = 0;
1841 if (second_best_edge) {
1842 int estimated_visits_to_change_best = std::numeric_limits<int>::max();
1843 if (best_without_u < second_best) {
1844 const auto n1 = current_nstarted[best_idx] + 1;
1845 estimated_visits_to_change_best = static_cast<int>(
1846 std::max(1.0f, std::min(current_pol[best_idx] * puct_mult /
1847 (second_best - best_without_u) -
1848 n1 + 1,
1849 1e9f)));
1850 }
1851 second_best_edge.Reset();
1852 max_limit = std::min(max_limit, estimated_visits_to_change_best);
1853 new_visits = std::min(cur_limit, estimated_visits_to_change_best);
1854 } else {
1855 // No second best - only one edge, so everything goes in here.
1856 new_visits = cur_limit;
1857 }
1858 if (best_idx >= vtp_last_filled.back()) {
1859 auto* vtp_array = visits_to_perform.back().get()->data();
1860 std::fill(vtp_array + (vtp_last_filled.back() + 1),
1861 vtp_array + best_idx + 1, 0);
1862 }
1863 (*visits_to_perform.back())[best_idx] += new_visits;
1864 cur_limit -= new_visits;
1865 Node* child_node = best_edge.GetOrSpawnNode(/* parent */ node);
1866
1867 // Probably best place to check for two-fold draws consistently.
1868 // Depth starts with 1 at root, so real depth is depth - 1.
1869 EnsureNodeTwoFoldCorrectForDepth(
1870 child_node, current_path.size() + base_depth + 1 - 1);
1871
1872 bool decremented = false;
1873 if (child_node->TryStartScoreUpdate()) {
1874 current_nstarted[best_idx]++;
1875 new_visits -= 1;
1876 decremented = true;
1877 if (child_node->GetN() > 0 && !child_node->IsTerminal()) {
1878 child_node->IncrementNInFlight(new_visits);
1879 current_nstarted[best_idx] += new_visits;
1880 }
1881 current_score[best_idx] = current_pol[best_idx] * puct_mult /
1882 (1 + current_nstarted[best_idx]) +
1883 current_util[best_idx];
1884 }
1885 if ((decremented &&
1886 (child_node->GetN() == 0 || child_node->IsTerminal()))) {
1887 // Reduce 1 for the visits_to_perform to ensure the collision created
1888 // doesn't include this visit.
1889 (*visits_to_perform.back())[best_idx] -= 1;
1890 receiver->push_back(NodeToProcess::Visit(
1891 child_node,
1892 static_cast<uint16_t>(current_path.size() + 1 + base_depth)));
1893 completed_visits++;
1894 receiver->back().moves_to_visit.reserve(moves_to_path.size() + 1);
1895 receiver->back().moves_to_visit = moves_to_path;
1896 receiver->back().moves_to_visit.push_back(best_edge.GetMove());
1897 }
1898 if (best_idx > vtp_last_filled.back() &&
1899 (*visits_to_perform.back())[best_idx] > 0) {
1900 vtp_last_filled.back() = best_idx;
1901 }
1902 }
1903 is_root_node = false;
1904 // Actively do any splits now rather than waiting for potentially long
1905 // tree walk to get there.
1906 for (int i = 0; i <= vtp_last_filled.back(); i++) {
1907 int child_limit = (*visits_to_perform.back())[i];
1908 if (task_workers_ > 0 &&
1909 child_limit > params_.GetMinimumWorkSizeForPicking() &&
1910 child_limit <
1911 ((collision_limit - passed_off - completed_visits) * 2 / 3) &&
1912 child_limit + passed_off + completed_visits <
1913 collision_limit -
1914 params_.GetMinimumRemainingWorkSizeForPicking()) {
1915 Node* child_node = cur_iters[i].GetOrSpawnNode(/* parent */ node);
1916 // Don't split if not expanded or terminal.
1917 if (child_node->GetN() == 0 || child_node->IsTerminal()) continue;
1918
1919 bool passed = false;
1920 {
1921 // Multiple writers, so need mutex here.
1922 Mutex::Lock lock(picking_tasks_mutex_);
1923 // Ensure not to exceed size of reservation.
1924 if (picking_tasks_.size() < MAX_TASKS) {
1925 moves_to_path.push_back(cur_iters[i].GetMove());
1926 picking_tasks_.emplace_back(
1927 child_node, current_path.size() - 1 + base_depth + 1,
1928 moves_to_path, child_limit);
1929 moves_to_path.pop_back();
1930 task_count_.fetch_add(1, std::memory_order_acq_rel);
1931 task_added_.notify_all();
1932 passed = true;
1933 passed_off += child_limit;
1934 }
1935 }
1936 if (passed) {
1937 (*visits_to_perform.back())[i] = 0;
1938 }
1939 }
1940 }
1941 // Fall through to select the first child.
1942 }
1943 int min_idx = current_path.back();
1944 bool found_child = false;
1945 if (vtp_last_filled.back() > min_idx) {
1946 int idx = -1;
1947 for (auto& child : node->Edges()) {
1948 idx++;
1949 if (idx > min_idx && (*visits_to_perform.back())[idx] > 0) {
1950 if (moves_to_path.size() != current_path.size() + base_depth) {
1951 moves_to_path.push_back(child.GetMove());
1952 } else {
1953 moves_to_path.back() = child.GetMove();
1954 }
1955 current_path.back() = idx;
1956 current_path.push_back(-1);
1957 node = child.GetOrSpawnNode(/* parent */ node);
1958 found_child = true;
1959 break;
1960 }
1961 if (idx >= vtp_last_filled.back()) break;
1962 }
1963 }
1964 if (!found_child) {
1965 node = node->GetParent();
1966 if (!moves_to_path.empty()) moves_to_path.pop_back();
1967 current_path.pop_back();
1968 vtp_buffer.push_back(std::move(visits_to_perform.back()));
1969 visits_to_perform.pop_back();
1970 vtp_last_filled.pop_back();
1971 }

1977void SearchWorker::ExtendNode(Node* node, int depth,
1978 const std::vector<Move>& moves_to_node,
1979 PositionHistory* history) {
1980 // Initialize position sequence with pre-move position.
1981 history->Trim(search_->played_history_.GetLength());
1982 for (size_t i = 0; i < moves_to_node.size(); i++) {
1983 history->Append(moves_to_node[i]);
1984 }
1985
1986 // We don't need the mutex because other threads will see that N=0 and
1987 // N-in-flight=1 and will not touch this node.
1988 const auto& board = history->Last().GetBoard();
1989 auto legal_moves = board.GenerateLegalMoves();
1990
1991 // Check whether it's a draw/lose by position. Importantly, we must check
1992 // these before doing the by-rule checks below.
1993 if (legal_moves.empty()) {
1994 // Could be a checkmate or a stalemate
1995 if (board.IsUnderCheck()) {
1996 node->MakeTerminal(GameResult::WHITE_WON);
1997 } else {
1998 node->MakeTerminal(GameResult::DRAW);
1999 }
2000 return;
2001 }
2002
2003 // We can shortcircuit these draws-by-rule only if they aren't root;
2004 // if they are root, then thinking about them is the point.
2005 if (node != search_->root_node_) {
2006 if (!board.HasMatingMaterial()) {
2007 node->MakeTerminal(GameResult::DRAW);
2008 return;
2009 }
2010
2011 if (history->Last().GetRule50Ply() >= 100) {
2012 node->MakeTerminal(GameResult::DRAW);
2013 return;
2014 }
2015
2016 const auto repetitions = history->Last().GetRepetitions();
2017 // Mark two-fold repetitions as draws according to settings.
2018 // Depth starts with 1 at root, so number of plies in PV is depth - 1.
2019 if (repetitions >= 2) {
2020 node->MakeTerminal(GameResult::DRAW);
2021 return;
2022 } else if (repetitions == 1 && depth - 1 >= 4 &&
2023 params_.GetTwoFoldDraws() &&
2024 depth - 1 >= history->Last().GetPliesSincePrevRepetition()) {
2025 const auto cycle_length = history->Last().GetPliesSincePrevRepetition();
2026 // use plies since first repetition as moves left; exact if forced draw.
2027 node->MakeTerminal(GameResult::DRAW, (float)cycle_length,
2028 Node::Terminal::TwoFold);
2029 return;
2030 }
2031
2032 // Neither by-position or by-rule termination, but maybe it's a TB position.
2033 if (search_->syzygy_tb_ && !search_->root_is_in_dtz_ &&
2034 board.castlings().no_legal_castle() &&
2035 history->Last().GetRule50Ply() == 0 &&
2036 (board.ours() | board.theirs()).count() <=
2037 search_->syzygy_tb_->max_cardinality()) {
2038 ProbeState state;
2039 const WDLScore wdl =
2040 search_->syzygy_tb_->probe_wdl(history->Last(), &state);
2041 // Only fail state means the WDL is wrong, probe_wdl may produce correct
2042 // result with a stat other than OK.
2043 if (state != FAIL) {
2044 // TB nodes don't have NN evaluation, assign M from parent node.
2045 float m = 0.0f;
2046 // Need a lock to access parent, in case MakeSolid is in progress.
2047 {
2048 SharedMutex::SharedLock lock(search_->nodes_mutex_);
2049 auto parent = node->GetParent();
2050 if (parent) {
2051 m = std::max(0.0f, parent->GetM() - 1.0f);
2052 }
2053 }
2054 // If the colors seem backwards, check the checkmate check above.
2055 if (wdl == WDL_WIN) {
2056 node->MakeTerminal(GameResult::BLACK_WON, m,
2057 Node::Terminal::Tablebase);
2058 } else if (wdl == WDL_LOSS) {
2059 node->MakeTerminal(GameResult::WHITE_WON, m,
2060 Node::Terminal::Tablebase);
2061 } else { // Cursed wins and blessed losses count as draws.
2062 node->MakeTerminal(GameResult::DRAW, m, Node::Terminal::Tablebase);
2063 }
2064 search_->tb_hits_.fetch_add(1, std::memory_order_acq_rel);
2065 return;
2066 }
2067 }
2068 }
2069
2070 // Add legal moves as edges of this node.
2071 node->CreateEdges(legal_moves);
2072}

2108// Prefetches up to @budget nodes into cache. Returns number of nodes
2109// prefetched.
2111int SearchWorker::PrefetchIntoCache(Node* node, int budget, bool is_odd_depth) {
2112 const float draw_score = search_->GetDrawScore(is_odd_depth);
2113 if (budget <= 0) return 0;
2114
2115 // We are in a leaf, which is not yet being processed.
2116 if (!node || node->GetNStarted() == 0) {
2117 if (search_->backend_->GetCachedEvaluation(
2118 EvalPosition{history_.GetPositions(), {}})) {
2119 // Make it return 0 to make it not use the slot, so that the function
2120 // tries hard to find something to cache even among unpopular moves.
2121 // In practice that slows things down a lot though, as it's not always
2122 // easy to find what to cache.
2123 return 1;
2124 }
2125 auto moves = history_.Last().GetBoard().GenerateLegalMoves();
2126 computation_->AddInput(EvalPosition{history_.GetPositions(), moves},
2127 EvalResultPtr{});
2128 return 1;
2129 }
2130
2131 assert(node);
2132 // n = 0 and n_in_flight_ > 0, that means the node is being extended.
2133 if (node->GetN() == 0) return 0;
2134 // The node is terminal; don't prefetch it.
2135 if (node->IsTerminal()) return 0;
2136
2137 // Populate all subnodes and their scores.
2138 typedef std::pair<float, EdgeAndNode> ScoredEdge;
2139 std::vector<ScoredEdge> scores;
2140 const float cpuct =
2141 ComputeCpuct(params_, node->GetN(), node == search_->root_node_);
2142 const float puct_mult =
2143 cpuct * std::sqrt(std::max(node->GetChildrenVisits(), 1u));
2144 const float fpu =
2145 GetFpu(params_, node, node == search_->root_node_, draw_score);
2146 for (auto& edge : node->Edges()) {
2147 if (edge.GetP() == 0.0f) continue;
2148 // Flip the sign of a score to be able to easily sort.
2149 // TODO: should this use logit_q if set??
2150 scores.emplace_back(-edge.GetU(puct_mult) - edge.GetQ(fpu, draw_score),
2151 edge);
2152 }
2153
2154 size_t first_unsorted_index = 0;
2155 int total_budget_spent = 0;
2156 int budget_to_spend = budget; // Initialize for the case where there's only
2157 // one child.
2158 for (size_t i = 0; i < scores.size(); ++i) {
2159 if (search_->stop_.load(std::memory_order_acquire)) break;
2160 if (budget <= 0) break;
2161
2162 // Sort next chunk of a vector. 3 at a time. Most of the time it's fine.
2163 if (first_unsorted_index != scores.size() &&
2164 i + 2 >= first_unsorted_index) {
2165 const int new_unsorted_index =
2166 std::min(scores.size(), budget < 2 ? first_unsorted_index + 2
2167 : first_unsorted_index + 3);
2168 std::partial_sort(scores.begin() + first_unsorted_index,
2169 scores.begin() + new_unsorted_index, scores.end(),
2170 [](const ScoredEdge& a, const ScoredEdge& b) {
2171 return a.first < b.first;
2172 });
2173 first_unsorted_index = new_unsorted_index;
2174 }
2175
2176 auto edge = scores[i].second;
2177 // Last node gets the same budget as prev-to-last node.
2178 if (i != scores.size() - 1) {
2179 // Sign of the score was flipped for sorting, so flip it back.
2180 const float next_score = -scores[i + 1].first;
2181 // TODO: As above - should this use logit_q if set?
2182 const float q = edge.GetQ(-fpu, draw_score);
2183 if (next_score > q) {
2184 budget_to_spend =
2185 std::min(budget, int(edge.GetP() * puct_mult / (next_score - q) -
2186 edge.GetNStarted()) +
2187 1);
2188 } else {
2189 budget_to_spend = budget;
2190 }
2191 }
2192 history_.Append(edge.GetMove());
2193 const int budget_spent =
2194 PrefetchIntoCache(edge.node(), budget_to_spend, !is_odd_depth);
2195 history_.Pop();
2196 budget -= budget_spent;
2197 total_budget_spent += budget_spent;
2198 }
2199 return total_budget_spent;
2200}

2281void SearchWorker::DoBackupUpdateSingleNode(
2282 const NodeToProcess& node_to_process) REQUIRES(search_->nodes_mutex_) {
2283 Node* node = node_to_process.node;
2284 if (node_to_process.IsCollision()) {
2285 // Collisions are handled via shared_collisions instead.
2286 return;
2287 }
2288
2289 // For the first visit to a terminal, maybe update parent bounds too.
2290 auto update_parent_bounds =
2291 params_.GetStickyEndgames() && node->IsTerminal() && !node->GetN();
2292
2293 // Backup V value up to a root. After 1 visit, V = Q.
2294 float v = node_to_process.eval->q;
2295 float d = node_to_process.eval->d;
2296 float m = node_to_process.eval->m;
2297 int n_to_fix = 0;
2298 float v_delta = 0.0f;
2299 float d_delta = 0.0f;
2300 float m_delta = 0.0f;
2301 uint32_t solid_threshold =
2302 static_cast<uint32_t>(params_.GetSolidTreeThreshold());
2303 for (Node *n = node, *p; n != search_->root_node_->GetParent(); n = p) {
2304 p = n->GetParent();
2305
2306 // Current node might have become terminal from some other descendant, so
2307 // backup the rest of the way with more accurate values.
2308 if (n->IsTerminal()) {
2309 v = n->GetWL();
2310 d = n->GetD();
2311 m = n->GetM();
2312 }
2313 n->FinalizeScoreUpdate(v, d, m, node_to_process.multivisit);
2314 if (n_to_fix > 0 && !n->IsTerminal()) {
2315 n->AdjustForTerminal(v_delta, d_delta, m_delta, n_to_fix);
2316 }
2317 if (n->GetN() >= solid_threshold) {
2318 if (n->MakeSolid() && n == search_->root_node_) {
2319 // If we make the root solid, the current_best_edge_ becomes invalid and
2320 // we should repopulate it.
2321 search_->current_best_edge_ =
2322 search_->GetBestChildNoTemperature(search_->root_node_, 0);
2323 }
2324 }
2325
2326 // Nothing left to do without ancestors to update.
2327 if (!p) break;
2328
2329 bool old_update_parent_bounds = update_parent_bounds;
2330 // If parent already is terminal further adjustment is not required.
2331 if (p->IsTerminal()) n_to_fix = 0;
2332 // Try setting parent bounds except the root or those already terminal.
2333 update_parent_bounds =
2334 update_parent_bounds && p != search_->root_node_ && !p->IsTerminal() &&
2335 MaybeSetBounds(p, m, &n_to_fix, &v_delta, &d_delta, &m_delta);
2336
2337 // Q will be flipped for opponent.
2338 v = -v;
2339 v_delta = -v_delta;
2340 m++;
2341
2342 // Update the stats.
2343 // Best move.
2344 // If update_parent_bounds was set, we just adjusted bounds on the
2345 // previous loop or there was no previous loop, so if n is a terminal, it
2346 // just became that way and could be a candidate for changing the current
2347 // best edge. Otherwise a visit can only change best edge if its to an edge
2348 // that isn't already the best and the new n is equal or greater to the old
2349 // n.
2350 if (p == search_->root_node_ &&
2351 ((old_update_parent_bounds && n->IsTerminal()) ||
2352 (n != search_->current_best_edge_.node() &&
2353 search_->current_best_edge_.GetN() <= n->GetN()))) {
2354 search_->current_best_edge_ =
2355 search_->GetBestChildNoTemperature(search_->root_node_, 0);
2356 }
2357 }
2358 search_->total_playouts_ += node_to_process.multivisit;
2359 if (node_to_process.nn_queried && !node_to_process.is_cache_hit) {
2360 search_->network_evaluations_++;
2361 }
2362 search_->cum_depth_ += node_to_process.depth * node_to_process.multivisit;
2363 search_->max_depth_ = std::max(search_->max_depth_, node_to_process.depth);
2364}

2367bool SearchWorker::MaybeSetBounds(Node* p, float m, int* n_to_fix,
2368 float* v_delta, float* d_delta,
2369 float* m_delta) const {
2370 auto losing_m = 0.0f;
2371 auto prefer_tb = false;
2372
2373 // Determine the maximum (lower, upper) bounds across all children.
2374 // (-1,-1) Loss (initial and lowest bounds)
2375 // (-1, 0) Can't Win
2376 // (-1, 1) Regular node
2377 // ( 0, 0) Draw
2378 // ( 0, 1) Can't Lose
2379 // ( 1, 1) Win (highest bounds)
2380 auto lower = GameResult::BLACK_WON;
2381 auto upper = GameResult::BLACK_WON;
2382 for (const auto& edge : p->Edges()) {
2383 const auto [edge_lower, edge_upper] = edge.GetBounds();
2384 lower = std::max(edge_lower, lower);
2385 upper = std::max(edge_upper, upper);
2386
2387 // Checkmate is the best, so short-circuit.
2388 const auto is_tb = edge.IsTbTerminal();
2389 if (edge_lower == GameResult::WHITE_WON && !is_tb) {
2390 prefer_tb = false;
2391 break;
2392 } else if (edge_upper == GameResult::BLACK_WON) {
2393 // Track the longest loss.
2394 losing_m = std::max(losing_m, edge.GetM(0.0f));
2395 }
2396 prefer_tb = prefer_tb || is_tb;
2397 }
2398
2399 // The parent's bounds are flipped from the children (-max(U), -max(L))
2400 // aggregated as if it was a single child (forced move) of the same bound.
2401 // Loss (-1,-1) -> ( 1, 1) Win
2402 // Can't Win (-1, 0) -> ( 0, 1) Can't Lose
2403 // Regular (-1, 1) -> (-1, 1) Regular
2404 // Draw ( 0, 0) -> ( 0, 0) Draw
2405 // Can't Lose ( 0, 1) -> (-1, 0) Can't Win
2406 // Win ( 1, 1) -> (-1,-1) Loss
2407
2408 // Nothing left to do for ancestors if the parent would be a regular node.
2409 if (lower == GameResult::BLACK_WON && upper == GameResult::WHITE_WON) {
2410 return false;
2411 } else if (lower == upper) {
2412 // Search can stop at the parent if the bounds can't change anymore, so make
2413 // it terminal preferring shorter wins and longer losses.
2414 *n_to_fix = p->GetN();
2415 assert(*n_to_fix > 0);
2416 float cur_v = p->GetWL();
2417 float cur_d = p->GetD();
2418 float cur_m = p->GetM();
2419 p->MakeTerminal(
2420 -upper,
2421 (upper == GameResult::BLACK_WON ? std::max(losing_m, m) : m) + 1.0f,
2422 prefer_tb ? Node::Terminal::Tablebase : Node::Terminal::EndOfGame);
2423 // Negate v_delta because we're calculating for the parent, but immediately
2424 // afterwards we'll negate v_delta in case it has come from the child.
