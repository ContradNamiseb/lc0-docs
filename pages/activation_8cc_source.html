37void SoftmaxActivation(const size_t size, const float* input, float* output) {
38 auto alpha = *std::max_element(input, input + size);
39
40 auto denom = 0.0f;
41 for (size_t i = 0; i < size; i++) {
42 auto val = std::exp(input[i] - alpha);
43 output[i] = val;
44 denom += val;
45 }
46 for (size_t i = 0; i < size; i++) {
47 output[i] = output[i] / denom;
48 }
49}

51static inline float mish(float val) {
52 auto e = expf(val);
53 auto n = e * e + 2.0f * e;
54 auto d = val / (n + 2.0f);
55 if (val <= -0.125f) {
56 return n * d;
57 } else {
58 return val - 2.0f * d;
59 }
60}

62static inline float selu(float val) {
63 float alpha = 1.67326324f, scale = 1.05070098f;
64 if (val > 0) {
65 return scale * val;
66 } else {
67 return scale * alpha * (expf(val) - 1.0f);
68 }
69}

71float Activate(const float val, const ActivationFunction activation) {
72 switch (activation) {
73 case ACTIVATION_RELU:
74 return val > 0 ? val : 0;
75 case ACTIVATION_RELU_2:
76 return val > 0 ? val * val : 0;
77 case ACTIVATION_MISH:
78 return mish(val);
79 case ACTIVATION_TANH:
80 return tanhf(val);
81 case ACTIVATION_SIGMOID:
82 return 1.0f / (1.0f + expf(-val));
83 case ACTIVATION_SELU:
84 return selu(val);
85 case ACTIVATION_SWISH:
86 return val / (1.0f + expf(-val));
87 case ACTIVATION_NONE:
88 // Nothing to do.
89 break;
90 default:
91 throw Exception("unsupported activation function");
92 }
93 return val;
94}

96void Activate(const size_t len, const float* data, const float* bias,
97 float* output, const ActivationFunction activation) {
98 if (activation == ACTIVATION_NONE) {
99 for (size_t b = 0; b < len; b++) {
100 output[b] = data[b] + bias[b];
101 }
102 } else if (activation == ACTIVATION_RELU) {
103#ifndef USE_ISPC
104 for (size_t b = 0; b < len; b++) {
105 float val = data[b] + bias[b];
106 output[b] = val > 0 ? val : 0;
107 }
108#else
109 ispc::ActivateRelu(len, 1.0f, data, bias, 0.0f, output);
110#endif
111 } else if (activation == ACTIVATION_MISH) {
112#ifndef USE_ISPC
113 for (size_t b = 0; b < len; b++) {
114 float val = data[b] + bias[b];
115 output[b] = mish(val);
116 }
117#else
118 ispc::ActivateMish(len, 1.0f, data, bias, 0.0f, output);
119#endif
120 } else if (activation == ACTIVATION_RELU_2) {
121#ifndef USE_ISPC
122 for (size_t b = 0; b < len; b++) {
123 float val = data[b] + bias[b];
124 output[b] = val > 0 ? val * val : 0;
125 }
126#else
127 ispc::ActivateRelu_2(len, data, bias, output);
128#endif
129 } else if (activation == ACTIVATION_SWISH) {
130#ifndef USE_ISPC
131 for (size_t b = 0; b < len; b++) {
132 float val = data[b] + bias[b];
133 output[b] = val / (1.0f + exp(-val));
134 ;
135 }
136#else
137 ispc::ActivateSwish(len, data, bias, output);
138#endif
139 } else if (activation == ACTIVATION_SELU) {
140#ifndef USE_ISPC
141 for (size_t b = 0; b < len; b++) {
142 float val = data[b] + bias[b];
143 output[b] = selu(val);
144 }
145#else
146 ispc::ActivateSelu(len, data, bias, output);
147#endif
148 } else {
149 for (size_t b = 0; b < len; b++) {
150 float val = data[b] + bias[b];
151 output[b] = Activate(val, activation);
152 }
153 }
154}

156void Activate(const size_t len, float gamma, const float* data,
157 const float* bias, float beta, float* output,
158 const ActivationFunction activation) {
159 if (activation == ACTIVATION_NONE) {
160 for (size_t b = 0; b < len; b++) {
161 float val = gamma * data[b] + bias[b] + beta;
162 output[b] = val;
163 }
164 } else if (activation == ACTIVATION_RELU) {
165#ifndef USE_ISPC
166 for (size_t b = 0; b < len; b++) {
167 float val = gamma * data[b] + bias[b] + beta;
168 output[b] = val > 0 ? val : 0;
169 }
170#else
171 ispc::ActivateRelu(len, gamma, data, bias, beta, output);
172#endif
173 } else if (activation == ACTIVATION_MISH) {
174#ifndef USE_ISPC
175 for (size_t b = 0; b < len; b++) {
176 float val = gamma * data[b] + bias[b] + beta;
177 output[b] = mish(val);
178 }
179#else
180 ispc::ActivateMish(len, gamma, data, bias, beta, output);
181#endif
182 } else {
183 for (size_t b = 0; b < len; b++) {
184 float val = gamma * data[b] + bias[b] + beta;

190void BiasResidual(const size_t batch_size, const size_t channels, float* data,
191 const float* biases, const float* eltwise,
192 const ActivationFunction activation) {
193 for (size_t i = 0; i < batch_size; i++) {
194 for (size_t c = 0; c < channels; ++c) {
195 auto bias = biases[c];
196 auto arr = &data[c * kSquares];
197 auto res = &eltwise[c * kSquares];
198 if (activation == ACTIVATION_NONE) {
199 for (size_t b = 0; b < kSquares; b++) {
200 float val = res[b] + arr[b] + bias;
201 arr[b] = val;
202 }
203 } else if (activation == ACTIVATION_RELU) {
204#ifndef USE_ISPC
205 for (size_t b = 0; b < kSquares; b++) {
206 float val = res[b] + arr[b] + bias;
207 arr[b] = val > 0 ? val : 0;
208 }
209#else
210 ispc::ActivateRelu(kSquares, 1.0f, res, arr, bias, arr);
211#endif
212 } else if (activation == ACTIVATION_MISH) {
213#ifndef USE_ISPC
214 for (size_t b = 0; b < kSquares; b++) {
215 float val = res[b] + arr[b] + bias;
216 arr[b] = mish(val);
217 }
218#else
219 ispc::ActivateMish(kSquares, 1.0f, res, arr, bias, arr);
220#endif
221 } else {
222 for (size_t b = 0; b < kSquares; b++) {
223 float val = res[b] + arr[b] + bias;
224 arr[b] = Activate(val, activation);
225 }
226 }
227 }
228 data += channels * kSquares;
229 eltwise += channels * kSquares;
230 }
