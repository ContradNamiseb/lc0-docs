82SelfPlayGame::SelfPlayGame(PlayerOptions white, PlayerOptions black,
83 bool shared_tree, const Opening& opening)
84 : options_{white, black},
85 chess960_{white.uci_options->Get<bool>(kUciChess960) ||
86 black.uci_options->Get<bool>(kUciChess960)},
87 training_data_(classic::SearchParams(*white.uci_options).GetHistoryFill(),
88 classic::SearchParams(*black.uci_options).GetHistoryFill(),
89 pblczero::NetworkFormat::INPUT_CLASSICAL_112_PLANE) {
90 orig_fen_ = opening.start_fen;
91 tree_[0] = std::make_shared<classic::NodeTree>();
92 tree_[0]->ResetToPosition(orig_fen_, {});
93
94 if (shared_tree) {
95 tree_[1] = tree_[0];
96 } else {
97 tree_[1] = std::make_shared<classic::NodeTree>();
98 tree_[1]->ResetToPosition(orig_fen_, {});
99 }
100 int ply = 0;
101 auto white_prob = white.uci_options->Get<float>(kOpeningStopProbId);
102 auto black_prob = black.uci_options->Get<float>(kOpeningStopProbId);
103 if (white_prob != black_prob && white_prob != 0 && black_prob != 0) {
104 throw Exception("Stop probabilities must be both equal or zero!");
105 }
106
107 for (Move m : opening.moves) {
108 // For early exit from the opening, we support two cases: a) where both
109 // sides have the same exit probability and b) where one side's exit
110 // probability is zero. In the following formula, `positions` is the number
111 // of possible exit points remaining, used for adjusting the exit
112 // probability (to avoid favoring the last position).
113 auto exit_prob_now = tree_[0]->IsBlackToMove() ? black_prob : white_prob;
114 auto exit_prob_next = tree_[0]->IsBlackToMove() ? white_prob : black_prob;
115 int positions = opening.moves.size() - ply + 1;
116 if (exit_prob_now > 0.0f &&
117 Random::Get().GetFloat(1.0f) <
118 std::max(exit_prob_now,
119 exit_prob_now / (exit_prob_now * ((positions + 1) / 2) +
120 exit_prob_next * (positions / 2)))) {
121 break;
122 }
123 if (tree_[0]->IsBlackToMove()) m.Flip();
124 tree_[0]->MakeMove(m);
125 if (tree_[0] != tree_[1]) tree_[1]->MakeMove(m);
126 ply++;
127 }
128 start_ply_ = ply;
129}

131void SelfPlayGame::Play(int white_threads, int black_threads, bool training,
132 SyzygyTablebase* syzygy_tb, bool enable_resign) {
133 bool blacks_move = tree_[0]->IsBlackToMove();
134
135 // Take syzygy tablebases from player1 options.
136 std::string tb_paths =
137 options_[0].uci_options->Get<std::string>(kSyzygyTablebaseId);
138 if (!tb_paths.empty()) { // && tb_paths != tb_paths_) {
139 syzygy_tb_ = std::make_unique<SyzygyTablebase>();
140 CERR << "Loading Syzygy tablebases from " << tb_paths;
141 if (!syzygy_tb_->init(tb_paths)) {
142 CERR << "Failed to load Syzygy tablebases!";
143 syzygy_tb_ = nullptr;
144 }
145 }
146 // Do moves while not end of the game. (And while not abort_)
147 while (!abort_) {
148 game_result_ = tree_[0]->GetPositionHistory().ComputeGameResult();
149
150 // If endgame, stop.
151 if (game_result_ != GameResult::UNDECIDED) break;
152 if (tree_[0]->GetPositionHistory().Last().GetGamePly() >= 450) {
153 adjudicated_ = true;
154 break;
155 }
156 // Initialize search.
157 const int idx = blacks_move ? 1 : 0;
158 if (!options_[idx].uci_options->Get<bool>(kReuseTreeId)) {
159 tree_[idx]->TrimTreeAtHead();
160 }
161 {
162 std::lock_guard<std::mutex> lock(mutex_);
163 if (abort_) break;
164 auto stoppers = options_[idx].search_limits.MakeSearchStopper();
165 classic::PopulateIntrinsicStoppers(stoppers.get(),
166 *options_[idx].uci_options);
167
168 std::unique_ptr<UciResponder> responder =
169 std::make_unique<CallbackUciResponder>(
170 options_[idx].best_move_callback, options_[idx].info_callback);
171
172 search_ = std::make_unique<classic::Search>(
173 *tree_[idx], options_[idx].backend, std::move(responder),
174 /* searchmoves */ MoveList(), std::chrono::steady_clock::now(),
175 std::move(stoppers), /* infinite */ false, /* ponder */ false,
176 *options_[idx].uci_options, syzygy_tb);
177 }
178
179 // Do search.
180 search_->RunBlocking(blacks_move ? black_threads : white_threads);
181 move_count_++;
182 nodes_total_ += search_->GetTotalPlayouts();
183 if (abort_) break;
184
185 Move best_move;
186 bool best_is_terminal;
187 const auto best_eval = search_->GetBestEval(&best_move, &best_is_terminal);
188 float eval = best_eval.wl;
189 eval = (eval + 1) / 2;
190 if (eval < min_eval_[idx]) min_eval_[idx] = eval;
191 const int move_number = tree_[0]->GetPositionHistory().GetLength() / 2 + 1;
192 auto best_w = (best_eval.wl + 1.0f - best_eval.d) / 2.0f;
193 auto best_d = best_eval.d;
194 auto best_l = best_w - best_eval.wl;
195 max_eval_[0] = std::max(max_eval_[0], blacks_move ? best_l : best_w);
196 max_eval_[1] = std::max(max_eval_[1], best_d);
197 max_eval_[2] = std::max(max_eval_[2], blacks_move ? best_w : best_l);
198 if (enable_resign && move_number >= options_[idx].uci_options->Get<int>(
199 kResignEarliestMoveId)) {
200 const float resignpct =
201 options_[idx].uci_options->Get<float>(kResignPercentageId) / 100;
202 if (options_[idx].uci_options->Get<bool>(kResignWDLStyleId)) {
203 auto threshold = 1.0f - resignpct;
204 if (best_w > threshold) {
205 game_result_ =
206 blacks_move ? GameResult::BLACK_WON : GameResult::WHITE_WON;
207 adjudicated_ = true;
208 break;
209 }
210 if (best_l > threshold) {
211 game_result_ =
212 blacks_move ? GameResult::WHITE_WON : GameResult::BLACK_WON;
213 adjudicated_ = true;
214 break;
215 }
216 if (best_d > threshold) {
217 game_result_ = GameResult::DRAW;
218 adjudicated_ = true;
219 break;
220 }
221 } else {
222 if (eval < resignpct) { // always false when resignpct == 0
223 game_result_ =
224 blacks_move ? GameResult::WHITE_WON : GameResult::BLACK_WON;
225 adjudicated_ = true;
226 break;
227 }
228 }
229 }
230
231 auto node = tree_[idx]->GetCurrentHead();
232 classic::Eval played_eval = best_eval;
233 Move move;
234 while (true) {
235 move = search_->GetBestMove().first;
236 uint32_t max_n = 0;
237 uint32_t cur_n = 0;
238
239 for (auto& edge : node->Edges()) {
240 if (edge.GetN() > max_n) {
241 max_n = edge.GetN();
242 }
243 if (edge.GetMove(tree_[idx]->IsBlackToMove()) == move) {
244 cur_n = edge.GetN();
245 played_eval.wl = edge.GetWL(-node->GetWL());
246 played_eval.d = edge.GetD(node->GetD());
247 played_eval.ml = edge.GetM(node->GetM() - 1) + 1;
248 }
249 }
250 // If 'best move' is less than allowed visits and not max visits,
251 // discard it and try again.
252 if (cur_n == max_n ||
253 static_cast<int>(cur_n) >=
254 options_[idx].uci_options->Get<int>(kMinimumAllowedVistsId)) {
255 break;
256 }
257 PositionHistory history_copy = tree_[idx]->GetPositionHistory();
258 Move move_for_history = move;
259 if (tree_[idx]->IsBlackToMove()) move_for_history.Flip();
260 history_copy.Append(move_for_history);
261 // Ensure not to discard games that are already decided.
262 if (history_copy.ComputeGameResult() == GameResult::UNDECIDED) {
263 auto move_list_to_discard = GetMoves();
264 move_list_to_discard.push_back(move);
265 options_[idx].discarded_callback({orig_fen_, move_list_to_discard});
266 }
267 search_->ResetBestMove();
268 }
269
270 if (training) {
271 bool best_is_proof = best_is_terminal; // But check for better moves.
272 if (best_is_proof && best_eval.wl < 1) {
273 auto best =
274 (best_eval.wl == 0) ? GameResult::DRAW : GameResult::BLACK_WON;
275 auto upper = best;
276 for (const auto& edge : node->Edges()) {
277 upper = std::max(edge.GetBounds().second, upper);
278 }
279 if (best < upper) {
280 best_is_proof = false;
281 }
282 }
283 // Append training data. The GameResult is later overwritten.
284 std::vector<Move> legal_moves = tree_[idx]
285 ->GetPositionHistory()
286 .Last()
287 .GetBoard()
288 .GenerateLegalMoves();
289 std::optional<EvalResult> nneval =
290 options_[idx].backend->GetCachedEvaluation(EvalPosition{
291 tree_[idx]->GetPositionHistory().GetPositions(), legal_moves});
292 training_data_.Add(tree_[idx]->GetCurrentHead(),
293 tree_[idx]->GetPositionHistory(), best_eval,
294 played_eval, best_is_proof, best_move, move,
295 legal_moves, nneval,
296 search_->GetParams().GetPolicySoftmaxTemp());
297 }
298 // Must reset the search before mutating the tree.
299 search_.reset();
300
301 // Add best move to the tree.
302 if (tree_[0]->IsBlackToMove()) move.Flip();
303 tree_[0]->MakeMove(move);
304 if (tree_[0] != tree_[1]) tree_[1]->MakeMove(move);
305 blacks_move = !blacks_move;
306 }
307}

309std::vector<Move> SelfPlayGame::GetMoves() const {
310 std::vector<Move> moves;
311 for (classic::Node* node = tree_[0]->GetCurrentHead();
312 node != tree_[0]->GetGameBeginNode(); node = node->GetParent()) {
313 moves.push_back(node->GetParent()->GetEdgeToNode(node)->GetMove());
314 }
315 std::vector<Move> result;
316 Position pos = tree_[0]->GetPositionHistory().Starting();
317 while (!moves.empty()) {
318 Move move = moves.back();
319 moves.pop_back();
320 pos = Position(pos, move);
321 // Position already flipped, therefore flip the move if white to move.
322 if (!pos.IsBlackToMove()) move.Flip();
323 result.push_back(move);
324 }
325 return result;
326}

328float SelfPlayGame::GetWorstEvalForWinnerOrDraw() const {
329 // TODO: This assumes both players have the same resign style.
330 // Supporting otherwise involves mixing the meaning of worst.
331 if (options_[0].uci_options->Get<bool>(kResignWDLStyleId)) {
332 if (game_result_ == GameResult::WHITE_WON) {
333 return std::max(max_eval_[1], max_eval_[2]);
334 } else if (game_result_ == GameResult::BLACK_WON) {
335 return std::max(max_eval_[1], max_eval_[0]);
336 } else {
337 return std::max(max_eval_[2], max_eval_[0]);
338 }
339 }
340 if (game_result_ == GameResult::WHITE_WON) return min_eval_[0];
341 if (game_result_ == GameResult::BLACK_WON) return min_eval_[1];
342 return std::min(min_eval_[0], min_eval_[1]);
343}

355std::unique_ptr<classic::ChainedSearchStopper>
356SelfPlayLimits::MakeSearchStopper() const {
357 auto result = std::make_unique<classic::ChainedSearchStopper>();
358
359 // always set VisitsStopper to avoid exceeding the limit 4000000000, the
360 // default value when visits = 0

362 if (playouts >= 0) {

365 }
366 if (movetime >= 0) {
367 result->AddStopper(std::make_unique<classic::TimeLimitStopper>(movetime));
368 }
369 return result;
