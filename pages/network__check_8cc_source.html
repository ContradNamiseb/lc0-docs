126 std::vector<float> PolicySoftMax(const NetworkComputation* comp, int sample,
127 const std::vector<Move>& moves) const {
128 float max_p = -std::numeric_limits<float>::infinity();
129 std::vector<float> policy;
130 policy.reserve(moves.size());
131 for (const auto move : moves) {
132 policy.emplace_back(comp->GetPVal(sample, MoveToNNIndex(move, 0)));
133 max_p = std::max(max_p, policy.back());
134 }
135 float total = 0;
136 for (auto& p : policy) {
137 p = std::exp(p - max_p);
138 total += p;
139 }
140 if (total > 0) {
141 for (auto& p : policy) {
142 p /= total;
143 }
144 }
145 return policy;
146 }

148 void CheckOnly() const {
149 bool valueAlmostEqual = true;
150 const int size = GetBatchSize();
151 for (int i = 0; i < size && valueAlmostEqual; i++) {
152 const float v1 = work_comp_->GetQVal(i);
153 const float v2 = check_comp_->GetQVal(i);
154 valueAlmostEqual &= IsAlmostEqual(v1, v2);
155 }
156
157 bool policyAlmostEqual = true;
158 for (int i = 0; i < size && policyAlmostEqual; i++) {
159 const auto work = PolicySoftMax(work_comp_.get(), i, moves_[i]);
160 const auto check = PolicySoftMax(check_comp_.get(), i, moves_[i]);
161 for (size_t j = 0; j < work.size(); j++) {
162 policyAlmostEqual &= IsAlmostEqual(work[j], check[j]);
163 }
164 }
165
166 if (valueAlmostEqual && policyAlmostEqual) {
167 CERR << "Check passed for a batch of " << size << ".";
168 return;
169 }
170
171 if (!valueAlmostEqual && !policyAlmostEqual) {
172 CERR << "*** ERROR check failed for a batch of " << size
173 << " both value and policy incorrect.";
174 return;
175 }
176
177 if (!valueAlmostEqual) {
178 CERR << "*** ERROR check failed for a batch of " << size
179 << " value incorrect (but policy ok).";
180 return;
181 }
182
183 CERR << "*** ERROR check failed for a batch of " << size
184 << " policy incorrect (but value ok).";
185 }

187 bool IsAlmostEqual(double a, double b) const {
188 return std::abs(a - b) <= std::max(params_.relative_tolerance *
189 std::max(std::abs(a), std::abs(b)),
190 params_.absolute_tolerance);
191 }

193 void DisplayHistogram() {
194 Histogram histogram(-15, 1, 5);
195
196 const int size = GetBatchSize();
197 for (int i = 0; i < size; i++) {
198 const float qv1 = work_comp_->GetQVal(i);
199 const float qv2 = check_comp_->GetQVal(i);
200 histogram.Add(qv2 - qv1);
201 const auto work = PolicySoftMax(work_comp_.get(), i, moves_[i]);
202 const auto check = PolicySoftMax(check_comp_.get(), i, moves_[i]);
203 for (size_t j = 0; j < work.size(); j++) {
204 histogram.Add(check[j] - work[j]);
205 }
206 }
207 CERR << "Absolute error histogram for a batch of " << size;
208 histogram.Dump();
209 }

211 // Compute maximum absolute/relative errors.
212 struct MaximumError {
213 double max_absolute_error = 0;
214 double max_relative_error = 0;
215
216 void Add(double a, double b) {
217 const double absolute_error = GetAbsoluteError(a, b);
218 if (absolute_error > max_absolute_error) {
219 max_absolute_error = absolute_error;
220 }
221 const double relative_error = GetRelativeError(a, b);
222 if (relative_error > max_relative_error) {
223 max_relative_error = relative_error;
224 }
225 }
226
227 void Dump(const char* name) {
228 CERR << std::scientific << std::setprecision(1) << name
229 << ": absolute: " << max_absolute_error
230 << ", relative: " << max_relative_error << ".";
231 }
232
233 static double GetRelativeError(double a, double b) {
234 const double max = std::max(std::abs(a), std::abs(b));
235 return max == 0 ? 0 : std::abs(a - b) / max;
236 }
237
238 static double GetAbsoluteError(double a, double b) {
239 return std::abs(a - b);
240 }
241 };
242
243 void DisplayError() {
244 MaximumError value_error;
245 const int size = GetBatchSize();
246 for (int i = 0; i < size; i++) {
247 const float v1 = work_comp_->GetQVal(i);
248 const float v2 = check_comp_->GetQVal(i);
249 value_error.Add(v1, v2);
250 }
251
252 MaximumError policy_error;
253 for (int i = 0; i < size; i++) {
254 const auto work = PolicySoftMax(work_comp_.get(), i, moves_[i]);
255 const auto check = PolicySoftMax(check_comp_.get(), i, moves_[i]);
256 for (size_t j = 0; j < work.size(); j++) {
257 policy_error.Add(work[j], check[j]);
258 }
259 }
260
261 CERR << "maximum error for a batch of " << size << ":";
262
263 value_error.Dump(" value");
264 policy_error.Dump(" policy");
265 }

280 CheckNetwork(const std::optional<WeightsFile>& weights,
281 const OptionsDict& options) {
282 params_.mode = kDefaultMode;
283 params_.absolute_tolerance = kDefaultAbsoluteTolerance;
284 params_.relative_tolerance = kDefaultRelativeTolerance;
285 check_frequency_ = kDefaultCheckFrequency;
286
287 OptionsDict dict1;
288 std::string backendName1 = "opencl";
289 OptionsDict& backend1_dict = dict1;
290
291 OptionsDict dict2;
292 std::string backendName2 = "eigen";
293 OptionsDict& backend2_dict = dict2;
294
295 const std::string mode = options.GetOrDefault<std::string>("mode", "check");
296 if (mode == "check") {
297 params_.mode = kCheckOnly;
298 } else if (mode == "histo") {
299 params_.mode = kHistogram;
300 } else if (mode == "display") {
301 params_.mode = kErrorDisplay;
302 }
303
304 params_.absolute_tolerance =
305 options.GetOrDefault<float>("atol", kDefaultAbsoluteTolerance);
306 params_.relative_tolerance =
307 options.GetOrDefault<float>("rtol", kDefaultRelativeTolerance);
308
309 const auto parents = options.ListSubdicts();
310 if (parents.size() > 0) {
311 backendName1 = parents[0];
312 backend1_dict = options.GetSubdict(backendName1);
313 backendName1 =
314 backend1_dict.GetOrDefault<std::string>("backend", backendName1);
315 }
316 if (parents.size() > 1) {
317 backendName2 = parents[1];
318 backend2_dict = options.GetSubdict(backendName2);
319 backendName2 =
320 backend2_dict.GetOrDefault<std::string>("backend", backendName2);
321 }
322 if (parents.size() > 2) {
323 CERR << "Warning, cannot check more than two backends";
324 }
325
326 CERR << "Working backend set to " << backendName1 << ".";
327 CERR << "Reference backend set to " << backendName2 << ".";
328
329 work_net_ =
330 NetworkFactory::Get()->Create(backendName1, weights, backend1_dict);
331 check_net_ =
332 NetworkFactory::Get()->Create(backendName2, weights, backend2_dict);
333
334 capabilities_ = work_net_->GetCapabilities();
335 capabilities_.Merge(check_net_->GetCapabilities());
336
337 params_.input_format = capabilities_.input_format;
338
339 check_frequency_ =
340 options.GetOrDefault<float>("freq", kDefaultCheckFrequency);
341 switch (params_.mode) {
342 case kCheckOnly:
343 CERR << std::scientific << std::setprecision(1)
344 << "Check mode: check only with relative tolerance "
345 << params_.relative_tolerance << ", absolute tolerance "
346 << params_.absolute_tolerance << ".";
347 break;
348 case kErrorDisplay:
349 CERR << "Check mode: error display.";
350 break;
351 case kHistogram:
352 CERR << "Check mode: histogram.";
353 break;
354 }
355 CERR << "Check rate: " << std::fixed << std::setprecision(0)
356 << 100 * check_frequency_ << "%.";
357 }
358
359 std::unique_ptr<NetworkComputation> NewComputation() override {
360 const double draw = Random::Get().GetDouble(1.0);
361 const bool check = draw < check_frequency_;
362 if (check) {
363 std::unique_ptr<NetworkComputation> work_comp =
364 work_net_->NewComputation();
365 std::unique_ptr<NetworkComputation> check_comp =
366 check_net_->NewComputation();
367 return std::make_unique<CheckComputation>(params_, std::move(work_comp),
368 std::move(check_comp));
369 }
370 return work_net_->NewComputation();
371 }
372
373 const NetworkCapabilities& GetCapabilities() const override {
374 return capabilities_;
375 }
