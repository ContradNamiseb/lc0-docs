139class Move {
140 public:
141 Move() = default;
142 static constexpr Move White(Square from, Square to) {
143 return Move((from.as_idx() << 6) | to.as_idx());
144 }
145 static constexpr Move WhitePromotion(Square from, Square to,
146 PieceType promotion_piece) {
147 return Move((from.as_idx() << 6) | to.as_idx() | kPromotion |
148 (promotion_piece.idx << 12));
149 }
150 static constexpr Move WhiteCastling(File king, File rook) {
151 return Move((king.idx << 6) | rook.idx | kCastling);
152 }
153 static constexpr Move WhiteEnPassant(Square from, Square to) {
154 return Move((from.as_idx() << 6) | to.as_idx() | kEnPassant);
155 }

157 bool operator==(const Move& other) const = default;
158 bool operator!=(const Move& other) const = default;

160 // Mirrors the ranks of the move.
161 void Flip() { data_ ^= kFlipMask; }
162 std::string ToString(bool is_chess960) const;
163
164 Square from() const { return Square::FromIdx((data_ & kFromMask) >> 6); }
165 Square to() const { return Square::FromIdx(data_ & kToMask); }
166 bool is_promotion() const { return data_ & kPromotion; }
167 PieceType promotion() const {
168 return PieceType::FromIdx((data_ & kPieceMask) >> 12);
169 }
170 bool is_castling() const { return (data_ & kSpecialMask) == kCastling; }
171 bool is_en_passant() const { return (data_ & kSpecialMask) == kEnPassant; }
172 // TODO remove this once UciReponder starts using std::optional for ponder.
173 bool is_null() const { return data_ == 0; }

175 uint16_t raw_data() const { return data_; }

177 private:
178 explicit constexpr Move(uint16_t data) : data_(data) {}

180 // Move encoding using 16 bits:
181 // - bits 0-5: "to" square (6 bits)
182 // - bits 6-11: "from" square (6 bits)
183 // - bits 12-13: if is_promotion: promotion piece type
184 // if !is_promotion: SpecialMove
185 // - bit 14: is_promotion flag
186 // - bit 15: reserved (potentially for side-to-move)
187 // Castling is always encoded as a "king takes rook" move.
188 uint16_t data_ = 0;
189
190 enum Masks : uint16_t {
191 // clang-format off
192 kToMask = 0b0000000000111111,
193 kFromMask = 0b0000111111000000,
194 kSpecialMask = 0b0111000000000000,
195 kCastling = 0b0001000000000000,
196 kEnPassant = 0b0010000000000000,
197 kPromotion = 0b0100000000000000,
198 kPieceMask = 0b0011000000000000,
199 // If/when we have side-to-move bit, also flip it here.
200 kFlipMask = 0b0000111000111000,
201 // clang-format on
202 };
203};

212inline PieceType PieceType::Parse(char c) {
213 switch (tolower(c)) {
214 case 'n':
215 return kKnight;
216 case 'q':
217 return kQueen;
218 case 'r':
219 return kRook;
220 case 'b':
221 return kBishop;
222 case 'p':
223 return kPawn;
224 case 'k':
225 return kKing;
226 default:
227 return PieceType{6};
228 }
229}

231inline std::string Move::ToString(bool is_chess960) const {
232 if (is_castling() && !is_chess960) {
233 return from().ToString() + (to().file() > from().file() ? "g" : "c") +
234 to().rank().ToString();
235 }
236 return from().ToString() + to().ToString() +
237 (is_promotion() ? promotion().ToString(false) : "");
