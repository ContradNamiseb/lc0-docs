103void SelfPlayTournament::PopulateOptions(OptionsParser* options) {
104 options->AddContext("player1");
105 options->AddContext("player2");
106 options->AddContext("white");
107 options->AddContext("black");
108 for (const auto context : {"player1", "player2"}) {
109 auto* dict = options->GetMutableOptions(context);
110 dict->AddSubdict("white")->AddAliasDict(&options->GetOptionsDict("white"));
111 dict->AddSubdict("black")->AddAliasDict(&options->GetOptionsDict("black"));
112 }
113
114 SharedBackendParams::Populate(options);
115 options->Add<IntOption>(kThreadsId, 1, 8) = 1;
116 classic::SearchParams::Populate(options);
117
118 options->Add<BoolOption>(kShareTreesId) = true;
119 options->Add<IntOption>(kTotalGamesId, -2, 999999) = -1;
120 options->Add<IntOption>(kParallelGamesId, 1, 256) = 8;
121 options->Add<IntOption>(kPlayoutsId, -1, 999999999) = -1;
122 options->Add<IntOption>(kVisitsId, -1, 999999999) = -1;
123 options->Add<IntOption>(kTimeMsId, -1, 999999999) = -1;
124 options->Add<BoolOption>(kTrainingId) = false;
125 options->Add<BoolOption>(kVerboseThinkingId) = false;
126 options->Add<IntOption>(kPolicyModeSizeId, 0, 1024) = 0;
127 options->Add<IntOption>(kValueModeSizeId, 0, 64) = 0;
128 options->Add<StringOption>(kTournamentResultsFileId) = "";
129 options->Add<BoolOption>(kMoveThinkingId) = false;
130 options->Add<FloatOption>(kResignPlaythroughId, 0.0f, 100.0f) = 0.0f;
131 options->Add<FloatOption>(kDiscardedStartChanceId, 0.0f, 100.0f) = 0.0f;
132 options->Add<StringOption>(kOpeningsFileId) = "";
133 options->Add<BoolOption>(kOpeningsMirroredId) = false;
134 std::vector<std::string> openings_modes = {"sequential", "shuffled",
135 "random"};
136 options->Add<ChoiceOption>(kOpeningsModeId, openings_modes) = "sequential";
137
138 options->Add<StringOption>(kSyzygyTablebaseId);
139 SelfPlayGame::PopulateUciParams(options);
140
141 auto defaults = options->GetMutableDefaultsOptions();
142 defaults->Set<int>(classic::SearchParams::kMiniBatchSizeId, 32);
143 defaults->Set<float>(classic::SearchParams::kCpuctId, 1.2f);
144 defaults->Set<float>(classic::SearchParams::kCpuctFactorId, 0.0f);
145 defaults->Set<float>(SharedBackendParams::kPolicySoftmaxTemp, 1.0f);
146 defaults->Set<int>(classic::SearchParams::kMaxCollisionVisitsId, 1);
147 defaults->Set<int>(classic::SearchParams::kMaxCollisionEventsId, 1);
148 defaults->Set<int>(classic::SearchParams::kCacheHistoryLengthId, 7);
149 defaults->Set<bool>(classic::SearchParams::kOutOfOrderEvalId, false);
150 defaults->Set<float>(classic::SearchParams::kTemperatureId, 1.0f);
151 defaults->Set<float>(classic::SearchParams::kNoiseEpsilonId, 0.25f);
152 defaults->Set<float>(classic::SearchParams::kFpuValueId, 0.0f);
153 defaults->Set<std::string>(SharedBackendParams::kHistoryFill, "no");
154 defaults->Set<std::string>(SharedBackendParams::kBackendId, "multiplexing");
155 defaults->Set<bool>(classic::SearchParams::kStickyEndgamesId, false);
156 defaults->Set<bool>(classic::SearchParams::kTwoFoldDrawsId, false);
157 defaults->Set<int>(classic::SearchParams::kTaskWorkersPerSearchWorkerId, 0);
158}

160SelfPlayTournament::SelfPlayTournament(const OptionsDict& options,
161 UciResponder* uci_responder,
162 GameInfo::Callback game_info,
163 TournamentInfo::Callback tournament_info)
164 : player_options_{{options.GetSubdict("player1").GetSubdict("white"),
165 options.GetSubdict("player1").GetSubdict("black")},
166 {options.GetSubdict("player2").GetSubdict("white"),
167 options.GetSubdict("player2").GetSubdict("black")}},
168 uci_responder_(uci_responder),
169 game_callback_(game_info),
170 tournament_callback_(tournament_info),
171 kTotalGames(options.Get<int>(kTotalGamesId)),
172 kShareTree(options.Get<bool>(kShareTreesId)),
173 kParallelism(options.Get<int>(kParallelGamesId)),
174 kTraining(options.Get<bool>(kTrainingId)),
175 kResignPlaythrough(options.Get<float>(kResignPlaythroughId)),
176 kPolicyGamesSize(options.Get<int>(kPolicyModeSizeId)),
177 kValueGamesSize(options.Get<int>(kValueModeSizeId)),
178 kTournamentResultsFile(
179 options.Get<std::string>(kTournamentResultsFileId)),
180 kDiscardedStartChance(options.Get<float>(kDiscardedStartChanceId)) {
181 multi_games_size_ = std::max(kPolicyGamesSize, kValueGamesSize);
182 std::string book = options.Get<std::string>(kOpeningsFileId);
183 if (!book.empty()) {
184 PgnReader book_reader;
185 book_reader.AddPgnFile(book);
186 openings_ = book_reader.ReleaseGames();
187 if (options.Get<std::string>(kOpeningsModeId) == "shuffled") {
188 Random::Get().Shuffle(openings_.begin(), openings_.end());
189 }
190 }
191 if (kPolicyGamesSize > 0 && kValueGamesSize > 0) {
192 throw Exception("Can't do both policy and value games at the same time.");
193 }
194 if (multi_games_size_ > 0 && openings_.size() == 0) {
195 throw Exception(
196 "Policy/Value games are deterministic, needs opening book to be "
197 "useful.");
198 }
199 if (multi_games_size_ > 0 &&
200 (kTotalGames == -1 ||
201 (kTotalGames > 0 &&
202 static_cast<size_t>(kTotalGames) > openings_.size() * 2))) {
203 throw Exception(
204 "Policy/Value games are deterministic, you do not want to go through "
205 "the "
206 "opening book more than once.");
207 }
208 // If playing just one game, the player1 is white, otherwise randomize.
209 if (kTotalGames != 1) {
210 first_game_black_ = Random::Get().GetBool();
211 }
212
213 static constexpr const char* kPlayerNames[2] = {"player1", "player2"};
214 static constexpr const char* kPlayerColors[2] = {"white", "black"};
215
216 // Initializing networks.
217 std::vector<std::shared_ptr<Backend>> backend_list;
218 for (int name_idx : {0, 1}) {
219 for (int color_idx : {0, 1}) {
220 const auto& name = kPlayerNames[name_idx];
221 const auto& color = kPlayerColors[color_idx];
222 const auto& opts = options.GetSubdict(name).GetSubdict(color);
223 for (const auto& backend : backend_list) {
224 if (backend->IsSameConfiguration(opts)) {
225 backends_[name_idx][color_idx] = backend;
226 break;
227 }
228 }
229 if (!backends_[name_idx][color_idx]) {
230 backends_[name_idx][color_idx] =
231 CreateMemCache(BackendManager::Get()->CreateFromParams(opts),
232 options.GetSubdict(name));
233 backend_list.emplace_back(backends_[name_idx][color_idx]);
234 }
235 }
236 }
237
238 // SearchLimits.
239 for (int name_idx : {0, 1}) {
240 for (int color_idx : {0, 1}) {
241 auto& limits = search_limits_[name_idx][color_idx];
242 const auto& dict = options.GetSubdict(kPlayerNames[name_idx])
243 .GetSubdict(kPlayerColors[color_idx]);
244 limits.playouts = dict.Get<int>(kPlayoutsId);
245 limits.visits = dict.Get<int>(kVisitsId);
246 limits.movetime = dict.Get<int>(kTimeMsId);
247
248 if (multi_games_size_ == 0 && limits.playouts == -1 &&
249 limits.visits == -1 && limits.movetime == -1) {
250 throw Exception(
251 "Please define --visits, --playouts or --movetime, otherwise it's "
252 "not clear when to stop search.");
253 } else if (multi_games_size_ > 0 &&
254 (limits.playouts != -1 || limits.visits != -1 ||
255 limits.movetime != -1)) {
256 throw Exception(
257 "Policy mode does not need --visits, --playouts or --movetime as "
258 "it has fixed 1 node behaviour.");
259 }
260 }
261 }
262
263 // Take syzygy tablebases from options.
264 std::string tb_paths = options.Get<std::string>(kSyzygyTablebaseId);
265 if (!tb_paths.empty()) {
266 syzygy_tb_ = std::make_unique<SyzygyTablebase>();
267 CERR << "Loading Syzygy tablebases from " << tb_paths;
268 if (!syzygy_tb_->init(tb_paths)) {
269 CERR << "Failed to load Syzygy tablebases!";
270 syzygy_tb_ = nullptr;
271 }
272 }
273}

275void SelfPlayTournament::PlayOneGame(int game_number) {
276 bool player1_black; // Whether player1 will player as black in this game.
277 Opening opening;
278 {
279 Mutex::Lock lock(mutex_);
280 player1_black = ((game_number % 2) == 1) != first_game_black_;
281 if (!openings_.empty()) {
282 if (player_options_[0][0].Get<bool>(kOpeningsMirroredId)) {
283 opening = openings_[(game_number / 2) % openings_.size()];
284 } else if (player_options_[0][0].Get<std::string>(kOpeningsModeId) ==
285 "random") {
286 opening = openings_[Random::Get().GetInt(0, openings_.size() - 1)];
287 } else {
288 opening = openings_[game_number % openings_.size()];
289 }
290 }
291 if (discard_pile_.size() > 0 &&
292 Random::Get().GetFloat(100.0f) < kDiscardedStartChance) {
293 const size_t idx = Random::Get().GetInt(0, discard_pile_.size() - 1);
294 if (idx != discard_pile_.size() - 1) {
295 std::swap(discard_pile_[idx], discard_pile_.back());
296 }
297 opening = discard_pile_.back();
298 discard_pile_.pop_back();
299 }
300 }
301 const int color_idx[2] = {player1_black ? 1 : 0, player1_black ? 0 : 1};
302
303 PlayerOptions options[2];
304
305 std::vector<ThinkingInfo> last_thinking_info;
306 for (int pl_idx : {0, 1}) {
307 const int color = color_idx[pl_idx];
308 const bool verbose_thinking =
309 player_options_[pl_idx][color].Get<bool>(kVerboseThinkingId);
310 const bool move_thinking =
311 player_options_[pl_idx][color].Get<bool>(kMoveThinkingId);
312 // Populate per-player options.
313 PlayerOptions& opt = options[color_idx[pl_idx]];
314 opt.backend = backends_[pl_idx][color].get();
315 opt.uci_options = &player_options_[pl_idx][color];
316 opt.search_limits = search_limits_[pl_idx][color];
317
318 // "bestmove" callback.
319 opt.best_move_callback = [this, game_number, pl_idx, player1_black,
320 verbose_thinking, move_thinking,
321 &last_thinking_info](const BestMoveInfo& info) {
322 if (!move_thinking) {
323 last_thinking_info.clear();
324 return;
325 }
326 // In non-verbose mode, output the last "info" message.
327 if (!verbose_thinking && !last_thinking_info.empty()) {
328 uci_responder_->OutputThinkingInfo(&last_thinking_info);
329 last_thinking_info.clear();
330 }
331 BestMoveInfo rich_info = info;
332 rich_info.player = pl_idx + 1;
333 rich_info.is_black = player1_black ? pl_idx == 0 : pl_idx != 0;
334 rich_info.game_id = game_number;
335 uci_responder_->OutputBestMove(&rich_info);
336 };
337
338 opt.info_callback =
339 [this, game_number, pl_idx, player1_black, verbose_thinking,
340 &last_thinking_info](const std::vector<ThinkingInfo>& infos) {
341 std::vector<ThinkingInfo> rich_info = infos;
342 for (auto& info : rich_info) {
343 info.player = pl_idx + 1;
344 info.is_black = player1_black ? pl_idx == 0 : pl_idx != 0;
345 info.game_id = game_number;
346 }
347 if (verbose_thinking) {
348 uci_responder_->OutputThinkingInfo(&rich_info);
349 } else {
350 // In non-verbose mode, remember the last "info" messages.
351 last_thinking_info = std::move(rich_info);
352 }
353 };
354 opt.discarded_callback = [this](const Opening& moves) {
355 // Only track discards if discard start chance is non-zero.
356 if (kDiscardedStartChance == 0.0f) return;
357 Mutex::Lock lock(mutex_);
358 discard_pile_.push_back(moves);
359 // 10k seems it should be enough to keep a good mix and avoid running out
360 // of ram.
361 if (discard_pile_.size() > 10000) {
362 // Swap a random element to end and pop it to avoid growing.
363 const size_t idx = Random::Get().GetInt(0, discard_pile_.size() - 1);
364 if (idx != discard_pile_.size() - 1) {
365 std::swap(discard_pile_[idx], discard_pile_.back());
366 }
367 discard_pile_.pop_back();
368 }
369 };
370 }
371
372 // Iterator to store the game in. Have to keep it so that later we can
373 // delete it. Need to expose it in games_ member variable only because
374 // of possible Abort() that should stop them all.
375 std::list<std::unique_ptr<SelfPlayGame>>::iterator game_iter;
376 SyzygyTablebase* syzygy_tb;
377 {
378 Mutex::Lock lock(mutex_);
379 games_.emplace_front(std::make_unique<SelfPlayGame>(options[0], options[1],
380 kShareTree, opening));
381 game_iter = games_.begin();
382 syzygy_tb = syzygy_tb_.get();
383 }
384 auto& game = **game_iter;
385
386 // If kResignPlaythrough == 0, then this comparison is unconditionally true
387 const bool enable_resign =
388 Random::Get().GetFloat(100.0f) >= kResignPlaythrough;
389
390 // PLAY GAME!
391 auto player1_threads = player_options_[0][color_idx[0]].Get<int>(kThreadsId);
392 auto player2_threads = player_options_[1][color_idx[1]].Get<int>(kThreadsId);
393 game.Play(player1_threads, player2_threads, kTraining, syzygy_tb,
394 enable_resign);
395
396 // If game was aborted, it's still undecided.
397 if (game.GetGameResult() != GameResult::UNDECIDED) {
398 // Game callback.
399 GameInfo game_info;
400 game_info.game_result = game.GetGameResult();
401 game_info.is_black = player1_black;
402 game_info.game_id = game_number;
403 game_info.initial_fen = opening.start_fen;
404 game_info.moves = game.GetMoves();
405 game_info.play_start_ply = game.GetStartPly();
406 if (!enable_resign) {
407 game_info.min_false_positive_threshold =
408 game.GetWorstEvalForWinnerOrDraw();
409 }
410 if (kTraining &&
411 game_info.play_start_ply < static_cast<int>(game_info.moves.size())) {
412 TrainingDataWriter writer(game_number);
413 game.WriteTrainingData(&writer);
414 writer.Finalize();
415 game_info.training_filename = writer.GetFileName();
416 }
417 game_callback_(game_info);
418
419 // Update tournament stats.
420 {
421 Mutex::Lock lock(mutex_);
422 int result = game.GetGameResult() == GameResult::DRAW ? 1
423 : game.GetGameResult() == GameResult::WHITE_WON ? 0
424 : 2;
425 if (player1_black) result = 2 - result;
426 ++tournament_info_.results[result][player1_black ? 1 : 0];
427 tournament_info_.move_count_ += game.move_count_;
428 tournament_info_.nodes_total_ += game.nodes_total_;
429 tournament_callback_(tournament_info_);
430 }
431 }
432
433 {
434 Mutex::Lock lock(mutex_);
435 games_.erase(game_iter);
436 }
437}

439void SelfPlayTournament::PlayMultiGames(int game_id, size_t game_count) {
440 bool use_value = kValueGamesSize > 0;
441 std::vector<Opening> openings;
442 openings.reserve(game_count / 2);
443 size_t opening_basis = game_id / 2;
444 {
445 Mutex::Lock lock(mutex_);
446 for (size_t i = 0; i < game_count / 2; i++) {
447 openings.push_back(openings_[(opening_basis + i) % openings_.size()]);
448 }
449 }
450
451 PlayerOptions options[2];
452 options[0].backend = backends_[0][0].get();
453 options[1].backend = backends_[1][1].get();
454
455 std::list<std::unique_ptr<MultiSelfPlayGames>>::iterator game1_iter;
456 auto aborted = false;
457 {
458 Mutex::Lock lock(mutex_);
459 multigames_.emplace_front(std::make_unique<MultiSelfPlayGames>(
460 options[0], options[1], openings, syzygy_tb_.get(), use_value));
461 game1_iter = multigames_.begin();
462 aborted = abort_;
463 }
464 auto& game1 = **game1_iter;
465
466 // PLAY GAMEs!
467 if (!aborted) game1.Play();
468
469 options[0].backend = backends_[0][1].get();
470 options[1].backend = backends_[1][0].get();
471
472 std::list<std::unique_ptr<MultiSelfPlayGames>>::iterator game2_iter;
473 {
474 Mutex::Lock lock(mutex_);
475 multigames_.emplace_front(std::make_unique<MultiSelfPlayGames>(
476 options[1], options[0], openings, syzygy_tb_.get(), use_value));
477 game2_iter = multigames_.begin();
478 aborted = abort_;
479 }
480 auto& game2 = **game2_iter;
481 // PLAY reverse GAMEs!
482 if (!aborted) game2.Play();
483
484 for (size_t i = 0; i < openings.size(); i++) {
485 auto game1_res = game1.GetGameResult(i);
486 if (game1_res != GameResult::UNDECIDED) {
487 // Game callback.
488 GameInfo game_info;
489 game_info.game_result = game1_res;
490 game_info.is_black = false;
491 game_info.game_id = game_id + 2 * i;
492 game_info.moves = game1.GetMoves(i);
493 game_info.initial_fen = openings[i].start_fen;
494 game_info.play_start_ply = openings[i].moves.size();
495 game_callback_(game_info);
496
497 // Update tournament stats.
498 {
499 Mutex::Lock lock(mutex_);
500 int result = game1_res == GameResult::DRAW ? 1
501 : game1_res == GameResult::WHITE_WON ? 0
502 : 2;
503 ++tournament_info_.results[result][0];
504 tournament_callback_(tournament_info_);
505 }
506 }
507 auto game2_res = game2.GetGameResult(i);
508 if (game2_res != GameResult::UNDECIDED) {
509 // Game callback.
510 GameInfo game_info;
511 game_info.game_result = game2_res;
512 game_info.is_black = true;
513 game_info.game_id = game_id + 2 * i + 1;
514 game_info.moves = game2.GetMoves(i);
515 game_info.initial_fen = openings[i].start_fen;
516 game_info.play_start_ply = openings[i].moves.size();
517 game_callback_(game_info);
518
519 // Update tournament stats.
520 {
521 Mutex::Lock lock(mutex_);
522 int result = game2_res == GameResult::DRAW ? 1
523 : game2_res == GameResult::WHITE_WON ? 2
524 : 0;
525 ++tournament_info_.results[result][1];
526 tournament_callback_(tournament_info_);
527 }
528 }
529 }
530 {
531 Mutex::Lock lock(mutex_);
532 multigames_.erase(game2_iter);
533 multigames_.erase(game1_iter);
534 }
535}

537void SelfPlayTournament::Worker() {
538 // Play games while game limit is not reached (or while not aborted).
539 while (true) {
540 int game_id;
541 int count = 0;
542 {
543 Mutex::Lock lock(mutex_);
544 if (abort_) break;
545 if (multi_games_size_) {
546 if (!player_options_[0][0].Get<bool>(kOpeningsMirroredId)) {
547 throw Exception(
548 "Policy/Value multi games mode only supports mirrored openings.");
549 }
550 if (kTotalGames > 0 && kTotalGames % 2 == 1) {
551 throw Exception(
552 "Policy/Value multi games can't support mirrored with an odd "
553 "number "
554 "of games.");
555 }
556 int to_take = 2 * multi_games_size_;
557 int max_take = 2 * multi_games_size_;
558 if (kTotalGames != -1) {
559 int cap = kTotalGames == -2 ? openings_.size() * 2 : kTotalGames;
560 to_take = std::min(max_take, cap - games_count_);
561 }
562 if (to_take <= 0) {
563 break;
564 }
565 game_id = games_count_;
566 count = to_take;
567 games_count_ += to_take;
568 } else {
569 bool mirrored = player_options_[0][0].Get<bool>(kOpeningsMirroredId);
570 if ((kTotalGames >= 0 && games_count_ >= kTotalGames) ||
571 (kTotalGames == -2 && !openings_.empty() &&
572 games_count_ >=
573 static_cast<int>(openings_.size()) * (mirrored ? 2 : 1)))
574 break;
575 game_id = games_count_++;
576 }
577 }
578 if (multi_games_size_) {
579 PlayMultiGames(game_id, count);
580 } else {
581 PlayOneGame(game_id);
582 }
583 }
584}

586void SelfPlayTournament::StartAsync() {
587 Mutex::Lock lock(threads_mutex_);
588 while (threads_.size() < kParallelism) {
589 threads_.emplace_back([&]() { Worker(); });
590 }

593void SelfPlayTournament::RunBlocking() {
594 if (kParallelism == 1) {
595 // No need for multiple threads if there is one worker.
