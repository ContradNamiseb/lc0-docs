55MoveList MakeRootMoveFilter(const MoveList& searchmoves,
56 SyzygyTablebase* syzygy_tb,
57 const PositionHistory& history, bool fast_play,
58 std::atomic<int>* tb_hits, bool* dtz_success) {
59 assert(tb_hits);
60 assert(dtz_success);
61 // Search moves overrides tablebase.
62 if (!searchmoves.empty()) return searchmoves;
63 const auto& board = history.Last().GetBoard();
64 MoveList root_moves;
65 if (!syzygy_tb || !board.castlings().no_legal_castle() ||
66 (board.ours() | board.theirs()).count() > syzygy_tb->max_cardinality()) {
67 return root_moves;
68 }
69 if (syzygy_tb->root_probe(
70 history.Last(), fast_play || history.DidRepeatSinceLastZeroingMove(),
71 false, &root_moves)) {
72 *dtz_success = true;
73 tb_hits->fetch_add(1, std::memory_order_acq_rel);
74 } else if (syzygy_tb->root_probe_wdl(history.Last(), &root_moves)) {
75 tb_hits->fetch_add(1, std::memory_order_acq_rel);
76 }
77 return root_moves;
78}

80class MEvaluator {
81 public:
82 MEvaluator()
83 : enabled_{false},
84 m_slope_{0.0f},
85 m_cap_{0.0f},
86 a_constant_{0.0f},
87 a_linear_{0.0f},
88 a_square_{0.0f},
89 q_threshold_{0.0f},
90 parent_m_{0.0f} {}
91
92 MEvaluator(const SearchParams& params, const Node* parent = nullptr)
93 : enabled_{true},
94 m_slope_{params.GetMovesLeftSlope()},
95 m_cap_{params.GetMovesLeftMaxEffect()},
96 a_constant_{params.GetMovesLeftConstantFactor()},
97 a_linear_{params.GetMovesLeftScaledFactor()},
98 a_square_{params.GetMovesLeftQuadraticFactor()},
99 q_threshold_{params.GetMovesLeftThreshold()},
100 parent_m_{parent ? parent->GetM() : 0.0f},
101 parent_within_threshold_{parent ? WithinThreshold(parent, q_threshold_)
102 : false} {}
103
104 void SetParent(const Node* parent) {
105 assert(parent);
106 if (enabled_) {
107 parent_m_ = parent->GetM();
108 parent_within_threshold_ = WithinThreshold(parent, q_threshold_);
109 }
110 }
111
112 // Calculates the utility for favoring shorter wins and longer losses.
113 float GetMUtility(Node* child, float q) const {
114 if (!enabled_ || !parent_within_threshold_) return 0.0f;
115 const float child_m = child->GetM();
116 float m = std::clamp(m_slope_ * (child_m - parent_m_), -m_cap_, m_cap_);
117 m *= FastSign(-q);
118 if (q_threshold_ > 0.0f && q_threshold_ < 1.0f) {
119 // This allows a smooth M effect with higher q thresholds, which is
120 // necessary for using MLH together with contempt.
121 q = std::max(0.0f, (std::abs(q) - q_threshold_)) / (1.0f - q_threshold_);
122 }
123 m *= a_constant_ + a_linear_ * std::abs(q) + a_square_ * q * q;
124 return m;
125 }
126
127 float GetMUtility(const EdgeAndNode& child, float q) const {
128 if (!enabled_ || !parent_within_threshold_) return 0.0f;
129 if (child.GetN() == 0) return GetDefaultMUtility();
130 return GetMUtility(child.node(), q);
131 }
132
133 // The M utility to use for unvisited nodes.
134 float GetDefaultMUtility() const { return 0.0f; }
135
136 private:
137 static bool WithinThreshold(const Node* parent, float q_threshold) {
138 return std::abs(parent->GetQ(0.0f)) > q_threshold;
139 }
140
141 const bool enabled_;
142 const float m_slope_;
143 const float m_cap_;
144 const float a_constant_;
145 const float a_linear_;
146 const float a_square_;
147 const float q_threshold_;
148 float parent_m_ = 0.0f;
149 bool parent_within_threshold_ = false;
150};

152// Unpack task_count_ atomic which holds both task_count_ and tasks_taken_. It
153// can unpack a value from an already read value or load it from the atomic
154// variable.
155// Variables are packed together because there is a potential race between task
156// workers and ResetTasks. A task worker can read tasks_taken_ and task_count
157// to a local register. A task worker can be suspended by kernel before tries
158// to acquire work. Other threads can process all tasks and main thread resets
159// tasks before the suspended thread resumes. The suspended thread now manages
160// to acquire work based on stale values if the stale tasks_taken was zero.
161// Packed values avoid the race because compare exchange is checking both when
162// incrementing tasks_taken_.
163template <typename T>
164std::tuple<int, int, int> ReadTaskCount(T& task_count) {
165 int packed;
166 if constexpr (std::is_same_v<T, std::atomic<int>>) {
167 packed = task_count.load(std::memory_order_acquire);
168 } else {
169 packed = task_count;
170 }
171 // The top half is tasks taken.
172 const int shift = SearchWorker::kTasksTakenShift;
173 int tasks_taken = packed >> shift;
174 // The bottom is task count. The first shift moves the sign bit from the lower
175 // half to the hardware sign bit. The second shift lowers bits back to the
176 // original positions and duplicates the sign bit if it is set.
177 int tc = (packed << shift) >> shift;
178 return {packed, tasks_taken, tc};
179}
180
181[[maybe_unused]]
182bool IsTasksCompleted(const std::atomic<int>& task_count,
183 const std::atomic<int>& completed_tasks) {
184 int tc = 0, nta = 0;
185 std::tie(std::ignore, nta, tc) = ReadTaskCount(task_count);
186 int ct = completed_tasks.load(std::memory_order_acquire);
187 return tc == ct || (nta == ct && tc == -1);
188}

250void ApplyDirichletNoise(LowNode* node, float eps, double alpha) {
251 float total = 0;
252 std::vector<float> noise;
253
254 for (int i = 0; i < node->GetNumEdges(); ++i) {
255 float eta = Random::Get().GetGamma(alpha, 1.0);
256 noise.emplace_back(eta);
257 total += eta;
258 }
259
260 if (total < std::numeric_limits<float>::min()) return;
261
262 int noise_idx = 0;
263 auto edges = node->GetEdges();
264 std::transform(edges, edges + node->GetNumEdges(), edges, [&](auto edge) {
265 edge.SetP(edge.GetP() * (1 - eps) + eps * noise[noise_idx++] / total);
266 return edge;
267 });
268}

272// WDL conversion formula based on random walk model.
273inline double WDLRescale(float& v, float& d, float wdl_rescale_ratio,
274 float wdl_rescale_diff, float sign, bool invert,
275 float max_reasonable_s) {
276 if (invert) {
277 wdl_rescale_diff = -wdl_rescale_diff;
278 wdl_rescale_ratio = 1.0f / wdl_rescale_ratio;
279 }
280 auto w = (1 + v - d) / 2;
281 auto l = (1 - v - d) / 2;
282 // Safeguard against numerical issues; skip WDL transformation if WDL is too
283 // extreme.
284 const float eps = 0.0001f;
285 if (w > eps && d > eps && l > eps && w < (1.0f - eps) && d < (1.0f - eps) &&
286 l < (1.0f - eps)) {
287 auto a = FastLog(1 / l - 1);
288 auto b = FastLog(1 / w - 1);
289 auto s = 2 / (a + b);
290 // Safeguard against unrealistically broad WDL distributions coming from
291 // the NN. Originally hardcoded, made into a parameter for piece odds.
292 if (!invert) s = std::min(max_reasonable_s, s);
293 auto mu = (a - b) / (a + b);
294 auto s_new = s * wdl_rescale_ratio;
295 if (invert) {
296 std::swap(s, s_new);
297 s = std::min(max_reasonable_s, s);
298 }
299 auto mu_new = mu + sign * s * s * wdl_rescale_diff;
300 auto w_new = FastLogistic((-1.0f + mu_new) / s_new);
301 auto l_new = FastLogistic((-1.0f - mu_new) / s_new);
302 v = w_new - l_new;
303 d = std::max(0.0f, 1.0f - w_new - l_new);
304 return mu_new;
305 }
306 return 0;
307}

310void Search::SendUciInfo(const classic::IterationStats& stats)
311 REQUIRES(nodes_mutex_) REQUIRES(counters_mutex_) {
312 const auto max_pv = params_.GetMultiPv();
313 const auto edges = GetBestChildrenNoTemperature(root_node_, max_pv, 0);
314 const auto score_type = params_.GetScoreType();
315 const auto per_pv_counters = params_.GetPerPvCounters();
316 const auto draw_score = GetDrawScore(false);
317
318 std::vector<ThinkingInfo> uci_infos;
319
320 // Info common for all multipv variants.
321 ThinkingInfo common_info;
322 common_info.depth = cum_depth_ / (total_playouts_ ? total_playouts_ : 1);
323 common_info.seldepth = max_depth_;
324 common_info.time = stats.time_since_movestart;
325 if (!per_pv_counters) {
326 common_info.nodes = total_playouts_ + initial_visits_;
327 }
328 if (stats.time_since_first_batch) {
329 const auto time_since_first_batch_ms = stats.time_since_first_batch;
330 if (time_since_first_batch_ms > 0) {
331 common_info.nps = total_playouts_ * 1000 / time_since_first_batch_ms;
332 common_info.eps = network_evaluations_ * 1000 / time_since_first_batch_ms;
333 }
334 }
335 common_info.tb_hits = tb_hits_.load(std::memory_order_acquire);
336
337 int multipv = 0;
338 const auto default_q = -root_node_->GetQ(-draw_score);
339 const auto default_wl = -root_node_->GetWL();
340 const auto default_d = root_node_->GetD();
341 for (const auto& edge : edges) {
342 ++multipv;
343 uci_infos.emplace_back(common_info);
344 auto& uci_info = uci_infos.back();
345 auto wl = edge.GetWL(default_wl);
346 auto d = edge.GetD(default_d);
347 float mu_uci = 0.0f;
348 if (score_type == "WDL_mu" || (params_.GetWDLRescaleDiff() != 0.0f &&
349 contempt_mode_ != ContemptMode::NONE)) {
350 auto sign = ((contempt_mode_ == ContemptMode::BLACK) ==
351 played_history_.IsBlackToMove())
352 ? 1.0f
353 : -1.0f;
354 mu_uci = WDLRescale(
355 wl, d, params_.GetWDLRescaleRatio(),
356 contempt_mode_ == ContemptMode::NONE
357 ? 0
358 : params_.GetWDLRescaleDiff() * params_.GetWDLEvalObjectivity(),
359 sign, true, params_.GetWDLMaxS());
360 }
361 const auto q = edge.GetQ(default_q, draw_score);
362 if (edge.IsTerminal() && wl != 0.0f) {
363 uci_info.mate = std::copysign(
364 std::round(edge.GetM(0.0f) + 1) / 2 + (edge.IsTbTerminal() ? 100 : 0),
365 wl);
366 } else if (score_type == "centipawn_with_drawscore") {
367 uci_info.score = 90 * tan(1.5637541897 * q);
368 } else if (score_type == "centipawn") {
369 uci_info.score = 90 * tan(1.5637541897 * wl);
370 } else if (score_type == "centipawn_2019") {
371 uci_info.score = 295 * wl / (1 - 0.976953126 * std::pow(wl, 14));
372 } else if (score_type == "centipawn_2018") {
373 uci_info.score = 290.680623072 * tan(1.548090806 * wl);
374 } else if (score_type == "win_percentage") {
375 uci_info.score = wl * 5000 + 5000;
376 } else if (score_type == "Q") {
377 uci_info.score = q * 10000;
378 } else if (score_type == "W-L") {
379 uci_info.score = wl * 10000;
380 } else if (score_type == "WDL_mu") {
381 // Reports the WDL mu value whenever it is reasonable, and defaults to
382 // centipawn otherwise.
383 const float centipawn_fallback_threshold = 0.996f;
384 float centipawn_score = 45 * tan(1.56728071628 * wl);
385 uci_info.score =
386 backend_attributes_.has_wdl && mu_uci != 0.0f &&
387 std::abs(wl) + d < centipawn_fallback_threshold &&
388 (std::abs(mu_uci) < 1.0f ||
389 std::abs(centipawn_score) < std::abs(100 * mu_uci))
390 ? 100 * mu_uci
391 : centipawn_score;
392 }
393
394 auto wdl_w =
395 std::max(0, static_cast<int>(std::round(500.0 * (1.0 + wl - d))));
396 auto wdl_l =
397 std::max(0, static_cast<int>(std::round(500.0 * (1.0 - wl - d))));
398 // Using 1000-w-l so that W+D+L add up to 1000.0.
399 auto wdl_d = 1000 - wdl_w - wdl_l;
400 if (wdl_d < 0) {
401 wdl_w = std::min(1000, std::max(0, wdl_w + wdl_d / 2));
402 wdl_l = 1000 - wdl_w;
403 wdl_d = 0;
404 }
405 uci_info.wdl = ThinkingInfo::WDL{wdl_w, wdl_d, wdl_l};
406 if (backend_attributes_.has_mlh) {
407 uci_info.moves_left = static_cast<int>(
408 (1.0f + edge.GetM(1.0f + root_node_->GetM())) / 2.0f);
409 }
410 if (max_pv > 1) uci_info.multipv = multipv;
411 if (per_pv_counters) uci_info.nodes = edge.GetN();
412 bool flip = played_history_.IsBlackToMove();
413 int depth = 0;
414 auto history = played_history_;
415 for (auto iter = edge; iter;
416 iter = GetBestChildNoTemperature(iter.node(), depth), flip = !flip) {
417 uci_info.pv.push_back(iter.GetMove(flip));
418 history.Append(iter.GetMove());
419 // Last edge was dangling or a draw by repetition, cannot continue.
420 if (!iter.node() || history.Last().GetRepetitions() >= 2) break;
421 depth += 1;
422 }
423 }
424
425 if (!uci_infos.empty()) last_outputted_uci_info_ = uci_infos.front();
426 if (current_best_edge_ && !edges.empty()) {
427 last_outputted_info_edge_ = current_best_edge_.edge();
428 }
429
430 uci_responder_->OutputThinkingInfo(&uci_infos);
431}

436 SharedMutex::Lock lock(nodes_mutex_);
437 Mutex::Lock counters_lock(counters_mutex_);
438 if (!bestmove_is_sent_ && current_best_edge_ &&

455 }

515std::vector<std::string> Search::GetVerboseStats(
516 const Node* node, std::optional<Move> move_to_node) const {
517 const bool is_root = (node == root_node_);
518 const bool is_odd_depth = !is_root;
519 const bool is_black_to_move = (played_history_.IsBlackToMove() == is_root);
520 const float draw_score = GetDrawScore(is_odd_depth);
521 const float fpu = GetFpu(params_, node, is_root, draw_score);
522 const float cpuct = ComputeCpuct(params_, node->GetTotalVisits(), is_root);
523 const float U_coeff =
524 cpuct * std::sqrt(std::max(node->GetChildrenVisits(), 1u));
525 std::vector<std::tuple<uint32_t, float, EdgeAndNode>> edges;
526 edges.reserve(node->GetNumEdges());
527 for (const auto& edge : node->Edges()) {
528 edges.emplace_back(edge.GetN(),
529 edge.GetQ(fpu, draw_score) + edge.GetU(U_coeff), edge);
530 }
531 std::sort(edges.begin(), edges.end());

579 if (is_edge) {
580 history.Append(n->GetMove());
581 }
582 std::optional<EvalResult> nneval = backend_->GetCachedEvaluation(
583 EvalPosition{history.GetPositions(), {}});
584 if (nneval) v = -nneval->q;
585 }
586 if (v) {
587 print(oss, "(V: ", sign * *v, ") ", 7, 4);
588 } else {
589 *oss << "(V: -.----) ";
590 }
591
592 if (n) {
593 auto [lo, up] = n->GetBounds();
594 if (sign == -1) {
595 lo = -lo;
596 up = -up;
597 std::swap(lo, up);
598 }
599 *oss << (lo == up ? "(T) "
600 : lo == GameResult::DRAW && up == GameResult::WHITE_WON ? "(W) "
601 : lo == GameResult::BLACK_WON && up == GameResult::DRAW ? "(L) "
602 : "");
603 }
604 };
605
606 std::vector<std::string> infos;
607 const auto m_evaluator =
608 backend_attributes_.has_mlh ? MEvaluator(params_, node) : MEvaluator();
609 for (const auto& edge_tuple : edges) {
610 const auto& edge = std::get<2>(edge_tuple);
611 float Q = edge.GetQ(fpu, draw_score);
612 float M = m_evaluator.GetMUtility(edge, Q);
613 std::ostringstream oss;
614 oss << std::left;
615 // TODO: should this be displaying transformed index?
616 print_head(&oss, edge.GetMove(is_black_to_move).ToString(true),
617 MoveToNNIndex(edge.GetMove(), 0), edge.GetN(),
618 edge.GetNInFlight(), edge.GetP());
619 print_stats(&oss, edge.node());
620 print(&oss, "(U: ", edge.GetU(U_coeff), ") ", 6, 5);
621 print(&oss, "(S: ", Q + edge.GetU(U_coeff) + M, ") ", 8, 5);
622 print_tail(&oss, edge.node(), true);
623 infos.emplace_back(oss.str());
624 }
625
626 // Include stats about the node in similar format to its children above.
627 std::ostringstream oss;
628 print_head(&oss, "node ", node->GetNumEdges(), node->GetN(),
629 node->GetNInFlight(), node->GetVisitedPolicy());
630 print_stats(&oss, node);
631 print_tail(&oss, node, false);
632 infos.emplace_back(oss.str());
633 return infos;

695 EnsureBestMoveKnown();
696 SendMovesStats();

698 uci_responder_->OutputBestMove(&info);

700 bestmove_is_sent_ = true;

703 }

744// Computes the best move, maybe with temperature (according to the settings).
745void Search::EnsureBestMoveKnown() REQUIRES(nodes_mutex_)
746 REQUIRES(counters_mutex_) {
747 if (bestmove_is_sent_) return;
748 if (root_node_->GetN() == 0) return;
749 if (!root_node_->HasChildren()) return;
750
751 float temperature = params_.GetTemperature();
752 const int cutoff_move = params_.GetTemperatureCutoffMove();
753 const int decay_delay_moves = params_.GetTempDecayDelayMoves();
754 const int decay_moves = params_.GetTempDecayMoves();
755 const int moves = played_history_.Last().GetGamePly() / 2;
756
757 if (cutoff_move && (moves + 1) >= cutoff_move) {
758 temperature = params_.GetTemperatureEndgame();
759 } else if (temperature && decay_moves) {
760 if (moves >= decay_delay_moves + decay_moves) {
761 temperature = 0.0;
762 } else if (moves >= decay_delay_moves) {
763 temperature *=
764 static_cast<float>(decay_delay_moves + decay_moves - moves) /
765 decay_moves;
766 }
767 // don't allow temperature to decay below endgame temperature
768 if (temperature < params_.GetTemperatureEndgame()) {
769 temperature = params_.GetTemperatureEndgame();
770 }
771 }
772
773 auto bestmove_edge = temperature
774 ? GetBestRootChildWithTemperature(temperature)
775 : GetBestChildNoTemperature(root_node_, 0);
776 final_bestmove_ = bestmove_edge.GetMove(played_history_.IsBlackToMove());
777
778 if (bestmove_edge.GetN() > 0 && bestmove_edge.node()->HasChildren()) {
779 final_pondermove_ = GetBestChildNoTemperature(bestmove_edge.node(), 1)
780 .GetMove(!played_history_.IsBlackToMove());
781 }
782}

784// Returns @count children with most visits.
785std::vector<EdgeAndNode> Search::GetBestChildrenNoTemperature(Node* parent,
786 int count,
787 int depth) const {
788 // Even if Edges is populated at this point, its a race condition to access
789 // the node, so exit quickly.
790 if (parent->GetN() == 0) return {};
791 const bool is_odd_depth = (depth % 2) == 1;
792 const float draw_score = GetDrawScore(is_odd_depth);
793 // Best child is selected using the following criteria:
794 // * Prefer shorter terminal wins / avoid shorter terminal losses.
795 // * Largest number of playouts.
796 // * If two nodes have equal number:
797 // * If that number is 0, the one with larger prior wins.
798 // * If that number is larger than 0, the one with larger eval wins.
799 std::vector<EdgeAndNode> edges;
800 for (auto& edge : parent->Edges()) {
801 if (parent == root_node_ && !root_move_filter_.empty() &&
802 std::find(root_move_filter_.begin(), root_move_filter_.end(),
803 edge.GetMove()) == root_move_filter_.end()) {
804 continue;
805 }
806 edges.push_back(edge);
807 }
808 const auto middle = (static_cast<int>(edges.size()) > count)
809 ? edges.begin() + count
810 : edges.end();
811 std::partial_sort(
812 edges.begin(), middle, edges.end(),
813 [draw_score](const auto& a, const auto& b) {
814 // The function returns "true" when a is preferred to b.
815
816 // Lists edge types from less desirable to more desirable.
817 enum EdgeRank {
818 kTerminalLoss,
819 kTablebaseLoss,
820 kNonTerminal, // Non terminal or terminal draw.
821 kTablebaseWin,
822 kTerminalWin,
823 };
824
825 auto GetEdgeRank = [](const EdgeAndNode& edge) {
826 // This default isn't used as wl only checked for case edge is
827 // terminal.
828 const auto wl = edge.GetWL(0.0f);
829 // Not safe to access IsTerminal if GetN is 0.
830 if (edge.GetN() == 0 || !edge.IsTerminal() || !wl) {
831 return kNonTerminal;
832 }
833 if (edge.IsTbTerminal()) {
834 return wl < 0.0 ? kTablebaseLoss : kTablebaseWin;
835 }
836 return wl < 0.0 ? kTerminalLoss : kTerminalWin;
837 };
838
839 // If moves have different outcomes, prefer better outcome.
840 const auto a_rank = GetEdgeRank(a);
841 const auto b_rank = GetEdgeRank(b);
842 if (a_rank != b_rank) return a_rank > b_rank;
843
844 // If both are terminal draws, try to make it shorter.
845 // Not safe to access IsTerminal if GetN is 0.
846 if (a_rank == kNonTerminal && a.GetN() != 0 && b.GetN() != 0 &&
847 a.IsTerminal() && b.IsTerminal()) {
848 if (a.IsTbTerminal() != b.IsTbTerminal()) {
849 // Prefer non-tablebase draws.
850 return a.IsTbTerminal() < b.IsTbTerminal();
851 }
852 // Prefer shorter draws.
853 return a.GetM(0.0f) < b.GetM(0.0f);
854 }
855
856 // Neither is terminal, use standard rule.
857 if (a_rank == kNonTerminal) {
858 // Prefer largest playouts then eval then prior.
859 if (a.GetN() != b.GetN()) return a.GetN() > b.GetN();
860 // Default doesn't matter here so long as they are the same as either
861 // both are N==0 (thus we're comparing equal defaults) or N!=0 and
862 // default isn't used.
863 if (a.GetQ(0.0f, draw_score) != b.GetQ(0.0f, draw_score)) {
864 return a.GetQ(0.0f, draw_score) > b.GetQ(0.0f, draw_score);
865 }
866 return a.GetP() > b.GetP();
867 }
868
869 // Both variants are winning, prefer shortest win.
870 if (a_rank > kNonTerminal) {
871 return a.GetM(0.0f) < b.GetM(0.0f);
872 }
873
874 // Both variants are losing, prefer longest losses.
875 return a.GetM(0.0f) > b.GetM(0.0f);
876 });
877
878 if (count < static_cast<int>(edges.size())) {
879 edges.resize(count);
880 }
881 return edges;
882}

890// Returns a child of a root chosen according to weighted-by-temperature visit
891// count.
892EdgeAndNode Search::GetBestRootChildWithTemperature(float temperature) const {
893 // Root is at even depth.
894 const float draw_score = GetDrawScore(/* is_odd_depth= */ false);
895
896 std::vector<float> cumulative_sums;
897 float sum = 0.0;
898 float max_n = 0.0;
899 const float offset = params_.GetTemperatureVisitOffset();
900 float max_eval = -1.0f;
901 const float fpu =
902 GetFpu(params_, root_node_, /* is_root= */ true, draw_score);
903
904 for (auto& edge : root_node_->Edges()) {
905 if (!root_move_filter_.empty() &&
906 std::find(root_move_filter_.begin(), root_move_filter_.end(),
907 edge.GetMove()) == root_move_filter_.end()) {
908 continue;
909 }
910 if (edge.GetN() + offset > max_n) {
911 max_n = edge.GetN() + offset;
912 max_eval = edge.GetQ(fpu, draw_score);
913 }
914 }
915
916 // TODO(crem) Simplify this code when samplers.h is merged.
917 const float min_eval =
918 max_eval - params_.GetTemperatureWinpctCutoff() / 50.0f;
919 for (auto& edge : root_node_->Edges()) {
920 if (!root_move_filter_.empty() &&
921 std::find(root_move_filter_.begin(), root_move_filter_.end(),
922 edge.GetMove()) == root_move_filter_.end()) {
923 continue;
924 }
925 if (edge.GetQ(fpu, draw_score) < min_eval) continue;
926 sum += std::pow(
927 std::max(0.0f,
928 (max_n <= 0.0f
929 ? edge.GetP()
930 : ((static_cast<float>(edge.GetN()) + offset) / max_n))),
931 1 / temperature);
932 cumulative_sums.push_back(sum);
933 }
934 assert(sum);
935
936 const float toss = Random::Get().GetFloat(cumulative_sums.back());
937 int idx =
938 std::lower_bound(cumulative_sums.begin(), cumulative_sums.end(), toss) -
939 cumulative_sums.begin();
940
941 for (auto& edge : root_node_->Edges()) {
942 if (!root_move_filter_.empty() &&
943 std::find(root_move_filter_.begin(), root_move_filter_.end(),
944 edge.GetMove()) == root_move_filter_.end()) {
945 continue;
946 }
947 if (edge.GetQ(fpu, draw_score) < min_eval) continue;
948 if (idx-- == 0) return edge;
949 }
950 assert(false);
951 return {};
952}

988void Search::PopulateCommonIterationStats(classic::IterationStats* stats) {
989 stats->time_since_movestart = GetTimeSinceStart();
990
991 SharedMutex::SharedLock nodes_lock(nodes_mutex_);
992 stats->time_since_first_batch = GetTimeSinceFirstBatch();
993 stats->total_nodes = total_playouts_ + initial_visits_;
994 stats->nodes_since_movestart = total_playouts_;
995 stats->batches_since_movestart = total_batches_;
996 stats->average_depth = cum_depth_ / (total_playouts_ ? total_playouts_ : 1);
997 stats->edge_n.clear();
998 stats->win_found = false;
999 stats->may_resign = true;
1000 stats->num_losing_edges = 0;
1001 stats->time_usage_hint_ = classic::IterationStats::TimeUsageHint::kNormal;
1002 stats->mate_depth = std::numeric_limits<int>::max();
1003
1004 // If root node hasn't finished first visit, none of this code is safe.
1005 if (root_node_->GetN() > 0) {
1006 const auto draw_score = GetDrawScore(true);
1007 const float fpu =
1008 GetFpu(params_, root_node_, /* is_root_node */ true, draw_score);
1009 float max_q_plus_m = -1000;
1010 uint64_t max_n = 0;
1011 bool max_n_has_max_q_plus_m = true;
1012 const auto m_evaluator = backend_attributes_.has_mlh
1013 ? MEvaluator(params_, root_node_)
1014 : MEvaluator();
1015 for (const auto& edge : root_node_->Edges()) {
1016 const auto n = edge.GetN();
1017 const auto q = edge.GetQ(fpu, draw_score);
1018 const auto m = m_evaluator.GetMUtility(edge, q);
1019 const auto q_plus_m = q + m;
1020 stats->edge_n.push_back(n);
1021 if (n > 0 && edge.IsTerminal() && edge.GetWL(0.0f) > 0.0f) {
1022 stats->win_found = true;
1023 }
1024 if (n > 0 && edge.IsTerminal() && edge.GetWL(0.0f) < 0.0f) {
1025 stats->num_losing_edges += 1;
1026 }
1027 if (n > 0 && edge.IsTerminal() && edge.GetWL(0.0f) == 1.0f &&
1028 !edge.IsTbTerminal()) {
1029 stats->mate_depth =
1030 std::min(stats->mate_depth,
1031 static_cast<int>(std::round(edge.GetM(0.0f))) / 2 + 1);
1032 }
1033
1034 // If game is resignable, no need for moving quicker. This allows
1035 // proving mate when losing anyway for better score output.
1036 // Hardcoded resign threshold, because there is no available parameter.
1037 if (n > 0 && q > -0.98f) {
1038 stats->may_resign = false;
1039 }
1040 if (max_n < n) {
1041 max_n = n;
1042 max_n_has_max_q_plus_m = false;
1043 }
1044 if (max_q_plus_m <= q_plus_m) {
1045 max_n_has_max_q_plus_m = (max_n == n);
1046 max_q_plus_m = q_plus_m;
1047 }
1048 }
1049 if (!max_n_has_max_q_plus_m) {
1050 stats->time_usage_hint_ =
1051 classic::IterationStats::TimeUsageHint::kNeedMoreTime;
1052 }
1053 }

1056void Search::WatchdogThread() {
1057 LOGFILE << "Start a watchdog thread.";
1058 classic::StoppersHints hints;
1059 classic::IterationStats stats;
1060 while (true) {
1061 PopulateCommonIterationStats(&stats);
1062 MaybeTriggerStop(stats, &hints);
1063 MaybeOutputInfo(stats);
1064
1065 constexpr auto kMaxWaitTimeMs = 100;
1066 constexpr auto kMinWaitTimeMs = 1;
1067
1068 Mutex::Lock lock(counters_mutex_);
1069 // Only exit when bestmove is responded. It may happen that search threads
1070 // already all exited, and we need at least one thread that can do that.
1071 if (bestmove_is_sent_) break;
1072
1073 auto remaining_time = hints.GetEstimatedRemainingTimeMs();
1074 if (remaining_time > kMaxWaitTimeMs) remaining_time = kMaxWaitTimeMs;
1075 if (remaining_time < kMinWaitTimeMs) remaining_time = kMinWaitTimeMs;
1076 // There is no real need to have max wait time, and sometimes it's fine
1077 // to wait without timeout at all (e.g. in `go nodes` mode), but we
1078 // still limit wait time for exotic cases like when pc goes to sleep
1079 // mode during thinking.
1080 // Minimum wait time is there to prevent busy wait and other threads
1081 // starvation.
1082 watchdog_cv_.wait_for(
1083 lock.get_raw(), std::chrono::milliseconds(remaining_time),
1084 [this]() { return stop_.load(std::memory_order_acquire); });
1085 }
1086 LOGFILE << "End a watchdog thread.";
1087}

1165std::tuple<SearchWorker::PickTask*, int, int>
1166SearchWorker::PickTaskToProcess() {
1167 auto [packed_value, nta, tc] = ReadTaskCount(task_count_);
1168
1169 // Check if tasks are queued and try increment taken count.
1170 while (nta < tc && !task_count_.compare_exchange_weak(
1171 packed_value, packed_value + kTasksTakenOne,
1172 std::memory_order_acq_rel)) {
1173 // Queue had tasks but another worker increment taken. We check
1174 // if new work was added to the queue. Then we try to increment
1175 // taken again.
1176 std::tie(packed_value, nta, tc) = ReadTaskCount(packed_value);
1177 }
1178 // We incremented taken if nta and tc are different
1179 if (nta < tc) {
1180 return {picking_tasks_.data() + nta, nta, tc};
1181 }
1182 return {nullptr, nta, tc};
1183}

1185void SearchWorker::ProcessTask(PickTask* task, int id,
1186 std::vector<NodeToProcess>* receiver,
1187 TaskWorkspace* workspace) {
1188 switch (task->task_type) {
1189 case PickTask::kGathering: {
1190 PickNodesToExtendTask(task->start_path, task->collision_limit,
1191 task->history, receiver, workspace);
1192 break;
1193 }
1194 case PickTask::kProcessing: {
1195 ProcessPickedTask(task->start_idx, task->end_idx);
1196 break;

1198 }
1199 picking_tasks_.data()[id].complete = true;
1200 completed_tasks_.fetch_add(1, std::memory_order_acq_rel);
1201}

1203void SearchWorker::RunTasks(int tid) {
1204 while (true) {
1205 PickTask* task = nullptr;
1206 int id = 0;
1207 int tc = 0;
1208 {
1209 int spins = 0;
1210 while (true) {
1211 std::tie(task, id, tc) = PickTaskToProcess();
1212 if (task) {
1213 break;
1214 } else if (tc != -1) {
1215 spins++;
1216 if (spins >= 512) {
1217 std::this_thread::yield();
1218 spins = 0;
1219 } else {
1220 SpinloopPause();
1221 }
1222 continue;
1223 }
1224 spins = 0;
1225 // Looks like sleep time.
1226 Mutex::Lock lock(picking_tasks_mutex_);
1227 // Refresh them now we have the lock.
1228 int tc, nta;
1229 std::tie(std::ignore, std::ignore, tc) = ReadTaskCount(task_count_);
1230 if (tc != -1) continue;
1231 if (exiting_) return;
1232 task_added_.wait(lock.get_raw());
1233 std::tie(std::ignore, nta, tc) = ReadTaskCount(task_count_);
1234 if (nta >= tc && exiting_) return;
1235 }
1236 }
1237 if (task != nullptr) {
1238 ProcessTask(task, id, &(task->results), &(task_workspaces_[tid]));
1239 }
1240 }

1243void SearchWorker::ExecuteOneIteration() {
1244 // 1. Initialize internal structures.
1245 InitializeIteration();
1246
1247 if (params_.GetMaxConcurrentSearchers() != 0) {
1248 std::unique_ptr<SpinHelper> spin_helper;
1249 if (params_.GetSearchSpinBackoff()) {
1250 spin_helper = std::make_unique<ExponentialBackoffSpinHelper>();
1251 } else {
1252 // This is a hard spin lock to reduce latency but at the expense of busy
1253 // wait cpu usage. If search worker count is large, this is probably a
1254 // bad idea.
1255 spin_helper = std::make_unique<SpinHelper>();
1256 }
1257
1258 while (true) {
1259 // If search is stopped, we've not gathered or done anything and we don't
1260 // want to, so we can safely skip all below. But make sure we have done
1261 // at least one iteration.
1262 if (search_->stop_.load(std::memory_order_acquire) &&
1263 search_->GetTotalPlayouts() + search_->initial_visits_ > 0) {
1264 return;
1265 }
1266
1267 int available =
1268 search_->pending_searchers_.load(std::memory_order_acquire);
1269 if (available == 0) {
1270 spin_helper->Wait();
1271 continue;
1272 }
1273
1274 if (search_->pending_searchers_.compare_exchange_weak(
1275 available, available - 1, std::memory_order_acq_rel)) {
1276 break;
1277 } else {
1278 spin_helper->Backoff();
1279 }
1280 }
1281 }
1282
1283 // 2. Gather minibatch.
1284 GatherMinibatch();
1285 assert(IsTasksCompleted(task_count_, completed_tasks_));
1286 task_count_.fetch_or(kTaskCountSuspend, std::memory_order_release);
1287 search_->backend_waiting_counter_.fetch_add(1, std::memory_order_relaxed);
1288
1289 if (params_.GetMaxConcurrentSearchers() != 0) {
1290 search_->pending_searchers_.fetch_add(1, std::memory_order_acq_rel);
1291 }
1292
1293 // 4. Run NN computation.
1294 RunNNComputation();
1295 search_->backend_waiting_counter_.fetch_add(-1, std::memory_order_relaxed);
1296
1297 // 5. Retrieve NN computations (and terminal values) into nodes.
1298 FetchMinibatchResults();
1299
1300 // 6. Propagate the new nodes' information to all their parents in the tree.
1301 DoBackupUpdate();
1302
1303 // 7. Update the Search's status and progress information.
1304 UpdateCounters();
1305
1306 // If required, waste time to limit nps.
1307 if (params_.GetNpsLimit() > 0 && iteration_stats_.time_since_first_batch) {
1308 while (search_->IsSearchActive()) {
1309 // GetTimeSinceFirstBatch is set only once. We check iteration_stats_ to
1310 // know if it was set and later read inside nodes_mutex_.
1311 int64_t time_since_first_batch_ms = search_->GetTimeSinceFirstBatch();
1312 auto nps = search_->GetTotalPlayouts() * 1e3f / time_since_first_batch_ms;
1313 if (nps > params_.GetNpsLimit()) {
1314 std::this_thread::sleep_for(std::chrono::milliseconds(1));
1315 } else {
1316 break;
1317 }
1318 }
1319 }
1320}

1359void SearchWorker::GatherMinibatch() {
1360 LCTRACE_FUNCTION_SCOPE;
1361 // Total number of nodes to process.
1362 int minibatch_size = 0;
1363 int cur_n = 0;
1364
1365 // Collision use atomic operations. We can cancel them outside the lock.
1366 struct CollisionsManager {
1367 SearchWorker& worker;
1368 CollisionsManager(SearchWorker& worker) : worker(worker) {}
1369 ~CollisionsManager() { worker.CancelCollisions(); }
1370 } cancel_collisions_object(*this);
1371 // We take the nodes_mutex_ only once to avoid bouncing between this thread
1372 // and a thread returning from RunNNComputation.
1373 SharedMutex::Lock lock(search_->nodes_mutex_);
1374 cur_n = search_->root_node_->GetN();
1375 // TODO: GetEstimatedRemainingPlayouts has already had smart pruning factor
1376 // applied, which doesn't clearly make sense to include here...
1377 int64_t remaining_n =
1378 latest_time_manager_hints_.GetEstimatedRemainingPlayouts();
1379 int collisions_left = CalculateCollisionsLeft(
1380 std::min(static_cast<int64_t>(cur_n), remaining_n), params_);
1381
1382 // Number of nodes processed out of order.
1383 number_out_of_order_ = 0;
1384
1385 int thread_count = search_->thread_count_.load(std::memory_order_acquire);
1386
1387 absl::Cleanup record_batch_start_time = [&] {
1388 if (minibatch_size) search_->RecordNPSStartTime();
1389 };
1390
1391 // Gather nodes to process in the current batch.
1392 // If we had too many nodes out of order, also interrupt the iteration so
1393 // that search can exit.
1394 while (minibatch_size < target_minibatch_size_ &&
1395 number_out_of_order_ < max_out_of_order_) {
1396 // If there's something to process without touching slow neural net, do it.
1397 if (minibatch_size > 0 && computation_->UsedBatchSize() == 0) return;
1398
1399 // If there is backend work to be done, and the backend is idle - exit
1400 // immediately.
1401 // Only do this fancy work if there are multiple threads as otherwise we
1402 // early exit from every batch since there is never another search thread to
1403 // be keeping the backend busy. Which would mean that threads=1 has a
1404 // massive nps drop.
1405 if (thread_count > 1 && minibatch_size > 0 &&
1406 static_cast<int>(computation_->UsedBatchSize()) >
1407 params_.GetIdlingMinimumWork() &&
1408 thread_count - search_->backend_waiting_counter_.load(
1409 std::memory_order_relaxed) >
1410 params_.GetThreadIdlingThreshold()) {
1411 return;
1412 }
1413
1414 int new_start = static_cast<int>(minibatch_.size());
1415
1416 PickNodesToExtend(
1417 std::min({collisions_left, target_minibatch_size_ - minibatch_size,
1418 max_out_of_order_ - number_out_of_order_}));
1419
1420 // Count the non-collisions.
1421 int non_collisions = 0;
1422 for (int i = new_start; i < static_cast<int>(minibatch_.size()); i++) {
1423 auto& picked_node = minibatch_[i];
1424 if (picked_node.IsCollision()) {
1425 continue;
1426 }
1427 ++non_collisions;
1428 ++minibatch_size;
1429 }
1430
1431 {
1432 bool needs_wait = false;
1433 int ppt_start = new_start;
1434 if (task_workers_ > 0 &&
1435 non_collisions >= params_.GetMinimumWorkSizeForProcessing()) {
1436 const int num_tasks = std::clamp(
1437 non_collisions / params_.GetMinimumWorkPerTaskForProcessing(), 2,
1438 task_workers_ + 1);
1439 // Round down, left overs can go to main thread so it waits less.
1440 int per_worker = non_collisions / num_tasks;
1441 needs_wait = true;
1442 ResetTasks();
1443 int found = 0;
1444 for (int i = new_start; i < static_cast<int>(minibatch_.size()); i++) {
1445 auto& picked_node = minibatch_[i];
1446 if (picked_node.IsCollision()) {
1447 continue;
1448 }
1449 ++found;
1450 if (found == per_worker) {
1451 picking_tasks_.emplace_back(ppt_start, i + 1);
1452 task_count_.fetch_add(1, std::memory_order_acq_rel);
1453 ppt_start = i + 1;
1454 found = 0;
1455 if (picking_tasks_.size() == static_cast<size_t>(num_tasks - 1)) {
1456 break;
1457 }
1458 }
1459 }
1460 }
1461 ProcessPickedTask(ppt_start, static_cast<int>(minibatch_.size()));
1462 if (needs_wait) {
1463 WaitForTasks();
1464 }
1465 }
1466 bool some_ooo = false;
1467 for (int i = static_cast<int>(minibatch_.size()) - 1; i >= new_start; i--) {
1468 if (minibatch_[i].ooo_completed) {
1469 some_ooo = true;
1470 break;
1471 }
1472 }
1473 if (some_ooo) {
1474 for (int i = static_cast<int>(minibatch_.size()) - 1; i >= new_start;
1475 i--) {
1476 // If there was any OOO, revert 'all' new collisions - it isn't possible
1477 // to identify exactly which ones are afterwards and only prune those.
1478 // This may remove too many items, but hopefully most of the time they
1479 // will just be added back in the same in the next gather.
1480 if (minibatch_[i].IsCollision()) {
1481 for (auto it = ++(minibatch_[i].path.crbegin());
1482 it != minibatch_[i].path.crend(); ++it) {
1483 std::get<0>(*it)->CancelScoreUpdate(minibatch_[i].multivisit);
1484 }
1485 minibatch_.erase(minibatch_.begin() + i);
1486 } else if (minibatch_[i].ooo_completed) {
1487 FetchSingleNodeResult(&minibatch_[i]);
1488 DoBackupUpdateSingleNode(minibatch_[i]);
1489 minibatch_.erase(minibatch_.begin() + i);
1490 --minibatch_size;
1491 ++number_out_of_order_;
1492 }
1493 }
1494 }
1495
1496 // Check for stop at the end so we have at least one node.
1497 for (size_t i = new_start; i < minibatch_.size(); i++) {
1498 auto& picked_node = minibatch_[i];
1499
1500 if (picked_node.IsCollision()) {
1501 // Check to see if we can upsize the collision to exit sooner.
1502 if (picked_node.maxvisit > 0 &&
1503 collisions_left > picked_node.multivisit) {
1504 int extra = std::min(picked_node.maxvisit, collisions_left) -
1505 picked_node.multivisit;
1506 picked_node.multivisit += extra;
1507 for (auto it = ++(picked_node.path.crbegin());
1508 it != picked_node.path.crend(); ++it) {
1509 std::get<0>(*it)->IncrementNInFlight(extra);
1510 }
1511 }
1512 if ((collisions_left -= picked_node.multivisit) <= 0) return;
1513 if (search_->stop_.load(std::memory_order_acquire)) return;
1514 }
1515 }
1516 }
1517}

1552int SearchWorker::WaitForTasks() REQUIRES(search_->nodes_mutex_) {
1553 // Process any outstanding tasks before checking if compelted. This avoids a
1554 // long polling loop when PickNodesToExtend scheduled many tasks.
1555 while (true) {
1556 PickTask* task = nullptr;
1557 int id = 0;
1558 std::tie(task, id, std::ignore) = PickTaskToProcess();
1559 if (task == nullptr) {
1560 break;
1561 }
1562 ProcessTask(task, id, &minibatch_, &main_workspace_);
1563 }

1565 while (true) {

1567 int todo, nta;

1569 std::ignore = nta;
1570 assert(nta <= todo);
1571 if (todo == completed) return completed;
1572 SpinloopPause();
1573 }

1601// Check if the situation described by @depth under root and @position is a
1602// safe two-fold or a draw by repetition and return the number of safe
1603// repetitions and moves_left.
1604// Depth starts with 0 at root, so number of plies in PV equals depth.
1605std::pair<int, int> SearchWorker::GetRepetitions(int depth,
1606 const Position& position) {
1607 const auto repetitions = position.GetRepetitions();
1608
1609 if (repetitions == 0) return {0, 0};
1610
1611 if (repetitions >= 2) return {repetitions, 0};
1612
1613 const auto plies = position.GetPliesSincePrevRepetition();
1614 if (params_.GetTwoFoldDraws() && /*repetitions == 1 &&*/ depth >= 4 &&
1615 depth >= plies) {
1616 return {1, plies};
1617 }
1618
1619 return {0, 0};
1620}

1622// Check if PickNodesToExtendTask should stop picking at this @node.
1623bool SearchWorker::ShouldStopPickingHere(Node* node, bool is_root_node,
1624 int repetitions) {
1625 constexpr double wl_diff_limit = 0.01f;
1626 constexpr float d_diff_limit = 0.01f;
1627 constexpr float m_diff_limit = 2.0f;
1628
1629 if (node->GetN() == 0 || node->IsTerminal()) return true;
1630
1631 // Only stop at root when there is no other option.
1632 assert(!is_root_node || node == search_->root_node_);
1633 if (is_root_node) return false;
1634
1635 // Stop at draws by repetition.
1636 if (repetitions >= 2) return true;
1637
1638 // Check if Node and LowNode differ significantly.
1639 auto low_node = node->GetLowNode().get();
1640 assert(low_node);
1641
1642 // Only known transpositions can differ.
1643 if (!low_node->IsTransposition()) return false;
1644
1645 // LowNode is terminal when Node is not.
1646 if (low_node->IsTerminal()) return true;
1647
1648 // Bounds differ (swap).
1649 auto [low_node_lower, low_node_upper] = low_node->GetBounds();
1650 auto [node_lower, node_upper] = node->GetBounds();
1651 if (low_node_lower != -node_upper || low_node_upper != -node_lower)
1652 return true;
1653
1654 // WL differs significantly (flip).
1655 auto wl_diff = std::abs(low_node->GetWL() + node->GetWL());
1656 if (wl_diff >= wl_diff_limit) return true;
1657
1658 // D differs significantly.
1659 auto d_diff = std::abs(low_node->GetD() - node->GetD());
1660 if (d_diff >= d_diff_limit) return true;
1661
1662 // M differs significantly (increment).
1663 auto m_diff = std::abs(low_node->GetM() + 1 - node->GetM());
1664 if (m_diff >= m_diff_limit) return true;
1665
1666 return false;
1667}

1669void SearchWorker::PickNodesToExtendTask(
1670 const BackupPath& path, int collision_limit, PositionHistory& history,
1671 std::vector<NodeToProcess>* receiver,
1672 TaskWorkspace* workspace) NO_THREAD_SAFETY_ANALYSIS {
1673 LCTRACE_FUNCTION_SCOPE;
1674 assert(path.size() == (size_t)history.GetLength() -
1675 search_->played_history_.GetLength() + 1);
1676
1677 // TODO: Bring back pre-cached nodes created outside locks in a way that works
1678 // with tasks.
1679 // TODO: pre-reserve visits_to_perform for expected depth and likely maximum
1680 // width. Maybe even do so outside of lock scope.
1681 auto& vtp_buffer = workspace->vtp_buffer;
1682 auto& visits_to_perform = workspace->visits_to_perform;
1683 visits_to_perform.clear();
1684 auto& vtp_last_filled = workspace->vtp_last_filled;
1685 vtp_last_filled.clear();
1686 auto& current_path = workspace->current_path;
1687 current_path.clear();
1688 auto& full_path = workspace->full_path;
1689 full_path = path;
1690 assert(full_path.size() > 0);
1691 auto [node, repetitions, moves_left] = full_path.back();
1692 // Sometimes receiver is reused, othertimes not, so only jump start if small.
1693 if (receiver->capacity() < 30) {
1694 receiver->reserve(receiver->size() + 30);
1695 }
1696
1697 // This 1 is 'filled pre-emptively'.
1698 std::array<float, 256> current_util;
1699
1700 // These 3 are 'filled on demand'.
1701 std::array<float, 256> current_score;
1702 std::array<int, 256> current_nstarted;
1703 auto& cur_iters = workspace->cur_iters;
1704
1705 Node::Iterator best_edge;
1706 Node::Iterator second_best_edge;
1707 // Fetch the current best root node visits for possible smart pruning.
1708 const int64_t best_node_n = search_->current_best_edge_.GetN();
1709
1710 int passed_off = 0;
1711 int completed_visits = 0;
1712
1713 bool is_root_node = node == search_->root_node_;
1714 const float even_draw_score = search_->GetDrawScore(false);
1715 const float odd_draw_score = search_->GetDrawScore(true);
1716 const auto& root_move_filter = search_->root_move_filter_;
1717 auto m_evaluator = moves_left_support_ ? MEvaluator(params_) : MEvaluator();
1718
1719 int max_limit = std::numeric_limits<int>::max();
1720
1721 current_path.push_back(-1);
1722 while (current_path.size() > 0) {
1723 assert(full_path.size() >= path.size());
1724 // First prepare visits_to_perform.
1725 if (current_path.back() == -1) {
1726 // Need to do n visits, where n is either collision_limit, or comes from
1727 // visits_to_perform for the current path.
1728 int cur_limit = collision_limit;
1729 if (current_path.size() > 1) {
1730 cur_limit =
1731 (*visits_to_perform.back())[current_path[current_path.size() - 2]];
1732 }
1733 // First check if node is terminal or not-expanded. If either than create
1734 // a collision of appropriate size and pop current_path.
1735 if (ShouldStopPickingHere(node, is_root_node, repetitions)) {
1736 if (is_root_node) {
1737 // Root node is special - since its not reached from anywhere else, so
1738 // it needs its own logic. Still need to create the collision to
1739 // ensure the outer gather loop gives up.
1740 if (node->TryStartScoreUpdate()) {
1741 cur_limit -= 1;
1742 minibatch_.push_back(
1743 NodeToProcess::Visit(full_path, search_->played_history_));
1744 completed_visits++;
1745 }
1746 }
1747 // Visits are created elsewhere, just need the collisions here.
1748 if (cur_limit > 0) {
1749 int max_count = 0;
1750 if (cur_limit == collision_limit && path.size() == 1 &&
1751 max_limit > cur_limit) {
1752 max_count = max_limit;
1753 }
1754 receiver->push_back(
1755 NodeToProcess::Collision(full_path, cur_limit, max_count));
1756 completed_visits += cur_limit;
1757 }
1758 history.Pop();
1759 full_path.pop_back();
1760 if (full_path.size() > 0) {
1761 std::tie(node, repetitions, moves_left) = full_path.back();
1762 } else {
1763 node = nullptr;
1764 repetitions = 0;
1765 }
1766 current_path.pop_back();
1767 continue;
1768 }
1769 if (is_root_node) {
1770 // Root node is again special - needs its n in flight updated separately
1771 // as its not handled on the path to it, since there isn't one.
1772 node->IncrementNInFlight(cur_limit);
1773 }
1774
1775 // Create visits_to_perform new back entry for this level.
1776 if (vtp_buffer.size() > 0) {
1777 visits_to_perform.push_back(std::move(vtp_buffer.back()));
1778 vtp_buffer.pop_back();
1779 } else {
1780 visits_to_perform.push_back(std::make_unique<std::array<int, 256>>());
1781 }
1782 vtp_last_filled.push_back(-1);
1783
1784 // Cache all constant UCT parameters.
1785
1786 int max_needed = node->GetNumEdges();
1787 for (int i = 0; i < max_needed; i++) {
1788 current_util[i] = std::numeric_limits<float>::lowest();
1789 }
1790 // Root depth is 1 here, while for GetDrawScore() it's 0-based, that's why
1791 // the weirdness.
1792 const float draw_score =
1793 (full_path.size() % 2 == 0) ? odd_draw_score : even_draw_score;
1794 m_evaluator.SetParent(node);
1795 float visited_pol = 0.0f;
1796 for (Node* child : node->VisitedNodes()) {
1797 int index = child->Index();
1798 visited_pol += child->GetP();
1799 float q = child->GetQ(draw_score);
1800 current_util[index] = q + m_evaluator.GetMUtility(child, q);
1801 }
1802 const float fpu =
1803 GetFpu(params_, node, is_root_node, draw_score, visited_pol);
1804 for (int i = 0; i < max_needed; i++) {
1805 if (current_util[i] == std::numeric_limits<float>::lowest()) {
1806 current_util[i] = fpu + m_evaluator.GetDefaultMUtility();
1807 }
1808 }
1809
1810 const float cpuct =
1811 ComputeCpuct(params_, node->GetTotalVisits(), is_root_node);
1812 const float puct_mult =
1813 cpuct * std::sqrt(std::max(node->GetChildrenVisits(), 1u));
1814 int cache_filled_idx = -1;
1815 while (cur_limit > 0) {
1816 // Perform UCT for current node.
1817 float best = std::numeric_limits<float>::lowest();
1818 int best_idx = -1;
1819 float best_without_u = std::numeric_limits<float>::lowest();
1820 float second_best = std::numeric_limits<float>::lowest();
1821 bool can_exit = false;
1822 best_edge.Reset();
1823 for (int idx = 0; idx < max_needed; ++idx) {
1824 if (idx > cache_filled_idx) {
1825 if (idx == 0) {
1826 cur_iters[idx] = node->Edges();
1827 } else {
1828 cur_iters[idx] = cur_iters[idx - 1];
1829 ++cur_iters[idx];
1830 }
1831 current_nstarted[idx] = cur_iters[idx].GetNStarted();
1832 }
1833 int nstarted = current_nstarted[idx];
1834 const float util = current_util[idx];
1835 if (idx > cache_filled_idx) {
1836 current_score[idx] =
1837 cur_iters[idx].GetP() * puct_mult / (1 + nstarted) + util;
1838 cache_filled_idx++;
1839 }
1840 if (is_root_node) {
1841 // If there's no chance to catch up to the current best node with
1842 // remaining playouts, don't consider it.
1843 // best_move_node_ could have changed since best_node_n was
1844 // retrieved. To ensure we have at least one node to expand, always
1845 // include current best node.
1846 if (cur_iters[idx] != search_->current_best_edge_ &&
1847 latest_time_manager_hints_.GetEstimatedRemainingPlayouts() <
1848 best_node_n - cur_iters[idx].GetN()) {
1849 continue;
1850 }
1851 // If root move filter exists, make sure move is in the list.
1852 if (!root_move_filter.empty() &&
1853 std::find(root_move_filter.begin(), root_move_filter.end(),
1854 cur_iters[idx].GetMove()) == root_move_filter.end()) {
1855 continue;
1856 }
1857 }
1858
1859 float score = current_score[idx];
1860 if (score > best) {
1861 second_best = best;
1862 second_best_edge = best_edge;
1863 best = score;
1864 best_idx = idx;
1865 best_without_u = util;
1866 best_edge = cur_iters[idx];
1867 } else if (score > second_best) {
1868 second_best = score;
1869 second_best_edge = cur_iters[idx];
1870 }
1871 if (can_exit) break;
1872 if (nstarted == 0) {
1873 // One more loop will get 2 unvisited nodes, which is sufficient to
1874 // ensure second best is correct. This relies upon the fact that
1875 // edges are sorted in policy decreasing order.
1876 can_exit = true;
1877 }
1878 }
1879 int new_visits = 0;
1880 if (second_best_edge) {
1881 int estimated_visits_to_change_best = std::numeric_limits<int>::max();
1882 if (best_without_u < second_best) {
1883 const auto n1 = current_nstarted[best_idx] + 1;
1884 estimated_visits_to_change_best = static_cast<int>(
1885 std::max(1.0f, std::min(cur_iters[best_idx].GetP() * puct_mult /
1886 (second_best - best_without_u) -
1887 n1 + 1,
1888 1e9f)));
1889 }
1890 second_best_edge.Reset();
1891 max_limit = std::min(max_limit, estimated_visits_to_change_best);
1892 new_visits = std::min(cur_limit, estimated_visits_to_change_best);
1893 } else {
1894 // No second best - only one edge, so everything goes in here.
1895 new_visits = cur_limit;
1896 }
1897 if (best_idx >= vtp_last_filled.back()) {
1898 auto* vtp_array = visits_to_perform.back().get()->data();
1899 std::fill(vtp_array + (vtp_last_filled.back() + 1),
1900 vtp_array + best_idx + 1, 0);
1901 }
1902 (*visits_to_perform.back())[best_idx] += new_visits;
1903 cur_limit -= new_visits;
1904
1905 Node* child_node = best_edge.GetOrSpawnNode(/* parent */ node);
1906 history.Append(best_edge.GetMove());
1907 auto [child_repetitions, child_moves_left] =
1908 GetRepetitions(full_path.size(), history.Last());
1909 full_path.push_back({child_node, child_repetitions, child_moves_left});
1910 if (child_node->TryStartScoreUpdate()) {
1911 current_nstarted[best_idx]++;
1912 new_visits -= 1;
1913 if (ShouldStopPickingHere(child_node, false, child_repetitions)) {
1914 // Reduce 1 for the visits_to_perform to ensure the collision
1915 // created doesn't include this visit.
1916 (*visits_to_perform.back())[best_idx] -= 1;
1917 receiver->push_back(NodeToProcess::Visit(full_path, history));
1918 completed_visits++;
1919 } else {
1920 child_node->IncrementNInFlight(new_visits);
1921 current_nstarted[best_idx] += new_visits;
1922 }
1923 current_score[best_idx] = cur_iters[best_idx].GetP() * puct_mult /
1924 (1 + current_nstarted[best_idx]) +
1925 current_util[best_idx];
1926 }
1927 if (best_idx > vtp_last_filled.back() &&
1928 (*visits_to_perform.back())[best_idx] > 0) {
1929 vtp_last_filled.back() = best_idx;
1930 }
1931 history.Pop();
1932 full_path.pop_back();
1933 }
1934 is_root_node = false;
1935 // Actively do any splits now rather than waiting for potentially long
1936 // tree walk to get there.
1937 for (int i = 0; i <= vtp_last_filled.back(); i++) {
1938 int child_limit = (*visits_to_perform.back())[i];
1939 if (task_workers_ > 0 &&
1940 child_limit > params_.GetMinimumWorkSizeForPicking() &&
1941 child_limit <
1942 ((collision_limit - passed_off - completed_visits) * 2 / 3) &&
1943 child_limit + passed_off + completed_visits <
1944 collision_limit -
1945 params_.GetMinimumRemainingWorkSizeForPicking()) {
1946 Node* child_node = cur_iters[i].GetOrSpawnNode(/* parent */ node);
1947 history.Append(cur_iters[i].GetMove());
1948 auto [child_repetitions, child_moves_left] =
1949 GetRepetitions(full_path.size(), history.Last());
1950 full_path.push_back(
1951 {child_node, child_repetitions, child_moves_left});
1952 // Don't split if not expanded or terminal.
1953 if (!ShouldStopPickingHere(child_node, false, child_repetitions)) {
1954 bool passed = false;
1955 {
1956 // Multiple writers, so need mutex here.
1957 Mutex::Lock lock(picking_tasks_mutex_);
1958 // Ensure not to exceed size of reservation.
1959 if (picking_tasks_.size() < MAX_TASKS) {
1960 picking_tasks_.emplace_back(full_path, history, child_limit);
1961 task_count_.fetch_add(1, std::memory_order_acq_rel);
1962 task_added_.notify_all();
1963 passed = true;
1964 passed_off += child_limit;
1965 }
1966 }
1967 if (passed) {
1968 (*visits_to_perform.back())[i] = 0;
1969 }
1970 }
1971 history.Pop();
1972 full_path.pop_back();
1973 }
1974 }
1975 // Fall through to select the first child.
1976 }
1977 int min_idx = current_path.back();
1978 bool found_child = false;
1979 if (vtp_last_filled.back() > min_idx) {
1980 int idx = -1;
1981 for (auto& child : node->Edges()) {
1982 idx++;
1983 if (idx > min_idx && (*visits_to_perform.back())[idx] > 0) {
1984 current_path.back() = idx;
1985 current_path.push_back(-1);
1986 node = child.GetOrSpawnNode(/* parent */ node);
1987 history.Append(child.GetMove());
1988 std::tie(repetitions, moves_left) =
1989 GetRepetitions(full_path.size(), history.Last());
1990 full_path.push_back({node, repetitions, moves_left});
1991 found_child = true;
1992 break;
1993 }
1994 if (idx >= vtp_last_filled.back()) break;
1995 }
1996 }
1997 if (!found_child) {
1998 history.Pop();
1999 full_path.pop_back();
2000 if (full_path.size() > 0) {
2001 std::tie(node, repetitions, moves_left) = full_path.back();
2002 } else {
2003 node = nullptr;
2004 repetitions = 0;
2005 }
2006 current_path.pop_back();
2007 vtp_buffer.push_back(std::move(visits_to_perform.back()));
2008 visits_to_perform.pop_back();
2009 vtp_last_filled.pop_back();
2010 }
2011 }

2016void SearchWorker::ExtendNode(NodeToProcess& picked_node) {
2017 const auto path = picked_node.path;
2018 assert(!std::get<0>(path.back())->GetLowNode());
2019
2020 const PositionHistory& history = picked_node.history;
2021
2022 // We don't need the mutex because other threads will see that N=0 and
2023 // N-in-flight=1 and will not touch this node.
2024 const auto& board = history.Last().GetBoard();
2025 std::vector<Move> legal_moves = board.GenerateLegalMoves();
2026
2027 // Check whether it's a draw/lose by position. Importantly, we must check
2028 // these before doing the by-rule checks below.
2029 auto node = picked_node.node;
2030 if (legal_moves.empty()) {
2031 // Could be a checkmate or a stalemate
2032 if (board.IsUnderCheck()) {
2033 node->MakeTerminal(GameResult::WHITE_WON);
2034 } else {
2035 node->MakeTerminal(GameResult::DRAW);
2036 }
2037 return;
2038 }
2039
2040 // We can shortcircuit these draws-by-rule only if they aren't root;
2041 // if they are root, then thinking about them is the point.
2042 if (node != search_->root_node_) {

2044 node->MakeTerminal(GameResult::DRAW);
2045 return;
2046 }

2049 node->MakeTerminal(GameResult::DRAW);
2050 return;

2052

2054 if (picked_node.repetitions >= 2) {
2055 // Not a real terminal, set low node.
2056 }
2057 // Neither by-position or by-rule termination, but maybe it's a TB
2058 // position.
2059 else if (search_->syzygy_tb_ && !search_->root_is_in_dtz_ &&
2060 board.castlings().no_legal_castle() &&
2061 history.Last().GetRule50Ply() == 0 &&
2062 (board.ours() | board.theirs()).count() <=
2063 search_->syzygy_tb_->max_cardinality()) {
2064 ProbeState state;
2065 const WDLScore wdl =
2066 search_->syzygy_tb_->probe_wdl(history.Last(), &state);

2071 float m = 0.0f;

2073 auto parent = std::get<0>(path[path.size() - 2]);
2074 m = std::max(0.0f, parent->GetM() - 1.0f);
2075 }
2076 // If the colors seem backwards, check the checkmate check above.
2077 if (wdl == WDL_WIN) {
2078 node->MakeTerminal(GameResult::BLACK_WON, m, Terminal::Tablebase);
2079 } else if (wdl == WDL_LOSS) {
2080 node->MakeTerminal(GameResult::WHITE_WON, m, Terminal::Tablebase);
2081 } else { // Cursed wins and blessed losses count as draws.
2082 node->MakeTerminal(GameResult::DRAW, m, Terminal::Tablebase);

2084 search_->tb_hits_.fetch_add(1, std::memory_order_acq_rel);

2086 }
2087 }
2088 }
2089
2090 // Check the transposition table first and NN cache second before asking for
2091 // NN evaluation.
2092 picked_node.hash = history.HashLast(params_.GetCacheHistoryLength() + 1);
2093 auto tt_iter = search_->tt_->find(picked_node.hash);
2094 // Transposition table entry might be expired.
2095 if (tt_iter != search_->tt_->end()) {
2096 picked_node.tt_low_node = tt_iter->second.lock();
2097 }
2098 if (picked_node.tt_low_node) {
2099 assert(!tt_iter->second.expired());
2100 picked_node.is_tt_hit = true;
2101 } else {
2102 picked_node.tt_low_node = std::make_shared<LowNode>(legal_moves);
2103 picked_node.nn_queried = true;
2104 picked_node.eval->p.resize(legal_moves.size());
2105 picked_node.is_cache_hit = computation_->AddInput(
2106 EvalPosition{
2107 .pos = history.GetPositions(),
2108 .legal_moves = legal_moves,
2109 },
2110 picked_node.eval->AsPtr()) ==
2111 BackendComputation::FETCHED_IMMEDIATELY;
2112 }

2389 search_->max_depth_ =
2390 std::max(search_->max_depth_, (uint16_t)node_to_process.path.size());

2401 // (-1,-1) Loss (initial and lowest bounds)

2403 // (-1, 1) Regular node
2404 // ( 0, 0) Draw
2405 // ( 0, 1) Can't Lose
2406 // ( 1, 1) Win (highest bounds)
2407 auto lower = GameResult::BLACK_WON;
2408 auto upper = GameResult::BLACK_WON;
2409 for (const auto& edge : p->Edges()) {
2410 const auto [edge_lower, edge_upper] = edge.GetBounds();
2411 lower = std::max(edge_lower, lower);
2412 upper = std::max(edge_upper, upper);
2413
2414 // Checkmate is the best, so short-circuit.
2415 const auto is_tb = edge.IsTbTerminal();
2416 if (edge_lower == GameResult::WHITE_WON && !is_tb) {
2417 prefer_tb = false;
2418 break;
2419 } else if (edge_upper == GameResult::BLACK_WON) {
2420 // Track the longest loss.
2421 losing_m = std::max(losing_m, edge.GetM(0.0f));
2422 }
2423 prefer_tb = prefer_tb || is_tb;
2424 }
2425
2426 // The parent's bounds are flipped from the children (-max(U), -max(L))
2427 // aggregated as if it was a single child (forced move) of the same bound.
2428 // Loss (-1,-1) -> ( 1, 1) Win
2429 // Can't Win (-1, 0) -> ( 0, 1) Can't Lose
2430 // Regular (-1, 1) -> (-1, 1) Regular
2431 // Draw ( 0, 0) -> ( 0, 0) Draw
2432 // Can't Lose ( 0, 1) -> (-1, 0) Can't Win
2433 // Win ( 1, 1) -> (-1,-1) Loss
2434
2435 // Nothing left to do for ancestors if the parent would be a regular node.
2436 const auto& pl = p->GetLowNode();
2437 if (lower == GameResult::BLACK_WON && upper == GameResult::WHITE_WON) {
2438 return false;
2439 } else if (lower == upper) {
2440 // Search can stop at the parent if the bounds can't change anymore, so make
2441 // it terminal preferring shorter wins and longer losses.
2442 *n_to_fix = p->GetN();
2443 assert(*n_to_fix > 0);
2444 pl->MakeTerminal(
2445 upper, (upper == GameResult::BLACK_WON ? std::max(losing_m, m) : m),
2446 prefer_tb ? Terminal::Tablebase : Terminal::EndOfGame);
2447 // v, d and m will be set in MaybeAdjustForTerminalOrTransposition.
2448 *v_delta = pl->GetWL() + p->GetWL();
2449 *d_delta = pl->GetD() - p->GetD();
2450 *m_delta = pl->GetM() + 1 - p->GetM();
2451 p->MakeTerminal(
2452 -upper,
2453 (upper == GameResult::BLACK_WON ? std::max(losing_m, m) : m) + 1.0f,
2454 prefer_tb ? Terminal::Tablebase : Terminal::EndOfGame);
2455 } else {
2456 pl->SetBounds(lower, upper);
2457 p->SetBounds(-upper, -lower);
2458 }
2459
2460 // Bounds were set, so indicate we should check the parent too.
2461 return true;
2462}

2464// 7. Update the Search's status and progress information.
2465//~~~~~~~~~~~~~~~~~~~~
2467void SearchWorker::UpdateCounters() {
2468 LCTRACE_FUNCTION_SCOPE;
2469 search_->PopulateCommonIterationStats(&iteration_stats_);
2470 search_->MaybeTriggerStop(iteration_stats_, &latest_time_manager_hints_);
2471 search_->MaybeOutputInfo(iteration_stats_);
2472
2473 // If this thread had no work, not even out of order, then sleep for some
2474 // milliseconds. Collisions don't count as work, so have to enumerate to find
2475 // out if there was anything done.
2476 bool work_done = number_out_of_order_ > 0;
2477 if (!work_done) {
2478 for (NodeToProcess& node_to_process : minibatch_) {
2479 if (!node_to_process.IsCollision()) {
2480 work_done = true;
2481 break;
2482 }
2483 }
2484 }
2485 if (!work_done) {
2486 std::this_thread::sleep_for(std::chrono::milliseconds(10));
2487 }
2488}
