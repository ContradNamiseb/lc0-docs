65float GetContempt(std::string name, std::string contempt_str,
66 float uci_rating_adv) {
67 float contempt = uci_rating_adv;
68 for (auto& entry : StrSplit(contempt_str, ",")) {
69 // The default contempt is an empty string, so skip empty entries.
70 if (entry.length() == 0) continue;
71 auto parts = StrSplit(entry, "=");
72 if (parts.size() == 1) {
73 try {
74 contempt = std::stof(parts[0]);
75 } catch (std::exception& e) {
76 throw Exception("Invalid default contempt: " + entry);
77 }
78 } else if (parts.size() == 2) {
79 if (std::search(name.begin(), name.end(), parts[0].begin(),
80 parts[0].end(), [](unsigned char a, unsigned char b) {
81 return std::tolower(a) == std::tolower(b);
82 }) != name.end()) {
83 try {
84 contempt = std::stof(parts[1]);
85 } catch (std::exception& e) {
86 throw Exception("Invalid contempt entry: " + entry);
87 }
88 break;
89 }
90 } else {
91 throw Exception("Invalid contempt entry:" + entry);
92 }
93 }
94 return contempt;
95}

97// Calculate ratio and diff for WDL conversion from the contempt settings.
98// More accurate model, allowing book bias dependent Elo calculation.
99// Doesn't take lower accuracy of opponent into account and needs clamping.
101BaseSearchParams::WDLRescaleParams AccurateWDLRescaleParams(
102 float contempt, float draw_rate_target, float draw_rate_reference,
103 float book_exit_bias, float contempt_max, float contempt_attenuation) {
104 // Catch accidental low positive values of draw_rate_target to guarantee
105 // somewhat reasonable behavior without numerical issues.
106 if (draw_rate_target > 0.0f && draw_rate_target < 0.001f) {
107 draw_rate_target = 0.001f;
108 }
109 float scale_reference = 1.0f / std::log((1.0f + draw_rate_reference) /
110 (1.0f - draw_rate_reference));
111 float scale_target =
112 (draw_rate_target == 0 ? scale_reference
113 : 1.0f / std::log((1.0f + draw_rate_target) /
114 (1.0f - draw_rate_target)));
115 float ratio = scale_target / scale_reference;
116 float diff =
117 scale_target / (scale_reference * scale_reference) /
118 (1.0f /
119 std::pow(std::cosh(0.5f * (1 - book_exit_bias) / scale_target), 2) +
120 1.0f /
121 std::pow(std::cosh(0.5f * (1 + book_exit_bias) / scale_target), 2)) *
122 std::log(10) / 200 * std::clamp(contempt, -contempt_max, contempt_max) *
123 contempt_attenuation;
124 return BaseSearchParams::WDLRescaleParams(ratio, diff);
125}

127// Converts regular Elo into ideal UHO game pair Elo based on the same Elo
128// dependent draw rate model used below. Necessary because regular Elo doesn't
129// behave well at higher level, while the ideal UHO game pair Elo calculated
130// from the decisive game pair ratio underestimates Elo differences by a
131// factor of 2 at lower levels.
132
134float ConvertRegularToGamePairElo(float elo_regular) {
135 const float transition_sharpness = 250.0f;
136 const float transition_midpoint = 2737.0f;
137 return elo_regular +
138 0.5f * transition_sharpness *
139 std::log(1.0f + std::exp((transition_midpoint - elo_regular) /
140 transition_sharpness));
141}

143// Calculate ratio and diff for WDL conversion from the contempt settings.
144// Less accurate Elo model, but automatically chooses draw rate and accuracy
145// based on the absolute Elo of both sides. Doesn't require clamping, but still
146// uses the parameter.
148BaseSearchParams::WDLRescaleParams SimplifiedWDLRescaleParams(
149 float contempt, float draw_rate_reference, float elo_active,
150 float contempt_max, float contempt_attenuation) {
151 // Scale parameter of the logistic WDL distribution is fitted as a sigmoid,
152 // predicting b/a for the WDL model fits for Stockfish levels at the Elo in
153 // https://github.com/official-stockfish/Stockfish/pull/4341
154 // Elo dependent mu is calculated from d(mu)/d(Elo) = c * s
155 // Sigmoid parameters for the Elo dependent scaling:
156 const float scale_zero = 15.0f;
157 const float elo_slope = 425.0f;
158 const float offset = 6.75f;
159
160 float scale_reference = 1.0f / std::log((1.0f + draw_rate_reference) /
161 (1.0f - draw_rate_reference));
162 float elo_opp =
163 elo_active - std::clamp(contempt, -contempt_max, contempt_max);
164 // Convert regular Elo input into internally used game pair Elo.
165 elo_active = ConvertRegularToGamePairElo(elo_active);
166 elo_opp = ConvertRegularToGamePairElo(elo_opp);
167 // Estimate draw rate from given Elo.
168 float scale_active =
169 1.0f / (1.0f / scale_zero + std::exp(elo_active / elo_slope - offset));
170 float scale_opp =
171 1.0f / (1.0f / scale_zero + std::exp(elo_opp / elo_slope - offset));
172 // Scale of target WDL distribution uses a sigmoid with Elo as input.
173 float scale_target =
174 std::sqrt((scale_active * scale_active + scale_opp * scale_opp) / 2.0f);
175 float ratio = scale_target / scale_reference;
176 // Mu is calculated as the integral over scale(Elo) between the Elo values.
177 float mu_active =
178 -std::log(10) / 200 * scale_zero * elo_slope *
179 std::log(1.0f + std::exp(-elo_active / elo_slope + offset) / scale_zero);
180 float mu_opp =
181 -std::log(10) / 200 * scale_zero * elo_slope *
182 std::log(1.0f + std::exp(-elo_opp / elo_slope + offset) / scale_zero);
183 float diff = 1.0f / (scale_reference * scale_reference) *
184 (mu_active - mu_opp) * contempt_attenuation;
185 return BaseSearchParams::WDLRescaleParams(ratio, diff);
186}
