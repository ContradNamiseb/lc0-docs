71int Perft(const ChessBoard& board, int max_depth, bool dump = false,
72 int depth = 0) {
73 if (depth == max_depth) return 1;
74 int total_count = 0;
75 auto moves = board.GeneratePseudolegalMoves();
76
77 auto legal_moves = board.GenerateLegalMoves();
78 auto iter = legal_moves.begin();
79
80 for (const auto& move : moves) {
81 auto new_board = board;
82 new_board.ApplyMove(move);
83 if (new_board.IsUnderCheck()) {
84 if (iter != legal_moves.end()) {
85 EXPECT_NE(*iter, move)
86 << board.DebugString() << "legal:[" << iter->ToString(true)
87 << "]==pseudo:(" << move.ToString(true) << ") Under check:\n"
88 << new_board.DebugString();
89 }
90 continue;
91 }
92
93 EXPECT_EQ(*iter, move) << board.DebugString() << "legal:["
94 << iter->ToString(true) << "]pseudo:("
95 << move.ToString(true) << ") after:\n"
96 << new_board.DebugString();
97
98 new_board.Mirror();
99 ++iter;
100 int count = Perft(new_board, max_depth, dump, depth + 1);
101 if (dump && depth == 0) {
102 Move m = move;
103 if (board.flipped()) m.Flip();
104 std::cerr << m.ToString(true) << ": " << count << '\n';
105 }
106 total_count += count;
107 }
108
109 EXPECT_EQ(iter, legal_moves.end());
110 return total_count;
111}
