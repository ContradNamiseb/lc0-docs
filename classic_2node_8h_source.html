47// Children of a node are stored the following way:
48// * Edges and Nodes edges point to are stored separately.
49// * There may be dangling edges (which don't yet point to any Node object yet)
50// * Edges are stored are a simple array on heap.
51// * Nodes are stored as a linked list, and contain index_ field which shows
52// which edge of a parent that node points to.
53// Or they are stored a contiguous array of Node objects on the heap if
54// solid_children_ is true. If the children have been 'solidified' their
55// sibling links are unused and left empty. In this state there are no
56// dangling edges, but the nodes may not have ever received any visits.
57//
58// Example:
59// Parent Node
60// |
61// +-------------+-------------+----------------+--------------+
62// | | | | |
63// Edge 0(Nf3) Edge 1(Bc5) Edge 2(a4) Edge 3(Qxf7) Edge 4(a3)
64// (dangling) | (dangling) | (dangling)
65// Node, Q=0.5 Node, Q=-0.2
66//
67// Is represented as:
68// +--------------+
69// | Parent Node |
70// +--------------+ +--------+
71// | edges_ | -------------------------------------> | Edge[] |
72// | | +------------+ +--------+
73// | child_ | -> | Node | | Nf3 |
74// +--------------+ +------------+ | Bc5 |
75// | index_ = 1 | | a4 |
76// | q_ = 0.5 | +------------+ | Qxf7 |
77// | sibling_ | -> | Node | | a3 |
78// +------------+ +------------+ +--------+
79// | index_ = 3 |
80// | q_ = -0.2 |
81// | sibling_ | -> nullptr
82// +------------+
83

282 // 8 byte fields.
283 // Average value (from value head of neural network) of all visited nodes in
284 // subtree. For terminal nodes, eval is stored. This is from the perspective
285 // of the player who "just" moved to reach this position, rather than from the
286 // perspective of the player-to-move for the position.
287 // WL stands for "W minus L". Is equal to Q if draw score is 0.

299 // Also null in the solid case.

304 // flipped depending on the side to move.

306 // Estimated remaining plies.

308 // How many completed visits this node had.

311 // but not finished). This value is added to n during selection which node
312 // to pick in MCTS, and also when selecting the best move.
313 uint32_t n_in_flight_ = 0;
314
315 // 2 byte fields.
316 // Index of this node is parent's edge list.
317 uint16_t index_;
318
319 // 1 byte fields.
320 // Number of edges in @edges_.
321 uint8_t num_edges_ = 0;
322
323 // Bit fields using parts of uint8_t fields initialized in the constructor.
324 // Whether or not this node end game (with a winning of either sides or draw).
325 Terminal terminal_type_ : 2;
326 // Best and worst result for this node.
327 GameResult lower_bound_ : 2;
328 GameResult upper_bound_ : 2;
329 // Whether the child_ is actually an array of equal length to edges.
330 bool solid_children_ : 1;

422// TODO(crem) Replace this with less hacky iterator once we support C++17.
423// This class has multiple hypostases within one class:
424// * Range (begin() and end() functions)
425// * Iterator (operator++() and operator*())
426// * Element, pointed by iterator (EdgeAndNode class mainly, but Edge_Iterator
427// is useful too when client wants to call GetOrSpawnNode).
428// It's safe to slice EdgeAndNode off Edge_Iterator.
429// It's more customary to have those as three classes, but
430// creating zoo of classes and copying them around while iterating seems
431// excessive.
432//
433// All functions are not thread safe (must be externally synchronized), but
434// it's fine if GetOrSpawnNode is called between calls to functions of the
435// iterator (e.g. advancing the iterator). Other functions that manipulate
436// child_ of parent or the sibling chain are not safe to call while iterating.
437template <bool is_const>
438class Edge_Iterator : public EdgeAndNode {
439 public:
440 using Ptr = std::conditional_t<is_const, const std::unique_ptr<Node>*,
441 std::unique_ptr<Node>*>;
442 using value_type = Edge_Iterator;
443 using iterator_category = std::forward_iterator_tag;
444 using difference_type = std::ptrdiff_t;
445 using pointer = Edge_Iterator*;
446 using reference = Edge_Iterator&;
447
448 // Creates "end()" iterator.
449 Edge_Iterator() {}
450
451 // Creates "begin()" iterator. Also happens to be a range constructor.
452 // child_ptr will be nullptr if parent_node is solid children.
453 Edge_Iterator(const Node& parent_node, Ptr child_ptr)
454 : EdgeAndNode(parent_node.edges_.get(), nullptr),
455 node_ptr_(child_ptr),
456 total_count_(parent_node.num_edges_) {
457 if (edge_ && child_ptr != nullptr) Actualize();
458 if (edge_ && child_ptr == nullptr) {
459 node_ = parent_node.child_.get();
460 }
461 }
462
463 // Function to support range interface.
464 Edge_Iterator<is_const> begin() { return *this; }
465 Edge_Iterator<is_const> end() { return {}; }
466
467 // Functions to support iterator interface.
468 // Equality comparison operators are inherited from EdgeAndNode.
469 void operator++() {
470 // If it was the last edge in array, become end(), otherwise advance.
471 if (++current_idx_ == total_count_) {
472 edge_ = nullptr;
473 } else {
474 ++edge_;
475 if (node_ptr_ != nullptr) {
476 Actualize();
477 } else {
478 ++node_;
479 }
480 }
481 }
482 Edge_Iterator& operator*() { return *this; }
483
484 // If there is node, return it. Otherwise spawn a new one and return it.
485 Node* GetOrSpawnNode(Node* parent) {
486 if (node_) return node_; // If there is already a node, return it.
487 // Should never reach here in solid mode.
488 assert(node_ptr_ != nullptr);
489 Actualize(); // But maybe other thread already did that.
490 if (node_) return node_; // If it did, return.
491 // Now we are sure we have to create a new node.
492 // Suppose there are nodes with idx 3 and 7, and we want to insert one with
493 // idx 5. Here is how it looks like:
494 // node_ptr_ -> &Node(idx_.3).sibling_ -> Node(idx_.7)
495 // Here is how we do that:
496 // 1. Store pointer to a node idx_.7:
497 // node_ptr_ -> &Node(idx_.3).sibling_ -> nullptr
498 // tmp -> Node(idx_.7)
499 std::unique_ptr<Node> tmp = std::move(*node_ptr_);
500 // 2. Create fresh Node(idx_.5):
501 // node_ptr_ -> &Node(idx_.3).sibling_ -> Node(idx_.5)
502 // tmp -> Node(idx_.7)
503 *node_ptr_ = std::make_unique<Node>(parent, current_idx_);
504 // 3. Attach stored pointer back to a list:
505 // node_ptr_ ->
506 // &Node(idx_.3).sibling_ -> Node(idx_.5).sibling_ -> Node(idx_.7)
507 (*node_ptr_)->sibling_ = std::move(tmp);
508 // 4. Actualize:
509 // node_ -> &Node(idx_.5)
510 // node_ptr_ -> &Node(idx_.5).sibling_ -> Node(idx_.7)
511 Actualize();
512 return node_;
513 }
514
515 private:
516 void Actualize() {
517 // This must never be called in solid mode.
518 assert(node_ptr_ != nullptr);
519 // If node_ptr_ is behind, advance it.
520 // This is needed (and has to be 'while' rather than 'if') as other threads
521 // could spawn new nodes between &node_ptr_ and *node_ptr_ while we didn't
522 // see.
523 while (*node_ptr_ && (*node_ptr_)->index_ < current_idx_) {
524 node_ptr_ = &(*node_ptr_)->sibling_;
525 }
526 // If in the end node_ptr_ points to the node that we need, populate node_
527 // and advance node_ptr_.
528 if (*node_ptr_ && (*node_ptr_)->index_ == current_idx_) {
529 node_ = (*node_ptr_).get();
530 node_ptr_ = &node_->sibling_;
531 } else {
532 node_ = nullptr;
533 }
534 }
535
536 // Pointer to a pointer to the next node. Has to be a pointer to pointer
537 // as we'd like to update it when spawning a new node.
538 Ptr node_ptr_;
539 uint16_t current_idx_ = 0;
540 uint16_t total_count_ = 0;
541};

543// TODO(crem) Replace this with less hacky iterator once we support C++17.
544// This class has multiple hypostases within one class:
545// * Range (begin() and end() functions)
546// * Iterator (operator++() and operator*())
547// It's more customary to have those as two classes, but
548// creating zoo of classes and copying them around while iterating seems
549// excessive.
550//
551// All functions are not thread safe (must be externally synchronized).
552template <bool is_const>
553class VisitedNode_Iterator {
554 public:
555 // Creates "end()" iterator.
556 VisitedNode_Iterator() {}
557
558 // Creates "begin()" iterator. Also happens to be a range constructor.
559 // child_ptr will be nullptr if parent_node is solid children.
560 VisitedNode_Iterator(const Node& parent_node, Node* child_ptr)
561 : node_ptr_(child_ptr),
562 total_count_(parent_node.num_edges_),
563 solid_(parent_node.solid_children_) {
564 if (node_ptr_ != nullptr && node_ptr_->GetN() == 0) {
565 operator++();
566 }
567 }
568 // These are technically wrong, but are usable to compare with end().
569 bool operator==(const VisitedNode_Iterator<is_const>& other) const {
570 return node_ptr_ == other.node_ptr_;
571 }
572 bool operator!=(const VisitedNode_Iterator<is_const>& other) const {
573 return node_ptr_ != other.node_ptr_;
574 }
575
576 // Function to support range interface.
577 VisitedNode_Iterator<is_const> begin() { return *this; }
578 VisitedNode_Iterator<is_const> end() { return {}; }
579
580 // Functions to support iterator interface.
581 // Equality comparison operators are inherited from EdgeAndNode.
582 void operator++() {
583 if (solid_) {
584 while (++current_idx_ != total_count_ &&
585 node_ptr_[current_idx_].GetN() == 0) {
586 if (node_ptr_[current_idx_].GetNInFlight() == 0) {
587 // Once there is not even n in flight, we can skip to the end. This is
588 // due to policy being in sorted order meaning that additional n in
589 // flight are always selected from the front of the section with no n
590 // in flight or visited.
591 current_idx_ = total_count_;
592 break;
593 }
594 }
595 if (current_idx_ == total_count_) {
596 node_ptr_ = nullptr;
597 }
598 } else {
599 do {
600 node_ptr_ = node_ptr_->sibling_.get();
601 // If n started is 0, can jump direct to end due to sorted policy
602 // ensuring that each time a new edge becomes best for the first time,
603 // it is always the first of the section at the end that has NStarted of
604 // 0.
605 if (node_ptr_ != nullptr && node_ptr_->GetN() == 0 &&
606 node_ptr_->GetNInFlight() == 0) {
607 node_ptr_ = nullptr;
608 break;
609 }
610 } while (node_ptr_ != nullptr && node_ptr_->GetN() == 0);
611 }
612 }
613 Node* operator*() {
614 if (solid_) {
615 return &(node_ptr_[current_idx_]);
616 } else {
617 return node_ptr_;
618 }
619 }
620
621 private:
622 // Pointer to current node.
623 Node* node_ptr_ = nullptr;
624 uint16_t current_idx_ = 0;
625 uint16_t total_count_ = 0;
626 bool solid_ = false;
627};
