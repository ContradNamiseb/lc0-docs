214// Single thread worker of the search engine.
215// That used to be just a function Search::Worker(), but to parallelize it
216// within one thread, have to split into stages.
217class SearchWorker {
218 public:
219 static constexpr int kTaskCountDigits = std::numeric_limits<int>::digits + 1;
220 static constexpr int kTasksTakenShift = kTaskCountDigits/2;
221 static constexpr int kTasksTakenOne = 1 << kTasksTakenShift;
222 // Suspend is -1 for the low half.
223 static constexpr int kTaskCountSuspend = kTasksTakenOne - 1;
224
225 SearchWorker(Search* search, const SearchParams& params)
226 : search_(search),
227 history_(search_->played_history_),
228 params_(params),
229 moves_left_support_(search_->backend_attributes_.has_mlh) {
230 task_workers_ = params.GetTaskWorkersPerSearchWorker();
231 if (task_workers_ < 0) {
232 if (search_->backend_attributes_.runs_on_cpu) {
233 task_workers_ = 0;
234 } else {
235 int working_threads = std::max(
236 search_->thread_count_.load(std::memory_order_acquire) - 1, 1);
237 task_workers_ = std::min(
238 std::thread::hardware_concurrency() / working_threads - 1, 4U);
239 }
240 }
241 for (int i = 0; i < task_workers_; i++) {
242 task_workspaces_.emplace_back();
243 task_threads_.emplace_back([this, i]() {
244 LOGFILE << "Task worker " << i << " starting.";
245 this->RunTasks(i);
246 LOGFILE << "Task worker " << i << " exiting.";
247 });
248 }
249 target_minibatch_size_ = params_.GetMiniBatchSize();
250 if (target_minibatch_size_ == 0) {
251 target_minibatch_size_ =
252 search_->backend_attributes_.recommended_batch_size;
253 }
254 max_out_of_order_ =
255 std::max(1, static_cast<int>(params_.GetMaxOutOfOrderEvalsFactor() *
256 target_minibatch_size_));
257 }
258
259 ~SearchWorker();
260
261 // Runs iterations while needed.
262 void RunBlocking() {
263 LOGFILE << "Started search thread.";
264 try {
265 // A very early stop may arrive before this point, so the test is at the
266 // end to ensure at least one iteration runs before exiting.
267 do {
268 ExecuteOneIteration();
269 } while (search_->IsSearchActive());
270 } catch (std::exception& e) {
271 std::cerr << "Unhandled exception in worker thread: " << e.what()
272 << std::endl;
273 abort();
274 }
275 }
276
277 // Does one full iteration of MCTS search:
278 // 1. Initialize internal structures.
279 // 2. Gather minibatch.
280 // 3.
281 // 4. Run NN computation.
282 // 5. Retrieve NN computations (and terminal values) into nodes.
283 // 6. Propagate the new nodes' information to all their parents in the tree.
284 // 7. Update the Search's status and progress information.
285 void ExecuteOneIteration();
286
287 // The same operations one by one:
288 // 1. Initialize internal structures.
289 // @computation is the computation to use on this iteration.
290 void InitializeIteration();
291
292 // 2. Gather minibatch.
293 void GatherMinibatch();
294
295 // 2b. Copy collisions into shared_collisions_.
296 void CollectCollisions();
297
298 // 4. Run NN computation.
299 void RunNNComputation();
300
301 // 5. Retrieve NN computations (and terminal values) into nodes.
302 void FetchMinibatchResults();
303
304 // 6. Propagate the new nodes' information to all their parents in the tree.
305 void DoBackupUpdate();
306
307 // 7. Update the Search's status and progress information.
308 void UpdateCounters();

311 struct NodeToProcess {
312 bool IsExtendable() const {
313 return !is_collision && !node->IsTerminal() && !node->GetLowNode();
314 }
315 bool IsCollision() const { return is_collision; }
316 bool CanEvalOutOfOrder() const {
317 return is_tt_hit || is_cache_hit || node->IsTerminal() ||
318 node->GetLowNode();
319 }

322 BackupPath path;
323 // The node to extend.
324 Node* node;
325 std::unique_ptr<EvalResult> eval;
326 int multivisit = 0;
327 // If greater than multivisit, and other parameters don't imply a lower
328 // limit, multivist could be increased to this value without additional
329 // change in outcome of next selection.
330 int maxvisit = 0;
331 bool nn_queried = false;
332 bool is_tt_hit = false;
333 bool is_cache_hit = false;
334 bool is_collision = false;

340 bool ooo_completed = false;
341

343 int repetitions = 0;

345 static NodeToProcess Collision(const BackupPath& path, int collision_count,
346 int max_count) {
347 return NodeToProcess(path, collision_count, max_count);
348 }

353
354 std::string DebugString() const {
355 std::ostringstream oss;
356 oss << "<NodeToProcess> This:" << this << " Depth:" << path.size()
357 << " Node:" << node << " Multivisit:" << multivisit
358 << " Maxvisit:" << maxvisit << " NNQueried:" << nn_queried
359 << " TTHit:" << is_tt_hit << " CacheHit:" << is_cache_hit
360 << " Collision:" << is_collision << " OOO:" << ooo_completed
361 << " Repetitions:" << repetitions << " Path:";
362 for (auto it = path.cbegin(); it != path.cend(); ++it) {
363 if (it != path.cbegin()) oss << "->";
364 auto n = std::get<0>(*it);
365 const auto& nl = n->GetLowNode();
366 oss << n << ":" << n->GetNInFlight();
367 if (nl) {
368 oss << "(" << nl << ")";
369 }
370 }
371 oss << " --- " << std::get<0>(path.back())->DebugString();
372 if (node->GetLowNode())
373 oss << " --- " << node->GetLowNode()->DebugString();
374
375 return oss.str();
376 }
